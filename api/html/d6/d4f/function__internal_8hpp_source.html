54  template<typename T>
55  std::vector<std::pair<std::string, T>> zip(const std::vector<std::string>& id,
56  const std::vector<T>& mat) {
57  casadi_assert_dev(id.size()==mat.size());
58  std::vector<std::pair<std::string, T>> r(id.size());
59  for (casadi_uint i=0; i<r.size(); ++i) r[i] = std::make_pair(id[i], mat[i]);
60  return r;
61  }

1474  template<typename M>
1475  void FunctionInternal::call(const std::vector<M>& arg, std::vector<M>& res,
1476  bool always_inline, bool never_inline) const {
1477  // If all inputs are scalar ...
1478  if (all_scalar()) {
1479  // ... and some arguments are matrix-valued with matching dimensions ...
1480  bool matrix_call = false;
1481  std::pair<casadi_int, casadi_int> sz;
1482  for (auto&& a : arg) {
1483  if (!a.is_scalar() && !a.is_empty()) {
1484  if (!matrix_call) {
1485  // Matrix call
1486  matrix_call = true;
1487  sz = a.size();
1488  } else if (a.size()!=sz) {
1489  // Not same dimensions
1490  matrix_call = false;
1491  break;
1492  }
1493  }
1494  }
1495 
1496  // ... then, call multiple times
1497  if (matrix_call) {
1498  // Start with zeros
1499  res.resize(n_out_);
1500  M z = M::zeros(sz);
1501  for (auto&& a : res) a = z;
1502  // Call multiple times
1503  std::vector<M> arg1 = arg, res1;
1504  for (casadi_int c=0; c<sz.second; ++c) {
1505  for (casadi_int r=0; r<sz.first; ++r) {
1506  // Get scalar arguments
1507  for (casadi_int i=0; i<arg.size(); ++i) {
1508  if (arg[i].size()==sz) arg1[i] = arg[i](r, c);
1509  }
1510  // Call recursively with scalar arguments
1511  call(arg1, res1, always_inline, never_inline);
1512  // Get results
1513  casadi_assert_dev(res.size() == res1.size());
1514  for (casadi_int i=0; i<res.size(); ++i) res[i](r, c) = res1[i];
1515  }
1516  }
1517  // All elements assigned
1518  return;
1519  }
1520  }
1521 
1522  // Check if inputs need to be replaced
1523  casadi_int npar = 1;
1524  if (!matching_arg(arg, npar)) {
1525  return call(replace_arg(arg, npar), res, always_inline, never_inline);
1526  }
1527 
1528  // Call the type-specific method
1529  call_gen(arg, res, npar, always_inline, never_inline);
