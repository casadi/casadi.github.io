54  template<typename T>
55  std::vector<std::pair<std::string, T>> zip(const std::vector<std::string>& id,
56  const std::vector<T>& mat) {
57  casadi_assert_dev(id.size()==mat.size());
58  std::vector<std::pair<std::string, T>> r(id.size());
59  for (casadi_uint i=0; i<r.size(); ++i) r[i] = std::make_pair(id[i], mat[i]);
60  return r;
61  }

1467  bool always_inline, bool never_inline) const {

1469  if (all_scalar()) {
1470  // ... and some arguments are matrix-valued with matching dimensions ...
1471  bool matrix_call = false;

1473  for (auto&& a : arg) {
1474  if (!a.is_scalar() && !a.is_empty()) {
1475  if (!matrix_call) {
1476  // Matrix call
1477  matrix_call = true;
1478  sz = a.size();
1479  } else if (a.size()!=sz) {
1480  // Not same dimensions
1481  matrix_call = false;
1482  break;

1484  }
1485  }
1486

1488  if (matrix_call) {

1496  for (casadi_int r=0; r<sz.first; ++r) {
1497  // Get scalar arguments
1498  for (casadi_int i=0; i<arg.size(); ++i) {
1499  if (arg[i].size()==sz) arg1[i] = arg[i](r, c);
1500  }

1502  call(arg1, res1, always_inline, never_inline);
1503  // Get results
1504  casadi_assert_dev(res.size() == res1.size());
1505  for (casadi_int i=0; i<res.size(); ++i) res[i](r, c) = res1[i];

1511  }
