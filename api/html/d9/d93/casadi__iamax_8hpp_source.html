20 // SYMBOL "iamax"
21 template<typename T1>
22 casadi_int casadi_iamax(casadi_int n, const T1* x, casadi_int inc_x) {
23  casadi_int largest_index, i;
24  T1 t, largest_value;
25  largest_value = -1.0;
26  largest_index = -1;
27  for (i=0; i<n; ++i) {
28  t = fabs(*x);
29  x += inc_x;
30  if (t>largest_value) {
31  largest_value = t;
32  largest_index = i;
33  }
34  }
35  return largest_index;
36 }
