441  enum ConstraintType {
442  OPTI_GENERIC_EQUALITY, // g1(x,p) == g2(x,p)
443  OPTI_GENERIC_INEQUALITY, // g1(x,p) <= g2(x,p)
444  OPTI_EQUALITY, // g(x,p) == bound(p)
445  OPTI_INEQUALITY, // g(x,p) <= bound(p)
446  OPTI_DOUBLE_INEQUALITY, // lb(p) <= g(x,p) <= ub(p)
