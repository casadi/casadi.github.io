41 template<typename T1>
42 void casadi_forward_diff_new(const T1* yk, T1* J, T1 h, casadi_int n_y) {
43  // Local variables
44  casadi_int i;
45  const double *yf, *yc;
46  T1 hinv;
47  // Inverse of step size
48  hinv = 1. / h;
49  // Get stencil
50  yc = yk;
51  yf = yk + n_y;
52  // Calculate FD approximation
53  for (i = 0; i < n_y; ++i) J[i] = hinv * (yf[i] - yc[i]);
54 }

56 template<typename T1>
57 void casadi_central_diff_new(const T1* yk, T1* J, T1 h, casadi_int n_y) {
58  // Local variables
59  casadi_int i;
60  const T1 *yf, *yc, *yb;
61  T1 hinv;
62  // Inverse of step size
63  hinv = 1. / h;
64  // Get stencil
65  yb = yk;
66  yc = yk + n_y;
67  yf = yk + 2 * n_y;
68  // Set u and stencils to zero (also supresses warnings)
69  for (i = 0; i < n_y; ++i) {
70  if (isfinite(yb[i])) {
71  if (isfinite(yf[i])) {
72  // Both forward and backward allowed
73  J[i] = 0.5 * hinv * (yf[i] - yb[i]);
74  } else {
75  // Backward but not forward allowed
76  J[i] = hinv * (yc[i] - yb[i]);
77  }
78  } else {
79  if (isfinite(yf[i])) {
80  // Forward but not backward allowed
81  J[i] = hinv * (yf[i] - yc[i]);
82  } else {
83  // Neither forward nor backward possible
84  J[i] = std::numeric_limits<T1>::quiet_NaN();
85  }
86  }
87  }
88 }

90 template<typename T1>
91 T1 casadi_central_diff_err(const T1* yk, T1 h, casadi_int n_y, casadi_int i,
92  T1 abstol, T1 reltol) {
93  // Local variables
94  const T1 *yf, *yc, *yb;
95  T1 err_trunc, err_round;
96  // Get stencil
97  yb = yk;
98  yc = yk + n_y;
99  yf = yk + 2 * n_y;
100  // Only consider points where both forward and backward allowed
101  if (isfinite(yb[i]) && isfinite(yf[i])) {
102  // Truncation error
103  err_trunc = yf[i] - 2*yc[i] + yb[i];
104  // Roundoff error
105  err_round = reltol / h * fmax(fabs(yf[i] - yc[i]), fabs(yc[i] - yb[i])) + abstol;
106  // Error quotient estimate
107  return err_trunc / err_round;
108  } else {
109  // Cannot be calculated
110  return std::numeric_limits<T1>::quiet_NaN();;
111  }
112 }

114 template<typename T1>
115 T1 casadi_smoothing_diff_weights(casadi_int k, T1 yb, T1 yc, T1 yf, T1 *J) {
116  // Calculate shifted finite difference approximation, weights
117  if (k == 0) {
118  // Backward shifted
119  // 7.10 in Conte & Carl de Boor: Elementary Numerical Analysis (1972)
120  // and 25.3.4 in Abramowitz and Stegun, Handbook of Mathematical Functions (1964)
121  if (J) *J = 3 * yf - 4 * yc + yb;
122  // Relative weight is 1
123  return 1;
124  } else if (k == 1) {
125  // Central
126  // We give this the relative weight 4 since if all weights are equal,
127  // this would amount to a five point formula for the derivative
128  // (yb2 - 8*yb + 8*yf - y_f2)/(12*h)
129  // cf. 25.3.6 in Abramowitz and Stegun, Handbook of Mathematical Functions (1964)
130  if (J) *J = yf - yb;
131  // Relative weight is 4
132  return 4;
133  } else {
134  // Forward shifted, cf. backward shifted above
135  if (J) *J = -3 * yb + 4 * yc - yf;
136  // Relative weight is 1
137  return 1;
138  }
139 }

141 template<typename T1>
142 void casadi_smoothing_diff_new(const T1* yk, T1* J, T1 h, casadi_int n_y, T1 smoothing) {
143  // Stencil
144  T1 yb, yc, yf;
145  // Local variables
146  T1 Jk, wk, sw, ui, sm;
147  casadi_int i, k;
148  // Set stencils to zero (also supresses warnings)
149  yf = yc = yb = 0;
150  for (i=0; i<n_y; ++i) {
151  // Reset derivative estimate, sum of weights, error estimate
152  J[i] = sw = ui = 0;
153  // For backward shifted, central and forward shifted
154  for (k = 0; k < 3; ++k) {
155  // Get stencil
156  yb = yk[i + k * n_y];
157  yc = yk[i + (k + 1) * n_y];
158  yf = yk[i + (k + 2) * n_y];
159  // No contribuation if any value is infinite
160  if (!isfinite(yb) || !isfinite(yc) || !isfinite(yf)) continue;
161  // Calculate weights
162  wk = casadi_smoothing_diff_weights(k, yb, yc, yf, &Jk);
163  // Smoothness measure (second order derivative)
164  sm = yf - 2*yc + yb;
165  sm /= h*h;
166  // Modify the weight according to smoothness
167  wk /= sm*sm + smoothing;
168  sw += wk;
169  // Added weighted contribution to weight and error
170  J[i] += wk * Jk;
171  }
172  // If sw is 0, no stencil worked
173  if (sw == 0) {
174  // Set component to 0, return -1
175  J[i] = std::numeric_limits<T1>::quiet_NaN();
176  } else {
177  // Finalize estimate using the sum of weights and the step length
178  J[i] /= 2*h*sw;
179  }
180  }
181 }

183 template<typename T1>
184 T1 casadi_smoothing_diff_err(const T1* yk, T1 h, casadi_int n_y, casadi_int i,
185  T1 abstol, T1 reltol, T1 smoothing) {
186  // Stencil
187  T1 yb, yc, yf;
188  // Local variables
189  T1 wk, sw, ui, err_trunc, err_round, sm;
190  casadi_int k;
191  // Set stencils to zero (also supresses warnings)
192  yf = yc = yb = 0;
193  // Reset derivative estimate, sum of weights, error estimate
194  sw = ui = 0;
195  // For backward shifted, central and forward shifted
196  for (k = 0; k < 3; ++k) {
197  // Get stencil
198  yb = yk[i + k * n_y];
199  yc = yk[i + (k + 1) * n_y];
200  yf = yk[i + (k + 2) * n_y];
201  // No contribuation if any value is infinite
202  if (!isfinite(yb) || !isfinite(yc) || !isfinite(yf)) continue;
203  // Calculate weights
204  wk = casadi_smoothing_diff_weights(k, yb, yc, yf, static_cast<T1*>(0));
205  // Truncation error
206  err_trunc = yf - 2*yc + yb;
207  // Roundoff error
208  err_round = reltol/h*fmax(fabs(yf - yc), fabs(yc - yb)) + abstol;
209  // We use the second order derivative as a smoothness measure
210  sm = err_trunc/(h*h);
211  // Modify the weight according to smoothness
212  wk /= sm*sm + smoothing;
213  sw += wk;
214  // Added weighted contribution to weight and error
215  ui += wk * fabs(err_trunc / err_round);
216  }
217  // If sw is 0, no stencil worked
218  if (sw == 0) {
219  // Cannot be calculated
220  return std::numeric_limits<T1>::quiet_NaN();;
221  } else {
222  // Finalize estimate using the sum of weights and the step length
223  return ui / sw;
224  }
225 }

293  T1 smoothing) {

304  }
