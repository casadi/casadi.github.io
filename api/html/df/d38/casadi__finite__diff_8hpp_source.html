37 // SYMBOL "forward_diff"
38 template<typename T1>
39 T1 casadi_forward_diff(T1** yk, T1* y0, T1* J,
40  T1 h, casadi_int n_y, const casadi_finite_diff_mem<T1>* m) {
41  casadi_int i;
42  for (i=0; i<n_y; ++i) {
43  J[i] = (yk[0][i]-y0[i])/h;
44  }
45  return -1;
46 }

48 // SYMBOL "central_diff"
49 template<typename T1>
50 T1 casadi_central_diff(T1** yk, T1* y0, T1* J,
51  T1 h, casadi_int n_y, const casadi_finite_diff_mem<T1>* m) {
52  // Return value
53  T1 u;
54  // Stencil
55  T1 yf, yc, yb;
56  // Local variables
57  T1 err_trunc, err_round;
58  casadi_int i;
59  // Set u and stencils to zero (also supresses warnings)
60  yf = yc = yb = u = 0;
61  for (i=0; i<n_y; ++i) {
62  // Copy to local variables, return -1 if invalid entry
63  if (!isfinite((yf=yk[1][i])) || !isfinite((yc=y0[i])) || !isfinite((yb=yk[0][i]))) {
64  J[i] = std::numeric_limits<T1>::quiet_NaN();
65  u = -1;
66  continue;
67  }
68  // Central difference approximation
69  J[i] = (yf - yb)/(2*h);
70  // Truncation error
71  err_trunc = yf - 2*yc + yb;
72  // Roundoff error
73  err_round = m->reltol/h*fmax(fabs(yf-yc), fabs(yc-yb)) + m->abstol;
74  // Update error estimate
75  if (u>=0) u = fmax(u, fabs(err_trunc/err_round));
76  }
77  return u;
78 }

80 // SYMBOL "smoothing_diff"
81 template<typename T1>
82 T1 casadi_smoothing_diff(T1** yk, T1* y0, T1* J,
83  T1 h, casadi_int n_y, const casadi_finite_diff_mem<T1>* m) {
84  // Return value
85  T1 u;
86  // Stencil
87  T1 yb, yc, yf;
88  // Local variables
89  T1 Jk, wk, sw, ui, err_trunc, err_round, sm;
90  casadi_int i, k;
91  // Set u and stencils to zero (also supresses warnings)
92  yf = yc = yb = u = 0;
93  for (i=0; i<n_y; ++i) {
94  // Reset derivative estimate, sum of weights, error estimate
95  J[i] = sw = ui = 0;
96  // For backward shifted, central and forward shifted
97  for (k=0; k<3; ++k) {
98  // Calculate shifted finite difference approximation
99  if (k==0) {
100  // Backward shifted
101  // 7.10 in Conte & Carl de Boor: Elementary Numerical Analysis (1972)
102  // and 25.3.4 in Abramowitz and Stegun, Handbook of Mathematical Functions (1964)
103  if (!isfinite((yc=yk[0][i]))) continue;
104  if (!isfinite((yb=yk[1][i]))) continue;
105  yf = y0[i];
106  Jk = 3*yf - 4*yc + yb;
107  wk = 1;
108  } else if (k==1) {
109  // Central
110  // We give this the "nominal weight" 4 since if all weights are equal,
111  // this would amount to a five point formula for the derivative
112  // (yb2 - 8*yb + 8*yf - y_f2)/(12*h)
113  // cf. 25.3.6 in Abramowitz and Stegun, Handbook of Mathematical Functions (1964)
114  if (!isfinite((yf=yk[2][i]))) continue;
115  if (!isfinite((yb=yc))) continue;
116  yc = y0[i];
117  Jk = yf-yb;
118  wk = 4;
119  } else {
120  // Forward shifted
121  if (!isfinite((yc=yf))) continue;
122  if (!isfinite((yf=yk[3][i]))) continue;
123  yb = y0[i];
124  Jk = -3*yb + 4*yc - yf;
125  wk = 1;
126  }
127  // Truncation error
128  err_trunc = yf - 2*yc + yb;
129  // Roundoff error
130  err_round = m->reltol/h*fmax(fabs(yf-yc), fabs(yc-yb)) + m->abstol;
131  // We use the second order derivative as a smoothness measure
132  sm = err_trunc/(h*h);
133  // Modify the weight according to smoothness
134  wk /= sm*sm + m->smoothing;
135  sw += wk;
136  // Added weighted contribution to weight and error
137  J[i] += wk * Jk;
138  ui += wk * fabs(err_trunc/err_round);
139  }
140  // If sw is 0, no stencil worked
141  if (sw==0) {
142  // Set component to 0, return -1
143  J[i] = std::numeric_limits<T1>::quiet_NaN();
144  u = -1;
145  } else {
146  // Finalize estimate using the sum of weights and the step length
147  J[i] /= 2*h*sw;
148  if (u>=0) u = fmax(u, ui/sw);
149  }
150  }
151  return u;
152 }
