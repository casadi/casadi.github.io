41  void Input::generate(CodeGenerator& g,
42  const std::vector<casadi_int>& arg, const std::vector<casadi_int>& res) const {
43  casadi_int nnz = this->nnz();
44  if (nnz==0) return; // quick return
45  std::string a = g.arg(ind_);
46  casadi_int i = res.front();
47  if (nnz==1) {
48  g << g.workel(i) << " = " << a << " ? " << a << "[" << offset_ << "] : 0;\n";
49  } else if (offset_==0) {
50  g << g.copy(a, nnz, g.work(i, nnz)) << "\n";
51  } else {
52  g << g.copy(a + " ? " + a + "+" + str(offset_) + " : 0",
53  nnz, g.work(i, nnz)) << "\n";
54  }
55  }

68  void Output::generate(CodeGenerator& g,
69  const std::vector<casadi_int>& arg, const std::vector<casadi_int>& res) const {
70  casadi_int nnz = dep().nnz();
71  if (nnz==0) return; // quick return
72  casadi_int i = arg.front();
73  std::string r = g.res(ind_);
74  if (nnz==1) {
75  g << "if (" << r << ") " << r << "[" << offset_ << "] = " << g.workel(i) << ";\n";
76  } else if (offset_==0) {
77  g << g.copy(g.work(i, nnz), nnz, r) << "\n";
78  } else {
79  g << "if (" << r << ") "
80  << g.copy(g.work(i, nnz), nnz, r + "+" + str(offset_)) << "\n";
81  }
