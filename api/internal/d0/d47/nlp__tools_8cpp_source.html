32 template <class T>
33 void detect_simple_bounds_gen(const T& x, const T& p,
34  const T& g, const T& lbg, const T& ubg,
35  std::vector<casadi_int>& gi,
36  T& lbx, T& ubx,
37  Function& lam_forward,
38  Function& lam_backward) {
39 
40  // Read dimensions
41  casadi_int ng = g.size1();
42  casadi_int nx = x.size1();
43 
44  casadi_assert(lbg.size1()==ng, "Dimension mismatch");
45  casadi_assert(ubg.size1()==ng, "Dimension mismatch");
46  casadi_assert(g.is_column(), "Dimension mismatch");
47  casadi_assert(lbg.is_column(), "Dimension mismatch");
48  casadi_assert(ubg.is_column(), "Dimension mismatch");
49  casadi_assert(x.is_column(), "Dimension mismatch");
50 
51  // Get constraint Jacobian sparsity
52  Function temp("temp_detect_simple_bounds_gen", {x, p}, {g});
53  Sparsity sp = temp.jac_sparsity().at(0);
54  Sparsity spT = sp.T();
55 
56  // Reset result vector
57  std::vector<bool> is_simple(ng, true);
58 
59  // Check nonlinearity
60  std::vector<bool> is_nonlin = which_depends(g, x, 2, true);
61 
62  const casadi_int* row = spT.colind();
63  for (casadi_int i=0;i<ng;++i) {
64  // Check if each row of jac_g_x only depends on one column
65  bool single_dependency = row[i+1]-row[i]==1;
66  is_simple[i] = single_dependency && !is_nonlin[i];
67  }
68 
69  // Full-indices of all simple constraints
70  std::vector<casadi_int> sgi = boolvec_to_index(is_simple);
71  gi = boolvec_to_index(boolvec_not(is_simple));
72  T g_bounds = g(sgi);
73  std::vector<casadi_int> sgi_to_sgsi = lookupvector(sgi, ng);
74 
75  // Detect f2(p)x+f1(p)==0
76  Function gf = Function("gf", std::vector<T>{x, p},
77  std::vector<T>{g_bounds, jtimes(g_bounds, x, T::ones(nx, 1))});
78 
79  std::vector<T> res;
80  gf.call(std::vector<T>{0, p}, res, true, false);
81  T f1 = res[0];
82  T f2 = res[1];
83 
84  T lb = (lbg(sgi)-f1)/abs(f2);
85  T ub = (ubg(sgi)-f1)/abs(f2);
86 
87  // Start without simple bounds
88  lbx = -T::inf(nx);
89  ubx = T::inf(nx);
90 
91  const casadi_int* xi = spT.row();
92 
93 
94  /* We will process in groups
95  * (=collection of bounds that have been re-defined a certain number of times)
96  * instead of element-wise to keep the MX graph size minimal.
97  */
98 
99  std::vector< std::vector<casadi_int> > sgsi_groups, sgi_groups, sxi_groups;
100  casadi_int n_groups = 0;
101 
102  // How often has a certain variable been encountered?
103  std::vector<casadi_int> group_count(nx);
104 
105  // Loop over all constraints
106  for (casadi_int i=0;i<ng;++i) {
107  // Only treat simple ones
108  if (!is_simple[i]) continue;
109 
110  casadi_int j = xi[row[i]];
111  casadi_int& k = group_count[j];
112  // Grow as needed
113  if (k==n_groups) {
114  sgsi_groups.emplace_back();
115  sgi_groups.emplace_back();
116  sxi_groups.emplace_back();
117  n_groups++;
118  }
119  sxi_groups[k].push_back(j);
120  sgi_groups[k].push_back(i);
121  sgsi_groups[k].push_back(sgi_to_sgsi[i]);
122  k++;
123  }
124 
125  casadi_assert(n_groups>=1, "mismatch");
126  // Take min/max group-wise to determine simple bounds
127  for (casadi_int i=0;i<n_groups;++i) {
128  lbx(sxi_groups[i]) = fmax(lbx(sxi_groups[i]), lb(sgsi_groups[i]));
129  ubx(sxi_groups[i]) = fmin(ubx(sxi_groups[i]), ub(sgsi_groups[i]));
130  }
131 
132 
133  // Choose the first group as reference
134  const std::vector<casadi_int>& sxi_ref = sxi_groups[0];
135  // Indices into the first group that reproduce the contents of other groups
136  std::vector< std::vector<casadi_int> > xsub(n_groups);
137  // Identity map for first entry of xsub
138  xsub[0] = range(sxi_ref.size());
139 
140  // Computations for other xsub entries
141  std::vector<casadi_int> lookup = lookupvector(sxi_ref, nx);
142  for (casadi_int i=1;i<n_groups;++i) {
143  const std::vector<casadi_int>& sxi = sxi_groups[i];
144  for (casadi_int j=0;j<sxi.size();++j) {
145  xsub[i].push_back(lookup[sxi[j]]);
146  }
147  }
148 
149  // Determine multiplier maps
150 
151  // Symbols
152  T lam_g_forward = T::sym("lam_g", ng);
153  T lam_sg_backward = T::sym("lam_g", gi.size());
154  T lam_x_backward = T::sym("lam_x", nx);
155 
156 
157  T lam_sg_forward = lam_g_forward(sgi); // NOLINT(cppcoreguidelines-slicing)
158  T lam_x_forward = T::zeros(nx, 1);
159  T lam_g_backward = T::zeros(ng, 1);
160  lam_g_backward(gi) = lam_sg_backward;
161 
162  // Comparison expression per group
163  std::vector<T> lcomp, ucomp;
164  for (casadi_int i=0;i<n_groups;++i) {
165  lcomp.push_back(lbx(sxi_groups[i])==lb(sgsi_groups[i]));
166  ucomp.push_back(ubx(sxi_groups[i])==ub(sgsi_groups[i]));
167  }
168 
169  // How many lb/ub are active?
170  T count_lb = T::zeros(nx);
171  T count_ub = T::zeros(nx);
172  for (casadi_int i=0;i<n_groups;++i) {
173  count_lb(sxi_groups[i]) += lcomp[i];
174  count_ub(sxi_groups[i]) += ucomp[i];
175  }
176 
177  // Compute lam_x from lam_g
178  for (casadi_int i=0;i<n_groups;++i) {
179  T l = lam_sg_forward(sgsi_groups[i]); // NOLINT(cppcoreguidelines-slicing)
180  T lt = (l<0);
181  T gt = !lt;
182  lam_x_forward(sxi_groups[i]) += l*(lt*lcomp[i]+gt*ucomp[i]);
183  }
184 
185  // Compute lam_g from lam_x
186  T l = lam_x_backward(sxi_ref); // NOLINT(cppcoreguidelines-slicing)
187  T lt = (l<0);
188  T gt = !lt;
189  T lam_xl = l*lt/count_lb(sxi_ref);
190  T lam_xu = l*gt/count_ub(sxi_ref);
191 
192  for (casadi_int i=0;i<n_groups;++i) {
193  lam_g_backward(sgi_groups[i]) = lam_xl(xsub[i])*lcomp[i]+lam_xu(xsub[i])*ucomp[i];
194  }
195 
196  // Construct Functions for mappings
197  lam_forward = Function("lam_forward",
198  {lam_g_forward, p}, {lam_g_forward(gi), lam_x_forward},// NOLINT(cppcoreguidelines-slicing)
199  {"lam_g", "p"}, {"lam_sg", "lam_x"});
200  casadi_assert_dev(!lam_forward.has_free());
201  lam_backward = Function("lam_backward",
202  {lam_sg_backward, lam_x_backward, p}, {lam_g_backward},
203  {"lam_sg", "lam_x", "p"}, {"lam_g"});
204  casadi_assert_dev(!lam_backward.has_free());
205  }
