61  int Multiplication::eval_gen(const T** arg, T** res, casadi_int* iw, T* w) const {
62  if (arg[0]!=res[0]) std::copy(arg[0], arg[0]+dep(0).nnz(), res[0]);
63  casadi_mtimes(arg[1], dep(1).sparsity(),
64  arg[2], dep(2).sparsity(),
65  res[0], sparsity(), w, false);

69  void Multiplication::ad_forward(const std::vector<std::vector<MX> >& fseed,
70  std::vector<std::vector<MX> >& fsens) const {
71  for (casadi_int d=0; d<fsens.size(); ++d) {
72  fsens[d][0] = fseed[d][0]
73  + mac(dep(1), fseed[d][2], MX::zeros(dep(0).sparsity()))
74  + mac(fseed[d][1], dep(2), MX::zeros(dep(0).sparsity()));
75  }
76  }

78  void Multiplication::ad_reverse(const std::vector<std::vector<MX> >& aseed,
79  std::vector<std::vector<MX> >& asens) const {
80  for (casadi_int d=0; d<aseed.size(); ++d) {
81  asens[d][1] += mac(aseed[d][0], dep(2).T(), MX::zeros(dep(1).sparsity()));
82  asens[d][2] += mac(dep(1).T(), aseed[d][0], MX::zeros(dep(2).sparsity()));
83  asens[d][0] += aseed[d][0];
84  }
85  }

102  Sparsity::mul_sparsityR(arg[1], dep(1).sparsity(),

109  void Multiplication::generate(CodeGenerator& g,
110  const std::vector<casadi_int>& arg,
111  const std::vector<casadi_int>& res) const {
112  // Copy first argument if not inplace
113  if (arg[0]!=res[0]) {
114  g << g.copy(g.work(arg[0], nnz()), nnz(), g.work(res[0], nnz())) << '\n';
115  }
116 
117  // Perform sparse matrix multiplication
118  g << g.mtimes(g.work(arg[1], dep(1).nnz()), dep(1).sparsity(),
119  g.work(arg[2], dep(2).nnz()), dep(2).sparsity(),
120  g.work(res[0], nnz()), sparsity(), "w", false) << '\n';
121  }

123  void DenseMultiplication::
124  generate(CodeGenerator& g,
125  const std::vector<casadi_int>& arg, const std::vector<casadi_int>& res) const {
126  // Copy first argument if not inplace
127  if (arg[0]!=res[0]) {
128  g << g.copy(g.work(arg[0], nnz()), nnz(),
129  g.work(res[0], nnz())) << '\n';
130  }
131 
132  casadi_int nrow_x = dep(1).size1(), nrow_y = dep(2).size1(), ncol_y = dep(2).size2();
133  g.local("rr", "casadi_real", "*");
134  g.local("ss", "casadi_real", "*");
135  g.local("tt", "casadi_real", "*");
136  g.local("i", "casadi_int");
137  g.local("j", "casadi_int");
138  g.local("k", "casadi_int");
139  g << "for (i=0, rr=" << g.work(res[0], nnz()) <<"; i<" << ncol_y << "; ++i)"
140  << " for (j=0; j<" << nrow_x << "; ++j, ++rr)"
141  << " for (k=0, ss=" << g.work(arg[1], dep(1).nnz()) << "+j, tt="
142  << g.work(arg[2], dep(2).nnz()) << "+i*" << nrow_y << "; k<" << nrow_y << "; ++k)"
143  << " *rr += ss[k*" << nrow_x << "]**tt++;\n";
144  }
