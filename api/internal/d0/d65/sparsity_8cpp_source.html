193  casadi_int Sparsity::add_nz(casadi_int rr, casadi_int cc) {
194  // If negative index, count from the back
195  if (rr<0) rr += size1();
196  if (cc<0) cc += size2();
197 
198  // Check consistency
199  casadi_assert(rr>=0 && rr<size1(), "Row index out of bounds");
200  casadi_assert(cc>=0 && cc<size2(), "Column index out of bounds");
201 
202  // Quick return if matrix is dense
203  if (is_dense()) return rr+cc*size1();
204 
205  // Get sparsity pattern
206  casadi_int size1=this->size1(), size2=this->size2(), nnz=this->nnz();
207  const casadi_int *colind = this->colind(), *row = this->row();
208 
209  // Quick return if we are adding an element to the end
210  if (colind[cc]==nnz || (colind[cc+1]==nnz && row[nnz-1]<rr)) {
211  std::vector<casadi_int> rowv(nnz+1);
212  std::copy(row, row+nnz, rowv.begin());
213  rowv[nnz] = rr;
214  std::vector<casadi_int> colindv(colind, colind+size2+1);
215  for (casadi_int c=cc; c<size2; ++c) colindv[c+1]++;
216  assign_cached(size1, size2, colindv, rowv);
217  return rowv.size()-1;
218  }
219 
220  // go to the place where the element should be
221  casadi_int ind;
222  for (ind=colind[cc]; ind<colind[cc+1]; ++ind) { // better: loop from the back to the front
223  if (row[ind] == rr) {
224  return ind; // element exists
225  } else if (row[ind] > rr) {
226  break; // break at the place where the element should be added
227  }
228  }
229 
230  // insert the element
231  std::vector<casadi_int> rowv = get_row(), colindv = get_colind();
232  rowv.insert(rowv.begin()+ind, rr);
233  for (casadi_int c=cc+1; c<size2+1; ++c) colindv[c]++;
234 
235  // Return the location of the new element
236  assign_cached(size1, size2, colindv, rowv);
237  return ind;
238  }

469  void Sparsity::append(const Sparsity& sp) {
470  if (sp.size1()==0 && sp.size2()==0) {
471  // Appending pattern is empty
472  return;
473  } else if (size1()==0 && size2()==0) {
474  // This is empty
475  *this = sp;
476  } else {
477  casadi_assert(size2()==sp.size2(),
478  "Sparsity::append: Dimension mismatch. "
479  "You attempt to append a shape " + sp.dim()
480  + " to a shape " + dim()
481  + ". The number of columns must match.");
482  if (sp.size1()==0) {
483  // No rows to add
484  return;
485  } else if (size1()==0) {
486  // No rows before
487  *this = sp;
488  } else if (is_column()) {
489  // Append to vector (inefficient)
490  *this = (*this)->_appendVector(*sp);
491  } else {
492  // Append to matrix (inefficient)
493  *this = vertcat({*this, sp});

620  Sparsity Sparsity::ldl(std::vector<casadi_int>& p, bool amd) const {
621  casadi_assert(is_symmetric(),
622  "LDL factorization requires a symmetric matrix");
623  // Recursive call if AMD
624  if (amd) {
625  // Get AMD reordering
626  p = this->amd();
627  // Permute sparsity pattern
628  std::vector<casadi_int> tmp;
629  Sparsity Aperm = sub(p, p, tmp);
630  // Call recursively
631  return Aperm.ldl(tmp, false);
632  }
633  // Dimension
634  casadi_int n=size1();
635  // Natural ordering
636  p = range(n);
637  // Work vector
638  std::vector<casadi_int> w(3*n);
639  // Elimination tree
640  std::vector<casadi_int> parent(n);
641  // Calculate colind in L (strictly lower entries only)
642  std::vector<casadi_int> L_colind(1+n);
643  SparsityInternal::ldl_colind(*this, get_ptr(parent), get_ptr(L_colind), get_ptr(w));
644  // Get rows in L (strictly lower entries only)
645  std::vector<casadi_int> L_row(L_colind.back());

806  void Sparsity::assign_cached(casadi_int nrow, casadi_int ncol,
807  const casadi_int* colind, const casadi_int* row, bool order_rows) {
808  // Scalars and empty patterns are handled separately
809  if (ncol==0 && nrow==0) {
810  // If empty
811  *this = getEmpty();
812  return;
813  } else if (ncol==1 && nrow==1) {
814  if (colind[ncol]==0) {
815  // If sparse scalar
816  *this = getScalarSparse();
817  return;
818  } else {
819  // If dense scalar
820  *this = getScalar();
821  return;
822  }
823  }
824 
825  // Make sure colind starts with zero
826  casadi_assert(colind[0]==0,
827  "Compressed Column Storage is not sane. "
828  "First element of colind must be zero.");
829 
830  // Make sure colind is montone
831  for (casadi_int c=0; c<ncol; c++) {
832  casadi_assert(colind[c+1]>=colind[c],
833  "Compressed Column Storage is not sane. "
834  "colind must be monotone.");
835  }
836 
837  // Check if rows correct and ordered without duplicates
838  bool rows_ordered = true;
839  for (casadi_int c=0; c<ncol; ++c) {
840  casadi_int last_r = -1;
841  for (casadi_int k=colind[c]; k<colind[c+1]; ++k) {
842  casadi_int r = row[k];
843  // Make sure values are in within the [0,ncol) range
844  casadi_assert(r>=0 && r<nrow,
845  "Compressed Column Storage is not sane.\n"
846  "row[ " + str(k) + " == " + str(r) + " not in range "
847  "[0, " + str(nrow) + ")");
848  // Check if ordered
849  if (r<=last_r) rows_ordered = false;
850  last_r = r;
851  }
852  }
853 
854  // If unordered, need to order
855  if (!rows_ordered) {
856  casadi_assert(order_rows,
857  "Compressed Column Storage is not sane.\n"
858  "Row indices not strictly monotonically increasing for each "
859  "column and reordering is not enabled.");
860  // Number of nonzeros
861  casadi_int nnz = colind[ncol];
862  // Get all columns
863  std::vector<casadi_int> col(nnz);
864  for (casadi_int c=0; c<ncol; ++c) {
865  for (casadi_int k=colind[c]; k<colind[c+1]; ++k) col[k] = c;
866  }
867  // Make sane via triplet format
868  *this = triplet(nrow, ncol, std::vector<casadi_int>(row, row+nnz), col);
869  return;
870  }
871 
872  // Hash the pattern
873  std::size_t h = hash_sparsity(nrow, ncol, colind, row);
874 
875  // Get a reference to the cache
876  CachingMap& cache = getCache();
877 
878  // Record the current number of buckets (for garbage collection below)
879  casadi_int bucket_count_before = cache.bucket_count();
880 
881  // WORKAROUND, functions do not appear to work when bucket_count==0
882  if (bucket_count_before>0) {
883 
884  // Find the range of patterns equal to the key (normally only zero or one)
885  std::pair<CachingMap::iterator, CachingMap::iterator> eq = cache.equal_range(h);
886 
887  // Loop over maching patterns
888  for (CachingMap::iterator i=eq.first; i!=eq.second; ++i) {
889 
890  // Get a weak reference to the cached sparsity pattern
891  WeakRef& wref = i->second;
892 
893  // Check if the pattern still exists
894  if (wref.alive()) {
895 
896  // Get an owning reference to the cached pattern
897  Sparsity ref = shared_cast<Sparsity>(wref.shared());
898 
899  // Check if the pattern matches
900  if (ref.is_equal(nrow, ncol, colind, row)) {
901 
902  // Found match!
903  own(ref.get());
904  return;
905 
906  } else { // There is a hash rowision (unlikely, but possible)
907  // Leave the pattern alone, continue to the next matching pattern
908  continue;
909  }
910  } else {
911 
912  // Check if one of the other cache entries indeed has a matching sparsity
913  CachingMap::iterator j=i;
914  j++; // Start at the next matching key
915  for (; j!=eq.second; ++j) {
916  if (j->second.alive()) {
917 
918  // Recover cached sparsity
919  Sparsity ref = shared_cast<Sparsity>(j->second.shared());
920 
921  // Match found if sparsity matches
922  if (ref.is_equal(nrow, ncol, colind, row)) {
923  own(ref.get());
924  return;
925  }
926  }
927  }
928 
929  // The cached entry has been deleted, create a new one
930  own(new SparsityInternal(nrow, ncol, colind, row));
931 
932  // Cache this pattern
933  wref = *this;
934 
935  // Return
936  return;
937  }
938  }
939  }
940 
941  // No matching sparsity pattern could be found, create a new one
942  own(new SparsityInternal(nrow, ncol, colind, row));
943 
944  // Cache this pattern
945  cache.insert(std::pair<std::size_t, WeakRef>(h, *this));
946 
947  // Garbage collection (currently only supported for unordered_multimap)
948  casadi_int bucket_count_after = cache.bucket_count();
949 
950  // We we increased the number of buckets, take time to garbage-collect deleted references
951  if (bucket_count_before!=bucket_count_after) {
952  CachingMap::const_iterator i=cache.begin();
953  while (i!=cache.end()) {
954  if (!i->second.alive()) {
955  i = cache.erase(i);
956  } else {
957  i++;
958  }
959  }
960  }
961  }

1116  casadi_assert_dev(ncol>=0);
1117  casadi_assert(col.size()==row.size(), "inconsistent lengths");
1118

1120  std::vector<casadi_int> r_colind(ncol+1, 0);
1121  std::vector<casadi_int> r_row;
1122  r_row.reserve(row.size());
1123 
1124  // Consistency check and check if elements are already perfectly ordered with no duplicates
1125  casadi_int last_col=-1, last_row=-1;
1126  bool perfectly_ordered=true;
1127  for (casadi_int k=0; k<col.size(); ++k) {
1128  // Consistency check
1129  casadi_assert(col[k]>=0 && col[k]<ncol,
1130  "Column index (" + str(col[k]) + ") out of bounds [0," + str(ncol) + "[");
1131  casadi_assert(row[k]>=0 && row[k]<nrow,
1132  "Row index out of bounds (" + str(row[k]) + ") out of bounds [0," + str(nrow) + "[");
1133 
1134  // Check if ordering is already perfect
1135  perfectly_ordered = perfectly_ordered && (col[k]<last_col ||
1136  (col[k]==last_col && row[k]<=last_row));
1137  last_col = col[k];
1138  last_row = row[k];
1139  }
1140 
1141  // Quick return if perfectly ordered
1142  if (perfectly_ordered) {
1143  // Save rows
1144  r_row.resize(row.size());
1145  std::copy(row.begin(), row.end(), r_row.begin());
1146 
1147  // Find offset index
1148  casadi_int el=0;
1149  for (casadi_int i=0; i<ncol; ++i) {
1150  while (el<col.size() && col[el]==i) el++;

1152  }
1153 
1154  // Identity mapping
1155  mapping.resize(row.size());
1156  for (casadi_int k=0; k<row.size(); ++k) mapping[k] = k;
1157 
1158  // Quick return
1159  return Sparsity(nrow, ncol, r_colind, r_row);
1160  }

1163  std::vector<casadi_int>& mapping1 = invert_mapping ? r_row : mapping;
1164  std::vector<casadi_int>& mapping2 = invert_mapping ? mapping : r_row;
1165 
1166  // Make sure that enough memory is allocated to use as a work vector
1167  mapping1.reserve(std::max(nrow+1, static_cast<casadi_int>(col.size())));
1168

1170  std::vector<casadi_int>& rowcount = mapping1; // reuse memory
1171  rowcount.resize(nrow+1);
1172  std::fill(rowcount.begin(), rowcount.end(), 0);
1173  for (std::vector<casadi_int>::const_iterator it=row.begin(); it!=row.end(); ++it) {
1174  rowcount[*it+1]++;
1175  }
1176 
1177  // Cumsum to get index offset for each row
1178  for (casadi_int i=0; i<nrow; ++i) {
1179  rowcount[i+1] += rowcount[i];
1180  }
1181 
1182  // New row for each old row
1183  mapping2.resize(row.size());
1184  for (casadi_int k=0; k<row.size(); ++k) {
1185  mapping2[rowcount[row[k]]++] = k;
1186  }
1187 
1188  // Number of elements in each col
1189  // reuse memory, r_colind is already the right size
1190  std::vector<casadi_int>& colcount = r_colind;
1191  // and is filled with zeros
1192  for (std::vector<casadi_int>::const_iterator it=mapping2.begin(); it!=mapping2.end(); ++it) {
1193  colcount[col[*it]+1]++;
1194  }
1195 
1196  // Cumsum to get index offset for each col
1197  for (casadi_int i=0; i<ncol; ++i) {
1198  colcount[i+1] += colcount[i];
1199  }
1200 
1201  // New col for each old col
1202  mapping1.resize(col.size());
1203  for (std::vector<casadi_int>::const_iterator it=mapping2.begin(); it!=mapping2.end(); ++it) {
1204  mapping1[colcount[col[*it]]++] = *it;
1205  }
1206 
1207  // Current element in the return matrix
1208  casadi_int r_el = 0;
1209  r_row.resize(col.size());
1210 
1211  // Current nonzero
1212  std::vector<casadi_int>::const_iterator it=mapping1.begin();
1213 
1214  // Loop over columns
1215  r_colind[0] = 0;
1216  for (casadi_int i=0; i<ncol; ++i) {
1217 
1218  // Previous row (to detect duplicates)
1219  casadi_int j_prev = -1;
1220 
1221  // Loop over nonzero elements of the col
1222  while (it!=mapping1.end() && col[*it]==i) {
1223
