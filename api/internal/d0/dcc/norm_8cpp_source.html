58  void NormF::ad_forward(const std::vector<std::vector<MX> >& fseed,
59  std::vector<std::vector<MX> >& fsens) const {
60  MX self = shared_from_this<MX>();
61  for (casadi_int d=0; d<fsens.size(); ++d) {
62  fsens[d][0] = dep(0)->get_dot(fseed[d][0]) / self;

66  void NormF::ad_reverse(const std::vector<std::vector<MX> >& aseed,
67  std::vector<std::vector<MX> >& asens) const {
68  MX self = shared_from_this<MX>();
69  for (casadi_int d=0; d<aseed.size(); ++d) {
70  asens[d][0] += (aseed[d][0]/self) * dep(0);
71  }
