78  void SXNode::disp(std::ostream& stream, bool more) const {
79  // Find out which noded can be inlined

81  can_inline(nodeind);
82 
83  // Print expression

87  // Print intermediate expressions
88  for (casadi_int i=0; i<intermed.size(); ++i)

103  void SXNode::can_inline(std::map<const SXNode*, casadi_int>& nodeind) const {
104  // Add or mark node in map
105  std::map<const SXNode*, casadi_int>::iterator it=nodeind.find(this);
106  if (it==nodeind.end()) {
107  // First time encountered, mark inlined
108  nodeind.insert(it, std::make_pair(this, 0));
109 
110  // Handle dependencies with recursion
111  for (casadi_int i=0; i<n_dep(); ++i) {
112  dep(i)->can_inline(nodeind);
113  }
114  } else if (it->second==0 && op()!=OP_PARAMETER) {
115  // Node encountered before, do not inline (except if symbolic primitive)
116  it->second = -1;
117  }
118  }

120  std::string SXNode::print_compact(std::map<const SXNode*, casadi_int>& nodeind,
121  std::vector<std::string>& intermed) const {
122  // Get reference to node index
123  casadi_int& ind = nodeind[this];
124 
125  // If positive, already in intermediate expressions
126  if (ind>0) {
127  std::stringstream ss;
128  ss << "@" << ind;
129  return ss.str();
130  }
131 
132  // Get expressions for dependencies
133  std::string arg[2];
134  for (casadi_int i=0; i<n_dep(); ++i) {
135  arg[i] = dep(i)->print_compact(nodeind, intermed);
136  }
137 
138  // Get expression for this
139  std::string s = print(arg[0], arg[1]);
140 
141  // Decide what to do with the expression
142  if (ind==0) {
143  // Inline expression
144  return s;
145  } else {
146  // Add to list of intermediate expressions and return reference
147  intermed.push_back(s);
148  ind = intermed.size(); // For subsequent references
149  std::stringstream ss;
150  ss << "@" << ind;
151  return ss.str();
152  }
153  }

155  void SXNode::safe_delete(SXNode* n) {
156  // Quick return if more owners
157  if (n->count>0) return;
158  // Delete straight away if it doesn't have any dependencies
159  if (!n->n_dep()) {
160  delete n;
161  return;
162  }
163  // Stack of expressions to be deleted
164  std::stack<SXNode*> deletion_stack;
165  // Add the node to the deletion stack
166  deletion_stack.push(n);
167  // Process stack
168  while (!deletion_stack.empty()) {
169  // Top element
170  SXNode *t = deletion_stack.top();
171  // Check if the top element has dependencies with dependencies
172  bool added_to_stack = false;
173  for (casadi_int c2=0; c2<t->n_dep(); ++c2) { // for all dependencies of the dependency
174  // Get the node of the dependency of the top element
175  // and remove it from the smart pointer
176  SXNode *n2 = t->dep(c2).assignNoDelete(casadi_limits<SXElem>::nan);
177  // Check if this is the only reference to the element
178  if (n2->count == 0) {
179  // Check if unary or binary
180  if (!n2->n_dep()) {
181  // Delete straight away if not binary
182  delete n2;
183  } else {
184  // Add to deletion stack
185  deletion_stack.push(n2);
186  added_to_stack = true;
187  }
188  }
189  }
190  // Delete and pop from stack if nothing added to the stack
191  if (!added_to_stack) {
192  delete deletion_stack.top();
193  deletion_stack.pop();
194  }
195  }
196  }

209  SXNode* SXNode::deserialize(DeserializingStream& s) {
210  casadi_int op;
211  s.unpack("SXNode::op", op);
212 
213  if (casadi_math<MX>::is_binary(op)) {
214  return BinarySX::deserialize(s, op);
215  } else if (casadi_math<MX>::is_unary(op)) {
216  return UnarySX::deserialize(s, op);
217  }
218 
219  auto it = SXNode::deserialize_map.find(op);
220  if (it==SXNode::deserialize_map.end()) {
221  casadi_error("Not implemented op " + str(casadi_int(op)));
222  } else {
223  return it->second(s);
224  }
225  }
