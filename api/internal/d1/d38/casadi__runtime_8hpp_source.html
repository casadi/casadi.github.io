232  // Dense matrix multiplication
233  #define CASADI_GEMM_NT(M, N, K, A, LDA, B, LDB, C, LDC) \
234  for (i=0, rr=C; i<M; ++i) \
235  for (j=0; j<N; ++j, ++rr) \
236  for (k=0, ss=A+i*LDA, tt=B+j*LDB; k<K; ++k) \
237  *rr += *ss++**tt++;
