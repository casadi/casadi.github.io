56  SharedObject& SharedObject::operator=(const SharedObject& ref) {
57  // quick return if the old and new pointers point to the same object
58  if (node == ref.node) return *this;
59 
60  // decrease the counter and delete if this was the last pointer
61  count_down();
62 
63  // save the new pointer
64  node = ref.node;
65  count_up();
66  return *this;
67  }

77  void SharedObject::count_up() {
78 #ifdef WITH_EXTRA_CHECKS
79  casadi_assert_dev(Function::call_depth_==0);

84  void SharedObject::count_down() {
85 #ifdef WITH_EXTRA_CHECKS
86  casadi_assert_dev(Function::call_depth_==0);
87 #endif // WITH_EXTRA_CHECKS
88  if (node && --node->count == 0) {
89  delete node;
90  node = nullptr;
91  }
92  }
