56  int SXFunction::eval(const double** arg, double** res,
57  casadi_int* iw, double* w, void* mem) const {
58  if (verbose_) casadi_message(name_ + "::eval");
59 
60  // Make sure no free parameters
61  if (!free_vars_.empty()) {
62  std::stringstream ss;
63  disp(ss, false);
64  casadi_error("Cannot evaluate \"" + ss.str() + "\" since variables "
65  + str(free_vars_) + " are free.");
66  }
67 
68  // NOTE: The implementation of this function is very delicate. Small changes in the
69  // class structure can cause large performance losses. For this reason,
70  // the preprocessor macros are used below
71 
72  // Evaluate the algorithm
73  for (auto&& e : algorithm_) {
74  switch (e.op) {
75  CASADI_MATH_FUN_BUILTIN(w[e.i1], w[e.i2], w[e.i0])
76 
77  case OP_CONST: w[e.i0] = e.d; break;
78  case OP_INPUT: w[e.i0] = arg[e.i1]==nullptr ? 0 : arg[e.i1][e.i2]; break;
79  case OP_OUTPUT: if (res[e.i0]!=nullptr) res[e.i0][e.i2] = w[e.i1]; break;
80  default:
81  casadi_error("Unknown operation" + str(e.op));
82  }
83  }
84  return 0;
85  }

97  void SXFunction::disp_more(std::ostream &stream) const {
98  stream << "Algorithm:";
99 
100  // Iterator to free variables
101  std::vector<SXElem>::const_iterator p_it = free_vars_.begin();
102 
103  // Normal, interpreted output
104  for (auto&& a : algorithm_) {
105  InterruptHandler::check();
106  stream << std::endl;
107  if (a.op==OP_OUTPUT) {
108  stream << "output[" << a.i0 << "][" << a.i2 << "] = @" << a.i1;
109  } else {
110  stream << "@" << a.i0 << " = ";
111  if (a.op==OP_INPUT) {
112  stream << "input[" << a.i1 << "][" << a.i2 << "]";
113  } else {
114  if (a.op==OP_CONST) {
115  stream << a.d;
116  } else if (a.op==OP_PARAMETER) {
117  stream << *p_it++;
118  } else {
119  casadi_int ndep = casadi_math<double>::ndeps(a.op);
120  stream << casadi_math<double>::pre(a.op);
121  for (casadi_int c=0; c<ndep; ++c) {
122  if (c==0) {
123  stream << "@" << a.i1;
124  } else {
125  stream << casadi_math<double>::sep(a.op);
126  stream << "@" << a.i2;
127  }
128 
129  }
130  stream << casadi_math<double>::post(a.op);
131  }
132  }
133  }
134  stream << ";";
135  }

147  void SXFunction::codegen_body(CodeGenerator& g) const {
148 
149  // Run the algorithm
150  for (auto&& a : algorithm_) {
151  if (a.op==OP_OUTPUT) {
152  g << "if (res[" << a.i0 << "]!=0) "
153  << g.res(a.i0) << "[" << a.i2 << "]=" << g.sx_work(a.i1);
154  } else {
155 
156  // Where to store the result
157  g << g.sx_work(a.i0) << "=";
158 
159  // What to store
160  if (a.op==OP_CONST) {
161  g << g.constant(a.d);
162  } else if (a.op==OP_INPUT) {
163  g << g.arg(a.i1) << "? " << g.arg(a.i1) << "[" << a.i2 << "] : 0";
164  } else {
165  casadi_int ndep = casadi_math<double>::ndeps(a.op);
166  casadi_assert_dev(ndep>0);
167  if (ndep==1) g << g.print_op(a.op, g.sx_work(a.i1));
168  if (ndep==2) g << g.print_op(a.op, g.sx_work(a.i1), g.sx_work(a.i2));
169  }
170  }
171  g << ";\n";
172  }

211  void SXFunction::init(const Dict& opts) {
212  // Call the init function of the base class
213  XFunction<SXFunction, SX, SXNode>::init(opts);
214  if (verbose_) casadi_message(name_ + "::init");
215 
216  // Default (temporary) options
217  live_variables_ = true;
218 
219  bool cse_opt = false;
220  bool allow_free = false;
221 
222  // Read options
223  for (auto&& op : opts) {
224  if (op.first=="default_in") {
225  default_in_ = op.second;
226  } else if (op.first=="live_variables") {
227  live_variables_ = op.second;
228  } else if (op.first=="just_in_time_opencl") {
229  just_in_time_opencl_ = op.second;
230  } else if (op.first=="just_in_time_sparsity") {
231  just_in_time_sparsity_ = op.second;
232  } else if (op.first=="cse") {
233  cse_opt = op.second;
234  } else if (op.first=="allow_free") {
235  allow_free = op.second;
236  }
237  }
238 
239  if (cse_opt) out_ = cse(out_);
240 
241  // Check/set default inputs
242  if (default_in_.empty()) {
243  default_in_.resize(n_in_, 0);
244  } else {
245  casadi_assert(default_in_.size()==n_in_,
246  "Option 'default_in' has incorrect length");
247  }
248 
249  // Stack used to sort the computational graph
250  std::stack<SXNode*> s;
251 
252  // All nodes
253  std::vector<SXNode*> nodes;
254 
255  // Add the list of nodes
256  casadi_int ind=0;
257  for (auto it = out_.begin(); it != out_.end(); ++it, ++ind) {
258  casadi_int nz=0;
259  for (auto itc = (*it)->begin(); itc != (*it)->end(); ++itc, ++nz) {
260  // Add outputs to the list
261  s.push(itc->get());
262  sort_depth_first(s, nodes);
263 
264  // A null pointer means an output instruction
265  nodes.push_back(static_cast<SXNode*>(nullptr));
266  }
267  }
268 
269  casadi_assert(nodes.size() <= std::numeric_limits<int>::max(), "Integer overflow");
270  // Set the temporary variables to be the corresponding place in the sorted graph
271  for (casadi_int i=0; i<nodes.size(); ++i) {
272  if (nodes[i]) {
273  nodes[i]->temp = static_cast<int>(i);
274  }
275  }
276 
277  // Sort the nodes by type
278  constants_.clear();
279  operations_.clear();
280  for (std::vector<SXNode*>::iterator it = nodes.begin(); it != nodes.end(); ++it) {
281  SXNode* t = *it;
282  if (t) {
283  if (t->is_constant())
284  constants_.push_back(SXElem::create(t));
285  else if (!t->is_symbolic())
286  operations_.push_back(SXElem::create(t));
287  }
288  }
289 
290  // Input instructions
291  std::vector<std::pair<int, SXNode*> > symb_loc;
292 
293  // Current output and nonzero, start with the first one
294  int curr_oind, curr_nz=0;
295  casadi_assert(out_.size() <= std::numeric_limits<int>::max(), "Integer overflow");
296  for (curr_oind=0; curr_oind<out_.size(); ++curr_oind) {
297  if (out_[curr_oind].nnz()!=0) {
298  break;
299  }
300  }
301 
302  // Count the number of times each node is used
303  std::vector<casadi_int> refcount(nodes.size(), 0);
304 
305  // Get the sequence of instructions for the virtual machine
306  algorithm_.resize(0);
307  algorithm_.reserve(nodes.size());
308  for (std::vector<SXNode*>::iterator it=nodes.begin(); it!=nodes.end(); ++it) {
309  // Current node
310  SXNode* n = *it;
311 
312  // New element in the algorithm
313  AlgEl ae;
314 
315  // Get operation
316  ae.op = n==nullptr ? static_cast<int>(OP_OUTPUT) : static_cast<int>(n->op());
317 
318  // Get instruction
319  switch (ae.op) {
320  case OP_CONST: // constant
321  ae.d = n->to_double();
322  ae.i0 = n->temp;
323  break;
324  case OP_PARAMETER: // a parameter or input
325  symb_loc.push_back(std::make_pair(algorithm_.size(), n));
326  ae.i0 = n->temp;
327  ae.d = 0; // value not used, but set here to avoid uninitialized data in serialization
328  break;
329  case OP_OUTPUT: // output instruction
330  ae.i0 = curr_oind;
331  ae.i1 = out_[curr_oind]->at(curr_nz)->temp;
332  ae.i2 = curr_nz;
333 
334  // Go to the next nonzero
335  casadi_assert(curr_nz < std::numeric_limits<int>::max(), "Integer overflow");
336  curr_nz++;
337  if (curr_nz>=out_[curr_oind].nnz()) {
338  curr_nz=0;
339  casadi_assert(curr_oind < std::numeric_limits<int>::max(), "Integer overflow");
340  curr_oind++;
341  for (; curr_oind<out_.size(); ++curr_oind) {
342  if (out_[curr_oind].nnz()!=0) {
343  break;
344  }
345  }
346  }
347  break;
348  default: // Unary or binary operation
349  ae.i0 = n->temp;
350  ae.i1 = n->dep(0).get()->temp;
351  ae.i2 = n->dep(1).get()->temp;
352  }
353 
354  // Number of dependencies
355  casadi_int ndeps = casadi_math<double>::ndeps(ae.op);
356 
357  // Increase count of dependencies
358  for (casadi_int c=0; c<ndeps; ++c) {
359  refcount.at(c==0 ? ae.i1 : ae.i2)++;
360  }
361  // Add to algorithm
362  algorithm_.push_back(ae);
363  }
364 
365  // Place in the work vector for each of the nodes in the tree (overwrites the reference counter)
366  std::vector<int> place(nodes.size());
367 
368  // Stack with unused elements in the work vector
369  std::stack<int> unused;
370 
371  // Work vector size
372  int worksize = 0;
373 
374  // Find a place in the work vector for the operation
375  for (auto&& a : algorithm_) {
376 
377  // Number of dependencies
378  casadi_int ndeps = casadi_math<double>::ndeps(a.op);
379 
380  // decrease reference count of children
381  // reverse order so that the first argument will end up at the top of the stack
382  for (casadi_int c=ndeps-1; c>=0; --c) {
383  casadi_int ch_ind = c==0 ? a.i1 : a.i2;
384  casadi_int remaining = --refcount.at(ch_ind);
385  if (remaining==0) unused.push(place[ch_ind]);
386  }
387 
388  // Find a place to store the variable
389  if (a.op!=OP_OUTPUT) {
390  if (live_variables_ && !unused.empty()) {
391  // Try to reuse a variable from the stack if possible (last in, first out)
392  a.i0 = place[a.i0] = unused.top();
393  unused.pop();
394  } else {
395  // Allocate a new variable
396  a.i0 = place[a.i0] = worksize++;
397  }
398  }
399 
400  // Save the location of the children
401  for (casadi_int c=0; c<ndeps; ++c) {
402  if (c==0) {
403  a.i1 = place[a.i1];
404  } else {
405  a.i2 = place[a.i2];
406  }
407  }
408 
409  // If binary, make sure that the second argument is the same as the first one
410  // (in order to treat all operations as binary) NOTE: ugly
411  if (ndeps==1 && a.op!=OP_OUTPUT) {
412  a.i2 = a.i1;
413  }
414  }
415 
416  worksize_ = worksize;
417 
418  if (verbose_) {
419  if (live_variables_) {
420  casadi_message("Using live variables: work array is " + str(worksize_)
421  + " instead of " + str(nodes.size()));
422  } else {
423  casadi_message("Live variables disabled.");
424  }
425  }
426 
427  // Allocate work vectors (symbolic/numeric)
428  alloc_w(worksize_);
429 
430  // Reset the temporary variables
431  for (casadi_int i=0; i<nodes.size(); ++i) {
432  if (nodes[i]) {
433  nodes[i]->temp = 0;
434  }
435  }
436 
437  // Now mark each input's place in the algorithm
438  for (auto it=symb_loc.begin(); it!=symb_loc.end(); ++it) {
439  it->second->temp = it->first+1;
440  }
441 
442  // Add input instructions
443  casadi_assert(in_.size() <= std::numeric_limits<int>::max(), "Integer overflow");
444  for (int ind=0; ind<in_.size(); ++ind) {
445  int nz=0;
446  for (auto itc = in_[ind]->begin(); itc != in_[ind]->end(); ++itc, ++nz) {
447  int i = itc->get_temp()-1;
448  if (i>=0) {
449  // Mark as input
450  algorithm_[i].op = OP_INPUT;
451 
452  // Location of the input
453  algorithm_[i].i1 = ind;
454  algorithm_[i].i2 = nz;
455 
456  // Mark input as read
457  itc->set_temp(0);
458  }
459  }
460  }
461 
462  // Locate free variables
463  free_vars_.clear();
464  for (std::vector<std::pair<int, SXNode*> >::const_iterator it=symb_loc.begin();
465  it!=symb_loc.end(); ++it) {
466  if (it->second->temp!=0) {
467  // Save to list of free parameters
468  free_vars_.push_back(SXElem::create(it->second));
469 
470  // Remove marker
471  it->second->temp=0;
472  }
473  }
474 
475  if (!allow_free && has_free()) {
476  casadi_error(name_ + "::init: Initialization failed since variables [" +
477  join(get_free(), "") + "] are free. These symbols occur in the output expressions "
478  "but you forgot to declare these as inputs. "
479  "Set option 'allow_free' to allow free variables.");
480  }
481 
482  // Initialize just-in-time compilation for numeric evaluation using OpenCL
483  if (just_in_time_opencl_) {
484  casadi_error("OpenCL is not supported in this version of CasADi");
485  }
486 
487  // Initialize just-in-time compilation for sparsity propagation using OpenCL
488  if (just_in_time_sparsity_) {
489  casadi_error("OpenCL is not supported in this version of CasADi");
490  }
491 
492  // Print
493  if (verbose_) casadi_message(str(algorithm_.size()) + " elementary operations");
494  }

496  SX SXFunction::instructions_sx() const {
497  std::vector<SXElem> ret(algorithm_.size(), casadi_limits<SXElem>::nan);
498 
499  std::vector<SXElem>::iterator it=ret.begin();
500 
501  // Iterator to the binary operations
502  std::vector<SXElem>::const_iterator b_it = operations_.begin();
503 
504  // Iterator to stack of constants
505  std::vector<SXElem>::const_iterator c_it = constants_.begin();
506 
507  // Iterator to free variables
508  std::vector<SXElem>::const_iterator p_it = free_vars_.begin();
509 
510  // Evaluate algorithm
511  if (verbose_) casadi_message("Evaluating algorithm forward");
512  for (auto&& a : algorithm_) {
513  switch (a.op) {
514  case OP_INPUT:
515  case OP_OUTPUT:
516  it++;
517  break;
518  case OP_CONST:
519  *it++ = *c_it++;
520  break;
521  case OP_PARAMETER:
522  *it++ = *p_it++;
523  break;
524  default:
525  *it++ = *b_it++;
526  }
527  }
528  casadi_assert(it==ret.end(), "Dimension mismatch");
529  return ret;
530  }

532  int SXFunction::
533  eval_sx(const SXElem** arg, SXElem** res, casadi_int* iw, SXElem* w, void* mem) const {
534  if (verbose_) casadi_message(name_ + "::eval_sx");
535 
536  // Iterator to the binary operations
537  std::vector<SXElem>::const_iterator b_it=operations_.begin();
538 
539  // Iterator to stack of constants
540  std::vector<SXElem>::const_iterator c_it = constants_.begin();
541 
542  // Iterator to free variables
543  std::vector<SXElem>::const_iterator p_it = free_vars_.begin();
544 
545  // Evaluate algorithm
546  if (verbose_) casadi_message("Evaluating algorithm forward");
547  for (auto&& a : algorithm_) {
548  switch (a.op) {
549  case OP_INPUT:
550  w[a.i0] = arg[a.i1]==nullptr ? 0 : arg[a.i1][a.i2];
551  break;
552  case OP_OUTPUT:
553  if (res[a.i0]!=nullptr) res[a.i0][a.i2] = w[a.i1];
554  break;
555  case OP_CONST:
556  w[a.i0] = *c_it++;
557  break;
558  case OP_PARAMETER:
559  w[a.i0] = *p_it++; break;
560  default:
561  {
562  // Evaluate the function to a temporary value
563  // (as it might overwrite the children in the work vector)
564  SXElem f;
565  switch (a.op) {
566  CASADI_MATH_FUN_BUILTIN(w[a.i1], w[a.i2], f)
567  }
568 
569  // If this new expression is identical to the expression used
570  // to define the algorithm, then reuse
571  const casadi_int depth = 2; // NOTE: a higher depth could possibly give more savings
572  f.assignIfDuplicate(*b_it++, depth);
573 
574  // Finally save the function value
575  w[a.i0] = f;
576  }
577  }
578  }
579  return 0;
580  }

582  void SXFunction::ad_forward(const std::vector<std::vector<SX> >& fseed,
583  std::vector<std::vector<SX> >& fsens) const {
584  if (verbose_) casadi_message(name_ + "::ad_forward");
585 
586  // Number of forward seeds
587  casadi_int nfwd = fseed.size();
588  fsens.resize(nfwd);
589 
590  // Quick return if possible
591  if (nfwd==0) return;
592 
593  // Check if seeds need to have dimensions corrected
594  casadi_int npar = 1;
595  for (auto&& r : fseed) {
596  if (!matching_arg(r, npar)) {
597  casadi_assert_dev(npar==1);
598  return ad_forward(replace_fseed(fseed, npar), fsens);
599  }
600  }
601 
602  // Make sure seeds have matching sparsity patterns
603  for (auto it=fseed.begin(); it!=fseed.end(); ++it) {
604  casadi_assert_dev(it->size()==n_in_);
605  for (casadi_int i=0; i<n_in_; ++i) {
606  if (it->at(i).sparsity()!=sparsity_in_[i]) {
607  // Correct sparsity
608  std::vector<std::vector<SX> > fseed2(fseed);
609  for (auto&& r : fseed2) {
610  for (casadi_int i=0; i<n_in_; ++i) r[i] = project(r[i], sparsity_in_[i]);
611  }
612  return ad_forward(fseed2, fsens);
613  }
614  }
615  }
616 
617  // Allocate results
618  for (casadi_int d=0; d<nfwd; ++d) {
619  fsens[d].resize(n_out_);
620  for (casadi_int i=0; i<fsens[d].size(); ++i)
621  if (fsens[d][i].sparsity()!=sparsity_out_[i])
622  fsens[d][i] = SX::zeros(sparsity_out_[i]);
623  }
624 
625  // Iterator to the binary operations
626  std::vector<SXElem>::const_iterator b_it=operations_.begin();
627 
628  // Tape
629  std::vector<TapeEl<SXElem> > s_pdwork(operations_.size());
630  std::vector<TapeEl<SXElem> >::iterator it1 = s_pdwork.begin();
631 
632  // Evaluate algorithm
633  if (verbose_) casadi_message("Evaluating algorithm forward");
634  for (auto&& e : algorithm_) {
635  switch (e.op) {
636  case OP_INPUT:
637  case OP_OUTPUT:
638  case OP_CONST:
639  case OP_PARAMETER:
640  break;
641  default:
642  {
643  const SXElem& f=*b_it++;
644  switch (e.op) {
645  CASADI_MATH_DER_BUILTIN(f->dep(0), f->dep(1), f, it1++->d)
646  }
647  }
648  }
649  }
650 
651  // Work vector
652  std::vector<SXElem> w(worksize_);
653 
654  // Calculate forward sensitivities
655  if (verbose_) casadi_message("Calculating forward derivatives");
656  for (casadi_int dir=0; dir<nfwd; ++dir) {
657  std::vector<TapeEl<SXElem> >::const_iterator it2 = s_pdwork.begin();
658  for (auto&& a : algorithm_) {
659  switch (a.op) {
660  case OP_INPUT:
661  w[a.i0] = fseed[dir][a.i1].nonzeros()[a.i2]; break;
662  case OP_OUTPUT:
663  fsens[dir][a.i0].nonzeros()[a.i2] = w[a.i1]; break;
664  case OP_CONST:
665  case OP_PARAMETER:
666  w[a.i0] = 0;
667  break;
668  case OP_IF_ELSE_ZERO:
669  w[a.i0] = if_else_zero(it2++->d[1], w[a.i2]);
670  break;
671  CASADI_MATH_BINARY_BUILTIN // Binary operation
672  w[a.i0] = it2->d[0] * w[a.i1] + it2->d[1] * w[a.i2];
673  it2++;
674  break;
675  default: // Unary operation
676  w[a.i0] = it2->d[0] * w[a.i1];
677  it2++;
678  }
679  }
680  }

683  void SXFunction::ad_reverse(const std::vector<std::vector<SX> >& aseed,
684  std::vector<std::vector<SX> >& asens) const {
685  if (verbose_) casadi_message(name_ + "::ad_reverse");
686 
687  // number of adjoint seeds
688  casadi_int nadj = aseed.size();
689  asens.resize(nadj);
690 
691  // Quick return if possible
692  if (nadj==0) return;
693 
694  // Check if seeds need to have dimensions corrected
695  casadi_int npar = 1;
696  for (auto&& r : aseed) {
697  if (!matching_res(r, npar)) {
698  casadi_assert_dev(npar==1);
699  return ad_reverse(replace_aseed(aseed, npar), asens);
700  }
701  }
702 
703  // Make sure matching sparsity of fseed
704  bool matching_sparsity = true;
705  for (casadi_int d=0; d<nadj; ++d) {
706  casadi_assert_dev(aseed[d].size()==n_out_);
707  for (casadi_int i=0; matching_sparsity && i<n_out_; ++i)
708  matching_sparsity = aseed[d][i].sparsity()==sparsity_out_[i];
709  }
710 
711  // Correct sparsity if needed
712  if (!matching_sparsity) {
713  std::vector<std::vector<SX> > aseed2(aseed);
714  for (casadi_int d=0; d<nadj; ++d)
715  for (casadi_int i=0; i<n_out_; ++i)
716  if (aseed2[d][i].sparsity()!=sparsity_out_[i])
717  aseed2[d][i] = project(aseed2[d][i], sparsity_out_[i]);
718  return ad_reverse(aseed2, asens);
719  }
720 
721  // Allocate results if needed
722  for (casadi_int d=0; d<nadj; ++d) {
723  asens[d].resize(n_in_);
724  for (casadi_int i=0; i<asens[d].size(); ++i) {
725  if (asens[d][i].sparsity()!=sparsity_in_[i]) {
726  asens[d][i] = SX::zeros(sparsity_in_[i]);
727  } else {
728  std::fill(asens[d][i]->begin(), asens[d][i]->end(), 0);
729  }
730  }
731  }
732 
733  // Iterator to the binary operations
734  std::vector<SXElem>::const_iterator b_it=operations_.begin();
735 
736  // Tape
737  std::vector<TapeEl<SXElem> > s_pdwork(operations_.size());
738  std::vector<TapeEl<SXElem> >::iterator it1 = s_pdwork.begin();
739 
740  // Evaluate algorithm
741  if (verbose_) casadi_message("Evaluating algorithm forward");
742  for (auto&& a : algorithm_) {
743  switch (a.op) {
744  case OP_INPUT:
745  case OP_OUTPUT:
746  case OP_CONST:
747  case OP_PARAMETER:
748  break;
749  default:
750  {
751  const SXElem& f=*b_it++;
752  switch (a.op) {
753  CASADI_MATH_DER_BUILTIN(f->dep(0), f->dep(1), f, it1++->d)
754  }
755  }
756  }
757  }
758 
759  // Calculate adjoint sensitivities
760  if (verbose_) casadi_message("Calculating adjoint derivatives");
761 
762  // Work vector
763  std::vector<SXElem> w(worksize_, 0);
764 
765  for (casadi_int dir=0; dir<nadj; ++dir) {
766  auto it2 = s_pdwork.rbegin();
767  for (auto it = algorithm_.rbegin(); it!=algorithm_.rend(); ++it) {
768  SXElem seed;
769  switch (it->op) {
770  case OP_INPUT:
771  asens[dir][it->i1].nonzeros()[it->i2] = w[it->i0];
772  w[it->i0] = 0;
773  break;
774  case OP_OUTPUT:
775  w[it->i1] += aseed[dir][it->i0].nonzeros()[it->i2];
776  break;
777  case OP_CONST:
778  case OP_PARAMETER:
779  w[it->i0] = 0;
780  break;
781  case OP_IF_ELSE_ZERO:
782  seed = w[it->i0];
783  w[it->i0] = 0;
784  w[it->i2] += if_else_zero(it2++->d[1], seed);
785  break;
786  CASADI_MATH_BINARY_BUILTIN // Binary operation
787  seed = w[it->i0];
788  w[it->i0] = 0;
789  w[it->i1] += it2->d[0] * seed;
790  w[it->i2] += it2++->d[1] * seed;
791  break;
792  default: // Unary operation
793  seed = w[it->i0];
794  w[it->i0] = 0;
795  w[it->i1] += it2++->d[0] * seed;
796  }
797  }
798  }

801  int SXFunction::
802  sp_forward(const bvec_t** arg, bvec_t** res, casadi_int* iw, bvec_t* w, void* mem) const {
803  // Fall back when forward mode not allowed
804  if (sp_weight()==1 || sp_weight()==-1)
805  return FunctionInternal::sp_forward(arg, res, iw, w, mem);
806  // Propagate sparsity forward
807  for (auto&& e : algorithm_) {
808  switch (e.op) {
809  case OP_CONST:
810  case OP_PARAMETER:
811  w[e.i0] = 0; break;
812  case OP_INPUT:
813  w[e.i0] = arg[e.i1]==nullptr ? 0 : arg[e.i1][e.i2];
814  break;
815  case OP_OUTPUT:
816  if (res[e.i0]!=nullptr) res[e.i0][e.i2] = w[e.i1];
817  break;
818  default: // Unary or binary operation
819  w[e.i0] = w[e.i1] | w[e.i2]; break;
820  }
821  }
822  return 0;
823  }

825  int SXFunction::sp_reverse(bvec_t** arg, bvec_t** res,
826  casadi_int* iw, bvec_t* w, void* mem) const {
827  // Fall back when reverse mode not allowed
828  if (sp_weight()==0 || sp_weight()==-1)
829  return FunctionInternal::sp_reverse(arg, res, iw, w, mem);
830  std::fill_n(w, sz_w(), 0);
831 
832  // Propagate sparsity backward
833  for (auto it=algorithm_.rbegin(); it!=algorithm_.rend(); ++it) {
834  // Temp seed
835  bvec_t seed;
836 
837  // Propagate seeds
838  switch (it->op) {
839  case OP_CONST:
840  case OP_PARAMETER:
841  w[it->i0] = 0;
842  break;
843  case OP_INPUT:
844  if (arg[it->i1]!=nullptr) arg[it->i1][it->i2] |= w[it->i0];
845  w[it->i0] = 0;
846  break;
847  case OP_OUTPUT:
848  if (res[it->i0]!=nullptr) {
849  w[it->i1] |= res[it->i0][it->i2];
850  res[it->i0][it->i2] = 0;
851  }
852  break;
853  default: // Unary or binary operation
854  seed = w[it->i0];
855  w[it->i0] = 0;
856  w[it->i1] |= seed;
857  w[it->i2] |= seed;
858  }
859  }
860  return 0;
861  }
