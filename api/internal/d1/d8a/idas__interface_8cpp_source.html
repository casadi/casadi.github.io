157 int IdasInterface::resF(double t, N_Vector xz, N_Vector xzdot, N_Vector rr, void *user_data) {
158  try {
159  auto m = to_mem(user_data);
160  auto& s = m->self;
161  if (s.calc_daeF(m, t, NV_DATA_S(xz), NV_DATA_S(xz) + s.nx_,
162  NV_DATA_S(rr), NV_DATA_S(rr) + s.nx_)) return 1;
163 
164  // Subtract state derivative to get residual
165  casadi_axpy(s.nx_, -1., NV_DATA_S(xzdot), NV_DATA_S(rr));
166  return 0;
167  } catch(std::exception& e) { // non-recoverable error
168  uerr() << "res failed: " << e.what() << std::endl;
169  return -1;
170  }

223 int IdasInterface::init_mem(void* mem) const {
224  if (SundialsInterface::init_mem(mem)) return 1;
225  auto m = to_mem(mem);
226 
227  // Create IDAS memory block
228  m->mem = IDACreate();
229  casadi_assert(m->mem!=nullptr, "IDACreate: Creation failed");
230 
231  // Set error handler function
232  THROWING(IDASetErrHandlerFn, m->mem, ehfun, m);
233 
234  // Set user data
235  THROWING(IDASetUserData, m->mem, m);
236 
237  // Allocate n-vectors for ivp
238  m->xzdot = N_VNew_Serial(nx_+nz_);
239 
240  // Initialize Idas
241  double t0 = 0;
242  N_VConst(0.0, m->xz);
243  N_VConst(0.0, m->xzdot);
244  IDAInit(m->mem, resF, t0, m->xz, m->xzdot);
245  if (verbose_) casadi_message("IDA initialized");
246 
247  // Include algebraic variables in error testing
248  THROWING(IDASetSuppressAlg, m->mem, suppress_algebraic_);
249 
250  // Maxinum order for the multistep method
251  THROWING(IDASetMaxOrd, m->mem, max_multistep_order_);
252 
253  // Initial step size
254  if (step0_!=0) THROWING(IDASetInitStep, m->mem, step0_);
255 
256  // Set maximum step size
257  if (max_step_size_!=0) THROWING(IDASetMaxStep, m->mem, max_step_size_);
258 
259  // Set constraints
260  if (!y_c_.empty()) {
261  N_Vector domain = N_VNew_Serial(nx_+nz_);
262  std::copy(y_c_.begin(), y_c_.end(), NV_DATA_S(domain));
263 
264  // Pass to IDA
265  int flag = IDASetConstraints(m->mem, domain);
266  casadi_assert_dev(flag==IDA_SUCCESS);
267 
268  // Free the temporary vector
269  N_VDestroy_Serial(domain);
270  }
271 
272  // Maximum order of method
273  if (max_order_) THROWING(IDASetMaxOrd, m->mem, max_order_);
274 
275  // Coeff. in the nonlinear convergence test
276  if (nonlin_conv_coeff_!=0) THROWING(IDASetNonlinConvCoef, m->mem, nonlin_conv_coeff_);
277 
278  // Scaling
279  if (!abstolv_.empty()) {
280  // Vector absolute tolerances
281  N_Vector nv_abstol = N_VNew_Serial(static_cast<long>(abstolv_.size()));
282  std::copy(abstolv_.begin(), abstolv_.end(), NV_DATA_S(nv_abstol));
283  THROWING(IDASVtolerances, m->mem, reltol_, nv_abstol);
284  N_VDestroy_Serial(nv_abstol);
285  } else if (scale_abstol_) {
286  // Scale absolute tolerances with nominal values
287  THROWING(IDASVtolerances, m->mem, reltol_, m->abstolv);
288  } else {
289  // Scalar absolute tolerances
290  THROWING(IDASStolerances, m->mem, reltol_, abstol_);
291  }
292 
293  // Maximum number of steps
294  THROWING(IDASetMaxNumSteps, m->mem, max_num_steps_);
295 
296  // Set algebraic components
297  N_Vector id = N_VNew_Serial(nx_+nz_);
298  std::fill_n(NV_DATA_S(id), nx_, 1);
299  std::fill_n(NV_DATA_S(id)+nx_, nz_, 0);
300 
301  // Pass this information to IDAS
302  THROWING(IDASetId, m->mem, id);
303 
304  // Delete the allocated memory
305  N_VDestroy_Serial(id);
306 
307  // attach a linear solver
308  if (newton_scheme_==SD_DIRECT) {
309  // Direct scheme
310  IDAMem IDA_mem = IDAMem(m->mem);
311  IDA_mem->ida_lmem = m;
312  IDA_mem->ida_lsetup = lsetupF;
313  IDA_mem->ida_lsolve = lsolveF;
314  IDA_mem->ida_setupNonNull = TRUE;
315  } else {
316  // Iterative scheme
317  switch (newton_scheme_) {
318  case SD_DIRECT: casadi_assert_dev(0);
319  case SD_GMRES: THROWING(IDASpgmr, m->mem, max_krylov_); break;
320  case SD_BCGSTAB: THROWING(IDASpbcg, m->mem, max_krylov_); break;
321  case SD_TFQMR: THROWING(IDASptfqmr, m->mem, max_krylov_); break;
322  }
323  THROWING(IDASpilsSetJacTimesVecFn, m->mem, jtimesF);
324  if (use_precon_) THROWING(IDASpilsSetPreconditioner, m->mem, psetupF, psolveF);
325  }
326 
327  // Quadrature equations
328  if (nq1_ > 0) {
329 
330  // Initialize quadratures in IDAS
331  THROWING(IDAQuadInit, m->mem, rhsQF, m->q);
332 
333  // Should the quadrature errors be used for step size control?
334  if (quad_err_con_) {
335  THROWING(IDASetQuadErrCon, m->mem, true);
336 
337  // Quadrature error tolerances
338  // TODO(Joel): vector absolute tolerances
339  THROWING(IDAQuadSStolerances, m->mem, reltol_, abstol_);
340  }
341  }
342 
343  if (verbose_) casadi_message("Attached linear solver");
344 
345  // Adjoint sensitivity problem
346  if (nadj_ > 0) {
347  m->rxzdot = N_VNew_Serial(nrx_+nrz_);
348  N_VConst(0.0, m->rxz);
349  N_VConst(0.0, m->rxzdot);
350  }
351  if (verbose_) casadi_message("Initialized adjoint sensitivities");
352 
353  // Initialize adjoint sensitivities
354  if (nadj_ > 0) {
355  int interpType = interp_==SD_HERMITE ? IDA_HERMITE : IDA_POLYNOMIAL;
356  THROWING(IDAAdjInit, m->mem, steps_per_checkpoint_, interpType);
357  }
358 
359  m->first_callB = true;
360  return 0;
361 }

390 void IdasInterface::advance(IntegratorMemory* mem,
391  const double* u, double* x, double* z, double* q) const {
392  auto m = to_mem(mem);
393 
394  // Set controls
395  casadi_copy(u, nu_, m->u);
396 
397  // Do not integrate past change in input signals or past the end
398  THROWING(IDASetStopTime, m->mem, m->t_stop);
399 
400  // Integrate, unless already at desired time
401  double ttol = 1e-9; // tolerance
402  if (fabs(m->t - m->t_next) >= ttol) {
403  // Integrate forward ...
404  double tret = m->t;
405  if (nrx_>0) { // ... with taping
406  THROWING(IDASolveF, m->mem, m->t_next, &tret, m->xz, m->xzdot, IDA_NORMAL, &m->ncheck);
407  } else { // ... without taping
408  THROWING(IDASolve, m->mem, m->t_next, &tret, m->xz, m->xzdot, IDA_NORMAL);
409  }
410  // Get quadratures
411  if (nq1_ > 0) THROWING(IDAGetQuad, m->mem, &tret, m->q);
412  }
413 
414  // Set function outputs
415  casadi_copy(NV_DATA_S(m->xz), nx_, x);
416  casadi_copy(NV_DATA_S(m->xz)+nx_, nz_, z);
417  casadi_copy(NV_DATA_S(m->q), nq_, q);
418 
419  // Get stats
420  THROWING(IDAGetIntegratorStats, m->mem, &m->nsteps, &m->nfevals, &m->nlinsetups,
421  &m->netfails, &m->qlast, &m->qcur, &m->hinused, &m->hlast, &m->hcur, &m->tcur);
422  THROWING(IDAGetNonlinSolvStats, m->mem, &m->nniters, &m->nncfails);
423 
424 }

440 void IdasInterface::z_impulseB(IdasMemory* m, const double* rz) const {
441  // Quick return if nothing to propagate
442  if (all_zero(rz, nrz_)) return;
443  // We have the following solved nonlinear system of equations:
444  // f_alg(x, z) == 0,
445  // which implicitly defines z as a function of x
446  // Linearize w.r.t. x:
447  // df_alg/dz * dz/dx + df_alg/dx == 0 <=> dz/dx = -inv(df_alg/dz)*df_alg/dx
448  // Want to calculate:
449  // adj_x = (dz/dx)^T * adj_z = -(df_alg/dx)^T * inv((df_alg/dz)^T) * adj_z
450  // = -(df_alg/dx)^T * w,
451  // where
452  // (df_alg/dz)^T * w = adj_z
453  // Augment linear system to get the system we are able to factorize
454  // [(df_ode/dx)^T - cj*I, (df_alg/dx)^T; (df_ode/dz)^T, (df_alg/dz)^T] * [v; w] = [0; adj_z]
455  // (Re)factorize linear system
456  if (psetupF(m->t, m->xz, m->xzdot, nullptr, m->cj_last, m, nullptr, nullptr, nullptr))
457  casadi_error("Linear system factorization for backwards initial conditions failed");
458  // Right-hand-side for linear system in m->v2
459  casadi_clear(m->v2, nrx_);
460  casadi_copy(rz, nrz_, m->v2 + nrx_);
461  // Solve transposed linear system of equations (Note: m->v2 not used since rxz null)
462  if (solve_transposed(m, m->t, NV_DATA_S(m->xz), nullptr, m->v2, m->v2)) {
463  casadi_error("Linear system solve for backwards initial conditions failed");
464  }
465  // Calculate: -adj_x = (df_alg/dx)^T * w
466  casadi_clear(m->v2, nrx_);
467  if (calc_daeB(m, m->t, NV_DATA_S(m->xz), NV_DATA_S(m->xz) + nx_,
468  m->v2, m->v2 + nrx_, nullptr, m->v1, m->v1 + nrx_)) {
469  casadi_error("Adjoint seed propagation for backwards initial conditions failed");
470  }
471  // Add contribution to backward state
472  casadi_axpy(nrx_, -1., m->v1, NV_DATA_S(m->rxz));
473 }

475 void IdasInterface::impulseB(IntegratorMemory* mem,
476  const double* rx, const double* rz, const double* rp) const {
477  auto m = to_mem(mem);
478 
479  // Call method in base class
480  SundialsInterface::impulseB(mem, rx, rz, rp);
481 
482  // Propagate impulse from rz to rx
483  z_impulseB(m, rz);
484 
485  if (m->first_callB) {
486  // Create backward problem
487  THROWING(IDACreateB, m->mem, &m->whichB);
488  THROWING(IDAInitB, m->mem, m->whichB, resB, m->t, m->rxz, m->rxzdot);
489  THROWING(IDASStolerancesB, m->mem, m->whichB, reltol_, abstol_);
490  THROWING(IDASetUserDataB, m->mem, m->whichB, m);
491  THROWING(IDASetMaxNumStepsB, m->mem, m->whichB, max_num_steps_);
492 
493 
494  // Set algebraic components
495  N_Vector id = N_VNew_Serial(nrx_+nrz_);
496  std::fill_n(NV_DATA_S(id), nrx_, 1);
497  std::fill_n(NV_DATA_S(id)+nrx_, nrz_, 0);
498  THROWING(IDASetIdB, m->mem, m->whichB, id);
499  N_VDestroy_Serial(id);
500 
501  // attach linear solver
502  if (newton_scheme_==SD_DIRECT) {
503  // Direct scheme
504  IDAMem IDA_mem = IDAMem(m->mem);
505  IDAadjMem IDAADJ_mem = IDA_mem->ida_adj_mem;
506  IDABMem IDAB_mem = IDAADJ_mem->IDAB_mem;
507  IDAB_mem->ida_lmem = m;
508  IDAB_mem->IDA_mem->ida_lmem = m;
509  IDAB_mem->IDA_mem->ida_lsetup = lsetupB;
510  IDAB_mem->IDA_mem->ida_lsolve = lsolveB;
511  IDAB_mem->IDA_mem->ida_setupNonNull = TRUE;
512  } else {
513  // Iterative scheme
514  switch (newton_scheme_) {
515  case SD_DIRECT: casadi_assert_dev(0);
516  case SD_GMRES: THROWING(IDASpgmrB, m->mem, m->whichB, max_krylov_); break;
517  case SD_BCGSTAB: THROWING(IDASpbcgB, m->mem, m->whichB, max_krylov_); break;
518  case SD_TFQMR: THROWING(IDASptfqmrB, m->mem, m->whichB, max_krylov_); break;
519  }
520  THROWING(IDASpilsSetJacTimesVecFnB, m->mem, m->whichB, jtimesB);
521  if (use_precon_) THROWING(IDASpilsSetPreconditionerB, m->mem, m->whichB, psetupB, psolveB);
522  }
523 
524  // Quadratures for the adjoint problem
525  THROWING(IDAQuadInitB, m->mem, m->whichB, rhsQB, m->ruq);
526  if (quad_err_con_) {
527  THROWING(IDASetQuadErrConB, m->mem, m->whichB, true);
528  THROWING(IDAQuadSStolerancesB, m->mem, m->whichB, reltol_, abstol_);
529  }
530 
531  // Mark initialized
532  m->first_callB = false;
533  } else {
534  // Re-initialize
535  THROWING(IDAReInitB, m->mem, m->whichB, m->t, m->rxz, m->rxzdot);
536  if (nrq_ > 0 || nuq_ > 0) {
537  // Workaround (bug in SUNDIALS)
538  // THROWING(IDAQuadReInitB, m->mem, m->whichB[dir], m->rq[dir]);
539  void* memB = IDAGetAdjIDABmem(m->mem, m->whichB);
540  THROWING(IDAQuadReInit, memB, m->ruq);
541  }
542  }
543 
544  // Correct initial values for the integration if necessary

650 int IdasInterface::psolveF(double t, N_Vector xz, N_Vector xzdot, N_Vector rr,
651  N_Vector rvec, N_Vector zvec, double cj, double delta, void *user_data, N_Vector tmp) {
652  try {
653  auto m = to_mem(user_data);
654  auto& s = m->self;
655 
656  // Get right-hand sides in m->v1, ordered by sensitivity directions
657  double* vx = NV_DATA_S(rvec);
658  double* vz = vx + s.nx_;
659  double* v_it = m->v1;
660  for (int d = 0; d <= s.nfwd_; ++d) {
661  casadi_copy(vx + d * s.nx1_, s.nx1_, v_it);
662  v_it += s.nx1_;
663  casadi_copy(vz + d * s.nz1_, s.nz1_, v_it);
664  v_it += s.nz1_;
665  }
666 
667  // Solve for undifferentiated right-hand-side, save to output
668  if (s.linsolF_.solve(m->jacF, m->v1, 1, false, m->mem_linsolF))
669  return 1;
670  vx = NV_DATA_S(zvec); // possibly different from rvec
671  vz = vx + s.nx_;
672  casadi_copy(m->v1, s.nx1_, vx);
673  casadi_copy(m->v1 + s.nx1_, s.nz1_, vz);
674 
675  // Sensitivity equations
676  if (s.nfwd_ > 0) {
677  // Second order correction
678  if (s.second_order_correction_) {
679  // The outputs will double as seeds for jtimesF
680  casadi_clear(vx + s.nx1_, s.nx_ - s.nx1_);
681  casadi_clear(vz + s.nz1_, s.nz_ - s.nz1_);
682  if (s.calc_jtimesF(m, t, NV_DATA_S(xz), NV_DATA_S(xz) + s.nx_,
683  vx, vz, m->v2, m->v2 + s.nx_)) return 1;
684 
685  // Subtract m->v2 (reordered) from m->v1
686  v_it = m->v1 + s.nx1_ + s.nz1_;
687  for (int d = 1; d <= s.nfwd_; ++d) {
688  casadi_axpy(s.nx1_, -1., m->v2 + d*s.nx1_, v_it);
689  v_it += s.nx1_;
690  casadi_axpy(s.nz1_, -1., m->v2 + s.nx_ + d*s.nz1_, v_it);
691  v_it += s.nz1_;
692  }
693  }
694 
695  // Solve for sensitivity right-hand-sides
696  if (s.linsolF_.solve(m->jacF, m->v1 + s.nx1_ + s.nz1_, s.nfwd_,
697  false, m->mem_linsolF)) return 1;
698 
699  // Save to output, reordered
700  v_it = m->v1 + s.nx1_ + s.nz1_;
701  for (int d = 1; d <= s.nfwd_; ++d) {
702  casadi_copy(v_it, s.nx1_, vx + d * s.nx1_);
703  v_it += s.nx1_;
704  casadi_copy(v_it, s.nz1_, vz + d * s.nz1_);
705  v_it += s.nz1_;
706  }
707  }
708 
709  return 0;
710  } catch(std::exception& e) { // non-recoverable error
711  uerr() << "psolve failed: " << e.what() << std::endl;
712  return -1;
713  }

716 int IdasInterface::solve_transposed(IdasMemory* m, double t, const double* xz, const double* rxz,
717  const double* rhs, double* sol) const {
718  // Get right-hand sides in m->v1, ordered by sensitivity directions
719  double* v_it = m->v1;
720  for (int d = 0; d <= nfwd_; ++d) {
721  for (int a = 0; a < nadj_; ++a) {
722  casadi_copy(rhs + (d * nadj_ + a) * nrx1_, nrx1_, v_it);
723  v_it += nrx1_;
724  casadi_copy(rhs + nrx_ + (d * nadj_ + a) * nrz1_, nrz1_, v_it);
725  v_it += nrz1_;
726  }
727  }
728 
729  // Solve for undifferentiated right-hand-side, save to output
730  if (linsolF_.solve(m->jacF, m->v1, nadj_, true, m->mem_linsolF)) return 1;
731  for (int a = 0; a < nadj_; ++a) {
732  casadi_copy(m->v1 + a * (nrx1_ + nrz1_), nrx1_, sol + a * nrx1_);
733  casadi_copy(m->v1 + a * (nrx1_ + nrz1_) + nrx1_, nrz1_, sol + nrx_ + a * nrz1_);
734  }
735 
736  // Sensitivity equations
737  if (nfwd_ > 0) {
738  // Second order correction
739  if (second_order_correction_ && rxz) {
740  // The outputs will double as seeds for calc_daeB
741  casadi_clear(sol + nrx1_ * nadj_, nrx_ - nrx1_ * nadj_);
742  casadi_clear(sol + nrx_ + nrz1_ * nadj_, nrz_ - nrz1_ * nadj_);
743 
744  // Get second-order-correction, save to m->v2
745  if (calc_daeB(m, t, xz, xz + nx_, sol, sol + nrx_, nullptr,
746  m->v2, m->v2 + nrx_)) return 1;
747 
748  // Subtract m->v2 (reordered) from m->v1
749  v_it = m->v1 + (nrx1_ + nrz1_) * nadj_;
750  for (int d = 1; d <= nfwd_; ++d) {
751  for (int a = 0; a < nadj_; ++a) {
752  casadi_axpy(nrx1_, -1., m->v2 + nrx1_ * (d * nadj_ + a), v_it);
753  v_it += nrx1_;
754  casadi_axpy(nrz1_, -1., m->v2 + nrx_ + nrz1_ * (d * nadj_ + a), v_it);
755  v_it += nrz1_;
756  }
757  }
758  }
759 
760  // Solve for sensitivity right-hand-sides
761  if (linsolF_.solve(m->jacF, m->v1 + nrx1_ * nadj_ + nrz1_ * nadj_,
762  nadj_ * nfwd_, true, m->mem_linsolF)) return 1;
763 
764  // Save to output, reordered
765  v_it = m->v1 + (nrx1_ + nrz1_) * nadj_;
766  for (int d = 1; d <= nfwd_; ++d) {
767  for (int a = 0; a < nadj_; ++a) {
768  casadi_copy(v_it, nrx1_, sol + nrx1_ * (d * nadj_ + a));
769  v_it += nrx1_;
770  casadi_copy(v_it, nrz1_, sol + nrx_ + nrz1_ * (d * nadj_ + a));
771  v_it += nrz1_;
772  }
773  }
774  }
775 
776  return 0;
777 }

795 template<typename T1>
796 void casadi_copy_block(const T1* x, const casadi_int* sp_x, T1* y, const casadi_int* sp_y,
797  casadi_int r_begin, casadi_int c_begin, T1* w) {
798  // x and y should be distinct
799  casadi_int nrow_x, ncol_x, ncol_y, i_x, i_y, j, el, r_end;
800  const casadi_int *colind_x, *row_x, *colind_y, *row_y;
801  nrow_x = sp_x[0];
802  ncol_x = sp_x[1];
803  colind_x = sp_x+2; row_x = sp_x + 2 + ncol_x+1;
804  ncol_y = sp_y[1];
805  colind_y = sp_y+2; row_y = sp_y + 2 + ncol_y+1;
806  // End of the rows to be copied
807  r_end = r_begin + nrow_x;
808  // w will correspond to a column of x, initialize to zero
809  casadi_clear(w, nrow_x);
810  // Loop over columns in x
811  for (i_x = 0; i_x < ncol_x; ++i_x) {
812  // Corresponding row in y
813  i_y = i_x + c_begin;
814  // Copy x entries to w
815  for (el=colind_x[i_x]; el<colind_x[i_x + 1]; ++el) w[row_x[el]] = x[el];
816  // Copy entries to y, if in interval
817  for (el=colind_y[i_y]; el<colind_y[i_y + 1]; ++el) {
818  j = row_y[el];
819  if (j >= r_begin && j < r_end) y[el] = w[j - r_begin];
820  }
821  // Restore w
822  for (el=colind_x[i_x]; el<colind_x[i_x + 1]; ++el) w[row_x[el]] = 0;
823  }
824 }

826 int IdasInterface::psetupF(double t, N_Vector xz, N_Vector xzdot, N_Vector rr,
827  double cj, void* user_data, N_Vector tmp1, N_Vector tmp2, N_Vector tmp3) {

832  // Sparsity patterns
833  const Function& jacF = s.get_function("jacF");
834  const Sparsity& sp_jac_ode_x = jacF.sparsity_out(JACF_ODE_X);
835  const Sparsity& sp_jac_alg_x = jacF.sparsity_out(JACF_ALG_X);
836  const Sparsity& sp_jac_ode_z = jacF.sparsity_out(JACF_ODE_Z);
837  const Sparsity& sp_jac_alg_z = jacF.sparsity_out(JACF_ALG_Z);
838  const Sparsity& sp_jacF = s.linsolF_.sparsity();
839 
840  // Calculate Jacobian blocks
841  if (s.calc_jacF(m, t, NV_DATA_S(xz), NV_DATA_S(xz) + s.nx_,
842  m->jac_ode_x, m->jac_alg_x, m->jac_ode_z, m->jac_alg_z)) return 1;
843 
844  // Copy to jacF structure
845  casadi_int nx_jac = sp_jac_ode_x.size1(); // excludes sensitivity equations

849  casadi_copy_block(m->jac_alg_z, sp_jac_alg_z, m->jacF, sp_jacF, nx_jac, nx_jac, m->w);
850 
851  // Shift diagonal corresponding to jac_ode_x
852  const casadi_int *colind = sp_jacF.colind(), *row = sp_jacF.row();
853  for (casadi_int c = 0; c < nx_jac; ++c) {

856  }
857  }
858 
859  // Factorize the linear system
860  if (s.linsolF_.nfact(m->jacF, m->mem_linsolF)) return 1;
861  m->cj_last = cj;
862 
863  return 0;
864  } catch(std::exception& e) { // non-recoverable error
865  uerr() << "psetup failed: " << e.what() << std::endl;
866  return -1;
867  }
