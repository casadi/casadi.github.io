131 void SundialsSimulator::init(const Dict& opts) {
132  // Call the base class method
133  Simulator::init(opts);
134 
135  // Default options
136  abstol_ = 1e-5;
137  reltol_ = 1e-5;
138  max_num_steps_ = 5000;
139  stop_at_end_ = true;
140  use_precon_ = true;
141  max_krylov_ = 10;
142  linear_solver_ = "qr";
143  newton_scheme_ = NewtonScheme::DIRECT;
144  quad_err_con_ = false;
145  interp_ = InterpType::HERMITE;
146  steps_per_checkpoint_ = 20;
147  disable_internal_warnings_ = false;
148  max_multistep_order_ = 5;
149  step0_ = 0;
150  max_step_size_ = 0;
151  max_order_ = 0;
152  nonlin_conv_coeff_ = 0;
153  scale_abstol_ = false;
154 
155  // Read options
156  for (auto&& op : opts) {
157  if (op.first=="abstol") {
158  abstol_ = op.second;
159  } else if (op.first=="reltol") {
160  reltol_ = op.second;
161  } else if (op.first=="max_num_steps") {
162  max_num_steps_ = op.second;
163  } else if (op.first=="stop_at_end") {
164  stop_at_end_ = op.second;
165  } else if (op.first=="use_preconditioner") {
166  use_precon_ = op.second;
167  } else if (op.first=="max_krylov") {
168  max_krylov_ = op.second;
169  } else if (op.first=="newton_scheme") {
170  newton_scheme_ = to_enum<NewtonScheme>(op.second.to_string());
171  } else if (op.first=="linear_solver") {
172  linear_solver_ = op.second.to_string();
173  } else if (op.first=="linear_solver_options") {
174  linear_solver_options_ = op.second;
175  } else if (op.first=="quad_err_con") {
176  quad_err_con_ = op.second;
177  } else if (op.first=="interpolation_type") {
178  interp_ = to_enum<InterpType>(op.second.to_string());
179  } else if (op.first=="steps_per_checkpoint") {
180  steps_per_checkpoint_ = op.second;
181  } else if (op.first=="disable_internal_warnings") {
182  disable_internal_warnings_ = op.second;
183  } else if (op.first=="max_multistep_order") {
184  max_multistep_order_ = op.second;
185  } else if (op.first=="step0") {
186  step0_ = op.second;
187  } else if (op.first=="max_step_size") {
188  max_step_size_ = op.second;
189  } else if (op.first=="max_order") {
190  max_order_ = op.second;
191  } else if (op.first=="nonlin_conv_coeff") {
192  nonlin_conv_coeff_ = op.second;
193  } else if (op.first=="scale_abstol") {
194  scale_abstol_ = op.second;
195  }
196  }
197 
198  // Get or create Jacobians and linear system solvers
199  Function J = oracle_.jacobian();
200  set_function(J, "jac", true);
201 
202  // Allocate work vectors
203  // alloc_w(nx_ + nz_, true); // xz
204  alloc_w(2 * (nx_+nz_), true); // v1, v2
205  // alloc_w(nx_ + nz_, true); // xz
206 
207  // Allocate Jacobians
208  jac_x_sp_ = J.sparsity_out("jac_ode_x");
209  jac_u_sp_ = J.sparsity_out("jac_ode_u");
210  jac_p_sp_ = J.sparsity_out("jac_ode_p");
211  alloc_w(jac_x_sp_.nnz(), true); // jac_x
212  alloc_w(jac_u_sp_.nnz(), true); // jac_u
213  alloc_w(jac_p_sp_.nnz(), true); // jac_p
214 
215  // Last input values for Jacobian evaluation
216  alloc_w(oracle_.nnz_in(), true); // last_jac_in
217 
218  // Allocate linear solver
219  lin_sp_ = jac_x_sp_ + Sparsity::diag(jac_x_sp_.size());
220  linsolF_ = Linsol("linsolF", linear_solver_, lin_sp_, linear_solver_options_);
221  alloc_w(lin_sp_.nnz(), true); // lin_nz
222 
223  // Absolute tolerances for sensitivity equations
224  if (!scale_abstol_) abstolS_.resize(nfwd_, abstol_);
225 }

227 int SundialsSimulator::init_mem(void* mem) const {
228  if (Simulator::init_mem(mem)) return 1;
229  auto m = static_cast<SundialsSimMemory*>(mem);
230 
231  // State as NVector
232  m->xz = N_VNew_Serial(nx_+nz_);
233 
234  // Sensitivities as NVector
235  m->fwd_xz.resize(nfwd_);
236  for (N_Vector& e : m->fwd_xz) e = N_VNew_Serial(nx_+nz_);
237 
238  // Absolute tolerances as NVector
239  if (scale_abstol_) {
240  m->abstolv = N_VNew_Serial(nx_+nz_);
241  // Sensitivity equations have the same tolerance
242  m->abstolvS.resize(nfwd_);
243  for (N_Vector& e : m->abstolvS) e = m->abstolv;
244  }
245 
246  // Checkout linear solver instance
247  m->mem_linsolF = linsolF_.checkout();
248 
249  return 0;
250 }

252 void SundialsSimulator::reset(SimulatorMemory* mem) const {
253  auto m = static_cast<SundialsSimMemory*>(mem);
254  // Set the state
255  casadi_copy(m->xk, nx_, NV_DATA_S(m->xz));
256  casadi_copy(m->zk, nz_, NV_DATA_S(m->xz) + nx_);
257  // Set forward sensitivities
258  for (size_t i = 0; i < nfwd_; ++i) {
259  casadi_copy(m->fwd_xk + i * nx_, nx_, NV_DATA_S(m->fwd_xz[i]));
260  casadi_copy(m->fwd_zk + i * nz_, nz_, NV_DATA_S(m->fwd_xz[i]) + nx_);
261  }
262  // Clear Jacobian input cache
263  casadi_fill(m->last_jac_in, oracle_.nnz_in(), casadi::nan);
264 }

354 void SundialsSimulator::add_diag(const casadi_int* sp, double *nz, double v,
355  const casadi_int* sp_new, double *w) {
356  // Local variables
357  casadi_int nrow, ncol, r, c, k;
358  const casadi_int *colind, *row, *colind_new, *row_new;
359  // Extract sparsities
360  nrow = sp[0];
361  ncol = sp[1];
362  colind = sp + 2;
363  row = colind + ncol + 1;
364  colind_new = sp_new + 2;
365  row_new = colind_new + ncol + 1;
366  // Clear work vector
367  for (r = 0; r < nrow; ++r) w[r] = 0;
368  // Loop over columns in reverse order
369  for (c = ncol; c-- > 0; ) {
370  // Copy content of column
371  for (k = colind[c]; k < colind[c + 1]; ++k) w[row[k]] = nz[k];
372  // Add diagonal shift
373  w[c] += v;
374  // Copy content of column to new matrix
375  for (k = colind_new[c]; k < colind_new[c + 1]; ++k) nz[k] = w[row_new[k]];
376  // Restore work vector
377  for (k = colind[c]; k < colind[c + 1]; ++k) w[row[k]] = 0;
378  w[c] = 0;
379  }
380 }

382 bool SundialsSimulator::update_jac_in(SundialsSimMemory* m) const {
383  // Are there any changes to the inputs?
384  bool changed = false;
385  // Last inputs
386  double* last_jac_in = m->last_jac_in;
387  // Loop over oracle/Jacobian inputs
388  for (casadi_int i = 0; i < oracle_.n_in(); ++i) {
389  casadi_int nnz = oracle_.nnz_in(i);
390  if (m->arg[i]) {
391  // Input vector given
392  for (casadi_int k = 0; k < nnz; ++k) {
393  changed = changed || *last_jac_in != m->arg[i][k];
394  *last_jac_in++ = m->arg[i][k];
395  }
396  } else {
397  // Null: All-zero
398  for (casadi_int k = 0; k < nnz; ++k) {
399  changed = changed || *last_jac_in != 0.;
400  *last_jac_in++ = 0.;
401  }
402  }
403  }
404  return changed;
405 }
