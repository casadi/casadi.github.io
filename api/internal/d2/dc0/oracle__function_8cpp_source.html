63 void OracleFunction::init(const Dict& opts) {
64 
65  FunctionInternal::init(opts);
66 
67  // Default options
68  bool expand = false;
69 
70  show_eval_warnings_ = true;
71 
72  max_num_threads_ = 1;
73 
74  // Read options
75  for (auto&& op : opts) {
76  if (op.first=="expand") {
77  expand = op.second;
78  } else if (op.first=="common_options") {
79  common_options_ = op.second;
80  } else if (op.first=="specific_options") {
81  specific_options_ = op.second;
82  for (auto&& i : specific_options_) {
83  casadi_assert(i.second.is_dict(),
84  "specific_option must be a nested dictionary."
85  " Type mismatch for entry '" + i.first+ "': "
86  " got type " + i.second.get_description() + ".");
87  }
88  } else if (op.first=="monitor") {
89  monitor_ = op.second;
90  } else if (op.first=="show_eval_warnings") {
91  show_eval_warnings_ = op.second;
92  }
93  }
94 
95  // Replace MX oracle with SX oracle?
96  if (expand) oracle_ = oracle_.expand();
97 
98  stride_arg_ = 0;
99  stride_res_ = 0;
100  stride_iw_ = 0;
101  stride_w_ = 0;
102 
103 }

105 void OracleFunction::finalize() {
106 
107  // Allocate space for (parallel) evaluations
108  // Lifted from set_function as max_num_threads_ is not known yet in that method
109  for (auto&& e : all_functions_) {
110  Function& fcn = e.second.f;
111  // Compute strides for multi threading
112  size_t sz_arg, sz_res, sz_iw, sz_w;
113  fcn.sz_work(sz_arg, sz_res, sz_iw, sz_w);
114  stride_arg_ = std::max(stride_arg_, sz_arg);
115  stride_res_ = std::max(stride_res_, sz_res);
116  stride_iw_ = std::max(stride_iw_, sz_iw);
117  stride_w_ = std::max(stride_w_, sz_w);
118  bool persistent = false;
119  alloc(fcn, persistent, max_num_threads_);
120  }
121 
122  // Set corresponding monitors
123  for (const std::string& fname : monitor_) {
124  auto it = all_functions_.find(fname);
125  if (it==all_functions_.end()) {
126  casadi_warning("Ignoring monitor '" + fname + "'."
127  " Available functions: " + join(get_function()) + ".");
128  } else {
129  if (it->second.monitored) casadi_warning("Duplicate monitor " + fname);
130  it->second.monitored = true;
131  }
132  }
133 
134  // Check specific options
135  for (auto&& i : specific_options_) {
136  if (all_functions_.find(i.first)==all_functions_.end())
137  casadi_warning("Ignoring specific_options entry '" + i.first+"'."
138  " Available functions: " + join(get_function()) + ".");
139  }
140 
141  // Recursive call
142  FunctionInternal::finalize();
143 }

163 Function OracleFunction::create_function(const Function& oracle, const std::string& fname,
164  const std::vector<std::string>& s_in,
165  const std::vector<std::string>& s_out,
166  const Function::AuxOut& aux) {
167  // Print progress
168  if (verbose_) {
169  casadi_message(name_ + "::create_function " + fname + ":" + str(s_in) + "->" + str(s_out));
170  }
171 
172  // Check if function is already in cache
173  Function ret;
174  if (incache(fname, ret)) {
175  // Consistency checks
176  casadi_assert(ret.n_in() == s_in.size(), fname + " has wrong number of inputs");
177  casadi_assert(ret.n_out() == s_out.size(), fname + " has wrong number of outputs");
178  } else {
179  // Retrieve specific set of options if available
180  Dict specific_options;
181  auto it = specific_options_.find(fname);
182  if (it!=specific_options_.end()) specific_options = it->second;
183 
184  // Combine specific and common options
185  Dict opt = combine(specific_options, common_options_);
186 
187  // Generate the function
188  ret = oracle.factory(fname, s_in, s_out, aux, opt);
189 
190  // Make sure that it's sound
191  if (ret.has_free()) {
192  casadi_error("Cannot create '" + fname + "' since " + str(ret.get_free()) + " are free.");
193  }
194 
195  // Add to cache
196  tocache(ret);
197  }
198 
199  // Save and return
200  set_function(ret, fname, true);
201  return ret;
202 }

220 int OracleFunction::
221 calc_function(OracleMemory* m, const std::string& fcn,
222  const double* const* arg, int thread_id) const {
223  auto ml = m->thread_local_mem.at(thread_id);
224  // Is the function monitored?
225  bool monitored = this->monitored(fcn);
226 
227  // Print progress
228  if (monitored) casadi_message("Calling \"" + fcn + "\"");
229 
230  // Respond to a possible Crl+C signals
231  // Python interrupt checker needs the GIL.
232  // We may not have access to it in a multi-threaded context
233  // See issue #2955
234  if (max_num_threads_==1) InterruptHandler::check();
235 
236  // Get function
237  const Function& f = get_function(fcn);
238 
239  // Get statistics structure
240  FStats& fstats = ml->fstats.at(fcn);
241 
242  // Number of inputs and outputs
243  casadi_int n_in = f.n_in(), n_out = f.n_out();
244 
245  // Prepare stats, start timer
246  ScopedTiming tic(fstats);
247 
248  // Input buffers
249  if (arg) {
250  std::fill_n(ml->arg, n_in, nullptr);
251  for (casadi_int i=0; i<n_in; ++i) ml->arg[i] = *arg++;
252  }
253 
254  // Print inputs nonzeros
255  if (monitored) {
256  std::stringstream s;
257  s << fcn << " input nonzeros:\n";
258  for (casadi_int i=0; i<n_in; ++i) {
259  s << " " << i << " (" << f.name_in(i) << "): ";
260  if (ml->arg[i]) {
261  // Print nonzeros
262  s << "[";
263  for (casadi_int k=0; k<f.nnz_in(i); ++k) {
264  if (k!=0) s << ", ";
265  DM::print_scalar(s, ml->arg[i][k]);
266  }
267  s << "]\n";
268  } else {
269  // All zero input
270  s << "0\n";
271  }
272  }
273  casadi_message(s.str());
274  }
275 
276  // Evaluate memory-less
277  try {
278  if (f(ml->arg, ml->res, ml->iw, ml->w)) {
279  // Recoverable error
280  if (monitored) casadi_message(name_ + ":" + fcn + " failed");
281  return 1;
282  }
283  } catch(std::exception& ex) {
284  // Fatal error: Generate stack trace
285  casadi_error("Error in " + name_ + ":" + fcn + ":" + std::string(ex.what()));
286  }
287 
288  // Print output nonzeros
289  if (monitored) {
290  std::stringstream s;
291  s << fcn << " output nonzeros:\n";
292  for (casadi_int i=0; i<n_out; ++i) {
293  s << " " << i << " (" << f.name_out(i) << "): ";
294  if (ml->res[i]) {
295  // Print nonzeros
296  s << "[";
297  for (casadi_int k=0; k<f.nnz_out(i); ++k) {
298  if (k!=0) s << ", ";
299  DM::print_scalar(s, ml->res[i][k]);
300  }
301  s << "]\n";
302  } else {
303  // Ignored output
304  s << " N/A\n";
305  }
306  }
307  casadi_message(s.str());
308  }
309 
310  // Make sure not NaN or Inf
311  for (casadi_int i=0; i<n_out; ++i) {
312  if (!ml->res[i]) continue;
313  if (!std::all_of(ml->res[i], ml->res[i]+f.nnz_out(i), [](double v) { return isfinite(v);})) {
314  std::stringstream ss;
315 
316  auto it = std::find_if(ml->res[i], ml->res[i] + f.nnz_out(i),
317  [](double v) { return !isfinite(v);});
318  casadi_int k = std::distance(ml->res[i], it);
319  bool is_nan = isnan(ml->res[i][k]);
320  ss << name_ << ":" << fcn << " failed: " << (is_nan? "NaN" : "Inf") <<
321  " detected for output " << f.name_out(i) << ", at " << f.sparsity_out(i).repr_el(k) << ".";
322 
323  if (regularity_check_) {
324  casadi_error(ss.str());
325  } else {
326  if (show_eval_warnings_) casadi_warning(ss.str());
327  }
328  return -1;
329  }
330  }
331 
332  // Success
333  return 0;
334 }

356 void OracleFunction::jit_dependencies(const std::string& fname) {
357  if (compiler_.is_null()) {
358  if (verbose_) casadi_message("compiling to "+ fname+"'.");
359  // JIT dependent functions
360  compiler_ = Importer(generate_dependencies(fname, Dict()),
361  compiler_plugin_, jit_options_);
362  }
363  // Replace the Oracle functions with generated functions
364  for (auto&& e : all_functions_) {
365  if (verbose_) casadi_message("loading '" + e.second.f.name() + "' from '" + fname + "'.");
366  if (e.second.jit) {

368  e.second.f = external(e.second.f.name(), compiler_);

510 OracleFunction::OracleFunction(DeserializingStream& s) : FunctionInternal(s) {
511 
512  int version = s.version("OracleFunction", 1, 3);
513  s.unpack("OracleFunction::oracle", oracle_);
514  s.unpack("OracleFunction::common_options", common_options_);
515  s.unpack("OracleFunction::specific_options", specific_options_);
516  s.unpack("OracleFunction::show_eval_warnings", show_eval_warnings_);
517 
518  if (version>=3) {
519  s.unpack("OracleFunction::max_num_threads", max_num_threads_);
520  } else {
521  max_num_threads_ = 1;
522  }
523 
524  size_t size;
525 
526  s.unpack("OracleFunction::all_functions::size", size);
527  for (casadi_int i=0;i<size;++i) {
528  std::string key;
529  s.unpack("OracleFunction::all_functions::key", key);
530  RegFun r;
531  if (version==1) {
532  s.unpack("OracleFunction::all_functions::value::f", r.f);
533  s.unpack("OracleFunction::all_functions::value::jit", r.jit);
534  } else {
535  s.unpack("OracleFunction::all_functions::value::jit", r.jit);
536  if (jit_ && r.jit) {
537  if (jit_serialize_=="source") {
538  s.unpack("OracleFunction::all_functions::value::f", r.f);
539  } else {
540  std::string f_name;
541  s.unpack("OracleFunction::all_functions::value::f_name", f_name);
542  r.f = Function(f_name, std::vector<MX>{}, std::vector<MX>{});
543  // FunctionInternal will set compiler_
544  }
545  } else {
546  s.unpack("OracleFunction::all_functions::value::f", r.f);
547  }
548  }
549  s.unpack("OracleFunction::all_functions::value::monitored", r.monitored);
550  all_functions_[key] = r;
551  }
552  s.unpack("OracleFunction::monitor", monitor_);
553  if (version>=3) {
554  s.unpack("OracleFunction::stride_arg", stride_arg_);
555  s.unpack("OracleFunction::stride_res", stride_res_);
556  s.unpack("OracleFunction::stride_iw", stride_iw_);
557  s.unpack("OracleFunction::stride_w", stride_w_);
558  } else {
559  stride_arg_ = 0;
560  stride_res_ = 0;
561  stride_iw_ = 0;
562  stride_w_ = 0;
563  }
564 }
