82  MXNode::~MXNode() {
83 
84  // Start destruction method if any of the dependencies has dependencies
85  for (std::vector<MX>::iterator cc=dep_.begin(); cc!=dep_.end(); ++cc) {
86  // Skip if constant
87  if (cc->is_constant()) continue;
88 
89  // Check if there are other "owners" of the node
90  if (cc->getCount()!= 1) {
91 
92  // Replace with a 0-by-0 matrix
93  *cc = MX();
94 
95  } else {
96  // Stack of expressions to be deleted
97  std::stack<MX> deletion_stack;
98 
99  // Move the child to the deletion stack
100  deletion_stack.push(*cc);
101  *cc = MX();
102 
103  // Process stack
104  while (!deletion_stack.empty()) {
105 
106  // Top element
107  MX t = deletion_stack.top();
108 
109  // Check if the top element has dependencies with dependencies
110  bool found_dep = false;
111 
112  // Start destruction method if any of the dependencies has dependencies
113  while (!t->dep_.empty()) {
114  const MX& ii = t->dep_.back();
115 
116  // Skip if constant
117  if (ii.is_constant()) {
118  t->dep_.pop_back();
119  continue;
120  }
121  // Check if this is the only reference to the element
122  if (ii.getCount()==1) {
123  // Remove and add to stack
124  deletion_stack.push(ii);
125  t->dep_.pop_back();
126  found_dep = true;
127  break;
128  } else {
129  t->dep_.pop_back();
130  }
131  }
132 
133  // Pop from stack if no dependencies found
134  if (!found_dep) {
135  deletion_stack.pop();
136  }
137  }
138  }
139  }
140  }

232  void MXNode::disp(std::ostream& stream, bool more) const {
233  // Find out which noded can be inlined
234  std::map<const MXNode*, casadi_int> nodeind;
235  can_inline(nodeind);
236 
237  // Print expression
238  std::vector<std::string> intermed;
239  std::string s = print_compact(nodeind, intermed);
240 
241  // Print intermediate expressions
242  for (casadi_int i=0; i<intermed.size(); ++i)
243  stream << "@" << (i+1) << "=" << intermed[i] << ", ";
244 
245  // Print this
246  stream << s;
247  }

249  void MXNode::can_inline(std::map<const MXNode*, casadi_int>& nodeind) const {
250  // Add or mark node in map
251  std::map<const MXNode*, casadi_int>::iterator it=nodeind.find(this);
252  if (it==nodeind.end()) {
253  // First time encountered, mark inlined
254  nodeind.insert(it, std::make_pair(this, 0));
255 
256  // Handle dependencies with recursion
257  for (casadi_int i=0; i<n_dep(); ++i) {
258  dep(i)->can_inline(nodeind);
259  }
260  } else if (it->second==0 && op()!=OP_PARAMETER) {
261  // Node encountered before, do not inline (except if symbolic primitive)
262  it->second = -1;
263  }
264  }

266  std::string MXNode::print_compact(std::map<const MXNode*, casadi_int>& nodeind,
267  std::vector<std::string>& intermed) const {
268  // Get reference to node index
269  casadi_int& ind = nodeind[this];
270 
271  // If positive, already in intermediate expressions
272  if (ind>0) return "@" + str(ind);
273 
274  // Get expressions for dependencies
275  std::vector<std::string> arg(n_dep());
276  for (casadi_int i=0; i<arg.size(); ++i) {
277  arg[i] = dep(i)->print_compact(nodeind, intermed);
278  }
279 
280  // Get expression for this
281  std::string s = disp(arg);
282 
283  // Decide what to do with the expression
284  if (ind==0) {
285  // Inline expression
286  return s;
287  } else {
288  // Add to list of intermediate expressions and return reference
289  intermed.push_back(s);
290  ind = intermed.size(); // For subsequent references
291  return "@" + str(ind);
292  }
293  }

327  int MXNode::sp_forward(const bvec_t** arg, bvec_t** res, casadi_int* iw, bvec_t* w) const {
328  // By default, everything depends on everything
329  bvec_t all_depend(0);
330 
331  // Get dependencies of all inputs
332  for (casadi_int k=0; k<n_dep(); ++k) {
333  const bvec_t* v = arg[k];
334  for (casadi_int i=0; i<dep(k).nnz(); ++i) {
335  all_depend |= v[i];
336  }
337  }
338 
339  // Propagate to all outputs
340  for (casadi_int k=0; k<nout(); ++k) {
341  bvec_t* v = res[k];
342  for (casadi_int i=0; i<sparsity(k).nnz(); ++i) {
343  v[i] = all_depend;
344  }
345  }
346  return 0;
347  }

468  MX MXNode::get_mac(const MX& y, const MX& z) const {
469  if (sparsity().is_orthonormal() && y.is_column() && y.is_dense()
470  && y.sparsity()==z.sparsity() && z.is_zero()) {
471  std::vector<casadi_int> perm = sparsity().permutation_vector();
472  MX nz = sparsity_cast(shared_from_this<MX>(), Sparsity::dense(nnz()));
473  return (nz*y)(perm);
474  }
475  // Get reference to transposed first argument
476  MX x = shared_from_this<MX>();
477 
478  casadi_assert(y.size2()==z.size2(),
479  "Dimension error x.mac(z). Got y=" + str(y.size2()) + " and z=" + z.dim() + ".");
480  casadi_assert(x.size1()==z.size1(),
481  "Dimension error x.mac(z). Got x=" + x.dim() + " and z=" + z.dim() + ".");
482  casadi_assert(y.size1()==x.size2(),
483  "Dimension error x.mac(z). Got y=" + str(y.size1()) + " and x" + x.dim() + ".");
484  if (x.is_dense() && y.is_dense() && z.is_dense()) {
485  return MX::create(new DenseMultiplication(z, x, y));
486  } else {
487  return MX::create(new Multiplication(z, x, y));
488  }

491  MX MXNode::get_einstein(const MX& A, const MX& B,
492  const std::vector<casadi_int>& dim_c, const std::vector<casadi_int>& dim_a,
493  const std::vector<casadi_int>& dim_b,
494  const std::vector<casadi_int>& c, const std::vector<casadi_int>& a,
495  const std::vector<casadi_int>& b) const {
496 
497  if (A.is_zero() || B.is_zero())
498  return shared_from_this<MX>();
499 
500  MX C = densify(shared_from_this<MX>());
501 
502  if (A.is_constant() && B.is_constant() && C.is_constant()) {
503  // Constant folding
504  DM Ac = A->get_DM();
505  DM Bc = B->get_DM();
506  DM Cc = C->get_DM();
507  return einstein(vec(densify(Ac)), vec(densify(Bc)), vec(densify(Cc)),
508  dim_a, dim_b, dim_c, a, b, c);
509  }
510 
511  return MX::create(new Einstein(C, densify(A), densify(B), dim_c, dim_a, dim_b, c, a, b));
512  }

697  MX MXNode::get_binary(casadi_int op, const MX& y) const {
698  // If-else-zero nodes are always simplified at top level to avoid NaN propagation
699  if (y.op() == OP_IF_ELSE_ZERO) {
700  if (op == OP_MUL) {
701  // (Rule 1.) x * if_else_zero(c, y), simplified to if_else_zero(c, x * y)
702  // Background: x is often a partial derivative and may evaluate to INF or NAN.
703  // The simplification ensures that the zero seed corresponding to an inactive branch does
704  // not give rise to any NaN contribution to the derivative due to NaN * 0 == NaN.
705  return if_else_zero(y.dep(0), shared_from_this<MX>() * y.dep(1));
706  } else if (op == OP_ADD && this->op() == OP_IF_ELSE_ZERO && MX::is_equal(dep(0), y.dep(0))) {
707  // (Rule 2.) if_else_zero(c, x) + if_else_zero(c, y) is simplified to if_else_zero(c, x + y)
708  // Background: During the backward propagation, seeds are added together. Without this rule,
709  // the addition node can prevent rule (1.) from working in subsequent steps.
710  return if_else_zero(y.dep(0), dep(1) + y.dep(1));
711  }
712  } else if (this->op() == OP_IF_ELSE_ZERO && op == OP_MUL) {
713  // Same as Rule 1. above, but with factors swapped. For symmetry.
714  return if_else_zero(dep(0), dep(1) * y);
715  }
716  // Create binary node
717  if (sparsity().is_scalar(false)) {
718  if (nnz()==0) {
719  if (operation_checker<F0XChecker>(op)) return MX::zeros(Sparsity(y.size()));
720  return to_matrix(MX(0)->_get_binary(op, y, true, false), y.sparsity());
721  } else {
722  return to_matrix(_get_binary(op, y, true, false), y.sparsity());
723  }
724  } else if (y.is_scalar()) {
725  if (y.nnz()==0) {
726  if (operation_checker<FX0Checker>(op)) return MX::zeros(Sparsity(size()));
727  return to_matrix(_get_binary(op, MX(0), false, true), sparsity());
728  } else {
729  return to_matrix(_get_binary(op, y, false, true), sparsity());
730  }
731  } else {
732  casadi_assert(sparsity().size() == y.sparsity().size(), "Dimension mismatch.");
733  if (sparsity()==y.sparsity()) {
734  // Matching sparsities
735  return _get_binary(op, y, false, false);
736  } else {
737  // Get the sparsity pattern of the result
738  // (ignoring structural zeros giving rise to nonzero result)
739  const Sparsity& x_sp = sparsity();
740  const Sparsity& y_sp = y.sparsity();
741  Sparsity r_sp = x_sp.combine(y_sp, operation_checker<F0XChecker>(op),
742  operation_checker<FX0Checker>(op));
743 
744  // Project the arguments to this sparsity
745  MX xx = project(shared_from_this<MX>(), r_sp);
746  MX yy = project(y, r_sp);
747  return xx->_get_binary(op, yy, false, false);
748  }
749  }
750  }

752  MX MXNode::_get_binary(casadi_int op, const MX& y, bool scX, bool scY) const {
753  casadi_assert_dev(sparsity()==y.sparsity() || scX || scY);
754 
755  if (GlobalOptions::simplification_on_the_fly) {
756 
757  // If identically zero due to one argument being zero
758  if ((operation_checker<F0XChecker>(op) && nnz()==0) ||
759  (operation_checker<FX0Checker>(op) && y->nnz()==0)) {
760  return MX::zeros(Sparsity(size()));
761  }
762 
763  if ((operation_checker<F0XChecker>(op) && is_zero()) ||
764  (operation_checker<FX0Checker>(op) && y->is_zero())) {
765  return MX::zeros(sparsity());
766  }
767 
768  // Handle special operations (independent of type)
769  switch (op) {
770  case OP_ADD:
771  if (MXNode::is_equal(y.get(), this, maxDepth())) return get_unary(OP_TWICE);
772  break;
773  case OP_SUB:
774  case OP_NE:
775  case OP_LT:
776  if (MXNode::is_equal(y.get(), this, maxDepth())) return MX::zeros(sparsity());
777  break;
778  case OP_DIV:
779  if (y->is_zero()) return MX::nan(sparsity());
780  // fall-through
781  case OP_EQ:
782  case OP_LE:
783  if (MXNode::is_equal(y.get(), this, maxDepth())) return MX::ones(sparsity());
784  break;
785  case OP_MUL:
786  if (MXNode::is_equal(y.get(), this, maxDepth())) return get_unary(OP_SQ);
787  break;
788  default: break; // no rule
789  }
790 
791  // Handle special cases for the second argument
792  switch (y->op()) {
793  case OP_CONST:
794  // Make the constant the first argument, if possible
795  if (this->op()!=OP_CONST && operation_checker<CommChecker>(op)) {
796  return y->_get_binary(op, shared_from_this<MX>(), scY, scX);
797  } else {
798  switch (op) {
799  case OP_POW:
800  return _get_binary(OP_CONSTPOW, y, scX, scY);
801  case OP_CONSTPOW:
802  if (y->is_value(-1)) return get_unary(OP_INV);
803  else if (y->is_value(0)) return MX::ones(sparsity());
804  else if (y->is_value(1)) return shared_from_this<MX>();
805  else if (y->is_value(2)) return get_unary(OP_SQ);
806  break;
807  case OP_ADD:
808  case OP_SUB:
809  if (y->is_zero())
810  return scX ? repmat(shared_from_this<MX>(), y.size()) : shared_from_this<MX>();
811  break;
812  case OP_MUL:
813  if (y->is_value(1)) return shared_from_this<MX>();
814  break;
815  case OP_DIV:
816  if (y->is_value(1)) return shared_from_this<MX>();
817  else if (y->is_value(0.5)) return get_unary(OP_TWICE);
818  break;
819  default: break; // no rule
820  }
821  }
822  break;
823  case OP_NEG:
824  if (op==OP_ADD) {
825  return _get_binary(OP_SUB, y->dep(), scX, scY);
826  } else if (op==OP_SUB) {
827  return _get_binary(OP_ADD, y->dep(), scX, scY);
828  } else if (op==OP_MUL) {
829  return -_get_binary(OP_MUL, y->dep(), scX, scY);
830  } else if (op==OP_DIV) {
831  return -_get_binary(OP_DIV, y->dep(), scX, scY);
832  }
833  break;
834  case OP_INV:
835  if (op==OP_MUL) {
836  return _get_binary(OP_DIV, y->dep(), scX, scY);
837  } else if (op==OP_DIV) {
838  return _get_binary(OP_MUL, y->dep(), scX, scY);
839  }
840  break;
841  default: break; // no rule
842  }
843 
844  }
845 
846  if (scX) {
847  // Check if it is ok to loop over nonzeros only
848  if (y.is_dense() || operation_checker<FX0Checker>(op) ||
849  (is_zero() && operation_checker<F00Checker>(op))) {
850  // Loop over nonzeros
851  return MX::create(new BinaryMX<true, false>(Operation(op), shared_from_this<MX>(), y));
852  } else {
853  // Put a densification node in between
854  return _get_binary(op, densify(y), true, false);
855  }
856  } else if (scY) {
857  // Check if it is ok to loop over nonzeros only
858  if (sparsity().is_dense() || operation_checker<F0XChecker>(op) ||
859  (y.is_zero() && operation_checker<F00Checker>(op))) {
860  // Loop over nonzeros
861  return MX::create(new BinaryMX<false, true>(Operation(op), shared_from_this<MX>(), y));
862  } else {
863  // Put a densification node in between
864  return densify(shared_from_this<MX>())->_get_binary(op, y, false, true);
865  }
866  } else {
867  // Loop over nonzeros only
868  MX rr = MX::create(new BinaryMX<false, false>(Operation(op), shared_from_this<MX>(), y));
869 
870  // Handle structural zeros giving rise to nonzero result, e.g. cos(0) == 1
871  if (!rr.is_dense() && !operation_checker<F00Checker>(op)) {
872  // Get the value for the structural zeros
873  double fcn_0(0);
874  casadi_math<double>::fun(op, 0, 0, fcn_0);
875  rr = densify(rr, fcn_0);
876  }
877  return rr;
878  }
