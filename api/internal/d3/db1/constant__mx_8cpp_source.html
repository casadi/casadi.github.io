158  ConstantMX* ConstantMX::create(const Sparsity& sp, double val) {
159  if (sp.is_empty(true)) {
160  return ZeroByZero::getInstance();
161  } else {
162  casadi_int intval = static_cast<casadi_int>(val);
163  if (static_cast<double>(intval)-val==0) {
164  return create(sp, intval);
165  } else {
166  return new Constant<RuntimeConst<double> >(sp, val);
167  }
168  }

171  ConstantMX* ConstantMX::create(const Matrix<double>& val) {
172  if (val.nnz()==0) {
173  return create(val.sparsity(), 0);
174  } else if (val.is_scalar()) {
175  return create(val.sparsity(), val.scalar());
176  } else {
177  // Check if all values are the same
178  const std::vector<double> vdata = val.nonzeros();
179  double v = vdata[0];
180  for (auto&& i : vdata) {
181  if (i!=v) {
182  // Values not all the same
183  return new ConstantDM(val);
184  }
185  }
186 
187  // All values identical if reached this point
188  return create(val.sparsity(), v);
189  }
190  }

238  bool ConstantDM::is_equal(const MXNode* node, casadi_int depth) const {
239  // Check if same node
240  const ConstantDM* n = dynamic_cast<const ConstantDM*>(node);
241  if (n==nullptr) return false;
242 
243  // Check sparsity
244  if (this->sparsity()!=node->sparsity()) return false;
245 
246  // Check indices
247  if (!std::equal(x_->begin(), x_->end(), n->x_->begin())) return false;
248 
249  return true;

312  MXNode* ConstantMX::deserialize(DeserializingStream& s) {
313  char t;
314  s.unpack("ConstantMX::type", t);
315  switch (t) {
316  case 'a': return new ConstantDM(s);
317  case 'f': return new ConstantFile(s);
318  case 'z': return ZeroByZero::getInstance();
319  case 'D':
320  return new Constant<RuntimeConst<double> >(s, RuntimeConst<double>::deserialize(s));
321  case 'I':
322  return new Constant<RuntimeConst<casadi_int> >(s,
323  RuntimeConst<casadi_int>::deserialize(s));
324  case '0':
325  return new Constant<CompiletimeConst<0> >(s, CompiletimeConst<0>::deserialize(s));
326  case '1':
327  return new Constant<CompiletimeConst<1> >(s, CompiletimeConst<1>::deserialize(s));
328  case 'm':
329  return new Constant<CompiletimeConst<(-1)> >(s, CompiletimeConst<(-1)>::deserialize(s));
330  default:
331  casadi_error("Error deserializing");
332  }
333  }
