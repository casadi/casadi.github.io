111 std::string OptiNode::format_stacktrace(const Dict& stacktrace, casadi_int indent) {
112  std::string s_indent;
113  for (casadi_int i=0;i<indent;++i) {
114  s_indent+= " ";
115  }
116  std::string description;
117  std::string filename = stacktrace.at("file").as_string();
118  casadi_int line = stacktrace.at("line").as_int();
119  description += "defined at " + filename +":"+str(line);
120  std::string name = stacktrace.at("name").as_string();
121  if (name!="Unknown" && name!= "<module>")
122  description += " in " + stacktrace.at("name").as_string();
123  try {
124  std::ifstream file(filename);
125  for (casadi_int i=0; i<line-1; ++i) {
126  file.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
127  }
128  std::string contents; std::getline(file, contents);
129  auto it = contents.find_first_not_of(" \n");
130  if (it!=std::string::npos) {
131  description += "\n" + s_indent + contents.substr(it);
132  }
133  } catch(...) {
134  // pass
135  }
136  return description;
137 }

139 std::string OptiNode::describe(const MX& expr, casadi_int indent) const {
140  if (problem_dirty()) return baked_copy().describe(expr, indent);
141  std::string s_indent;
142  for (casadi_int i=0;i<indent;++i) {
143  s_indent+= " ";
144  }
145  std::string description = s_indent;
146  if (expr.is_symbolic()) {
147  if (has(expr)) {
148  description += "Opti " + variable_type_to_string(meta(expr).type) + " '" + expr.name() +
149  "' of shape " + expr.dim();
150  const Dict& extra = meta(expr).extra;
151  auto it = extra.find("stacktrace");
152  if (it!=extra.end()) {
153  const Dict& stacktrace = it->second.as_dict();
154  description += ", " + format_stacktrace(stacktrace, indent+1);
155  }
156  } else {
157  VariableType vt;
158  if (parse_opti_name(expr.name(), vt)) {
159  description += "Opti " + variable_type_to_string(vt) + " '" + expr.name() +
160  "' of shape " + expr.dim()+
161  ", belonging to a different instance of Opti.";
162  } else {
163  description += "MX symbol '" + expr.name() + "' of shape " + expr.dim();
164  description += ", declared outside of Opti.";
165  }
166  }
167  } else {
168  if (has_con(expr)) {
169  description = "Opti constraint of shape " + expr.dim();
170  const Dict& extra = meta_con(expr).extra;
171  auto it = extra.find("stacktrace");
172  if (it!=extra.end()) {
173  const Dict& stacktrace = it->second.as_dict();
174  description += ", " + format_stacktrace(stacktrace, indent+1);
175  }
176  } else {
177  std::vector<MX> s = symvar(expr);
178  if (s.empty()) {
179  description+= "Constant epxression.";
180  } else {
181  description+= "General expression, dependent on " + str(s.size()) + " symbols:";
182  for (casadi_int i=0;i<s.size();++i) {
183  description+= "\n"+describe(s[i], indent+1);
184  if (i>5) {
185  description+= "\n...";
186  break;
187  }
188  }
189  }
190  }
191  }
192 
193  return description;
194 }

261 MX OptiNode::variable(casadi_int n, casadi_int m, const std::string& attribute) {
262 
263  // Prepare metadata
264  MetaVar meta_data;
265  meta_data.attribute = attribute;
266  meta_data.n = n;
267  meta_data.m = m;
268  meta_data.type = OPTI_VAR;
269  meta_data.count = count_++;
270  meta_data.i = count_var_++;
271  meta_data.domain = OPTI_DOMAIN_REAL;
272 
273  MX symbol, ret;
274 
275  if (attribute=="symmetric") {
276  casadi_assert(n==m, "You specified attribute 'symmetric', "
277  "while matrix is not even square, but " + str(n) + "-by-" + str(m) + ".");
278  symbol = MX::sym(name_prefix() + "x_" + str(count_var_), n*(n+1)/2);
279  ret = tril2symm(MX(Sparsity::lower(n), symbol));
280  } else if (attribute=="full") {
281  symbol = MX::sym(name_prefix() + "x_" + str(count_var_), n, m);
282  ret = symbol;
283  } else {
284  casadi_error("Unknown attribute '" + attribute + "'. Choose from 'full' or 'symmetric'.");
285  }
286 
287  // Store the symbol; preventing it from going ut of scope
288  symbols_.push_back(symbol);
289  store_initial_[OPTI_VAR].push_back(DM::zeros(symbol.sparsity()));
290  store_latest_[OPTI_VAR].push_back(DM::nan(symbol.sparsity()));
291 
292  set_meta(symbol, meta_data);
293  return ret;
294 }

304 void OptiNode::register_dual(MetaCon& c) {
305 
306  // Prepare metadata
307  MetaVar meta_data;
308  meta_data.attribute = "full";
309  meta_data.n = c.canon.size1();
310  meta_data.m = c.canon.size2();;
311  meta_data.type = OPTI_DUAL_G;
312  meta_data.count = count_++;
313  meta_data.i = count_dual_++;
314 
315  MX symbol, ret;
316  if (c.type==OPTI_PSD) {
317  symbol = MX();
318  ret = MX();
319  } else {
320  symbol = MX::sym(name_prefix()+"lam_g_"+str(count_dual_), c.canon.sparsity());
321 
322  casadi_assert_dev(c.canon.is_dense());
323 
324  Sparsity ret_sp = repmat(c.original.sparsity(), 1, c.n);
325 
326  casadi_int N = c.canon.sparsity().nnz();
327 
328  MX flat = vec(symbol);
329  if (c.type==OPTI_DOUBLE_INEQUALITY) {
330  MX v = MX::sym("v");
331  MX decide_left_right = vertcat(if_else_zero(v<0, -v), if_else_zero(v>=0, v));
332  Function sign = Function("sign", {v}, {decide_left_right});
333  Function sign_map = sign.map(c.canon.sparsity().nnz());
334  ret = MX(ret_sp, sign_map((c.flipped ? -1 : 1)*flat)[0].T());
335  } else {
336  casadi_int block_size = N / c.n;
337  std::vector<MX> original_blocks = vertsplit(fabs(flat), block_size);
338  std::vector<MX> blocks(N);
339  for (casadi_int i=0;i<c.n;++i) {
340  casadi_int p = c.flipped? c.n-i-1: i;
341  blocks[p] = original_blocks[i];
342  }
343  ret = MX(ret_sp, vertcat(blocks));
344  }
345  }
346 
347  symbols_.push_back(symbol);
348  store_initial_[OPTI_DUAL_G].push_back(DM::zeros(symbol.sparsity()));
349  store_latest_[OPTI_DUAL_G].push_back(DM::nan(symbol.sparsity()));
350 
351  c.dual = ret;
352  c.dual_canon = symbol;
353 
354  set_meta(symbol, meta_data);
355 }

579 void OptiNode::bake() {
580  casadi_assert(!f_.is_empty() || !g_.empty(),
581  "You need to specify at least an objective (y calling 'minimize'), "
582  "or a constraint (by calling 'subject_to').");
583 
584  symbol_active_.clear();
585  symbol_active_.resize(symbols_.size());
586 
587  // Gather all expressions
588  MX total_expr = vertcat(f_, veccat(g_));
589 
590  // Categorize the symbols appearing in those expressions
591  for (const auto& d : symvar(total_expr))
592  symbol_active_[meta(d).count] = true;
593 
594  std::vector<MX> x = active_symvar(OPTI_VAR);
595  for (casadi_int i=0;i<x.size();++i) meta(x[i]).active_i = i;
596 
597  casadi_int offset = 0;
598  for (const auto& v : x) {
599  meta(v).start = offset;
600  offset+= v.nnz();
601  meta(v).stop = offset;
602  }
603  std::vector<MX> p = active_symvar(OPTI_PAR);
604  for (casadi_int i=0;i<p.size();++i) meta(p[i]).active_i = i;
605 
606  // Fill the nlp definition
607  nlp_["x"] = veccat(x);
608  nlp_["p"] = veccat(p);
609 
610  discrete_.clear();
611  for (const MX& e : x) {
612  discrete_.insert(discrete_.end(), e.nnz(), meta(e).domain == OPTI_DOMAIN_INTEGER);
613  }
614 
615  nlp_["f"] = f_;
616 
617  offset = 0;
618  for (casadi_int i=0;i<g_.size();++i) {
619  MetaCon& r = meta_con(g_[i]);
620  MetaVar& r2 = meta(r.dual_canon);
621  symbol_active_[r2.count] = true;
622 
623  // Compute offsets for this constraint:
624  // location into the global constraint variable
625  r.start = offset;
626  offset+= r.canon.nnz();
627  r.stop = offset;
628 
629  r2.start = r.start;
630  r2.stop = r.stop;
631 
632  }
633 
634  std::vector<MX> lam = active_symvar(OPTI_DUAL_G);
635  for (casadi_int i=0;i<lam.size();++i) meta(lam[i]).active_i = i;
636 
637  lam_ = veccat(lam);
638 
639  // Collect bounds and canonical form of constraints
640  std::vector<MX> g_all;
641  std::vector<MX> h_all;
642  std::vector<MX> lbg_all;
643  std::vector<MX> ubg_all;
644  equality_.clear();
645  for (const auto& g : g_) {
646  if (meta_con(g).type==OPTI_PSD) {
647  h_all.push_back(meta_con(g).canon);
648  } else {

708 std::vector<MX> OptiNode::ineq_unchain(const MX& a, bool& flipped) {
709  flipped = false;
710  casadi_assert_dev(a.is_op(OP_LE) || a.is_op(OP_LT));
711 
712  // Is there inequalities in the left or right leaf?
713  bool left = a.dep(0).is_op(OP_LE) || a.dep(0).is_op(OP_LT);
714  bool right = a.dep(1).is_op(OP_LE) || a.dep(1).is_op(OP_LT);
715  casadi_assert_dev(!left || !right);
716 
717  if (!left && !right)
718  return {a.dep(0), a.dep(1)}; // Simple inequality
719 
720  // We have a chain of inequalities
721  bool ineq = !left;
722  std::vector<MX> ret = {a.dep(!ineq)};
723  MX e = a.dep(ineq);
724  while (e.is_op(OP_LE) || e.is_op(OP_LT)) {
725  casadi_assert_dev(!e.is_op(OP_EQ));
726  casadi_assert_dev(!e.dep(!ineq).is_op(OP_LE) && !e.dep(!ineq).is_op(OP_LT));
727  ret.push_back(e.dep(!ineq));
728  e = e.dep(ineq);
729  }
730  ret.push_back(e);
731  if (left) std::reverse(ret.begin(), ret.end());
732  flipped = !left;
733 
734  return ret;
735 }

754 MetaCon OptiNode::canon_expr(const MX& expr) const {
755  MX c = expr;
756 
757  MetaCon con;
758  con.original = expr;
759 
760  if (c.is_op(OP_LE) || c.is_op(OP_LT)) { // Inequalities
761  std::vector<MX> ret;
762  bool flipped;
763  std::vector<MX> args = ineq_unchain(c, flipped);
764  std::vector<bool> parametric;
765  for (auto &a : args) parametric.push_back(is_parametric(a));
766 
767  if (args.size()==2 && (parametric[0] || parametric[1])) {
768  // case: g(x,p) <= bound(p)
769  MX e = args[0]-args[1];
770  if (e.is_vector()) {
771  casadi_assert(!parametric[0] || !parametric[1],
772  "Constraint must contain decision variables.");
773  if (problem_type_=="conic") {
774  if (args[0].op()==OP_NORMF || args[0].op()==OP_NORM2) {
775  args[0] = -soc(args[0].dep(), args[1]);
776  args[1] = 0;
777  }
778  } else {
779  con.type = OPTI_INEQUALITY;
780  if (parametric[0]) {
781  con.lb = args[0]*DM::ones(e.sparsity());
782  con.ub = inf*DM::ones(e.sparsity());
783  con.canon = args[1]*DM::ones(e.sparsity());
784  } else {
785  con.lb = -inf*DM::ones(e.sparsity());
786  con.ub = args[1]*DM::ones(e.sparsity());
787  con.canon = args[0]*DM::ones(e.sparsity());
788  }
789  return con;
790  }
791  }
792  // Fall through to generic inequalities
793  } else if (args.size()==3 && parametric[0] && parametric[2]) {
794  // lb(p) <= g(x,p) <= ub(p)
795  con.type = OPTI_DOUBLE_INEQUALITY;
796  con.lb = args[0]*DM::ones(args[1].sparsity());
797  con.ub = args[2]*DM::ones(args[1].sparsity());
798  con.canon = args[1]*DM::ones(args[1].sparsity());
799  con.flipped = flipped;
800  con.n = 2;
801  return con;
802  }
803 
804  bool type_known = false;
805  for (casadi_int j=0;j<args.size()-1;++j) {
806  MX e = args[j]-args[j+1];
807  if (problem_type_=="conic") {
808  if (args[j].op()==OP_NORMF || args[j].op()==OP_NORM2) {
809  args[j] = -soc(args[j].dep(), args[j+1]);
810  args[j+1] = 0;
811  e = args[j]-args[j+1];
812  }
813  }
814  if (e.is_vector()) {
815  // g1(x,p) <= g2(x,p)
816  ret.push_back(e);
817  casadi_assert_dev(!type_known || con.type==OPTI_GENERIC_INEQUALITY);
818  type_known = true;
819  con.type = OPTI_GENERIC_INEQUALITY;
820  con.flipped = flipped;
821  } else {
822  // A(x,p) >= b(p)
823  MX a = args[j+1];
824  MX b = args[j];
825  e = a-b;
826 
827  casadi_assert(e.size1()==e.size2(),
828  "Matrix inequalities must be square. Did you mean element-wise inequality instead?");
829  if (a.is_scalar()) a*= MX::eye(e.size1());
830  if (b.is_scalar()) b*= MX::eye(e.size1());
831  e = a-b;
832 
833  ret.push_back(e);
834  casadi_assert_dev(!type_known || con.type==OPTI_PSD);
835  type_known = true;
836  con.type = OPTI_PSD;
837  }
838  }
839 
840  if (con.type==OPTI_GENERIC_INEQUALITY) {
841  con.canon = veccat(ret);
842  con.lb = -inf*DM::ones(con.canon.sparsity());
843  con.ub = DM::zeros(con.canon.sparsity());
844  con.n = ret.size();
845  } else {
846  con.canon = diagcat(ret);
847  con.n = ret.size();
848  }
849  return con;
850  } else if (c.is_op(OP_EQ)) { // Inequalities
851  casadi_assert(!is_parametric(c.dep(0)) || !is_parametric(c.dep(1)),
852  "Constraint must contain decision variables.");
853  MX e = c.dep(0)-c.dep(1);
854  if (is_parametric(c.dep(0))) {
855  con.canon = c.dep(1)*DM::ones(e.sparsity());
856  con.lb = c.dep(0)*DM::ones(e.sparsity());
857  con.type = OPTI_EQUALITY;
858  casadi_assert(c.dep(0).size1()<=c.dep(1).size1() && c.dep(0).size2()<=c.dep(1).size2(),
859  "Constraint shape mismatch.");
860  } else if (is_parametric(c.dep(1))) {
861  con.canon = c.dep(0)*DM::ones(e.sparsity());
862  con.lb = c.dep(1)*DM::ones(e.sparsity());
863  con.type = OPTI_EQUALITY;
864  casadi_assert(c.dep(1).size1()<=c.dep(0).size1() && c.dep(1).size2()<=c.dep(0).size2(),
865  "Constraint shape mismatch.");
866  } else {
867  con.lb = DM::zeros(e.sparsity());
868  con.canon = e;
869  con.type = OPTI_GENERIC_EQUALITY;
870  }
871  con.ub = con.lb;
872  return con;
873  } else { // Something else
874  con.type = OPTI_UNKNOWN;
875  con.canon = c;
876  return con;
877  }
