567  template<typename MatType>
568  MatType SparsityInterface<MatType>::repmat(const MatType& x, casadi_int n, casadi_int m) {
569  if (n==1 && m==1) return x;
570  MatType allrows = vertcat(std::vector<MatType>(n, x));
571  if (n==0) allrows = MatType(0, x.size2());
572  MatType ret = horzcat(std::vector<MatType>(m, allrows));
573  if (m==0) ret = MatType(allrows.size1(), 0);
574  return ret;
575  }

577  template<typename MatType>
578  std::vector< std::vector< MatType > >
579  SparsityInterface<MatType>::blocksplit(const MatType& x,
580  const std::vector<casadi_int>& vert_offset,
581  const std::vector<casadi_int>& horz_offset) {
582  std::vector<MatType> rows = MatType::vertsplit(x, vert_offset);
583  std::vector< std::vector< MatType > > ret;
584  for (auto&& r : rows) ret.push_back(MatType::horzsplit(r, horz_offset));
585  return ret;
586  }

588  template<typename MatType>
589  std::vector< std::vector< MatType > >
590  SparsityInterface<MatType>::blocksplit(const MatType& x,
591  casadi_int vert_incr, casadi_int horz_incr) {
592  casadi_assert_dev(horz_incr>=1);
593  casadi_assert_dev(vert_incr>=1);
594  casadi_int sz1 = x.size1();
595  std::vector<casadi_int> offset1 = range(0, sz1, vert_incr);
596  offset1.push_back(sz1);
597  casadi_int sz2 = x.size2();
598  std::vector<casadi_int> offset2 = range(0, sz2, horz_incr);
599  offset2.push_back(sz2);
600  return blocksplit(x, offset1, offset2);
601  }

603  template<typename MatType>
604  std::vector<casadi_int>
605  SparsityInterface<MatType>::offset(const std::vector< MatType > &v, bool vert) {
606  std::vector<casadi_int> ret(v.size()+1);
607  ret[0]=0;
608  for (casadi_int i=0; i<v.size(); ++i) {
609  ret[i+1] = ret[i] + (vert ? v[i].size1() : v[i].size2());
610  }
611  return ret;
612  }

614  template<typename MatType>
615  MatType SparsityInterface<MatType>::veccat(const std::vector< MatType >& x) {
616  std::vector< MatType > x_vec = x;
617  for (typename std::vector< MatType >::iterator it=x_vec.begin();
618  it!=x_vec.end(); ++it) {
619  *it = vec(*it);
620  }
621  if (x_vec.empty()) {
622  return MatType(0, 1);
623  } else {
624  return vertcat(x_vec);
625  }
626  }
