81  }

83  g << "cvx_config.type_in = CVX_SYMM;\n";
84  } else if (d.config.type_in==CVX_TRIL) {

87  g << "cvx_config.type_in = CVX_TRIU;\n";

90  g << "cvx_config.Hrsp = " << g.sparsity(d.Hrsp) << ";\n";

94  g << "cvx_config.scc_offset = " << g.constant(d.scc_offset) << ";\n";
95  g << "cvx_config.scc_mapping = " << g.constant(d.scc_mapping) << ";\n";
96  g << "cvx_config.scc_offset_size = " << d.scc_offset.size() << ";\n";
97  g << "cvx_config.max_iter_eig = " << d.config.max_iter_eig << ";\n";
98  g << "cvx_config.verbose = " << d.config.verbose << ";\n";
99  return "convexify_eval(&cvx_config, " + Hin + "," + Hout + "," + iw + "," + "w)";

164  Sparsity Convexify::setup(ConvexifyData& d, const Sparsity& H, const Dict& opts, bool inplace) {
165  // Validate and categorize matrix input sparsity
166  casadi_assert(H.is_square(), "Convexify ");
167  if (H.is_symmetric()) {
168  d.config.type_in = CVX_SYMM;
169  } else if (H.is_tril()) {
170  d.config.type_in = CVX_TRIL;
171  } else if (H.is_triu()) {
172  d.config.type_in = CVX_TRIU;
173  } else {
174  casadi_error("Convexify operation requires symmetric or triangular input");
175  }
176 
177  // Read options
178  d.config.margin = 1e-7;
179  d.config.max_iter_eig = 200;
180  std::string strategy = "eigen-clip";
181  d.config.verbose = false;
182 
183  for (auto&& op : opts) {
184  if (op.first=="strategy") {
185  strategy = op.second.to_string();
186  } else if (op.first=="margin") {
187  d.config.margin = op.second;
188  casadi_assert(d.config.margin>=0, "Margin must be >=0");
189  } else if (op.first=="max_iter_eig") {
190  d.config.max_iter_eig = op.second;
191  } else if (op.first=="verbose") {
192  d.config.verbose = op.second;
193  } else {
194  casadi_error("Unknown option '" + op.first + "'.");
195  }
196  }
197 
198  // Interpret strategy
199  if (strategy=="regularize") {
200  d.config.strategy = CVX_REGULARIZE;
201  casadi_assert(d.config.type_in==CVX_SYMM, "Only truly symmetric matrices supported");
202  } else if (strategy=="eigen-reflect") {
203  d.config.strategy = CVX_EIGEN_REFLECT;
204  } else if (strategy=="eigen-clip") {
205  d.config.strategy = CVX_EIGEN_CLIP;
206  } else {
207  casadi_error("Invalid convexify strategy. "
208  "Choose from regularize|eigen-reflect|eigen-clip. Got '" + strategy + "'.");
209  }
210 
211  d.Hrsp = H;
212 
213  d.config.scc_transform = 0;
214 
215  Sparsity Hrsp = H+H.T();
216 
217  casadi_int block_size = 0;
218  Sparsity& Hsp = d.Hsp;
219  if (d.config.strategy==CVX_EIGEN_REFLECT || d.config.strategy==CVX_EIGEN_CLIP) {
220  // Uncover strongly connected components
221  std::vector<casadi_int> scc_index;
222  casadi_int scc_nb = Hrsp.scc(scc_index, d.scc_offset);
223 
224  // Represent Hessian as block-dense in permuted space
225  std::vector<Sparsity> sp;
226  for (casadi_int i=0;i<scc_nb;++i) {
227  casadi_int block = d.scc_offset.at(i+1)-d.scc_offset.at(i);
228  Sparsity stencil;
229  if (d.config.type_in==CVX_SYMM) {
230  stencil = Sparsity::dense(block, block);
231  } else if (d.config.type_in==CVX_TRIL) {
232  stencil = Sparsity::lower(block);
233  } else {
234  stencil = Sparsity::upper(block);
235  }
236  sp.push_back(stencil);
237  }
238 
239  std::vector<casadi_int> ssc_perm = lookupvector(scc_index);
240  std::vector<casadi_int> mapping_dummy;
241  Hsp = diagcat(sp).sub(ssc_perm, ssc_perm, mapping_dummy);
242  Hsp.sub(scc_index, scc_index, d.scc_mapping);
243 
244  // Find out size of maximum block
245  for (casadi_int i=0;i<scc_nb;++i) {
246  casadi_int block = d.scc_offset.at(i+1)-d.scc_offset.at(i);
247  if (block>block_size) block_size = block;
248  }
249 
250  d.config.scc_transform = d.scc_offset!=range(H.size1());
251 
252  if (d.config.verbose) casadi_message("Identified " + str(scc_nb) + " blocks "
253  "with maximum size " + str(block_size) + ".");
254  } else if (d.config.strategy==CVX_REGULARIZE) {
255  Hsp = Hrsp + Sparsity::diag(H.size1());
256  } else {
257  Hsp = Hrsp;
258  }
259  d.config.Hsp_project = Hsp!=Hrsp;
260 
261  if (d.config.type_in==CVX_TRIL) {
262  Hsp = Sparsity::tril(Hsp);
263  } else if (d.config.type_in==CVX_TRIU) {
264  Hsp = Sparsity::triu(Hsp);
265  }
266 
267  d.sz_iw = 0;
268  if (d.config.strategy==CVX_EIGEN_REFLECT || d.config.strategy==CVX_EIGEN_CLIP) {
269  d.sz_iw = 1+3* d.config.max_iter_eig;
270  }
271 
272  d.sz_w = 0;
273  if (d.config.strategy==CVX_EIGEN_REFLECT || d.config.strategy==CVX_EIGEN_CLIP) {
274  d.sz_w = std::max(block_size, 2*(block_size-1)*d.config.max_iter_eig);
275  if (d.config.Hsp_project) d.sz_w = std::max(d.sz_w, Hsp.size1());
276  if (d.config.scc_transform) d.sz_w += block_size*block_size;
277  if (inplace) d.sz_w = std::max(d.sz_w, Hsp.size1()+d.Hrsp.nnz());
278  }
279  d.sz_w = Hsp.size1()+d.sz_w;
280 
281  d.config.scc_offset_size = d.scc_offset.size();
282 
283  // Set pointers
284  d.config.Hsp = Hsp;
285  d.config.Hrsp = d.Hrsp;
286  d.config.scc_offset = get_ptr(d.scc_offset);
287  d.config.scc_mapping = get_ptr(d.scc_mapping);
288 
289  return Hsp;
290  }
