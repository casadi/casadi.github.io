61  int Split::eval_gen(const T** arg, T** res, casadi_int* iw, T* w) const {
62  // Number of derivatives
63  casadi_int nx = offset_.size()-1;
64 
65  for (casadi_int i=0; i<nx; ++i) {
66  casadi_int nz_first = offset_[i];
67  casadi_int nz_last = offset_[i+1];
68  if (res[i]!=nullptr) {
69  std::copy(arg[0]+nz_first, arg[0]+nz_last, res[i]);
70  }
71  }
72  return 0;
73  }

75  int Split::sp_forward(const bvec_t** arg, bvec_t** res, casadi_int* iw, bvec_t* w) const {
76  casadi_int nx = offset_.size()-1;
77  for (casadi_int i=0; i<nx; ++i) {
78  if (res[i]!=nullptr) {
79  const bvec_t *arg_ptr = arg[0] + offset_[i];
80  casadi_int n_i = sparsity(i).nnz();
81  bvec_t *res_i_ptr = res[i];
82  for (casadi_int k=0; k<n_i; ++k) {
83  *res_i_ptr++ = *arg_ptr++;
84  }
85  }
86  }
87  return 0;
88  }

90  int Split::sp_reverse(bvec_t** arg, bvec_t** res, casadi_int* iw, bvec_t* w) const {
91  casadi_int nx = offset_.size()-1;
92  for (casadi_int i=0; i<nx; ++i) {
93  if (res[i]!=nullptr) {
94  bvec_t *arg_ptr = arg[0] + offset_[i];
95  casadi_int n_i = sparsity(i).nnz();
96  bvec_t *res_i_ptr = res[i];
97  for (casadi_int k=0; k<n_i; ++k) {
98  *arg_ptr++ |= *res_i_ptr;
99  *res_i_ptr++ = 0;
100  }
101  }
102  }
103  return 0;
104  }

106  void Split::generate(CodeGenerator& g,
107  const std::vector<casadi_int>& arg,
108  const std::vector<casadi_int>& res) const {
109  casadi_int nx = nout();
110  for (casadi_int i=0; i<nx; ++i) {
111  casadi_int nz_first = offset_[i];
112  casadi_int nz_last = offset_[i+1];
113  casadi_int nz = nz_last-nz_first;
114  if (res[i]>=0 && nz>0) { // if anything to assign
115  if (nz==1) { // assign scalar
116  g << g.workel(res[i]) << " = ";
117  if (dep(0).nnz()==1) {
118  // rhs is also scalar
119  casadi_assert_dev(nz_first==0);
120  g << g.workel(arg[0]) << ";\n";
121  } else {
122  // rhs is an element in a vector
123  g << g.work(arg[0], dep(0).nnz()) << "[" << nz_first << "];\n";
124  }
125  } else {
126  // assign vector
127  std::string r = g.work(arg[0], dep(0).nnz());
128  if (nz_first!=0) r = r + "+" + str(nz_first);
129  g << g.copy(r, nz, g.work(res[i], nnz(i))) << "\n";
130  }
131  }
132  }
133  }

347  MX Horzsplit::get_horzcat(const std::vector<MX>& x) const {
348  std::vector<MX> X;
349  for (auto&& e : x) {
350  if (e.nnz()!=0) X.push_back(e);
351  }
352 
353  // Check x length
354  if (X.size()!=nout()) {
355  return MXNode::get_horzcat(x);
356  }
357 
358  // Check x content
359  for (casadi_int i=0; i<X.size(); ++i) {
360  if (!(X[i]->is_output() && X[i]->which_output()==i && X[i]->dep().get()==this)) {
361  return MXNode::get_horzcat(x);
362  }
363  }
364 
365  return sparsity_cast(dep(), MXNode::get_horzcat(x).sparsity());
366  }
