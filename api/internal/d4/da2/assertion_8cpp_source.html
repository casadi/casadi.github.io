87  int Assertion::sp_reverse(bvec_t** arg, bvec_t** res, casadi_int* iw, bvec_t* w) const {
88  bvec_t *a = arg[0];
89  bvec_t *r = res[0];
90  casadi_int n = nnz();
91  if (a != r) {
92  for (casadi_int i=0; i<n; ++i) {
93  *a++ |= *r;
94  *r++ = 0;
95  }
96  }
97  return 0;
98  }
