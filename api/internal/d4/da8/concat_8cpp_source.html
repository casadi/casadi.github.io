47  int Concat::eval_gen(const T* const* arg, T* const* res, casadi_int* iw, T* w) const {
48  T* r = res[0];

57  int Concat::sp_forward(const bvec_t** arg, bvec_t** res, casadi_int* iw, bvec_t* w) const {
58  bvec_t *res_ptr = res[0];
59  for (casadi_int i=0; i<n_dep(); ++i) {
60  casadi_int n_i = dep(i).nnz();
61  const bvec_t *arg_i_ptr = arg[i];
62  std::copy(arg_i_ptr, arg_i_ptr+n_i, res_ptr);
63  res_ptr += n_i;
64  }

68  int Concat::sp_reverse(bvec_t** arg, bvec_t** res, casadi_int* iw, bvec_t* w) const {
69  bvec_t *res_ptr = res[0];
70  for (casadi_int i=0; i<n_dep(); ++i) {
71  casadi_int n_i = dep(i).nnz();
72  bvec_t *arg_i_ptr = arg[i];
73  for (casadi_int k=0; k<n_i; ++k) {
74  *arg_i_ptr++ |= *res_ptr;
75  *res_ptr++ = 0;
76  }
77  }
78  return 0;
79  }

81  void Concat::generate(CodeGenerator& g,
82  const std::vector<casadi_int>& arg,
83  const std::vector<casadi_int>& res) const {
84  g.local("rr", "casadi_real", "*");
85  g << "rr=" << g.work(res[0], nnz()) << ";\n";
86  for (casadi_int i=0; i<arg.size(); ++i) {
87  casadi_int nz = dep(i).nnz();
88  if (nz==1) {
89  g << "*rr++ = " << g.workel(arg[i]) << ";\n";
90  } else if (nz!=0) {
91  g.local("i", "casadi_int");
92  g.local("cs", "const casadi_real", "*");
93  g << "for (i=0, " << "cs=" << g.work(arg[i], nz) << "; "
94  << "i<" << nz << "; ++i) *rr++ = *cs++;\n";
95  }
96  }
97  }

99  MX Concat::get_nzref(const Sparsity& sp, const std::vector<casadi_int>& nz) const {
100  // Get the first nonnegative nz
101  casadi_int nz_test = -1;
102  for (auto&& i : nz) {
103  if (i>=0) {
104  nz_test = i;
105  break;
106  }
107  }
108 
109  // Quick return if none
110  if (nz_test<0) return MX::zeros(sp);
111 
112  // Find out to which dependency it might depend
113  casadi_int begin=0, end=0;
114  casadi_int i;
115  for (i=0; i<n_dep(); ++i) {
116  begin = end;
117  end += dep(i).nnz();
118  if (nz_test < end) break;
119  }
120 
121  // Check if any nz refer to a different nonzero
122  for (auto&& j : nz) {
123  if (j>=0 && (j < begin || j >= end)) {
124 
125  // Fallback to the base class
126  return MXNode::get_nzref(sp, nz);
127  }
128  }
129 
130  // All nz refer to the same dependency, update the nonzero indices
131  if (begin==0) {
132  return dep(i)->get_nzref(sp, nz);
133  } else {
134  std::vector<casadi_int> nz_new(nz);
135  for (auto&& j : nz_new) if (j>=0) j -= begin;
136  return dep(i)->get_nzref(sp, nz_new);
137  }
138  }

166  std::pair<std::vector<casadi_int>, std::vector<casadi_int> > Diagcat::off() const {
167  std::vector<casadi_int> offset1(n_dep()+1, 0);
168  std::vector<casadi_int> offset2(n_dep()+1, 0);
169  for (casadi_int i=0; i<n_dep(); ++i) {
170  casadi_int ncol = dep(i).sparsity().size2();
171  casadi_int nrow = dep(i).sparsity().size1();
172  offset2[i+1] = offset2[i] + ncol;
173  offset1[i+1] = offset1[i] + nrow;
174  }
175  return std::make_pair(offset1, offset2);
176  }

178  void Diagcat::ad_reverse(const std::vector<std::vector<MX> >& aseed,
179  std::vector<std::vector<MX> >& asens) const {
180  // Get offsets for each row and column
181  auto off = this->off();
182 
183  // Adjoint sensitivities
184  casadi_int nadj = aseed.size();
185  for (casadi_int d=0; d<nadj; ++d) {
186  std::vector<MX> s = diagsplit(aseed[d][0], off.first, off.second);
187  for (casadi_int i=0; i<n_dep(); ++i) {
188  asens[d][i] += s[i];
189  }
190  }
191  }
