72  DM Linsol::solve(const DM& A, const DM& B, bool tr) const {
73  casadi_assert(A.size1()==B.size1(),
74  "Linsol::solve: Dimension mismatch. A and b must have matching row count. "
75  "Got " + A.dim() + " and " + B.dim() + ".");
76 
77  scoped_checkout<Linsol> mem(*this);
78  auto m = static_cast<LinsolMemory*>((*this)->memory(mem));
79 
80  // Reset statistics
81  for (auto&& s : m->fstats) s.second.reset();
82  if (m->t_total) m->t_total->tic();
83  // Symbolic factorization
84  if (sfact(A.ptr(), mem)) casadi_error("Linsol::solve: 'sfact' failed");
85 
86  // Numeric factorization
87  if (nfact(A.ptr(), mem)) casadi_error("Linsol::solve: 'nfact' failed");
88 
89  // Solve
90  DM x = densify(B);
91  if (solve(A.ptr(), x.ptr(), x.size2(), false, mem))
92  casadi_error("Linsol::solve: 'solve' failed");
93  // Show statistics
94  if (m->t_total) m->t_total->toc();
95 
96  (*this)->print_time(m->fstats);
97  return x;
98  }

109  int Linsol::sfact(const double* A, int mem) const {
110  if (A==nullptr) return 1;
111  auto m = static_cast<LinsolMemory*>((*this)->memory(mem));
112 
113  // Factorization will be needed after this step
114  m->is_sfact = m->is_nfact = false;
115 
116  if (m->t_total) m->fstats.at("sfact").tic();
117  // Perform pivoting
118  if ((*this)->sfact(m, A)) return 1;
119  if (m->t_total) m->fstats.at("sfact").toc();
120 
121  // Mark as (successfully) pivoted
122  m->is_sfact = true;
123  return 0;
124  }

131  int Linsol::nfact(const double* A, int mem) const {
132  if (A==nullptr) return 1;
133  auto m = static_cast<LinsolMemory*>((*this)->memory(mem));
134 
135  // Perform pivoting, if required
136  if (!m->is_sfact) {
137  if (sfact(A, mem)) return 1;
138  }
139 
140  m->is_nfact = false;
141  if (m->t_total) m->fstats.at("nfact").tic();
142  int flag = (*this)->nfact(m, A);
143  if (m->t_total) m->fstats.at("nfact").toc();
144  if (flag && (*this)->regularity_check_) {
145  // Collect nonzeros
146  std::vector<std::string> nonzeros(sparsity().nnz());
147  for (size_t nz = 0; nz < nonzeros.size(); ++nz)
148  nonzeros[nz] = std::to_string(A[nz]);
149  // Create .m file
150  std::ofstream mfile;
151  std::string fname = (*this)->class_name() + "_" + (*this)->name_ + "_debug.m";
152  mfile.open(fname.c_str());
153  Dict opts;
154  opts["name"] = "A";
155  opts["nonzeros"] = nonzeros;
156  sparsity().export_code("matlab", mfile, opts);
157  mfile.close();
158  casadi_error("Numerical factorization failed for " + (*this)->name_
159  + "[" + (*this)->class_name() + "]. Linear system saved to '" + fname + "'");
160  }
161  m->is_nfact = true;
162  return flag;
163  }

187  int Linsol::solve(const double* A, double* x, casadi_int nrhs, bool tr, int mem) const {
188  auto m = static_cast<LinsolMemory*>((*this)->memory(mem));
189  casadi_assert(m->is_nfact, "Linear system has not been factorized");
190  if (m->t_total) m->fstats.at("solve").tic();
191  int ret = (*this)->solve(m, A, x, nrhs, tr);
192  if (m->t_total) m->fstats.at("solve").toc();
193  return ret;
