70  // Print comment

75  if (i!=0) uout() << ", ";

94  int Monitor::sp_reverse(bvec_t** arg, bvec_t** res, casadi_int* iw, bvec_t* w) const {
95  bvec_t *a = arg[0];
96  bvec_t *r = res[0];
97  casadi_int n = nnz();
98  if (a != r) {
99  for (casadi_int i=0; i<n; ++i) {
100  *a++ |= *r;
101  *r++ = 0;
102  }
103  }
104  return 0;
105  }

107  void Monitor::generate(CodeGenerator& g,
108  const std::vector<casadi_int>& arg,
109  const std::vector<casadi_int>& res) const {
110  // Print comment
111  g.local("rr", "casadi_real", "*");
112  g.local("i", "casadi_int");
113  g << g.printf(comment_ + "\\n[") << "\n"
114  << " for (i=0, rr=" << g.work(arg[0], dep(0).nnz())
115  << "; i!=" << nnz() << "; ++i) {\n"
116  << " if (i!=0) " << g.printf(", ") << "\n"
117  << " " << g.printf("%g", "*rr++") << "\n"
118  << " }\n"
119  << " " << g.printf("]\\n") << "\n";
120 
121  // Copy if not inplace
122  if (arg[0]!=res[0]) {
123  if (nnz()==1) {
124  g << g.workel(res[0]) << " = " << g.workel(arg[0]) << ";\n";
125  } else {
126  g << g.copy(g.work(arg[0], nnz()), nnz(), g.work(res[0], nnz())) << "\n";
127  }
128  }
129  }
