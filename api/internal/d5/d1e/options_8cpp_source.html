69  double Options::word_distance(const std::string &a, const std::string &b) {
71  if (a == b) return 0;
72  casadi_int na = a.size();
73  casadi_int nb = b.size();
74  if (na == 0) return static_cast<double>(nb);
75  if (nb == 0) return static_cast<double>(na);
76 
77  std::vector<casadi_int> v0(nb+1, 0);
78  std::vector<casadi_int> v1(nb+1, 0);
79 
80  for (casadi_int i=0;i<nb+1;++i)
81  v0[i] = i;
82 
83  char s;
84  char t;
85  std::locale loc;
86  for (casadi_int i=0;i<na;i++) {
87  v1[0] = i + 1;
88  for (casadi_int j=0; j<nb; j++) {
89  s = std::tolower(a[i], loc);
90  t = std::tolower(b[j], loc);
91  casadi_int cost = 0;
92  if (s != t)
93  cost = 1;
94 
95  v1[j+1] = std::min(std::min(v1[j] + 1, v0[j+1] + 1), v0[j] + cost);
96  }
97 
98  for (casadi_int j=0; j<nb+1; j++)
99  v0[j] = v1[j];
100  }
101 
102  return static_cast<double>(v1[nb]);
103  }

105  std::vector<std::string> Options::suggestions(const std::string& word, casadi_int amount) const {
106  // Best distances so far
107  const double inf = std::numeric_limits<double>::infinity();
108  std::vector<std::pair<double, std::string>> best(amount, {inf, ""});
109 
110  // Iterate over elements
111  best_matches(word, best);
112 
113  // Sort the elements in ascending order
114  stable_sort(best.begin(), best.end());
115 
116  // Collect the values that are non-infinite
117  std::vector<std::string> ret;
118  ret.reserve(amount);
119  for (auto&& e : best) {
120  if (e.first!=inf) {
121  ret.push_back(e.second);
122  }
123  }
124  return ret;
125  }

127  void Options::best_matches(const std::string& word,
128  std::vector<std::pair<double, std::string> >& best) const {
129  // Iterate over bases
130  for (auto&& b : bases) {
131  b->best_matches(word, best);
132  }
133 
134  // Worst match so far
135  auto worst = max_element(best.begin(), best.end());
136 
137  // Loop over entries
138  for (auto&& e : entries) {
139  // Get word distance
140  double d = word_distance(e.first, word);
141 
142  // Keep if better than the worst amongst the suggestions
143  if (d < worst->first) {
144  worst->first = d;
145  worst->second = e.first;
146  worst = max_element(best.begin(), best.end());
147  }
148  }
149  }

171  Dict Options::sanitize(const Dict& opts) {
172  // Drop nulls
173  if (has_null(opts)) {
174  // Create a new dictionary without the null entries
175  Dict ret;
176  for (auto&& op : opts) {
177  if (!op.second.is_null()) ret[op.first] = op.second;
178  }
179  return ret;
180  }
181 
182  // Treat the case where any of the options have a dot (dictionary shorthand)
183  if (has_dot(opts)) {
184  // New options dictionary being constructed
185  Dict ret;
186 
187  // Sub-dictionary and corresponding name being constructed
188  Dict sopts;
189  std::string sname;
190 
191  // Process options
192  for (auto&& op : opts) {
193  // Find the dot if any
194  std::string::size_type dotpos = op.first.find('.'), dotpos_end;
195  if (dotpos==std::string::npos) {
196  dotpos = op.first.find("__");
197  if (dotpos!=std::string::npos) dotpos_end = dotpos+2;
198  } else {
199  dotpos_end = dotpos+1;
200  }
201 
202  // Flush last sub-dictionary
203  if (!sname.empty() && (dotpos==std::string::npos
204  || op.first.compare(0, dotpos, sname)!=0)) {
205  ret[sname] = sopts;
206  sname.clear();
207  sopts.clear();
208  }
209 
210  // Add to dictionary
211  if (dotpos != std::string::npos) {
212  sname = op.first.substr(0, dotpos);
213  sopts[op.first.substr(dotpos_end)] = op.second;
214  } else {
215  ret[op.first] = op.second;
216  }
217  }
218 
219  // Flush trailing sub-dictionary
220  if (!sname.empty()) ret[sname] = sopts;
221 
222  return ret;
223  }
224 
225  // Nothing to do
226  return opts;
227  }

229  void Options::check(const Dict& opts) const {
230  // Make sure all options exist and have the correct type
231  for (auto&& op : opts) {
232  const Options::Entry* entry = find(op.first);
233 
234  // Informative error message if option does not exist
235  if (entry==nullptr) {
236  std::stringstream ss;
237  ss << "Unknown option: " << op.first << std::endl;
238  ss << std::endl;
239  ss << "Did you mean one of the following?" << std::endl;
240  for (auto&& s : suggestions(op.first)) {
241  print_one(s, ss);
242  }
243  ss << "Use print_options() to get a full list of options." << std::endl;
244  casadi_error(ss.str());
245  }
246 
247  // Check type
248  casadi_assert(op.second.can_cast_to(entry->type),
249  "Illegal type for " + op.first + ": " +
250  op.second.get_description() +
251  " cannot be cast to " +
252  GenericType::get_type_description(entry->type) + ".");
253 
254  }
255  }
