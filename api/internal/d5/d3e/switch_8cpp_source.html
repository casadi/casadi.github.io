73  if (i==0) {

104  void Switch::init(const Dict& opts) {
105  // Call the initialization method of the base class
106  FunctionInternal::init(opts);
107 
108  // Buffer for mismatching sparsities
109  size_t sz_buf=0;
110 
111  // Keep track of sparsity projections
112  project_in_ = project_out_ = false;
113 
114  // Get required work
115  for (casadi_int k=0; k<=f_.size(); ++k) {
116  const Function& fk = k<f_.size() ? f_[k] : f_def_;
117  if (fk.is_null()) continue;
118 
119  // Memory for evaluation
120  alloc(fk);
121 
122  // Required work vectors
123  size_t sz_buf_k=0;
124 
125  // Add size for input buffers
126  for (casadi_int i=1; i<n_in_; ++i) {
127  const Sparsity& s = fk.sparsity_in(i-1);
128  if (s!=sparsity_in_[i]) {
129  project_in_ = true;
130  alloc_w(s.size1()); // for casadi_project
131  sz_buf_k += s.nnz();
132  }
133  }
134 
135  // Add size for output buffers
136  for (casadi_int i=0; i<n_out_; ++i) {
137  const Sparsity& s = fk.sparsity_out(i);
138  if (s!=sparsity_out_[i]) {
139  project_out_ = true;
140  alloc_w(s.size1()); // for casadi_project
141  sz_buf_k += s.nnz();
142  }
143  }
144 
145  // Only need the largest of these work vectors
146  sz_buf = std::max(sz_buf, sz_buf_k);
147  }
148 
149  // Memory for the work vectors
150  alloc_w(sz_buf, true);
151  }

153  int Switch::eval(const double** arg, double** res, casadi_int* iw, double* w, void* mem) const {
154  // Get the function to be evaluated
155  casadi_int k = arg[0] ? static_cast<casadi_int>(*arg[0]) : 0;
156  const Function& fk = k>=0 && k<f_.size() ? f_[k] : f_def_;
157 
158  // Project arguments with different sparsity
159  const double** arg1;
160  if (project_in_) {
161  // Project one or more argument
162  arg1 = arg + n_in_;
163  for (casadi_int i=0; i<n_in_-1; ++i) {
164  const Sparsity& f_sp = fk.sparsity_in(i);
165  const Sparsity& sp = sparsity_in_[i+1];
166  arg1[i] = arg[i+1];
167  if (arg1[i] && f_sp!=sp) {
168  casadi_project(arg1[i], sp, w, f_sp, w + f_sp.nnz());
169  arg1[i] = w; w += f_sp.nnz();
170  }
171  }
172  } else {
173  // No inputs projected
174  arg1 = arg + 1;
175  }
176 
177  // Temporary memory for results with different sparsity
178  double** res1;
179  if (project_out_) {
180  // Project one or more results
181  res1 = res + n_out_;
182  for (casadi_int i=0; i<n_out_; ++i) {
183  const Sparsity& f_sp = fk.sparsity_out(i);
184  const Sparsity& sp = sparsity_out_[i];
185  res1[i] = res[i];
186  if (res1[i] && f_sp!=sp) {
187  res1[i] = w;
188  w += f_sp.nnz();
189  }
190  }
191  } else {
192  // No outputs projected
193  res1 = res;
194  }
195 
196  // Evaluate the corresponding function
197  if (fk(arg1, res1, iw, w, 0)) return 1;
198 
199  // Project results with different sparsity
200  if (project_out_) {
201  for (casadi_int i=0; i<n_out_; ++i) {
202  const Sparsity& f_sp = fk.sparsity_out(i);
203  const Sparsity& sp = sparsity_out_[i];
204  if (res[i] && f_sp!=sp) {
205  casadi_project(res1[i], f_sp, res[i], sp, w);
206  }
207  }
208  }
209  return 0;
210  }

212  Function Switch
213  ::get_forward(casadi_int nfwd, const std::string& name,
214  const std::vector<std::string>& inames,
215  const std::vector<std::string>& onames,
216  const Dict& opts) const {
217  // Derivative of each case
218  std::vector<Function> der(f_.size());
219  for (casadi_int k=0; k<f_.size(); ++k) {
220  if (!f_[k].is_null()) der[k] = f_[k].forward(nfwd);
221  }
222 
223  // Default case
224  Function der_def;
225  if (!f_def_.is_null()) der_def = f_def_.forward(nfwd);
226 
227  // New Switch for derivatives
228  Function sw = Function::conditional("switch_" + name, der, der_def);
229 
230  // Get expressions for the derivative switch
231  std::vector<MX> arg = sw.mx_in();
232  std::vector<MX> res = sw(arg);
233 
234  // Ignore seed for ind
235  arg.insert(arg.begin() + n_in_ + n_out_, MX(1, nfwd));
236 
237  Dict options = opts;
238  options["allow_duplicate_io_names"] = true;
239  // Create wrapper
240  return Function(name, arg, res, inames, onames, options);
241  }

243  Function Switch
244  ::get_reverse(casadi_int nadj, const std::string& name,
245  const std::vector<std::string>& inames,
246  const std::vector<std::string>& onames,
247  const Dict& opts) const {
248  // Derivative of each case
249  std::vector<Function> der(f_.size());
250  for (casadi_int k=0; k<f_.size(); ++k) {
251  if (!f_[k].is_null()) der[k] = f_[k].reverse(nadj);
252  }
253 
254  // Default case
255  Function der_def;
256  if (!f_def_.is_null()) der_def = f_def_.reverse(nadj);
257 
258  // New Switch for derivatives
259  Function sw = Function::conditional("switch_" + name, der, der_def);
260 
261  // Get expressions for the derivative switch
262  std::vector<MX> arg = sw.mx_in();
263  std::vector<MX> res = sw(arg);
264 
265  // No derivatives with respect to index
266  res.insert(res.begin(), MX(1, nadj));
267 
268  Dict options = opts;
269  options["allow_duplicate_io_names"] = true;
270 
271  // Create wrapper
272  return Function(name, arg, res, inames, onames, options);
273  }

298  int Switch::eval_sx(const SXElem** arg, SXElem** res,
299  casadi_int* iw, SXElem* w, void* mem) const {
300  // Input and output buffers
301  const SXElem** arg1 = arg + n_in_;
302  SXElem** res1 = res + n_out_;
303 
304  // Extra memory needed for chaining if_else calls
305  std::vector<SXElem> w_extra(nnz_out());
306  std::vector<SXElem*> res_tempv(n_out_);
307  SXElem** res_temp = get_ptr(res_tempv);
308 
309  for (casadi_int k=0; k<f_.size()+1; ++k) {
310 
311  // Local work vector
312  SXElem* wl = w;
313 
314  // Local work vector
315  SXElem* wll = get_ptr(w_extra);
316 
317  if (k==0) {
318  // For the default case, redirect the temporary results to res
319  std::copy_n(res, n_out_, res_temp);
320  } else {
321  // For the other cases, store the temporary results
322  for (casadi_int i=0; i<n_out_; ++i) {
323  res_temp[i] = wll;
324  wll += nnz_out(i);
325  }
326  }
327 
328  std::copy_n(arg+1, n_in_-1, arg1);
329  std::copy_n(res_temp, n_out_, res1);
330 
331  const Function& fk = k==0 ? f_def_ : f_[k-1];
332 
333  // Project arguments with different sparsity
334  for (casadi_int i=0; i<n_in_-1; ++i) {
335  if (arg1[i]) {
336  const Sparsity& f_sp = fk.sparsity_in(i);
337  const Sparsity& sp = sparsity_in_[i+1];
338  if (f_sp!=sp) {
339  SXElem *t = wl; wl += f_sp.nnz(); // t is non-const
340  casadi_project(arg1[i], sp, t, f_sp, wl);
341  arg1[i] = t;
342  }
343  }
344  }
345 
346  // Temporary memory for results with different sparsity
347  for (casadi_int i=0; i<n_out_; ++i) {
348  if (res1[i]) {
349  const Sparsity& f_sp = fk.sparsity_out(i);
350  const Sparsity& sp = sparsity_out_[i];
351  if (f_sp!=sp) { res1[i] = wl; wl += f_sp.nnz();}
352  }
353  }
354 
355  // Evaluate the corresponding function
356  if (fk(arg1, res1, iw, wl, 0)) return 1;
357 
358  // Project results with different sparsity
359  for (casadi_int i=0; i<n_out_; ++i) {
360  if (res1[i]) {
361  const Sparsity& f_sp = fk.sparsity_out(i);
362  const Sparsity& sp = sparsity_out_[i];
363  if (f_sp!=sp) casadi_project(res1[i], f_sp, res_temp[i], sp, wl);
364  }
365  }
366 
367  if (k>0) { // output the temporary results via an if_else
368  SXElem cond = k-1==arg[0][0];
369  for (casadi_int i=0; i<n_out_; ++i) {
370  if (res[i]) {
371  for (casadi_int j=0; j<nnz_out(i); ++j) {
372  res[i][j] = if_else(cond, res_temp[i][j], res[i][j]);
373  }
374  }
375  }
376  }
377 
378  }
379  return 0;
380  }

382  void Switch::codegen_body(CodeGenerator& g) const {
383  // Project arguments with different sparsity
384  if (project_in_) {
385  // Project one or more argument
386  g.local("i", "casadi_int");
387  g << "const casadi_real** arg1 = arg + " << n_in_ << ";\n";
388  }
389 
390  // Temporary memory for results with different sparsity
391  if (project_out_) {
392  // Project one or more results
393  g.local("i", "casadi_int");
394  g << "casadi_real** res1 = res + " << n_out_ << ";\n";
395  }
396 
397  if (project_in_)
398  g << "for (i=0; i<" << n_in_-1 << "; ++i) arg1[i]=arg[i+1];\n";
399 
400  if (project_out_)
401  g << "for (i=0; i<" << n_out_ << "; ++i) res1[i]=res[i];\n";
402 
403  // Codegen condition
404  bool if_else = f_.size()==1;
405  g.add_auxiliary(CodeGenerator::AUX_TO_INT);
406  g << (if_else ? "if" : "switch") << " (arg[0] ? casadi_to_int(*arg[0]) : 0) {\n";
407 
408  // Loop over cases/functions
409  for (casadi_int k=0; k<=f_.size(); ++k) {
410 
411  // For if, reverse order
412  casadi_int k1 = if_else ? 1-k : k;
413 
414  if (!if_else) {
415  // Codegen cases
416  if (k1<f_.size()) {
417  g << "case " << k1 << ":\n";
418  } else {
419  g << "default:\n";
420  }
421  } else if (k1==0) {
422  // Else
423  g << "} else {\n";
424  }
425 
426  // Get the function:
427  const Function& fk = k1<f_.size() ? f_[k1] : f_def_;
428  if (fk.is_null()) {
429  g << "return 1;\n";
430  } else {
431  // Project arguments with different sparsity
432  for (casadi_int i=0; i<n_in_-1; ++i) {
433  const Sparsity& f_sp = fk.sparsity_in(i);
434  const Sparsity& sp = sparsity_in_[i+1];
435  if (f_sp!=sp) {
436  if (f_sp.nnz()==0) {
437  g << "arg1[" << i << "]=0;\n";
438  } else {
439  g.local("t", "casadi_real", "*");
440  g << "t=w, w+=" << f_sp.nnz() << ";\n"
441  << g.project("arg1[" + str(i) + "]", sp, "t", f_sp, "w") << "\n"
442  << "arg1[" << i << "]=t;\n";
443  }
444  }
445  }
446 
447  // Temporary memory for results with different sparsity
448  for (casadi_int i=0; i<n_out_; ++i) {
449  const Sparsity& f_sp = fk.sparsity_out(i);
450  const Sparsity& sp = sparsity_out_[i];
451  if (f_sp!=sp) {
452  if (f_sp.nnz()==0) {
453  g << "res1[" << i << "]=0;\n";
454  } else {
455  g << "res1[" << i << "]=w, w+=" << f_sp.nnz() << ";\n";
456  }
457  }
458  }
459 
460  // Function call
461  g << "if (" << g(fk, project_in_ ? "arg1" : "arg+1",
462  project_out_ ? "res1" : "res",
463  "iw", "w") << ") return 1;\n";
464 
465  // Project results with different sparsity
466  for (casadi_int i=0; i<n_out_; ++i) {
467  const Sparsity& f_sp = fk.sparsity_out(i);
468  const Sparsity& sp = sparsity_out_[i];
469  if (f_sp!=sp) {
470  g << g.project("res1[" + str(i) + "]", f_sp,
471  g.res(i), sp, "w") << "\n";
472  }
473  }
474 
475  // Break (if switch)
476  if (!if_else) g << "break;\n";
477  }
478  }
479 
480  // End switch/else
481  g << "}\n";
482  }
