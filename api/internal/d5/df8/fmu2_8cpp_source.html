67 void Fmu2::init(const DaeBuilderInternal* dae) {
68  // Mark input indices
69  size_t numel = 0;
70  std::vector<bool> lookup(dae->n_variables(), false);
71  for (auto&& n : scheme_in_) {
72  for (size_t i : scheme_.at(n)) {
73  casadi_assert(!lookup.at(i), "Duplicate variable: " + dae->variable(i).name);
74  lookup.at(i) = true;
75  numel++;
76  }
77  }
78  // Input mappings
79  iind_.reserve(numel);
80  iind_map_.reserve(lookup.size());
81  for (size_t k = 0; k < lookup.size(); ++k) {
82  if (lookup[k]) {
83  iind_map_.push_back(iind_.size());
84  iind_.push_back(k);
85  } else {
86  iind_map_.push_back(-1);
87  }
88  }
89  // Mark output indices
90  numel = 0;
91  std::fill(lookup.begin(), lookup.end(), false);
92  for (auto&& n : scheme_out_) {
93  for (size_t i : scheme_.at(n)) {
94  casadi_assert(!lookup.at(i), "Duplicate variable: " + dae->variable(i).name);
95  lookup.at(i) = true;
96  numel++;
97  }
98  }
99  // Construct mappings
100  oind_.reserve(numel);
101  oind_map_.reserve(lookup.size());
102  for (size_t k = 0; k < lookup.size(); ++k) {
103  if (lookup[k]) {
104  oind_map_.push_back(oind_.size());
105  oind_.push_back(k);
106  } else {
107  oind_map_.push_back(-1);
108  }
109  }
110  // Inputs
111  ired_.resize(scheme_in_.size());
112  for (size_t i = 0; i < ired_.size(); ++i) {
113  auto&& s = scheme_.at(scheme_in_[i]);
114  ired_[i].resize(s.size());
115  for (size_t k = 0; k < s.size(); ++k) {
116  ired_[i][k] = iind_map_.at(s[k]);
117  }
118  }
119  // Outputs
120  ored_.resize(scheme_out_.size());
121  for (size_t i = 0; i < ored_.size(); ++i) {
122  auto&& s = scheme_.at(scheme_out_[i]);
123  ored_[i].resize(s.size());
124  for (size_t k = 0; k < s.size(); ++k) {
125  ored_[i][k] = oind_map_.at(s[k]);
126  }
127  }
128 
129  // Collect meta information for inputs
130  nominal_in_.reserve(iind_.size());
131  min_in_.reserve(iind_.size());
132  max_in_.reserve(iind_.size());
133  vn_in_.reserve(iind_.size());
134  vr_in_.reserve(iind_.size());
135  for (size_t i : iind_) {
136  const Variable& v = dae->variable(i);
137  nominal_in_.push_back(v.nominal);
138  min_in_.push_back(v.min);
139  max_in_.push_back(v.max);
140  vn_in_.push_back(v.name);
141  vr_in_.push_back(v.value_reference);
142  }
143  // Collect meta information for outputs
144  nominal_out_.reserve(oind_.size());
145  min_out_.reserve(oind_.size());
146  max_out_.reserve(oind_.size());
147  vn_out_.reserve(oind_.size());
148  vr_out_.reserve(oind_.size());
149  for (size_t i : oind_) {
150  const Variable& v = dae->variable(i);
151  nominal_out_.push_back(v.nominal);
152  min_out_.push_back(v.min);
153  max_out_.push_back(v.max);
154  vn_out_.push_back(v.name);
155  vr_out_.push_back(v.value_reference);
156  }
157 
158  // Numerical values for inputs
159  value_in_.resize(iind_.size());
160 
161  // Collect input and parameter values
162  vr_real_.clear();
163  vr_integer_.clear();
164  vr_boolean_.clear();
165  vr_string_.clear();
166  init_real_.clear();
167  init_integer_.clear();
168  init_boolean_.clear();
169  init_string_.clear();
170  for (size_t i = 0; i < dae->n_variables(); ++i) {
171  const Variable& v = dae->variable(i);
172  casadi_assert(v.numel == 1, "Vector variable support not implemented");
173  // Skip if the wrong type
174  if (v.causality != Causality::PARAMETER && v.causality != Causality::INPUT) continue;
175  // If nan - variable has not been set - keep default value
176  if (std::isnan(v.value.front())) continue;
177  // Value reference
178  fmi2ValueReference vr = v.value_reference;
179  // Get value
180  switch (to_fmi2(v.type)) {
181  case TypeFmi2::REAL:
182  init_real_.push_back(static_cast<fmi2Real>(v.value.front()));
183  vr_real_.push_back(vr);
184  break;
185  case TypeFmi2::INTEGER:
186  case TypeFmi2::ENUM:
187  init_integer_.push_back(static_cast<fmi2Integer>(v.value.front()));
188  vr_integer_.push_back(vr);
189  break;
190  case TypeFmi2::BOOLEAN:
191  init_boolean_.push_back(static_cast<fmi2Boolean>(v.value.front()));
192  vr_boolean_.push_back(vr);
193  break;
194  case TypeFmi2::STRING:
195  init_string_.push_back(v.stringvalue);
196  vr_string_.push_back(vr);
197  break;
198  default:
199  casadi_warning("Ignoring " + v.name + ", type: " + to_string(v.type));
200  }
201  }
202 
203  // Collect auxilliary variables
204  vn_aux_real_.clear();
205  vn_aux_integer_.clear();
206  vn_aux_boolean_.clear();
207  vn_aux_string_.clear();
208  vr_aux_real_.clear();
209  vr_aux_integer_.clear();
210  vr_aux_boolean_.clear();
211  vr_aux_string_.clear();
212  for (auto&& s : aux_) {
213  const Variable& v = dae->variable(s);
214  // Convert to expected type
215  fmi2ValueReference vr = v.value_reference;
216  // Sort by type
217  switch (to_fmi2(v.type)) {
218  case TypeFmi2::REAL:
219  // Real
220  vn_aux_real_.push_back(v.name);
221  vr_aux_real_.push_back(vr);
222  break;
223  case TypeFmi2::INTEGER:
224  case TypeFmi2::ENUM:
225  // Integer or enum
226  vn_aux_integer_.push_back(v.name);
227  vr_aux_integer_.push_back(vr);
228  break;
229  case TypeFmi2::BOOLEAN:
230  // Boolean
231  vn_aux_boolean_.push_back(v.name);
232  vr_aux_boolean_.push_back(vr);
233  break;
234  case TypeFmi2::STRING:
235  // String
236  vn_aux_string_.push_back(v.name);
237  vr_aux_string_.push_back(vr);
238  break;
239  default:
240  casadi_warning("Ignoring " + v.name + ", type: " + to_string(v.type));
241  }
242  }
243 
245  aux_value_.v_real.resize(vn_aux_real_.size());
246  aux_value_.v_integer.resize(vn_aux_integer_.size());
247  aux_value_.v_boolean.resize(vn_aux_boolean_.size());
248  aux_value_.v_string.resize(vn_aux_string_.size());
249 
250  // Get Jacobian sparsity information
251  jac_sp_ = dae->jac_sparsity(oind_, iind_);
252 
253  // Get Hessian sparsity information
254  hess_sp_ = dae->hess_sparsity(oind_, iind_);
255 
256  // Load DLL
257  std::string instance_name_no_dot = dae->model_identifier_;
258  std::replace(instance_name_no_dot.begin(), instance_name_no_dot.end(), '.', '_');
259  std::string dll_path = dae->path_ + "/binaries/" + system_infix()
260  + "/" + instance_name_no_dot + dll_suffix();
261  li_ = Importer(dll_path, "dll");
262 
263  declared_ad_ = dae->provides_directional_derivative_;
264 
265  if (dae->provides_directional_derivative_) {
266 
267  }
268 
269  // Path to resource directory
270  resource_loc_ = "file://" + dae->path_ + "/resources";
271 
272  // Copy info from DaeBuilder
273  fmutol_ = dae->fmutol_;
274  instance_name_ = dae->model_identifier_;
275  guid_ = dae->guid_;
276  logging_on_ = dae->debug_;
277 }

279 void Fmu2::finalize() {
280  // Get FMI C functions
281  instantiate_ = load_function<fmi2InstantiateTYPE>("fmi2Instantiate");
282  free_instance_ = load_function<fmi2FreeInstanceTYPE>("fmi2FreeInstance");
283  reset_ = load_function<fmi2ResetTYPE>("fmi2Reset");
284  setup_experiment_ = load_function<fmi2SetupExperimentTYPE>("fmi2SetupExperiment");
285  enter_initialization_mode_ = load_function<fmi2EnterInitializationModeTYPE>(
286  "fmi2EnterInitializationMode");
287  exit_initialization_mode_ = load_function<fmi2ExitInitializationModeTYPE>(
288  "fmi2ExitInitializationMode");
289  enter_continuous_time_mode_ = load_function<fmi2EnterContinuousTimeModeTYPE>(
290  "fmi2EnterContinuousTimeMode");
291  get_real_ = load_function<fmi2GetRealTYPE>("fmi2GetReal");
292  set_real_ = load_function<fmi2SetRealTYPE>("fmi2SetReal");
293  get_integer_ = load_function<fmi2GetIntegerTYPE>("fmi2GetInteger");
294  set_integer_ = load_function<fmi2SetIntegerTYPE>("fmi2SetInteger");
295  get_boolean_ = load_function<fmi2GetBooleanTYPE>("fmi2GetBoolean");
296  set_boolean_ = load_function<fmi2SetBooleanTYPE>("fmi2SetBoolean");
297  get_string_ = load_function<fmi2GetStringTYPE>("fmi2GetString");
298  set_string_ = load_function<fmi2SetStringTYPE>("fmi2SetString");
299 
300  if (declared_ad_) {
301  get_directional_derivative_ =
302  load_function<fmi2GetDirectionalDerivativeTYPE>("fmi2GetDirectionalDerivative");
303  }
304 
305  // Callback functions
306  functions_.logger = logger;
307  functions_.allocateMemory = calloc;
308  functions_.freeMemory = free;
309  functions_.stepFinished = 0;
310  functions_.componentEnvironment = 0;
311 
312  // Create a temporary instance
313  fmi2Component c = instantiate();
314  // Reset solver
315  setup_experiment(c);
316  // Set all values
317  if (set_values(c)) {
318  casadi_error("Fmu2::set_values failed");
319  }
320  // Initialization mode begins
321  if (enter_initialization_mode(c)) {
322  casadi_error("Fmu2::enter_initialization_mode failed");
323  }
324  // Get input values
325  if (get_in(c, &value_in_)) {
326  casadi_error("Fmu2::get_in failed");
327  }
328  // Get auxilliary variables
329  if (get_aux(c, &aux_value_)) {
330  casadi_error("Fmu2::get_aux failed");
331  }
332  // Free memory
333  free_instance(c);
334 }

336 void Fmu2::logger(fmi2ComponentEnvironment componentEnvironment,
337  fmi2String instanceName,
338  fmi2Status status,
339  fmi2String category,
340  fmi2String message, ...) {
341  // Variable number of arguments
342  va_list args;
343  va_start(args, message);
344  // Static & dynamic buffers
345  char buf[256];
346  size_t buf_sz = sizeof(buf);
347  char* buf_dyn = nullptr;
348  // Try to print with a small buffer
349  int n = vsnprintf(buf, buf_sz, message, args);
350  // Need a larger buffer?
351  if (n > buf_sz) {
352  buf_sz = n + 1;
353  buf_dyn = new char[buf_sz];
354  n = vsnprintf(buf_dyn, buf_sz, message, args);
355  }
356  // Print buffer content
357  if (n >= 0) {
358  uout() << "[" << instanceName << ":" << category << "] "
359  << (buf_dyn ? buf_dyn : buf) << std::endl;
360  }
361  // Cleanup
362  delete[] buf_dyn;
363  va_end(args);
364  // Throw error if failure
365  casadi_assert(n>=0, "Print failure while processing '" + std::string(message) + "'");
366 }

389 int Fmu2::init_mem(FmuMemory* m) const {
390  // Ensure not already instantiated
391  casadi_assert(m->instance == 0, "Already instantiated");
392  // Create instance
393  m->instance = instantiate();
394  // Reset solver
395  setup_experiment(m->instance);
396  // Set all values
397  if (set_values(m->instance)) {
398  casadi_warning("Fmu2::set_values failed");
399  return 1;
400  }
401  // Initialization mode begins
402  if (enter_initialization_mode(m->instance)) return 1;
403  // Initialization mode ends
404  if (exit_initialization_mode(m->instance)) return 1;
405  // Allocate/reset input buffer
406  m->ibuf_.resize(iind_.size());
407  std::fill(m->ibuf_.begin(), m->ibuf_.end(), casadi::nan);
408  // Allocate/reset output buffer
409  m->obuf_.resize(oind_.size());
410  std::fill(m->obuf_.begin(), m->obuf_.end(), casadi::nan);
411  // Allocate/reset seeds
412  m->seed_.resize(iind_.size());
413  std::fill(m->seed_.begin(), m->seed_.end(), 0);
414  // Allocate/reset sensitivities
415  m->sens_.resize(oind_.size());
416  std::fill(m->sens_.begin(), m->sens_.end(), 0);
417  // Allocate/reset changed
418  m->changed_.resize(iind_.size());
419  std::fill(m->changed_.begin(), m->changed_.end(), false);
420  // Allocate/reset requested
421  m->requested_.resize(oind_.size());
422  std::fill(m->requested_.begin(), m->requested_.end(), false);
423  // Also allocate memory for corresponding Jacobian entry (for debugging)
424  m->wrt_.resize(oind_.size());
425  // Successful return
426  return 0;
427 }

482  fmi2Status status = set_boolean_(c, get_ptr(vr_boolean_), vr_boolean_.size(),
483  get_ptr(init_boolean_));
484  if (status != fmi2OK) {
485  casadi_warning("fmi2SetBoolean failed");

496  }
497  }
498  // Successful return
499  return 0;

555 void Fmu2::get_stats(FmuMemory* m, Dict* stats,
556  const std::vector<std::string>& name_in, const InputStruct* in) const {
557  // To do: Use auxillary variables from last evaluation
558  (void)m; // unused
559  // Auxilliary values to be copied
560  const Value& v = aux_value_;
561  // Collect auxilliary variables
562  Dict aux;
563  // Real
564  for (size_t k = 0; k < vn_aux_real_.size(); ++k) {
565  aux[vn_aux_real_[k]] = static_cast<double>(v.v_real[k]);
566  }
567  // Integer
568  for (size_t k = 0; k < vn_aux_integer_.size(); ++k) {
569  aux[vn_aux_integer_[k]] = static_cast<casadi_int>(v.v_integer[k]);
570  }
571  // Boolean
572  for (size_t k = 0; k < vn_aux_boolean_.size(); ++k) {
573  aux[vn_aux_boolean_[k]] = static_cast<bool>(v.v_boolean[k]);
574  }
575  // String
576  for (size_t k = 0; k < vn_aux_string_.size(); ++k) {
577  aux[vn_aux_string_[k]] = v.v_string[k];
578  }
579  // Copy to stats
580  (*stats)["aux"] = aux;
581  // Loop over input variables
582  for (size_t k = 0; k < name_in.size(); ++k) {
583  // Only consider regular inputs
584  if (in[k].type == InputType::REG) {
585  // Get the indices
586  const std::vector<size_t>& iind = ired_.at(in[k].ind);
587  // Collect values
588  std::vector<double> v(iind.size());
589  for (size_t i = 0; i < v.size(); ++i) v[i] = value_in_.at(iind[i]);
590  // Save to stats
591  (*stats)[name_in[k]] = v;
592  }
593  }
594 }

596 int Fmu2::eval(FmuMemory* m) const {
597  // Gather inputs and outputs
598  gather_io(m);
599  // Number of inputs and outputs
600  size_t n_set = m->id_in_.size();
601  size_t n_out = m->id_out_.size();
602  // Fmi return flag
603  fmi2Status status;
604  // Set all variables
605  status = set_real_(m->instance, get_ptr(m->vr_in_), n_set, get_ptr(m->v_in_));
606  if (status != fmi2OK) {
607  casadi_warning("fmi2SetReal failed");
608  return 1;
609  }
610  // Quick return if nothing requested
611  if (n_out == 0) return 0;
612  // Calculate all variables
613  m->v_out_.resize(n_out);

656 int Fmu2::eval_fd(FmuMemory* m, bool independent_seeds) const {
657  // Number of inputs and outputs
658  size_t n_known = m->id_in_.size();
659  size_t n_unknown = m->id_out_.size();
660  // Quick return if nothing to be calculated
661  if (n_unknown == 0) return 0;
662  // Evalute (should not be necessary)
663  fmi2Status status = get_real_(m->instance, get_ptr(m->vr_out_), n_unknown, get_ptr(m->v_out_));
664  if (status != fmi2OK) {
665  casadi_warning("fmi2GetReal failed");
666  return 1;
667  }
668  // Make outputs dimensionless
669  for (size_t k = 0; k < n_unknown; ++k) m->v_out_[k] /= nominal_out_[m->id_out_[k]];
670  // Number of points in FD stencil
671  casadi_int n_points = n_fd_points(m->self.fd_);
672  // Offset for points
673  casadi_int offset = fd_offset(m->self.fd_);
674  // Memory for perturbed outputs
675  m->fd_out_.resize(n_points * n_unknown);
676  // Which inputs are in bounds
677  m->in_bounds_.resize(n_known);
678  // Memory for perturbed inputs
679  m->v_pert_.resize(n_known);
680  // Do any any inputs need flipping?
681  m->flip_.resize(n_known);
682  size_t first_flip = -1;
683  for (size_t i = 0; i < n_known; ++i) {
684  // Try to take step
685  double test = m->v_in_[i] + m->self.step_ * m->d_in_[i];
686  // Check if in bounds
687  size_t id = m->id_in_[i];
688  if (test >= min_in_[id] && test <= max_in_[id]) {
689  // Positive perturbation is fine
690  m->flip_[i] = false;
691  } else {
692  // Try negative direction instead
693  test = m->v_in_[i] - m->self.step_ * m->d_in_[i];
694  casadi_assert(test >= min_in_[id] && test <= max_in_[id],
695  "Cannot perturb " + vn_in_[id] + " at " + str(m->v_in_[i]) + ", min " + str(min_in_[id])
696  + ", max " + str(max_in_[id]) + ", nominal " + str(nominal_in_[id]));
697  m->flip_[i] = true;
698  if (first_flip == size_t(-1)) first_flip = i;
699  }
700  }
701  // If seeds are not independent, we have to flip the sign for all of the seeds or none
702  if (first_flip != size_t(-1) && !independent_seeds) {
703  // Flip the rest of the seeds
704  for (size_t i = 0; i < n_known; ++i) {
705  if (!m->flip_[i]) {
706  // Test negative direction
707  double test = m->v_in_[i] - m->self.step_ * m->d_in_[i];
708  size_t id = m->id_in_[i];
709  casadi_assert(test >= min_in_[id] && test <= max_in_[id],
710  "Cannot perturb both " + vn_in_[id] + " and " + vn_in_[first_flip]);
711  // Flip it too
712  m->flip_[i] = true;
713  }
714  }
715  }
716  // All perturbed outputs
717  const fmi2Real* yk_all[5] = {0};
718 
719  // Calculate all perturbed outputs
720  for (casadi_int k = 0; k < n_points; ++k) {
721  // Where to save the perturbed outputs
722  double* yk = &m->fd_out_[n_unknown * k];
723  casadi_assert_dev(k < 5);
724  yk_all[k] = yk;
725  // If unperturbed output, quick return
726  if (k == offset) {
727  casadi_copy(get_ptr(m->v_out_), n_unknown, yk);
728  continue;
729  }
730  // Perturbation size
731  double pert = (k - offset) * m->self.step_;
732  // Perturb inputs, if allowed
733  for (size_t i = 0; i < n_known; ++i) {
734  // Try to take step
735  double sign = m->flip_[i] ? -1 : 1;
736  double test = m->v_in_[i] + pert * sign * m->d_in_[i];
737  // Check if in bounds
738  size_t id = m->id_in_[i];
739  m->in_bounds_[i] = test >= min_in_[id] && test <= max_in_[id];
740  // Take step, if allowed
741  m->v_pert_[i] = m->in_bounds_[i] ? test : m->v_in_[i];
742  }
743  // Pass perturbed inputs to FMU
744  status = set_real_(m->instance, get_ptr(m->vr_in_), n_known, get_ptr(m->v_pert_));
745  if (status != fmi2OK) {
746  casadi_warning("fmi2SetReal failed");
747  return 1;
748  }
749  // Evaluate perturbed FMU
750  status = get_real_(m->instance, get_ptr(m->vr_out_), n_unknown, yk);
751  if (status != fmi2OK) {
752  casadi_warning("fmi2GetReal failed");
753  return 1;
754  }
755  // Post-process yk
756  for (size_t i = 0; i < n_unknown; ++i) {
757  // Variable id
758  size_t id = m->id_out_[i];
759  // Differentiation with respect to what variable
760  size_t wrt_id = m->wrt_.at(id);
761  // Find the corresponding input variable
762  size_t wrt_i;
763  for (wrt_i = 0; wrt_i < n_known; ++wrt_i) {
764  if (m->id_in_[wrt_i] == wrt_id) break;
765  }
766  // Check if in bounds
767  if (m->in_bounds_.at(wrt_i)) {
768  // Input was in bounds: Keep output, make dimensionless
769  yk[i] /= nominal_out_[m->id_out_[i]];
770  } else {
771  // Input was out of bounds: Discard output
772  yk[i] = nan;
773  }
774  }
775  }
776  // Restore FMU inputs
777  status = set_real_(m->instance, get_ptr(m->vr_in_), n_known, get_ptr(m->v_in_));
778  if (status != fmi2OK) {
779  casadi_warning("fmi2SetReal failed");
780  return 1;
781  }
782  // Step size
783  double h = m->self.step_;
784 
785  // Calculate FD approximation
786  finite_diff(m->self.fd_, yk_all, get_ptr(m->d_out_), h, n_unknown, eps);
787 
788  // Collect requested variables
789  for (size_t ind = 0; ind < m->id_out_.size(); ++ind) {
790  // Variable id
791  size_t id = m->id_out_[ind];
792  // With respect to what variable
793  size_t wrt = m->wrt_[id];
794  // Find the corresponding input variable
795  size_t wrt_i;
796  for (wrt_i = 0; wrt_i < n_known; ++wrt_i) {
797  if (m->id_in_[wrt_i] == wrt) break;
798  }
799  // Nominal value
800  double n = nominal_out_[id];
801  // Get the value
802  double d_fd = m->d_out_[ind] * n;
803  // Correct sign, if necessary
804  if (m->flip_[wrt_i]) d_fd = -d_fd;
805  // Use FD instead of AD or to compare with AD
806  if (m->self.validate_ad_) {
807  // Value to compare with
808  double d_ad = m->sens_[id];
809  // Nominal value used as seed
810  d_ad /= nominal_in_[wrt];
811  d_fd /= nominal_in_[wrt];
812  // Is it a not a number?
813  bool d_is_nan = d_ad != d_ad;
814  // Magnitude of derivatives
815  double d_max = std::fmax(std::fabs(d_fd), std::fabs(d_ad));
816  // Check if NaN or error exceeds thresholds
817  if (d_is_nan || (d_max > n * m->self.abstol_
818  && std::fabs(d_ad - d_fd) > d_max * m->self.reltol_)) {
819  // Offset for printing the stencil
820  double off = m->fd_out_.at(ind + offset * n_unknown);
821  // Warning or add to file
822  std::stringstream ss;
823  if (m->self.validate_ad_file_.empty()) {
824  // Issue warning
825  ss << (d_is_nan ? "NaN" : "Inconsistent") << " derivatives of " << vn_out_[id]
826  << " w.r.t. " << desc_in(m, wrt) << ", got " << d_ad
827  << " for AD vs. " << d_fd << " for FD[" << to_string(m->self.fd_) << "].";
828  // Print the stencil:
829  ss << "\nValues for step size " << h << ": " << (n * off) << " + [";
830  for (casadi_int k = 0; k < n_points; ++k) {
831  if (k > 0) ss << ", ";
832  ss << (n * (m->fd_out_.at(ind + k * n_unknown) - off));
833  }
834  ss << "]";
835  // Issue warning
836  casadi_warning(ss.str());
837  } else {
838  // Output
839  ss << vn_out_[id] << " ";
840  // Input
841  ss << vn_in_[wrt] << " ";
842  // Value
843  ss << m->ibuf_[wrt] << " ";
844  // Noninal
845  ss << nominal_in_[wrt] << " ";
846  // Min
847  ss << min_in_[wrt] << " ";
848  // Max
849  ss << max_in_[wrt] << " ";
850  // AD
851  ss << d_ad << " ";
852  // FD
853  ss << d_fd << " ";
854  // Step
855  ss << h << " ";
856  // Offset
857  ss << off << " ";
858  // Stencil
859  ss << "[";
860  for (casadi_int k = 0; k < n_points; ++k) {
861  if (k > 0) ss << ",";
862  ss << (n * (m->fd_out_.at(ind + k * n_unknown) - off));
863  }
864  ss << "]" << std::endl;
865  // Append to file
866  std::ofstream valfile;
867  valfile.open(m->self.validate_ad_file_, std::ios_base::app);
868  valfile << ss.str();
869  }
870  }
871  } else {
872  // Use instead of AD
873  m->sens_[id] = d_fd;
874  }
875  }
876  // Successful return
877  return 0;
878 }
