317 int Integrator::eval(const double** arg, double** res,
318  casadi_int* iw, double* w, void* mem) const {
319  auto m = static_cast<IntegratorMemory*>(mem);
320 
321  // Read inputs
322  const double* x0 = arg[INTEGRATOR_X0];
323  const double* z0 = arg[INTEGRATOR_Z0];
324  const double* p = arg[INTEGRATOR_P];
325  const double* u = arg[INTEGRATOR_U];
326  const double* rx0 = arg[INTEGRATOR_ADJ_XF];
327  const double* rz0 = arg[INTEGRATOR_ADJ_ZF];
328  const double* rp = arg[INTEGRATOR_ADJ_QF];
329  arg += INTEGRATOR_NUM_IN;
330 
331  // Read outputs
332  double* x = res[INTEGRATOR_XF];
333  double* z = res[INTEGRATOR_ZF];
334  double* q = res[INTEGRATOR_QF];
335  double* rx = res[INTEGRATOR_ADJ_X0];
336  double* rq = res[INTEGRATOR_ADJ_P];
337  double* uq = res[INTEGRATOR_ADJ_U];
338  res += INTEGRATOR_NUM_OUT;
339 
340  // Setup memory object
341  setup(m, arg, res, iw, w);
342 
343  // Reset solver, take time to t0
344  m->t = t0_;
345  reset(m, x0, z0, p);
346 
347  // Next stop time due to step change in input
348  casadi_int k_stop = next_stop(0, u);
349 
350  // Integrate forward
351  for (m->k = 0; m->k < nt(); ++m->k) {
352  // Update stopping time, if needed
353  if (m->k > k_stop) k_stop = next_stop(m->k, u);
354  // Advance solution
355  m->t_next = tout_[m->k];
356  m->t_stop = tout_[k_stop];
357  if (verbose_) casadi_message("Integrating forward to output time " + str(m->k) + ": t_next = "
358  + str(m->t_next) + ", t_stop = " + str(m->t_stop));
359  advance(m, u, x, z, q);
360  if (x) x += nx_;
361  if (z) z += nz_;
362  if (q) q += nq_;
363  if (u) u += nu_;
364  m->t = m->t_next;
365  }
366 
367  // Backwards integration, if needed
368  if (nrx_ > 0) {
369  // Take rx0, rz0, rp past the last grid point
370  if (rx0) rx0 += nrx_ * nt();
371  if (rz0) rz0 += nrz_ * nt();
372  if (rp) rp += nrp_ * nt();
373  if (uq) uq += nuq_ * nt();
374  // Next stop time due to step change in input
375  k_stop = nt();
376  // Reset the solver
377  resetB(m);
378  // Integrate backward
379  for (m->k = nt(); m->k-- > 0; ) {
380  m->t = tout_[m->k];
381  // Add impulse to backwards integration
382  if (rx0) rx0 -= nrx_;
383  if (rz0) rz0 -= nrz_;
384  if (rp) rp -= nrp_;
385  if (uq) uq -= nuq_;
386  if (u) u -= nu_;
387  impulseB(m, rx0, rz0, rp);
388  // Next output time, or beginning
389  casadi_int k_next = m->k - 1;
390  m->t_next = k_next < 0 ? t0_ : tout_[k_next];
391  // Update integrator stopping time
392  if (k_next < k_stop) k_stop = next_stopB(m->k, u);
393  m->t_stop = k_stop < 0 ? t0_ : tout_[k_stop];
394  // Proceed to the previous time point or t0
395  if (verbose_) casadi_message("Integrating backward from output time " + str(m->k)
396  + ": t_next = " + str(m->t_next) + ", t_stop = " + str(m->t_stop));
397  if (m->k > 0) {
398  retreat(m, u, 0, 0, uq);
399  } else {
400  retreat(m, u, rx, rq, uq);
401  }
402  }
403  // uq should contain the contribution from the grid point, not cumulative
404  if (uq) {
405  for (m->k = 0; m->k < nt() - 1; ++m->k) {
406  casadi_axpy(nuq_, -1., uq + nuq_, uq);
407  uq += nuq_;
408  }
409  }
410  }
411 
412  // Collect oracle statistics
413  join_results(m);
414 
415  // Print integrator statistics
416  if (print_stats_) print_stats(m);
417 
418  return 0;
419 }

453 void Integrator::init(const Dict& opts) {
454  // Default (temporary) options
455  double t0 = 0, tf = 1;
456  bool expand = false;
457  bool output_t0 = false;
458  std::vector<double> grid;
459  bool uses_legacy_options = false;
460 
461  // Read options
462  for (auto&& op : opts) {
463  if (op.first=="expand") {
464  expand = op.second;
465  } else if (op.first=="output_t0") {
466  output_t0 = op.second;
467  uses_legacy_options = true;
468  } else if (op.first=="print_stats") {
469  print_stats_ = op.second;
470  } else if (op.first=="nfwd") {
471  nfwd_ = op.second;
472  } else if (op.first=="nadj") {
473  nadj_ = op.second;
474  } else if (op.first=="grid") {
475  grid = op.second;
476  uses_legacy_options = true;
477  } else if (op.first=="augmented_options") {
478  augmented_options_ = op.second;
479  } else if (op.first=="t0") {
480  t0 = op.second;
481  uses_legacy_options = true;
482  } else if (op.first=="tf") {
483  tf = op.second;
484  uses_legacy_options = true;
485  }
486  }
487 
488  // Replace MX oracle with SX oracle?
489  if (expand) this->expand();
490 
491  // Store a copy of the options, for creating augmented integrators
492  opts_ = opts;
493 
494  // Construct t0_ and tout_ gbased on legacy options
495  if (uses_legacy_options) {
496  static bool first_encounter = true;
497  if (first_encounter) {
498  // Deprecation warning
499  casadi_warning("The options 't0', 'tf', 'grid' and 'output_t0' have been deprecated.\n"
500  "The same functionality is provided by providing additional input arguments to "
501  "the 'integrator' function, in particular:\n"
502  " * Call integrator(..., t0, tf, options) for a single output time, or\n"
503  " * Call integrator(..., t0, grid, options) for multiple grid points.\n"
504  "The legacy 'output_t0' option can be emulated by including or excluding 't0' in 'grid'.\n"
505  "Backwards compatibility is provided in this release only.");
506  first_encounter = false;
507  }
508 
509  // If grid unset, default to [t0, tf]
510  if (grid.empty()) grid = {t0, tf};
511 
512  // Construct t0 and tout from grid and output_t0
513  t0_ = grid.front();
514  tout_ = grid;
515  if (!output_t0) tout_.erase(tout_.begin());
516  }
517 
518  // Consistency checks: Sensitivities
519  casadi_assert(nfwd_ >= 0, "Number of forward sensitivities must be non-negative");
520  casadi_assert(nadj_ >= 0, "Number of adjoint sensitivities must be non-negative");
521 
522  // Consistency check: Valid oracle
523  casadi_assert(oracle_.n_in() == DYN_NUM_IN, "DAE has wrong number of inputs");
524  casadi_assert(oracle_.n_out() == DYN_NUM_OUT, "DAE has wrong number of outputs");
525 
526  // Consistency checks, input sparsities
527  for (casadi_int i = 0; i < DYN_NUM_IN; ++i) {
528  const Sparsity& sp = oracle_.sparsity_in(i);
529  if (i == DYN_T) {
530  casadi_assert(sp.is_empty() || sp.is_scalar(), "DAE time variable must be empty or scalar. "
531  "Got dimension " + str(sp.size()));
532  } else {
533  casadi_assert(sp.is_vector(), "DAE inputs must be vectors. "
534  + dyn_in(i) + " has dimension " + str(sp.size()) + ".");
535  }
536  casadi_assert(sp.is_dense(), "DAE inputs must be dense . "
537  + dyn_in(i) + " is sparse.");
538  }
539 
540  // Consistency checks, output sparsities
541  for (casadi_int i = 0; i < DYN_NUM_OUT; ++i) {
542  const Sparsity& sp = oracle_.sparsity_out(i);
543  casadi_assert(sp.is_vector(), "DAE outputs must be vectors. "
544  + dyn_out(i) + " has dimension " + str(sp.size()));
545  casadi_assert(sp.is_dense(), "DAE outputs must be dense . "
546  + dyn_out(i) + " is sparse.");
547  }
548 
549  // Get dimensions (excluding sensitivity equations), forward problem
550  nx1_ = oracle_.numel_in(DYN_X);
551  nz1_ = oracle_.numel_in(DYN_Z);
552  nq1_ = oracle_.numel_out(DYN_QUAD);
553  np1_ = oracle_.numel_in(DYN_P);
554  nu1_ = oracle_.numel_in(DYN_U);
555 
556  // Consistency checks
557  casadi_assert(nx1_ > 0, "Ill-posed ODE - no state");
558  casadi_assert(nx1_ == oracle_.numel_out(DYN_ODE), "Dimension mismatch for 'ode'");
559  casadi_assert(nz1_ == oracle_.numel_out(DYN_ALG), "Dimension mismatch for 'alg'");
560 
561  // Backward problem, if any
562  if (nadj_ > 0) {
563  // Generate backward DAE
564  rdae_ = oracle_.reverse(nadj_);
565  // Consistency checks
566  casadi_assert(rdae_.n_in() == BDYN_NUM_IN, "Backward DAE has wrong number of inputs");
567  casadi_assert(rdae_.n_out() == BDYN_NUM_OUT, "Backward DAE has wrong number of outputs");
568  casadi_assert(rdae_.numel_in(BDYN_X) == nx1_, "Dimension mismatch");
569  casadi_assert(rdae_.numel_in(BDYN_Z) == nz1_, "Dimension mismatch");
570  casadi_assert(rdae_.numel_in(BDYN_P) == np1_, "Dimension mismatch");
571  casadi_assert(rdae_.numel_in(BDYN_U) == nu1_, "Dimension mismatch");
572  casadi_assert(rdae_.numel_in(BDYN_ADJ_ODE) == nx1_ * nadj_, "Inconsistent dimensions");
573  casadi_assert(rdae_.numel_in(BDYN_ADJ_ALG) == nz1_ * nadj_, "Inconsistent dimensions");
574  casadi_assert(rdae_.numel_in(BDYN_ADJ_QUAD) == nq1_ * nadj_, "Inconsistent dimensions");
575  casadi_assert(rdae_.numel_out(BDYN_ADJ_P) == np1_ * nadj_, "Inconsistent dimensions");
576  casadi_assert(rdae_.numel_out(BDYN_ADJ_U) == nu1_ * nadj_, "Inconsistent dimensions");
577 
578  // Dimensions (excluding sensitivity equations), backward problem
579  nrx1_ = nx1_;
580  nrz1_ = nz1_;
581  nrp1_ = nq1_;
582  nrq1_ = np1_;
583  nuq1_ = nu1_;
584  } else {
585  // No backward problem
586  nrx1_ = nrz1_ = nrp1_ = nrq1_ = nuq1_ = 0;
587  }
588 
589  // Get dimensions (including sensitivity equations)
590  nx_ = nx1_ * (1 + nfwd_);
591  nz_ = nz1_ * (1 + nfwd_);
592  nq_ = nq1_ * (1 + nfwd_);
593  np_ = np1_ * (1 + nfwd_);
594  nu_ = nu1_ * (1 + nfwd_);
595  nrx_ = nrx1_ * nadj_ * (1 + nfwd_);
596  nrz_ = nrz1_ * nadj_ * (1 + nfwd_);
597  nrp_ = nrp1_ * nadj_ * (1 + nfwd_);
598  nrq_ = nrq1_ * nadj_ * (1 + nfwd_);
599  nuq_ = nuq1_ * nadj_ * (1 + nfwd_);
600 
601  // Call the base class method
602  OracleFunction::init(opts);
603 
604  // Create problem functions, forward problem
605  create_function("daeF", dyn_in(), dae_out());
606  if (nq_ > 0) create_function("quadF", dyn_in(), quad_out());
607  if (nfwd_ > 0) {
608  // one direction to conserve memory, symbolic processing time
609  create_forward("daeF", 1);
610  if (nq_ > 0) create_forward("quadF", 1);
611  }
612 
613  // Create problem functions, backward problem
614  if (nadj_ > 0) {
615  create_function(rdae_, "daeB", bdyn_in(), bdae_out());
616  if (nrq_ > 0 || nuq_ > 0) create_function(rdae_, "quadB", bdyn_in(), bquad_out());
617  if (nfwd_ > 0) {
618  // one direction to conserve memory, symbolic processing time
619  create_forward("daeB", 1);
620  if (nrq_ > 0 || nuq_ > 0) create_forward("quadB", 1);
621  }
622  }
623 
624  // Nominal values for states
625  nom_x_ = oracle_.nominal_in(DYN_X);
626  nom_z_ = oracle_.nominal_in(DYN_Z);
627 
628  // Get the sparsities of the forward and reverse DAE
629  sp_jac_dae_ = sp_jac_dae();
630  casadi_assert(!sp_jac_dae_.is_singular(),
631  "Jacobian of the forward problem is structurally rank-deficient. "
632  "sprank(J)=" + str(sprank(sp_jac_dae_)) + "<" + str(nx_+nz_));
633  if (nadj_ > 0) {
634  sp_jac_rdae_ = sp_jac_rdae();
635  casadi_assert(!sp_jac_rdae_.is_singular(),
636  "Jacobian of the backward problem is structurally rank-deficient. "
637  "sprank(J)=" + str(sprank(sp_jac_rdae_)) + "<" + str(nrx_+nrz_));
638  }

641  alloc_w(nx_, true); // x
642  alloc_w(nz_, true); // z
643  alloc_w(nx_, true); // x_prev
644  alloc_w(nrx_, true); // rx
645  alloc_w(nrz_, true); // rz
646  alloc_w(nrx_, true); // rx_prev
647  alloc_w(nrq_, true); // rq
648  alloc_w(nx_+nz_); // Sparsity::sp_solve

682 template<typename MatType>
683 Function Integrator::get_forward_dae(const std::string& name) const {
684  if (verbose_) casadi_message(name_ + "::get_forward_dae");
685 
686  // Get input and output expressions
687  std::vector<MatType> arg = MatType::get_input(oracle_);
688  std::vector<MatType> res = oracle_(arg);
689 
690  // Symbolic expression for augmented DAE
691  std::vector<std::vector<MatType>> aug_in(DYN_NUM_IN);
692  for (casadi_int i = 0; i < DYN_NUM_IN; ++i) aug_in[i].push_back(arg.at(i));
693  std::vector<std::vector<MatType>> aug_out(DYN_NUM_OUT);
694  for (casadi_int i = 0; i < DYN_NUM_OUT; ++i) aug_out[i].push_back(res.at(i));
695 
696  // Zero of time dimension
697  MatType zero_t = MatType::zeros(oracle_.sparsity_in(DYN_T));
698 
699  // Augment aug_in with forward sensitivity seeds
700  std::vector<std::vector<MatType>> seed(nfwd_, std::vector<MatType>(DYN_NUM_IN));
701  for (casadi_int d = 0; d < nfwd_; ++d) {
702  // Create expressions for augmented states
703  std::string pref = "aug" + str(d) + "_";
704  for (casadi_int i = 0; i < DYN_NUM_IN; ++i) {
705  if (i == DYN_T) {
706  seed[d][i] = zero_t;
707  } else {
708  seed[d][i] = MatType::sym(pref + dyn_in(i), oracle_.sparsity_in(i));
709  }
710  }
711  // Save to augmented function inputs
712  for (casadi_int i = 0; i < DYN_NUM_IN; ++i) {
713  if (i != DYN_T) aug_in[i].push_back(seed[d][i]);
714  }
715  }
716 
717  // Calculate directional derivatives
718  std::vector<std::vector<MatType>> sens;
719  bool always_inline = oracle_.is_a("SXFunction") || oracle_.is_a("MXFunction");
720  oracle_->call_forward(arg, res, seed, sens, always_inline, false);
721 
722  // Augment aug_out with forward sensitivity equations
723  casadi_assert_dev(sens.size() == nfwd_);
724  for (casadi_int d = 0; d < nfwd_; ++d) {
725  casadi_assert_dev(sens[d].size() == DYN_NUM_OUT);
726  for (casadi_int i = 0; i < DYN_NUM_OUT; ++i) {
727  aug_out[i].push_back(project(sens[d][i], oracle_.sparsity_out(i)));
728  }
729  }
730 
731  // Concatenate arrays
732  for (casadi_int i = 0; i < DYN_NUM_IN; ++i) arg.at(i) = vertcat(aug_in[i]);
733  for (casadi_int i = 0; i < DYN_NUM_OUT; ++i) res.at(i) = vertcat(aug_out[i]);
734 
735  // Convert to oracle function and return
736  return Function(name, arg, res, dyn_in(), dyn_out());
737 }

875 int Integrator::sp_forward(const bvec_t** arg, bvec_t** res,
876  casadi_int* iw, bvec_t* w, void* mem) const {
877  if (verbose_) casadi_message(name_ + "::sp_forward");
878 
879  // Inputs
880  const bvec_t* x0 = arg[INTEGRATOR_X0];
881  const bvec_t* p = arg[INTEGRATOR_P];
882  const bvec_t* u = arg[INTEGRATOR_U];
883  const bvec_t* rx0 = arg[INTEGRATOR_ADJ_XF];
884  const bvec_t* rp = arg[INTEGRATOR_ADJ_QF];
885  arg += n_in_;
886 
887  // Outputs
888  bvec_t* xf = res[INTEGRATOR_XF];
889  bvec_t* zf = res[INTEGRATOR_ZF];
890  bvec_t* qf = res[INTEGRATOR_QF];
891  bvec_t* rxf = res[INTEGRATOR_ADJ_X0];
892  bvec_t* rqf = res[INTEGRATOR_ADJ_P];
893  bvec_t* uqf = res[INTEGRATOR_ADJ_U];
894  res += n_out_;
895 
896  // Work vectors
897  bvec_t *x = w; w += nx_;
898  bvec_t *z = w; w += nz_;
899  bvec_t *x_prev = w; w += nx_;
900  bvec_t *rx = w; w += nrx_;
901  bvec_t *rz = w; w += nrz_;
902  bvec_t *rx_prev = w; w += nrx_;
903  bvec_t *rq = w; w += nrq_;
904 
905  // Memory struct for function calls below
906  SpForwardMem m = {arg, res, iw, w};
907 
908  // Copy initial guess to x_prev
909  std::copy_n(x0, nx_, x_prev);
910 
911  // Propagate forward
912  for (casadi_int k = 0; k < nt(); ++k) {
913  // Propagate through DAE function
914  if (fdae_sp_forward(&m, x_prev, p, u, x, z)) return 1;
915  for (casadi_int i = 0; i < nx_; ++i) x[i] |= x_prev[i];
916 
917  // "Solve" in order to resolve interdependencies (cf. Rootfinder)
918  std::copy_n(x, nx_, w);
919  std::copy_n(z, nz_, w + nx_);
920  std::fill_n(x, nx_ + nz_, 0);
921  sp_jac_dae_.spsolve(x, w, false);
922 
923  // Get xf and zf
924  if (xf) std::copy_n(x, nx_, xf);
925  if (zf) std::copy_n(z, nz_, zf);
926 
927  // Propagate to quadratures
928  if (nq_ > 0 && qf) {
929  if (fquad_sp_forward(&m, x, z, p, u, qf)) return 1;
930  }
931 
932  // Shift time
933  std::copy_n(x, nx_, x_prev);
934  if (xf) xf += nx_;
935  if (zf) zf += nz_;
936  if (qf) qf += nq_;
937  if (u) u += nu_;
938  }
939 
940  if (nrx_ > 0) {
941  // Clear rx_prev, rqf
942  std::fill_n(rx_prev, nrx_, 0);
943  if (rqf) std::fill_n(rqf, nrq_, 0);
944 
945  // Take rx0, rp, uqf past the last grid point
946  if (rx0) rx0 += nrx_ * nt();
947  if (rp) rp += nrp_ * nt();
948  if (uqf) uqf += nuq_ * nt();
949 
950  // Integrate backward
951  for (casadi_int k = nt(); k-- > 0; ) {
952  // Shift time
953  if (rx0) rx0 -= nrx_;
954  if (rp) rp -= nrp_;
955  if (uqf) uqf -= nuq_;
956  if (u) u -= nu_;
957 
958  // Add impulse from rx0
959  if (rx0) {
960  for (casadi_int i = 0; i < nrx_; ++i) rx_prev[i] |= rx0[i];
961  }
962 
963  // Propagate through DAE function
964  if (bdae_sp_forward(&m, x, z, p, u, rx_prev, rp, rx, rz)) return 1;
965  for (casadi_int i = 0; i < nrx_; ++i) rx[i] |= rx_prev[i];
966 
967  // "Solve" in order to resolve interdependencies (cf. Rootfinder)
968  std::copy_n(rx, nrx_ + nrz_, w);
969  std::fill_n(rx, nrx_ + nrz_, 0);
970  sp_jac_rdae_.spsolve(rx, w, false);
971 
972  // Propagate to quadratures
973  if ((nrq_ > 0 && rqf) || (nuq_ > 0 && uqf)) {
974  if (bquad_sp_forward(&m, x, z, p, u, rx, rz, rp, rq, uqf)) return 1;
975  // Sum contributions to rqf
976  if (rqf) {
977  for (casadi_int i = 0; i < nrq_; ++i) rqf[i] |= rq[i];
978  }
979  }

1157  bvec_t *rz = w; w += nrz_;
1158  bvec_t *rx_prev = w; w += nrx_;

1165  std::fill_n(x, nx_, 0);
1166  std::fill_n(z, nz_, 0);

1170  if (rxf) {
1171  std::copy_n(rxf, nrx_, rx);
1172  std::fill_n(rxf, nrx_, 0);
1173  } else {
1174  std::fill_n(rx, nrx_, 0);
1175  }
1176  // Reset rz
1177  std::fill_n(rz, nrz_, 0);
1178 
1179  // Save rqf: See note below
1180  if (rqf) std::copy_n(rqf, nrq_, rq);
1181 
1182  // Step backwards through backward problem
1183  for (casadi_int k = 0; k < nt(); ++k) {
1184  // Restore rqf: See note below
1185  if (rqf) std::copy_n(rq, nrq_, rqf);
1186 
1187  // Add impulse from rx0
1188  if (rx0) {
1189  for (casadi_int i = 0; i < nrx_; ++i) rx[i] |= rx0[i];
1190  std::fill_n(rx0, nrx_, 0);
1191  }
1192 
1193  // Get dependencies from backward quadratures
1194  if ((nrq_ > 0 && rqf) || (nuq_ > 0 && uqf)) {
1195  if (bquad_sp_reverse(&m, x, z, p, u, rx, rz, rp, rqf, uqf)) return 1;
1196  }
1197 
1198  // Propagate interdependencies
1199  std::fill_n(w, nrx_+nrz_, 0);
1200  sp_jac_rdae_.spsolve(w, rx, true);
1201  std::copy_n(w, nrx_+nrz_, rx);

1204  std::copy_n(rx, nrx_, rx_prev);
1205 
1206  // Indirect dependency via g
1207  if (bdae_sp_reverse(&m, x, z, p, u, rx_prev, rp, rx, rz)) return 1;
1208 
1209  // Update rx, rz
1210  std::copy_n(rx_prev, nrx_, rx);
1211  std::fill_n(rz, nrz_, 0);
1212 
1213  // Shift time
1214  if (rx0) rx0 += nrx_;
1215  if (rp) rp += nrp_;
1216  if (uqf) uqf += nuq_;
1217  if (u) u += nu_;
1218  }
1219  } else {
1220  // Take u past the last grid point
1221  if (u) u += nu_ * nt();
1222  }
1223 
1224  // Take xf, zf, qf past the last grid point
1225  if (xf) xf += nx_ * nt();
1226  if (zf) zf += nz_ * nt();
1227  if (qf) qf += nq_ * nt();
1228 
1229  // Step backwards through forward problem
1230  for (casadi_int k = nt(); k-- > 0; ) {
1231  // Shift time
1232  if (xf) xf -= nx_;
1233  if (zf) zf -= nz_;
1234  if (qf) qf -= nq_;
1235  if (u) u -= nu_;
1236 
1237  // Add impulse from outputs
1238  if (xf) {
1239  for (casadi_int i = 0; i < nx_; ++i) x[i] |= xf[i];
1240  std::fill_n(xf, nx_, 0);
1241  }
1242  if (zf) {
1243  for (casadi_int i = 0; i < nz_; ++i) z[i] |= zf[i];
1244  std::fill_n(zf, nz_, 0);
1245  }
1246 
1247  // Get dependencies from forward quadratures, if any
1248  if (nq_ > 0 && qf) {
1249  if (fquad_sp_reverse(&m, x, z, p, u, qf)) return 1;
1250  }
1251 
1252  // Propagate interdependencies
1253  std::fill_n(w, nx_ + nz_, 0);
1254  sp_jac_dae_.spsolve(w, x, true);
1255  std::copy_n(w, nx_ + nz_, x);
1256 
1257  // Direct dependency x_prev -> x
1258  std::copy_n(x, nx_, x_prev);
1259 
1260  // Indirect dependency through f
1261  if (fdae_sp_reverse(&m, x_prev, p, u, x, z)) return 1;
1262 
1263  // Update x, z
1264  std::copy_n(x_prev, nx_, x);
1265  std::fill_n(z, nz_, 0);
1266  }
1267 
1268  // Direct dependency x0 -> x
1269  if (x0) {
1270  for (casadi_int i = 0; i < nx_; ++i) x0[i] |= x_prev[i];

1309  for (casadi_int i = 0; i < INTEGRATOR_NUM_OUT; ++i) {
1310  ret_in.push_back(MX::sym("out_" + integrator_out(i), Sparsity(size_out(i))));
1311  }
1312

1314  std::vector<std::vector<MX>> aug_in(INTEGRATOR_NUM_IN);

1317  for (casadi_int d = 0; d < nfwd; ++d) {
1318  v[d] = MX::sym("fwd" + str(d) + "_" + integrator_in(i), sparsity_in(i));
1319  aug_in[i].push_back(v[d]);
1320  }
1321  ret_in.push_back(horzcat(v));
1322  }
1323

1325  std::vector<MX> integrator_in(INTEGRATOR_NUM_IN);
1326  for (casadi_int i = 0; i < INTEGRATOR_NUM_IN; ++i) {

1332  for (casadi_int d = 0; d < nfwd; ++d) {

1338  v.push_back(ret_in_split.at(k));
1339  for (casadi_int d = 0; d < nfwd; ++d) {
1340  v.push_back(aug_in_split[d].at(k));
1341  }
1342  }
1343  } else {

1345  v = aug_in[i];
1346  v.insert(v.begin(), ret_in[i]);
1347  }
1348  // Flatten all elements
1349  for (MX& e : v) e = vec(e);
1350  integrator_in[i] = horzcat(v);
1351  }

1353

1355  std::vector<MX> ret_out;
1356  ret_out.reserve(INTEGRATOR_NUM_OUT);

1362  for (casadi_int d = 0; d <= nfwd; ++d) {

1364  }
1365  }

1367  reshape(integrator_out[i], size1_out(i), offset.back()), offset);
1368  // Collect sensitivity blocks in the right order
1369  std::vector<MX> ret_out_split;
1370  ret_out_split.reserve(n_grid * nfwd);
1371  for (casadi_int d = 0; d < nfwd; ++d) {
1372  for (casadi_int k = 0; k < n_grid; ++k) {
1373  ret_out_split.push_back(integrator_out_split.at((nfwd + 1) * k + d + 1));
1374  }
1375  }
1376  ret_out.push_back(horzcat(ret_out_split));
1377  }
1378 
1379  Dict options = opts;
1380  options["allow_duplicate_io_names"] = true;

1406  aug_opts["nadj"] = nadj;
1407  aug_opts["nfwd"] = 0;
1408  } else {
1409  // Reformulate as forward-over-reverse
1410  aug_opts["nfwd"] = nadj;
1411  aug_opts["nadj"] = nadj_;
1412  }
1413  Function aug_int = integrator(aug_prefix + name_, plugin_name(),
1414  this_dae, t0_, tout_, aug_opts);
1415 
1416  // All inputs of the return function
1417  std::vector<MX> ret_in;
1418  ret_in.reserve(INTEGRATOR_NUM_IN + INTEGRATOR_NUM_OUT + INTEGRATOR_NUM_IN);
1419 
1420  // Add nondifferentiated inputs to ret_in
1421  for (casadi_int i = 0; i < INTEGRATOR_NUM_IN; ++i) {
1422  ret_in.push_back(MX::sym(integrator_in(i), sparsity_in(i)));
1423  }
1424 
1425  // Add nondifferentiated outputs (unused) to ret_in
1426  for (casadi_int i = 0; i < INTEGRATOR_NUM_OUT; ++i) {
1427  ret_in.push_back(MX::sym("out_" + integrator_out(i), Sparsity(size_out(i))));
1428  }
1429 
1430  // Create symbolic expressions for augmented problem, add adjoint seeds to ret_in
1431  std::vector<std::vector<MX>> aug_in(INTEGRATOR_NUM_OUT);
1432  std::vector<MX> v(nadj);
1433  for (casadi_int i = 0; i < INTEGRATOR_NUM_OUT; ++i) {
1434  for (casadi_int d=0; d<nadj; ++d) {
1435  v[d] = MX::sym("adj" + str(d) + "_" + integrator_out(i), sparsity_out(i));
1436  aug_in[i].push_back(v[d]);
1437  }
1438  ret_in.push_back(horzcat(v));
1439  }
1440 
1441  // Call the augmented integrator
1442  std::vector<MX> integrator_in(INTEGRATOR_NUM_IN);
1443  for (casadi_int i = 0; i < INTEGRATOR_NUM_IN; ++i) {
1444  // Output index contributing to adjoint seeds
1445  casadi_int j = adjmap_out(i);
1446  // Number of grid points for this integrator input
1447  casadi_int n_grid = grid_in(i) ? nt() : 1;
1448  // Split input and seeds by grid points, if necessary
1449  std::vector<MX> ret_in_split;
1450  std::vector<std::vector<MX>> aug_in_split(nadj);
1451  if (size1_in(i) > 0 && grid_in(i) && n_grid > 1) {
1452  // Split nondifferentiated input by grid point
1453  ret_in_split = horzsplit(ret_in[i], ret_in[i].size2() / nt());
1454  // Split augmented input by grid point
1455  for (casadi_int d = 0; d < nadj; ++d) {
1456  aug_in_split[d] = horzsplit(aug_in[j][d], aug_in[j][d].size2() / nt());
1457  }
1458  } else {
1459  // No reordering necessary
1460  ret_in_split = {ret_in[i]};
1461  for (casadi_int d = 0; d < nadj; ++d) aug_in_split[d] = {aug_in[j][d]};
1462  }
1463  // Vectorize all inputs to allow concatenation (unlike forward sensitivities,
1464  // number of rows for sensitivities may be different from original inputs)
1465  for (auto&& e : ret_in_split) e = vec(e);
1466  for (auto&& e1 : aug_in_split) {
1467  for (auto&& e2 : e1) e2 = vec(e2);
1468  }
1469  // Assemble input argument
1470  v.clear();
1471  for (casadi_int k = 0; k < ret_in_split.size(); ++k) {
1472  v.push_back(ret_in_split.at(k));
1473  for (casadi_int d = 0; d < nadj; ++d) {
1474  v.push_back(aug_in_split[d].at(k));
1475  }
1476  }
1477  integrator_in[i] = reshape(vertcat(v), aug_int.size_in(i));
1478  }
1479  std::vector<MX> integrator_out = aug_int(integrator_in);
1480 
1481  // Collect adjoint sensitivites
1482  std::vector<MX> ret_out;
1483  ret_out.reserve(INTEGRATOR_NUM_IN);
1484  for (casadi_int i = 0; i < INTEGRATOR_NUM_IN; ++i) {
1485  casadi_int j = adjmap_out(i);
1486  // Split return by grid points and sensitivities
1487  casadi_int n_grid = grid_out(j) ? nt() : 1;
1488  std::vector<casadi_int> offset = {0};
1489  for (casadi_int k = 0; k < n_grid; ++k) {
1490  offset.push_back(offset.back() + numel_out(j) / n_grid);
1491  for (casadi_int d = 0; d < nadj; ++d) {
1492  offset.push_back(offset.back() + numel_in(i) / n_grid);
1493  }
1494  }
1495  std::vector<MX> integrator_out_split = vertsplit(vec(integrator_out[j]), offset);
1496  // Collect sensitivity blocks in the right order
1497  std::vector<MX> ret_out_split;
1498  ret_out_split.reserve(n_grid * nadj);
1499  for (casadi_int d = 0; d < nadj; ++d) {
1500  for (casadi_int k = 0; k < n_grid; ++k) {
1501  ret_out_split.push_back(reshape(integrator_out_split.at((nadj + 1) * k + d + 1),
1502  size1_in(i), size2_in(i) / n_grid));
1503  }
1504  }
1505  ret_out.push_back(horzcat(ret_out_split));
1506  }
1507 
1508  Dict options = opts;
1509  options["allow_duplicate_io_names"] = true;
1510 
1511  // Create derivative function and return
1512  return Function(name, ret_in, ret_out, inames, onames, options);
1513 }
