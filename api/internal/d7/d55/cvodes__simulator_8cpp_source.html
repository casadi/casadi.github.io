73 void CvodesSimulator::init(const Dict& opts) {
74  if (verbose_) casadi_message(name_ + "::init");
75 
76  // Initialize the base classes
77  SundialsSimulator::init(opts);
78 
79  // Default options
80  std::string linear_multistep_method = "bdf";
81  std::string nonlinear_solver_iteration = "newton";
82  min_step_size_ = 0;
83 
84  // Read options
85  for (auto&& op : opts) {
86  if (op.first=="linear_multistep_method") {
87  linear_multistep_method = op.second.to_string();
88  } else if (op.first=="min_step_size") {
89  min_step_size_ = op.second;
90  } else if (op.first=="nonlinear_solver_iteration") {
91  nonlinear_solver_iteration = op.second.to_string();
92  }
93  }
94 
95  // Algebraic variables not supported
96  casadi_assert(nz_==0, "CVODES does not support algebraic variables");
97 
98  if (linear_multistep_method=="adams") {
99  lmm_ = CV_ADAMS;
100  } else if (linear_multistep_method=="bdf") {
101  lmm_ = CV_BDF;
102  } else {
103  casadi_error("Unknown linear multistep method: " + linear_multistep_method);
104  }
105 
106  if (nonlinear_solver_iteration=="newton") {
107  iter_ = CV_NEWTON;
108  } else if (nonlinear_solver_iteration=="functional") {
109  iter_ = CV_FUNCTIONAL;
110  } else {
111  casadi_error("Unknown nonlinear solver iteration: " + nonlinear_solver_iteration);
112  }
113 
114  // Attach functions for jacobian information
115  if (newton_scheme_ != NewtonScheme::DIRECT) {
116  set_function(oracle_.forward(1), "jtimes");
117  }
118 }

120 int CvodesSimulator::init_mem(void* mem) const {
121  if (SundialsSimulator::init_mem(mem)) return 1;
122  auto m = to_mem(mem);
123 
124  // Create CVodes memory block
125  m->mem = CVodeCreate(lmm_, iter_);
126  casadi_assert(m->mem!=nullptr, "CVodeCreate: Creation failed");
127 
128  // Set error handler function
129  THROWING(CVodeSetErrHandlerFn, m->mem, ehfun, m);
130 
131  // Set user data
132  THROWING(CVodeSetUserData, m->mem, m);
133 
134  // Initialize CVodes
135  double t0 = 0;
136  THROWING(CVodeInit, m->mem, rhs, t0, m->xz);
137 
138  // Absolute tolerances for each component
139  if (m->abstolv) {
140  casadi_copy(get_ptr(nom_x_), nx_, NV_DATA_S(m->abstolv));
141  casadi_copy(get_ptr(nom_z_), nz_, NV_DATA_S(m->abstolv) + nx_);
142  casadi_scal(nx_+nz_, abstol_, NV_DATA_S(m->abstolv));
143  }
144 
145  // Set tolerances
146  if (scale_abstol_) {
147  THROWING(CVodeSVtolerances, m->mem, reltol_, m->abstolv);
148  } else {
149  THROWING(CVodeSStolerances, m->mem, reltol_, abstol_);
150  }
151 
152  // Maximum number of steps
153  THROWING(CVodeSetMaxNumSteps, m->mem, max_num_steps_);
154 
155  // Initial step size
156  if (step0_!=0) THROWING(CVodeSetInitStep, m->mem, step0_);
157 
158  // Min step size
159  if (min_step_size_!=0) THROWING(CVodeSetMinStep, m->mem, min_step_size_);
160 
161  // Max step size
162  if (max_step_size_!=0) THROWING(CVodeSetMaxStep, m->mem, max_step_size_);
163 
164  // Maximum order of method
165  if (max_order_) THROWING(CVodeSetMaxOrd, m->mem, max_order_);
166 
167  // Coeff. in the nonlinear convergence test
168  if (nonlin_conv_coeff_!=0) THROWING(CVodeSetNonlinConvCoef, m->mem, nonlin_conv_coeff_);
169 
170  // attach a linear solver
171  if (newton_scheme_ == NewtonScheme::DIRECT) {
172  // Direct scheme
173  CVodeMem cv_mem = static_cast<CVodeMem>(m->mem);
174  cv_mem->cv_lmem = m;
175  cv_mem->cv_lsetup = lsetup;
176  cv_mem->cv_lsolve = lsolve;
177  cv_mem->cv_setupNonNull = TRUE;
178  } else {
179  // Iterative scheme
180  casadi_int pretype = use_precon_ ? PREC_LEFT : PREC_NONE;
181  switch (newton_scheme_) {
182  case NewtonScheme::DIRECT: casadi_assert_dev(0);
183  case NewtonScheme::GMRES: THROWING(CVSpgmr, m->mem, pretype, max_krylov_); break;
184  case NewtonScheme::BCGSTAB: THROWING(CVSpbcg, m->mem, pretype, max_krylov_); break;
185  case NewtonScheme::TFQMR: THROWING(CVSptfqmr, m->mem, pretype, max_krylov_); break;
186  default: casadi_error("No such Newton scheme");
187  }
188  THROWING(CVSpilsSetJacTimesVecFn, m->mem, jtimes);
189  if (use_precon_) THROWING(CVSpilsSetPreconditioner, m->mem, psetup, psolve);
190  }
191 
192  // Forward sensitivities
193  if (nfwd_ > 0) {
194  // Initialize sensitivity problem
195  int ism = CV_SIMULTANEOUS; // or CV_STAGGERED
196  THROWING(CVodeSensInit, m->mem, nfwd_, ism, sens_rhs, get_ptr(m->fwd_xz));
197 
198  // Set tolerances
199  if (scale_abstol_) {
200  THROWING(CVodeSensSVtolerances, m->mem, reltol_, get_ptr(m->abstolvS));
201  } else {
202  casadi_assert(!abstolS_.empty(), "here");
203  THROWING(CVodeSensSStolerances, m->mem, reltol_, const_cast<double*>(get_ptr(abstolS_)));
204  }
205  }
206 
207  return 0;
208 }

211  try {
212  casadi_assert_dev(user_data);

215  std::fill_n(m->arg, enum_traits<DynIn>::n_enum, nullptr);

246 void CvodesSimulator::advance(SimulatorMemory* mem, double t, double t_stop) const {
247  auto m = to_mem(mem);
248  // Do not integrate past change in input signals or past the end
249  THROWING(CVodeSetStopTime, m->mem, t_stop);
250  // Integrate
251  THROWING(CVode, m->mem, t, m->xz, &m->t, CV_NORMAL);
252  casadi_copy(NV_DATA_S(m->xz), nx_, m->xk);
253  //Get forward sensitivities
254  if (nfwd_ > 0) {
255  THROWING(CVodeGetSens, m->mem, &m->t, get_ptr(m->fwd_xz));
256  for (size_t i = 0; i < nfwd_; ++i) {
257  casadi_copy(NV_DATA_S(m->fwd_xz[i]), nx_, m->fwd_xk + i * nx_);
258  }
259  }
260  // Get stats
261  THROWING(CVodeGetIntegratorStats, m->mem, &m->nsteps, &m->nfevals, &m->nlinsetups,
262  &m->netfails, &m->qlast, &m->qcur, &m->hinused,
263  &m->hlast, &m->hcur, &m->tcur);
264  THROWING(CVodeGetNonlinSolvStats, m->mem, &m->nniters, &m->nncfails);
265 }

320 int CvodesSimulator::psolve(double t, N_Vector x, N_Vector xdot, N_Vector r,
321  N_Vector z, double gamma, double delta, int lr,
322  void *user_data, N_Vector tmp) {
323  try {
324  auto m = to_mem(user_data);
325  auto& s = m->self;
326 
327  // Get right-hand sides in m->v1
328  double* v = NV_DATA_S(r);
329  casadi_copy(v, s.nx_, m->v1);
330 
331  // Solve for undifferentiated right-hand-side, save to output
332  if (s.linsolF_.solve(m->lin_nz, m->v1, 1, false, m->mem_linsolF))
333  casadi_error("Linear system solve failed");
334  v = NV_DATA_S(z); // possibly different from r
335  casadi_copy(m->v1, s.nx_, v);
336 
337  return 0;
338  } catch(int flag) { // recoverable error
339  return flag;
340  } catch(std::exception& e) { // non-recoverable error
341  uerr() << "psolve failed: " << e.what() << std::endl;
342  return -1;
343  }

346 int CvodesSimulator::psetup(double t, N_Vector x, N_Vector xdot, booleantype jok,
347  booleantype *jcurPtr, double gamma, void *user_data,
348  N_Vector tmp1, N_Vector tmp2, N_Vector tmp3) {
349  try {
350  auto m = to_mem(user_data);
351  auto& s = m->self;
352  // Store gamma for later
353  m->gamma = gamma;
354 
355  // Calculate Jacobian
356  if (s.calculate_jac(m, t, x, xdot)) casadi_error("'jac' calculation failed");
357 
358  // Jacobian is now current
359  *jcurPtr = 1;
360 
361  // Copy Jacobian and scale with -gamma
362  casadi_copy(m->jac_x, s.jac_x_sp_.nnz(), m->lin_nz);
363  casadi_scal(s.jac_x_sp_.nnz(), -gamma, m->lin_nz);
364 
365  // Add diagonal contribution, project to correct sparsity
366  add_diag(s.jac_x_sp_, m->lin_nz, 1., s.lin_sp_, NV_DATA_S(tmp1));
367 
368  // Prepare the solution of the linear system (e.g. factorize)
369  if (s.linsolF_.nfact(m->lin_nz, m->mem_linsolF)) casadi_error("'jac' factorization failed");
370 
371  return 0;
372  } catch(int flag) { // recoverable error
373  return flag;
374  } catch(std::exception& e) { // non-recoverable error
375  uerr() << "psetup failed: " << e.what() << std::endl;
376  return -1;
377  }
378 }

380 int CvodesSimulator::calculate_jac(CvodesSimMemory* m, double t, N_Vector x, N_Vector xdot) const {
381  // Input buffers
382  std::fill_n(m->arg, enum_traits<DynIn>::n_enum + enum_traits<DynOut>::n_enum, nullptr);
383  m->arg[DYN_T] = &t; // t
384  m->arg[DYN_X] = NV_DATA_S(x); // x
385  m->arg[DYN_U] = m->u; // u
386  m->arg[DYN_P] = m->p; // p
387  m->arg[enum_traits<DynIn>::n_enum + DYN_ODE] = NV_DATA_S(xdot); // ode
388 
389  // Output buffers
390  std::fill_n(m->res, enum_traits<DynIn>::n_enum * enum_traits<DynOut>::n_enum, nullptr);
391  m->res[DYN_X + enum_traits<DynIn>::n_enum * DYN_ODE] = m->jac_x; // jac:ode:x
392  m->res[DYN_U + enum_traits<DynIn>::n_enum * DYN_ODE] = m->jac_u; // jac:ode:u
393  m->res[DYN_P + enum_traits<DynIn>::n_enum * DYN_ODE] = m->jac_p; // jac:ode:p
394 
395 #if 0
396  // Update input cache
397  bool changed = update_jac_in(m);
398 
399  // Quick return if inputs unchanged from last call
400  if (!changed) return 0;
401 #endif
402 
403  // Evaluate
404  if (calc_function(m, "jac")) return 1;
405 
406  // Successful return
407  return 0;
408 }

466 int CvodesSimulator::sens_rhs(int Ns, realtype t, N_Vector x, N_Vector xdot, N_Vector *fwd_x,
467  N_Vector *fwd_xdot, void *user_data, N_Vector tmp1, N_Vector tmp2) {
468  try {
469  auto m = to_mem(user_data);
470  auto& s = m->self;
471  // Calculate Jacobian
472  if (s.calculate_jac(m, t, x, xdot)) casadi_error("'jac' calculation failed");
473  // Perform Jacobian-vector multiplications (can be parallelized if it matters)
474  for (int k = 0; k < Ns; ++k) {
475  // Consistency checks
476  casadi_assert(NV_LENGTH_S(fwd_x[k]) == s.nx_, "Dimension mismatch");
477  casadi_assert(NV_LENGTH_S(fwd_xdot[k]) == s.nx_, "Dimension mismatch");
478  // Get pointers to data
479  double *fwd_x_k = NV_DATA_S(fwd_x[k]), *fwd_xdot_k = NV_DATA_S(fwd_xdot[k]);
480  // Clear fwd_xdot
481  casadi_clear(fwd_xdot_k, s.nx_);
482  // Matrix-vector multiplications
483  casadi_mv(m->jac_x, s.jac_x_sp_, fwd_x_k, fwd_xdot_k, false);
484  casadi_mv(m->jac_u, s.jac_u_sp_, m->fwd_u + s.nu_ * (s.ng_ - 1) * k, fwd_xdot_k, false);
485  casadi_mv(m->jac_p, s.jac_p_sp_, m->fwd_p, fwd_xdot_k, false);
486  }
487  return 0;
488  } catch(int flag) { // recoverable error
489  return flag;
490  } catch(std::exception& e) { // non-recoverable error
491  uerr() << "CvodesSimulator::sens_rhs failed: " << e.what() << std::endl;
492  return -1;
493  }
