238  void MX::set(const MX& m, bool ind1, const Matrix<casadi_int>& rr, const Matrix<casadi_int>& cc) {
239  // Row vector rr (e.g. in MATLAB) is transposed to column vector
240  if (rr.size1()==1 && rr.size2()>1) {
241  return set(m, ind1, rr.T(), cc);
242  }
243 
244  // Row vector cc (e.g. in MATLAB) is transposed to column vector
245  if (cc.size1()==1 && cc.size2()>1) {
246  return set(m, ind1, rr, cc.T());
247  }
248 
249  // Make sure rr and cc are dense vectors
250  casadi_assert(rr.is_dense() && rr.is_column(),
251  "MX::set: First index not dense vector");
252  casadi_assert(cc.is_dense() && cc.is_column(),
253  "MX::set: Second index not dense vector");
254 
255  // Assert dimensions of assigning matrix
256  if (rr.size1() != m.size1() || cc.size1() != m.size2()) {
257  if (m.is_scalar()) {
258  // m scalar means "set all"
259  return set(repmat(m, rr.size1(), cc.size1()), ind1, rr, cc);
260  } else if (rr.size1() == m.size2() && cc.size1() == m.size1()
261  && std::min(m.size1(), m.size2()) == 1) {
262  // m is transposed if necessary
263  return set(m.T(), ind1, rr, cc);
264  } else {
265  // Error otherwise
266  casadi_error("Dimension mismatch. lhs is " + str(rr.size1()) + "-by-"
267  + str(cc.size1()) + ", while rhs is " + str(m.size()));
268  }
269  }
270 
271  // Dimensions
272  casadi_int sz1 = size1(), sz2 = size2();
273 
274  // Report out-of-bounds
275  casadi_assert_in_range(rr.nonzeros(), -sz1+ind1, sz1+ind1);
276  casadi_assert_in_range(cc.nonzeros(), -sz2+ind1, sz2+ind1);
277 
278  // If we are assigning with something sparse, first remove existing entries
279  if (!m.is_dense()) {
280  erase(rr.nonzeros(), cc.nonzeros(), ind1);
281  }
282 
283  // Collect all assignments
284  IM el = IM::zeros(m.sparsity());
285  for (casadi_int j=0; j<el.size2(); ++j) { // Loop over columns of m
286  casadi_int this_j = cc->at(j) - ind1; // Corresponding column in this
287  if (this_j<0) this_j += sz2;
288  for (casadi_int k=el.colind(j); k<el.colind(j+1); ++k) { // Loop over rows of m
289  casadi_int i = m.row(k);
290  casadi_int this_i = rr->at(i) - ind1; // Corresponding row in this
291  if (this_i<0) this_i += sz1;
292  el->at(k) = this_i + this_j*sz1;
293  }
294  }
295  return set(m, false, el);
296  }

303  void MX::set(const MX& m, bool ind1, const Matrix<casadi_int>& rr) {
304  // Assert dimensions of assigning matrix
305  if (rr.sparsity() != m.sparsity()) {
306  if (rr.size() == m.size()) {
307  // Remove submatrix to be replaced
308  erase(rr.nonzeros(), ind1);
309 
310  // Find the intersection between rr's and m's sparsity patterns
311  Sparsity sp = rr.sparsity() * m.sparsity();
312 
313  // Project both matrices to this sparsity
314  return set(project(m, sp), ind1, Matrix<casadi_int>::project(rr, sp));
315  } else if (m.is_scalar()) {
316  // m scalar means "set all"
317  if (m.is_dense()) {
318  return set(MX(rr.sparsity(), m), ind1, rr);
319  } else {
320  return set(MX(rr.size()), ind1, rr);
321  }
322  } else if (rr.size1() == m.size2() && rr.size2() == m.size1()
323  && std::min(m.size1(), m.size2()) == 1) {
324  // m is transposed if necessary
325  return set(m.T(), ind1, rr);
326  } else {
327  // Error otherwise
328  casadi_error("Dimension mismatch. lhs is " + str(rr.size())
329  + ", while rhs is " + str(m.size()));
330  }
331  }
332 
333  // Dimensions of this
334  casadi_int sz1 = size1(), sz2 = size2(), sz = nnz(), nel = numel(), rrsz = rr.nnz();
335 
336  // Quick return if nothing to set
337  if (rrsz==0) return;
338 
339  // Check bounds
340  casadi_assert_in_range(rr.nonzeros(), -nel+ind1, nel+ind1);
341 
342  // Dense mode
343  if (is_dense() && m.is_dense()) {
344  return set_nz(m, ind1, rr);
345  }
346 
347  // Construct new sparsity pattern
348  std::vector<casadi_int> new_row=sparsity().get_row(), new_col=sparsity().get_col();
349  std::vector<casadi_int> nz(rr.nonzeros());
350  new_row.reserve(sz+rrsz);
351  new_col.reserve(sz+rrsz);
352  nz.reserve(rrsz);
353  for (std::vector<casadi_int>::iterator i=nz.begin(); i!=nz.end(); ++i) {
354  if (ind1) (*i)--;
355  if (*i<0) *i += nel;
356  new_row.push_back(*i % sz1);
357  new_col.push_back(*i / sz1);
358  }
359  Sparsity sp = Sparsity::triplet(sz1, sz2, new_row, new_col);
360 
361  // If needed, update pattern
362  if (sp != sparsity()) *this = project(*this, sp);
363 
364  // Find the nonzeros corresponding to rr
365  sparsity().get_nz(nz);
366 
367  // Create a nonzero assignment node
368  *this = m->get_nzassign(*this, nz);
369  }

389  void MX::get_nz(MX& m, bool ind1, const Matrix<casadi_int>& kk) const {
390  // If indexed matrix was a row/column vector, make sure that the result is too
391  bool tr = (is_column() && kk.is_row()) || (is_row() && kk.is_column());
392 
393  // Quick return if no entries
394  if (kk.nnz()==0) {
395  m = MX::zeros(tr ? kk.sparsity().T() : kk.sparsity());
396  return;
397  }
398 
399  // Check bounds
400  casadi_int sz = nnz();
401  casadi_assert_in_range(kk.nonzeros(), -sz+ind1, sz+ind1);
402 
403  // Handle index-1, negative indices
404  if (ind1 || *std::min_element(kk->begin(), kk->end())<0) {
405  Matrix<casadi_int> kk_mod = kk;
406  for (auto&& i : kk_mod.nonzeros()) {
407  casadi_assert(!(ind1 && i<=0),
408  "Matlab is 1-based, but requested index " + str(i) + ". "
409  "Note that negative slices are disabled in the Matlab interface. "
410  "Possibly you may want to use 'end'.");
411  if (ind1) i--;
412  if (i<0) i += sz;
413  }
414  get_nz(m, false, kk_mod); // Call recursively
415  return;
416  }
417 
418  // Return reference to the nonzeros
419  m = (*this)->get_nzref(tr ? kk.sparsity().T() : kk.sparsity(), kk.nonzeros());
420  }

447  void MX::set_nz(const MX& m, bool ind1, const Matrix<casadi_int>& kk) {
448  casadi_assert(kk.nnz()==m.nnz() || m.nnz()==1,
449  "MX::set_nz: length of non-zero indices (" + str(kk.nnz()) + ") " +
450  "must match size of rhs (" + str(m.nnz()) + ").");
451 
452  // Assert dimensions of assigning matrix
453  if (kk.sparsity() != m.sparsity()) {
454  if (m.is_scalar()) {
455  // m scalar means "set all"
456  if (!m.is_dense()) return; // Nothing to set
457  return set_nz(MX(kk.sparsity(), m), ind1, kk);
458  } else if (kk.size() == m.size()) {
459  // Project sparsity if needed
460  return set_nz(project(m, kk.sparsity()), ind1, kk);
461  } else if (kk.size1() == m.size2() && kk.size2() == m.size1()
462  && std::min(m.size1(), m.size2()) == 1) {
463  // m is transposed if necessary
464  return set_nz(m.T(), ind1, kk);
465  } else {
466  // Error otherwise
467  casadi_error("Dimension mismatch. lhs is " + str(kk.size())
468  + ", while rhs is " + str(m.size()));
469  }
470  }
471 
472  // Call recursively if points both objects point to the same node
473  if (this==&m) {
474  MX m_copy = m;
475  return set_nz(m_copy, ind1, kk);
476  }
477 
478  // Check bounds
479  casadi_int sz = nnz();
480  casadi_assert_in_range(kk.nonzeros(), -sz+ind1, sz+ind1);
481 
482  // Quick return if no assignments to be made
483  if (kk.nnz()==0) return;
484 
485  // Handle index-1, negative indices
486  if (ind1 || *std::min_element(kk->begin(), kk->end())<0) {
487  Matrix<casadi_int> kk_mod = kk;
488  for (auto&& i : kk_mod.nonzeros()) {
489  casadi_assert(!(ind1 && i<=0),
490  "Matlab is 1-based, but requested index " + str(i) + ". "
491  "Note that negative slices are disabled in the Matlab interface. "
492  "Possibly you may want to use 'end'.");
493  if (ind1) i--;
494  if (i<0) i += sz;
495  }
496  return set_nz(m, false, kk_mod); // Call recursively
497  }

500  *this = m->get_nzassign(*this, kk.nonzeros());

507  MX MX::binary(casadi_int op, const MX &x, const MX &y) {
508  // Check, correct dimensions
509  if (x.size()!=y.size() && !x.is_scalar() && !y.is_scalar()) {
510  // x and y are horizontal multiples of each other?
511  if (!x.is_empty() && !y.is_empty()) {
512  if (x.size1() == y.size1() && x.size2() % y.size2() == 0) {
513  return binary(op, x, repmat(y, 1, x.size2() / y.size2()));
514  } else if (y.size1() == x.size1() && y.size2() % x.size2() == 0) {
515  return binary(op, repmat(x, 1, y.size2() / x.size2()), y);
516  }
517  }
518  // x and y are empty horizontal multiples of each other?
519  if (x.size1()==0 && y.size1()==0 && x.size2()>0 && y.size2()>0) {
520  if (x.size2() % y.size2() == 0) {
521  return MX(0, x.size2());
522  } else if (y.size2() % x.size2() == 0) {
523  return MX(0, y.size2());
524  }
525  }
526  // Dimension mismatch
527  casadi_error("Dimension mismatch for " + casadi_math<double>::print(op, "x", "y") +
528  ", x is " + x.dim() + ", while y is " + y.dim());
529  }
530  // Call internal class
531  return x->get_binary(op, y);
532  }

1556  void MX::extract(std::vector<MX>& ex, std::vector<MX>& v,
1557  std::vector<MX>& vdef, const Dict& opts) {
1558  try {
1559  // Read options
1560  std::string v_prefix = "v_", v_suffix = "";
1561  bool lift_shared = true, lift_calls = false;
1562  casadi_int v_ind = 0;
1563  for (auto&& op : opts) {
1564  if (op.first == "prefix") {
1565  v_prefix = std::string(op.second);
1566  } else if (op.first == "suffix") {
1567  v_suffix = std::string(op.second);
1568  } else if (op.first == "lift_shared") {
1569  lift_shared = op.second;
1570  } else if (op.first == "lift_calls") {
1571  lift_calls = op.second;
1572  } else if (op.first == "offset") {
1573  v_ind = op.second;
1574  } else {
1575  casadi_error("No such option: " + std::string(op.first));
1576  }
1577  }
1578  // Sort the expression
1579  Function f("tmp_extract", std::vector<MX>{}, ex, Dict{{"max_io", 0}, {"allow_free", true}});
1580  auto *ff = f.get<MXFunction>();
1581  // Get references to the internal data structures
1582  const std::vector<MXAlgEl>& algorithm = ff->algorithm_;
1583  std::vector<MX> work(ff->workloc_.size()-1);
1584  // Count how many times an expression has been used
1585  std::vector<casadi_int> usecount(work.size(), 0);
1586  // Remember the origin of every calculation
1587  std::vector<std::pair<casadi_int, casadi_int> > origin(work.size(), std::make_pair(-1, -1));
1588  // Which evaluations to replace
1589  std::vector<std::pair<casadi_int, casadi_int> > replace;
1590  // Evaluate the algorithm to identify which evaluations to replace
1591  casadi_int k=0;
1592  for (auto it=algorithm.begin(); it<algorithm.end(); ++it, ++k) {
1593  // Increase usage counters
1594  switch (it->op) {
1595  case OP_CONST:
1596  case OP_PARAMETER:
1597  break;
1598  default: // Unary operation, binary operation or output
1599  for (casadi_int c=0; c<it->arg.size(); ++c) {
1600  // Identify nodes used more than once
1601  if (lift_calls && it->op == OP_CALL) {
1602  // If not already marked for replacing
1603  if (usecount.at(it->arg[c]) >= 0) {
1604  replace.push_back(origin.at(it->arg[c]));
1605  usecount.at(it->arg[c]) = -1; // Do not replace again
1606  }
1607  } else if (lift_shared && work[it->arg[c]].op() != OP_PARAMETER
1608  && work[it->arg[c]].op() != OP_CONST) {
1609  if (usecount.at(it->arg[c]) == 0) {
1610  // First time node is used
1611  usecount.at(it->arg[c]) = 1;
1612  } else if (usecount.at(it->arg[c]) == 1) {
1613  // Second time node is used
1614  replace.push_back(origin.at(it->arg[c]));
1615  usecount.at(it->arg[c]) = -1; // Do not replace again
1616  }
1617  }
1618  }
1619  }
1620  // Perform the operation
1621  switch (it->op) {
1622  case OP_OUTPUT:
1623  break;
1624  case OP_CONST:
1625  usecount[it->res.front()] = -1; // Never extract constants
1626  break;
1627  default:
1628  for (casadi_int c=0; c<it->res.size(); ++c) {
1629  if (it->res[c]>=0) {
1630  work[it->res[c]] = it->data.get_output(c);
1631  origin[it->res[c]] = std::make_pair(k, c);
1632  if (lift_calls && it->op == OP_CALL) {
1633  // If function call, replace right away
1634  replace.push_back(origin.at(it->res[c]));
1635  usecount.at(it->res[c]) = -1; // Do not replace again
1636  } else {
1637  usecount.at(it->res[c]) = 0; // Not (yet) extracted
1638  }
1639  }
1640  }
1641  break;
1642  }
1643  }
1644  // New variables and definitions
1645  v.clear();
1646  v.reserve(replace.size());
1647  vdef.clear();
1648  vdef.reserve(replace.size());
1649  // Quick return
1650  if (replace.empty()) return;
1651  // Sort the elements to be replaced in the order of appearence in the algorithm
1652  sort(replace.begin(), replace.end());
1653  std::vector<std::pair<casadi_int, casadi_int> >::const_iterator replace_it=replace.begin();
1654  // Arguments for calling the atomic operations
1655  std::vector<MX> oarg, ores;
1656  // Evaluate the algorithm
1657  k = 0;
1658  for (auto it=algorithm.begin(); it<algorithm.end(); ++it, ++k) {
1659  switch (it->op) {
1660  case OP_OUTPUT:
1661  casadi_assert(it->data->segment()==0, "Not implemented");
1662  ex[it->data->ind()] = work[it->arg.front()];
1663  break;
1664  case OP_CONST:
1665  work[it->res.front()] = it->data;
1666  break;
1667  default:
1668  {
1669  if (it->op == OP_PARAMETER) {
1670  // Free parameter
1671  work[it->res.front()] = it->data;
1672  } else {
1673  // Arguments of the operation
1674  oarg.resize(it->arg.size());
1675  for (casadi_int i=0; i<oarg.size(); ++i) {
1676  casadi_int el = it->arg[i];
1677  oarg[i] = el<0 ? MX(it->data->dep(i).size()) : work.at(el);
1678  }
1679  // Perform the operation
1680  ores.resize(it->res.size());
1681  it->data->eval_mx(oarg, ores);
1682  // Get the result
1683  for (casadi_int i=0; i<ores.size(); ++i) {
1684  casadi_int el = it->res[i];
1685  if (el>=0) work.at(el) = ores[i];
1686  }
1687  }
1688  // Possibly replace results with new variables
1689  for (casadi_int c=0; c<it->res.size(); ++c) {
1690  // Output index
1691  casadi_int ind = it->res[c];
1692  // In the list of nodes for replacing?
1693  bool replace_node = replace_it != replace.end()
1694  && replace_it->first==k && replace_it->second==c;
1695  // Call node (introduce variable for outputs, even if unused)
1696  bool output_node = lift_calls && it->op == OP_CALL;
1697  // Skip if no reason to replace
1698  if (!replace_node && !output_node) continue;
1699  // Create a new variable
1700  Sparsity v_sp = it->op == OP_PARAMETER ? it->data.sparsity() : ores.at(c).sparsity();
1701  v.push_back(MX::sym(v_prefix + std::to_string(v_ind++) + v_suffix, v_sp));
1702  // Add definition of new variable
1703  if (ind >= 0) {
1704  // Replace existing call
1705  casadi_assert(replace_node, "Consistency check");
1706  // Store the result
1707  vdef.push_back(work[ind]);
1708  // Use in calculations
1709  work[ind] = v.back();
1710  // Go to the next element to be replaced
1711  replace_it++;
1712  } else {
1713  // New node corresponding to an output
1714  casadi_assert(output_node, "Consistency check");
1715  // Store the result
1716  vdef.push_back(ores.at(c));
1717  }
1718  }
1719  }
1720  }
1721  }
1722  // Ensure all nodes have been replaced
1723  casadi_assert(replace_it == replace.end(), "Consistency check failed");
1724  } catch (std::exception& e) {
1725  CASADI_THROW_ERROR("extract", e.what());
1726  }
1727  }
