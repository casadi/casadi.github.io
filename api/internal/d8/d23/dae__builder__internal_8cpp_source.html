247 Initial Variable::default_initial(Causality causality, Variability variability) {
248  // According to table in FMI 2.0.2 specification, section 2.2.7
249  switch (variability) {
250  case Variability::CONSTANT:
251  if (causality == Causality::OUTPUT || causality == Causality::LOCAL)
252  return Initial::EXACT;
253  break;
254  case Variability::FIXED:
255  // Fall-through
256  case Variability::TUNABLE:
257  if (causality == Causality::PARAMETER)
258  return Initial::EXACT;
259  else if (causality == Causality::CALCULATED_PARAMETER || causality == Causality::LOCAL)
260  return Initial::CALCULATED;
261  break;
262  case Variability::DISCRETE:
263  // Fall-through
264  case Variability::CONTINUOUS:
265  if (causality == Causality::OUTPUT || causality == Causality::LOCAL)
266  return Initial::CALCULATED;
267  break;
268  default: break;
269  }
270  // Initial value not available
271  return Initial::NA;
272 }

293 XmlNode Variable::export_xml(const DaeBuilderInternal& self) const {
294  // Create new XmlNode
295  XmlNode r;
296  r.name = to_string(type);
297  // Name of variable
298  r.set_attribute("name", name);
299  // Value reference
300  r.set_attribute("valueReference", value_reference);
301  // Description, if any
302  if (!description.empty()) r.set_attribute("description", description);
303  // Causality
304  if (causality != Causality::LOCAL)
305  r.set_attribute("causality", to_string(causality));
306  // Variability (real variables are continuous by default)
307  if (!(is_real() && variability == Variability::CONTINUOUS))
308  r.set_attribute("variability", to_string(variability));
309  // Minimum attribute
310  if (min != -inf) {
311  if (is_real()) {
312  r.set_attribute("min", min);
313  } else {
314  r.set_attribute("min", static_cast<casadi_int>(min));
315  }
316  }
317  // Maximum attribute
318  if (max != inf) {
319  if (is_real()) {
320  r.set_attribute("max", max);
321  } else {
322  r.set_attribute("max", static_cast<casadi_int>(max));
323  }
324  }
325  // Unit
326  if (!unit.empty()) r.set_attribute("unit", unit);
327  // Display unit
328  if (!display_unit.empty()) r.set_attribute("displayUnit", display_unit);
329  // Nominal value, only for floats
330  if (is_real() && nominal != 1.) r.set_attribute("nominal", nominal);
331  // Start attribute, if any
332  if (has_start()) {
333  if (type == Type::BINARY || type == Type::STRING) {
334  casadi_warning("Start attribute for String, Binary not implemented.");
335  } else {
336  // Convert to string
337  std::stringstream ss;
338  for (size_t i = 0; i < start.size(); ++i) {
339  if (i > 0) ss << " ";
340  if (is_real()) {
341  ss << start.at(i);
342  } else {
343  ss << static_cast<casadi_int>(start.at(i));
344  }
345  }
346  r.set_attribute("start", ss.str());
347  }
348  }
349  // Derivative attribute, if any
350  if (der_of >= 0) {
351  r.set_attribute("derivative", self.variable(der_of).value_reference);
352  }
353  // Return XML representation
354  return r;
355 }

395 void DaeBuilderInternal::load_fmi_description(const std::string& filename) {
396  // Ensure no variables already
397  casadi_assert(n_variables() == 0, "Instance already has variables");
398 
399  // Parse XML file
400  XmlFile xml_file("tinyxml");
401  XmlNode fmi_desc = xml_file.parse(filename)[0]; // One child; fmiModelDescription
402 
403  // Read attributes
404  fmi_version_ = fmi_desc.attribute<std::string>("fmiVersion", "");
405  model_name_ = fmi_desc.attribute<std::string>("modelName", "");
406  guid_ = fmi_desc.attribute<std::string>("guid", "");
407  description_ = fmi_desc.attribute<std::string>("description", "");
408  author_ = fmi_desc.attribute<std::string>("author", "");
409  copyright_ = fmi_desc.attribute<std::string>("copyright", "");
410  license_ = fmi_desc.attribute<std::string>("license", "");
411  generation_tool_ = fmi_desc.attribute<std::string>("generationTool", "");
412  generation_date_and_time_ = fmi_desc.attribute<std::string>("generationDateAndTime", "");
413  variable_naming_convention_ = fmi_desc.attribute<std::string>("variableNamingConvention", "");
414  number_of_event_indicators_ = fmi_desc.attribute<casadi_int>("numberOfEventIndicators", 0);
415 
416  // Process ModelExchange
417  if (fmi_desc.has_child("ModelExchange"))
418  import_model_exchange(fmi_desc["ModelExchange"]);
419 
420  // Process ModelVariables
421  casadi_assert(fmi_desc.has_child("ModelVariables"), "Missing 'ModelVariables'");
422  import_model_variables(fmi_desc["ModelVariables"]);
423 
424  // Process model structure
425  if (fmi_desc.has_child("ModelStructure"))
426  import_model_structure(fmi_desc["ModelStructure"]);
427 
428  // **** Add binding equations ****
429  if (fmi_desc.has_child("equ:BindingEquations")) {
430  // Get a reference to the BindingEquations node
431  const XmlNode& bindeqs = fmi_desc["equ:BindingEquations"];
432  // Loop over binding equations
433  for (casadi_int i = 0; i < bindeqs.size(); ++i) {
434  // Reference to the binding equation
435  const XmlNode& beq_node = bindeqs[i];
436  // Get the variable and binding expression
437  Variable& var = read_variable(beq_node[0]);
438  if (beq_node[1].size() == 1) {
439  // Regular expression
440  var.beq = read_expr(beq_node[1][0]);
441  } else {
442  // OpenModelica 1.17 occationally generates integer values without type specifier (bug?)
443  casadi_assert(beq_node[1].size() == 0, "Not implemented");
444  casadi_int val;
445  beq_node[1].get(&val);
446  casadi_warning(var.name + " has binding equation without type specifier: " + str(val));
447  var.beq = val;
448  }
449  }
450  }
451 
452  // **** Add dynamic equations, initial equations ****
453  symbolic_ = false; // use DLL by default
454  for (bool init_eq : {true, false}) {
455  const char* equ = init_eq ? "equ:InitialEquations" : "equ:DynamicEquations";
456  if (fmi_desc.has_child(equ)) {
457  // Symbolic model equations available
458  symbolic_ = true;
459  // Get a reference to the DynamicEquations node
460  const XmlNode& dyneqs = fmi_desc[equ];
461  // Add equations
462  for (casadi_int i = 0; i < dyneqs.size(); ++i) {
463  // Get a reference to the variable
464  const XmlNode& n = dyneqs[i];
465  try {
466  // Consistency checks
467  casadi_assert_dev(n.name == "equ:Equation");
468  casadi_assert_dev(n.size() == 1 && n[0].name == "exp:Sub");
469  // Ensure not empty
470  if (n[0].size() == 0) {
471  casadi_warning(str(equ) + "#" + str(i) + " is empty, ignored.");
472  continue;
473  }
474  // Get the left-hand-sides and right-hand-sides
475  const XmlNode& lhs = n[0][0];
476  const XmlNode& rhs = n[0][1];
477  // Left-hand-side needs to be a variable
478  Variable& v = read_variable(lhs);
479  // Right-hand-side is the binding equation
480  MX beq = read_expr(rhs);
481  // Set the equation
482  w_.push_back(find(v.name));
483  v.beq = beq;
484  // Also add to list of initial equations
485  if (init_eq) {
486  init_lhs_.push_back(v.v);
487  init_rhs_.push_back(beq);
488  }
489  } catch (std::exception& e) {
490  uerr() << "Failed to read " << equ << "#" << i << ": " << e.what() << std::endl;
491  }
492  }
493  }
494  }

632 void DaeBuilderInternal::update_dependencies() const {
633  // Get oracle function
634  const Function& oracle = this->oracle();
635  // Dependendencies of the ODE right-hand-side
636  Sparsity dode_dxT = oracle.jac_sparsity(oracle.index_out("ode"), oracle.index_in("x")).T();
637  Sparsity dode_duT = oracle.jac_sparsity(oracle.index_out("ode"), oracle.index_in("u")).T();
638  for (casadi_int i = 0; i < x_.size(); ++i) {
639  // Get output variable
640  const Variable& xdot = variable(variable(x_.at(i)).der);
641  // Clear dependencies
642  xdot.dependencies.clear();
643  // Dependencies on states

837 MX DaeBuilderInternal::read_expr(const XmlNode& node) {
838  try {
839 
840  const std::string& fullname = node.name;
841  if (fullname.find("exp:")== std::string::npos) {
842  casadi_error("DaeBuilderInternal::read_expr: unknown - expression is supposed to "
843  "start with 'exp:' , got " + fullname);
844  }
845 
846  // Chop the 'exp:'
847  std::string name = fullname.substr(4);
848 
849  // The switch below is alphabetical, and can be thus made more efficient,
850  // for example by using a switch statement of the first three letters,
851  // if it would ever become a bottleneck
852  if (name=="Add") {
853  return read_expr(node[0]) + read_expr(node[1]);
854  } else if (name=="Acos") {
855  return acos(read_expr(node[0]));
856  } else if (name=="Asin") {
857  return asin(read_expr(node[0]));
858  } else if (name=="Atan") {
859  return atan(read_expr(node[0]));
860  } else if (name=="Cos") {
861  return cos(read_expr(node[0]));
862  } else if (name=="Der") {
863  return variable(read_variable(node[0]).der_of).v;
864  } else if (name=="Div") {
865  return read_expr(node[0]) / read_expr(node[1]);
866  } else if (name=="Exp") {
867  return exp(read_expr(node[0]));
868  } else if (name=="Identifier") {
869  return read_variable(node).v;
870  } else if (name=="IntegerLiteral" || name=="BooleanLiteral") {
871  casadi_int val;
872  node.get(&val);
873  return val;
874  } else if (name=="Instant") {
875  double val;
876  node.get(&val);
877  return val;
878  } else if (name=="Log") {
879  return log(read_expr(node[0]));
880  } else if (name=="LogLeq") { // Logical less than equal
881  return read_expr(node[0]) <= read_expr(node[1]);
882  } else if (name=="LogGeq") { // Logical greater than equal
883  return read_expr(node[0]) >= read_expr(node[1]);
884  } else if (name=="LogLt") { // Logical less than
885  return read_expr(node[0]) < read_expr(node[1]);
886  } else if (name=="LogGt") { // Logical greater than
887  return read_expr(node[0]) > read_expr(node[1]);
888  } else if (name=="Max") {
889  return fmax(read_expr(node[0]), read_expr(node[1]));
890  } else if (name=="Min") {
891  return fmin(read_expr(node[0]), read_expr(node[1]));
892  } else if (name=="Mul") { // Multiplication
893  return read_expr(node[0]) * read_expr(node[1]);
894  } else if (name=="Neg") {
895  return -read_expr(node[0]);
896  } else if (name=="NoEvent") {
897  // NOTE: This is a workaround, we assume that whenever NoEvent occurs,
898  // what is meant is a switch
899  casadi_int n = node.size();
900 
901  // Default-expression
902  MX ex = read_expr(node[n-1]);
903 
904  // Evaluate ifs
905  for (casadi_int i=n-3; i>=0; i -= 2) {
906  ex = if_else(read_expr(node[i]), read_expr(node[i+1]), ex);
907  }
908 
909  return ex;
910  } else if (name=="Pow") {
911  return pow(read_expr(node[0]), read_expr(node[1]));
912  } else if (name=="RealLiteral") {
913  double val;
914  node.get(&val);
915  return val;
916  } else if (name=="Sin") {
917  return sin(read_expr(node[0]));
918  } else if (name=="Sqrt") {
919  return sqrt(read_expr(node[0]));
920  } else if (name=="StringLiteral") {
921  casadi_error(node.text);
922  } else if (name=="Sub") {
923  return read_expr(node[0]) - read_expr(node[1]);
924  } else if (name=="Tan") {
925  return tan(read_expr(node[0]));
926  } else if (name=="Time") {
927  return var(t_.at(0));
928  } else if (name=="TimedVariable") {
929  return read_variable(node[0]).v;
930  } else if (name=="FunctionCall") {
931  // Get the name of the function
932  std::string fname = qualified_name(node["exp:Name"]);
933  casadi_warning("Function call to '" + fname + "' incomplete");
934  // Collect the arguments
935  const XmlNode& args = node["exp:Arguments"];
936  std::vector<MX> farg(args.size());
937  for (casadi_int i = 0; i < args.size(); ++i) {
938  // Lift input arguments
939  Variable& v = new_variable("w_" + str(w_.size()));
940  v.v = MX::sym(v.name);
941  // Add to list of variables
942  w_.push_back(v.index);
943  // Set binding expression
944  v.beq = read_expr(args[i]);
945  // Add to list of function arguments
946  farg[i] = v.v;
947  }
948  // Return argument (scalar for now)
949  Variable& r = new_variable("w_" + str(w_.size()));
950  r.v = MX::sym(r.name);
951  // Add to list of variables
952  w_.push_back(r.index);
953  // Return output variable
954  return r.v;
955  } else if (name=="Array") {
956  // Array of arguments
957  std::vector<MX> v(node.size());
958  for (casadi_int i = 0; i < v.size(); ++i) v[i] = read_expr(node[i]);
959  return vertcat(v);
960  }
961 
962  // throw error if reached this point
963  casadi_error("Unknown node: " + name);
964  } catch (std::exception& e) {
965  THROW_ERROR_NODE("read_expr", node, e.what());
966  return {};
967  }
968 }

1777 MX DaeBuilderInternal::jac_vdef_v_from_calls(std::map<MXNode*, CallIO>& call_nodes,
1778  const std::vector<casadi_int>& h_offsets) const {
1779  // Calculate all Jacobian expressions
1780  for (auto call_it = call_nodes.begin(); call_it != call_nodes.end(); ++call_it) {
1781  call_it->second.calc_jac();
1782  }
1783  // Row offsets in jac_vdef_v
1784  casadi_int voffset_begin = 0, voffset_end = 0, voffset_last = 0;
1785  // Vertical and horizontal slices of jac_vdef_v
1786  std::vector<MX> vblocks, hblocks;
1787  // All blocks for this block row
1788  std::map<size_t, MX> jac_brow;
1789  // Definitions of w
1790  std::vector<MX> wdef = this->wdef();
1791  // Collect all Jacobian blocks
1792  for (size_t vdefind = 0; vdefind < wdef.size(); ++vdefind) {
1793  // Current element handled
1794  const MX& vdefref = wdef.at(vdefind);
1795  // Update vertical offset
1796  voffset_begin = voffset_end;
1797  voffset_end += vdefref.numel();
1798  // Handle function call nodes
1799  if (vdefref.is_output()) {
1800  // Which output of the function are we calculating?
1801  casadi_int oind = vdefref.which_output();

1804  // Find data about inputs and outputs
1805  auto call_it = call_nodes.find(c.get());
1806  casadi_assert_dev(call_it != call_nodes.end());

1814  }

1816  if (voffset_last != voffset_begin) {

1828  // Update offsets

1830  }
1831  // Add trailing empty block, if needed
1832  if (hoffset < h_offsets.back())
1833  hblocks.push_back(MX(vdefref.numel(), h_offsets.back() - hoffset));

1836  // Keep track of the offset handled in jac_brow
1837  voffset_last = voffset_end;
1838  }
1839  }
1840  // Add empty trailing row to vblocks, if any

1842  vblocks.push_back(MX(voffset_end - voffset_last, h_offsets.back()));

2815 Sparsity DaeBuilderInternal::hess_sparsity(const std::vector<size_t>& oind,
2816  const std::vector<size_t>& iind) const {
2817  // Mark inputs
2818  std::vector<casadi_int> lookup(n_variables(), -1);
2819  for (size_t i = 0; i < iind.size(); ++i) lookup.at(iind[i]) = i;
2820  // Which variables enter as a nonlinear dependency in any variable in oind
2821  std::vector<bool> nonlin(iind.size(), false);
2822  // List of nonlinearly entering variables for the specific output
2823  std::vector<casadi_int> nonlin_list;
2824  // Rows and columns of the Hessian
2825  std::vector<casadi_int> row, col;
2826  // Loop over output variables
2827  for (casadi_int j = 0; j < oind.size(); ++j) {
2828  const Variable& v = variable(oind[j]);
2829  // Loop over dependencies
2830  for (size_t k = 0; k < v.dependencies.size(); ++k) {
2831  if (v.dependenciesKind.empty() || v.dependenciesKind.at(k) == DependenciesKind::DEPENDENT) {
2832  casadi_int i = lookup.at(v.dependencies[k]);
2833  if (i >= 0 && !nonlin.at(i)) {
2834  // Add to list
2835  nonlin_list.push_back(i);
2836  nonlin.at(i) = true;
2837  }
2838  }
2839  }
2840  // Add all combinations to sparsity pattern
2841  for (casadi_int k1 : nonlin_list) {
2842  for (casadi_int k2 : nonlin_list) {
2843  row.push_back(k1);
2844  col.push_back(k2);
2845  }
2846  }
2847  // If row/col vectors grow too large, remove duplicates
2848  if (col.size() > 2 * iind.size() * iind.size()) {
2849  Sparsity r = Sparsity::triplet(iind.size(), iind.size(), row, col);
2850  row = r.get_row();
2851  col = r.get_col();
2852  }
2853  // Reset nonlin, nonlin_list for next iteration
2854  for (casadi_int k : nonlin_list) nonlin[k] = false;
2855  nonlin_list.clear();
2856  }
2857  // Create sparsity pattern
2858  return Sparsity::triplet(iind.size(), iind.size(), row, col);
2859 }
