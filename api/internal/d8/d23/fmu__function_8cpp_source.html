50 int FmuFunction::init_mem(void* mem) const {
51  casadi_assert(mem != 0, "Memory is null");
52  // Instantiate base classes
53  if (FunctionInternal::init_mem(mem)) return 1;
54  // Number of memory instances needed
55  casadi_int n_mem = std::max(static_cast<casadi_int>(1),
56  std::max(max_jac_tasks_, max_hess_tasks_));
57  // Initialize master and all slaves
58  FmuMemory* m = static_cast<FmuMemory*>(mem);
59  for (casadi_int i = 0; i < n_mem; ++i) {
60  // Initialize the memory object itself or a slave
61  FmuMemory* m1 = i == 0 ? m : m->slaves.at(i - 1);
62  if (fmu_.init_mem(m1)) return 1;
63  }
64  return 0;
65 }

67 void* FmuFunction::alloc_mem() const {
68  // Create (master) memory object
69  FmuMemory* m = new FmuMemory(*this);
70  // Attach additional (slave) memory objects
71  for (casadi_int i = 1; i < max_jac_tasks_; ++i) {
72  m->slaves.push_back(new FmuMemory(*this));
73  }
74  return m;

77 void FmuFunction::free_mem(void *mem) const {
78  // Consistency check
79  casadi_assert(mem != nullptr, "Memory is null");
80  FmuMemory* m = static_cast<FmuMemory*>(mem);
81  // Free slave memory
82  for (FmuMemory*& s : m->slaves) {
83  if (!s) continue;
84  // Free FMU memory
85  if (s->instance) {
86  fmu_.free_instance(s->instance);
87  s->instance = nullptr;
88  }
89  // Free the slave
90  delete s;
91  }
92  // Free FMI memory
93  if (m->instance) {
94  fmu_.free_instance(m->instance);
95  m->instance = nullptr;
96  }
97  // Free the memory object
98  delete m;
99 }

101 FmuFunction::FmuFunction(const std::string& name, const Fmu& fmu,
102  const std::vector<std::string>& name_in,
103  const std::vector<std::string>& name_out)
104  : FunctionInternal(name), fmu_(fmu) {
105  // Parse input IDs
106  in_.resize(name_in.size());
107  for (size_t k = 0; k < name_in.size(); ++k) {
108  try {
109  in_[k] = InputStruct::parse(name_in[k], &fmu);
110  } catch (std::exception& e) {
111  casadi_error("Cannot process input " + name_in[k] + ": " + std::string(e.what()));
112  }
113  }
114  // Parse output IDs
115  out_.resize(name_out.size());
116  for (size_t k = 0; k < name_out.size(); ++k) {
117  try {
118  out_[k] = OutputStruct::parse(name_out[k], &fmu);
119  } catch (std::exception& e) {
120  casadi_error("Cannot process output " + name_out[k] + ": " + std::string(e.what()));
121  }
122  }
123  // Set input/output names
124  name_in_ = name_in;
125  name_out_ = name_out;
126  // Default options
127  enable_ad_ = fmu.has_ad();
128  validate_ad_ = false;
129  validate_ad_file_ = "";
130  make_symmetric_ = true;
131  check_hessian_ = false;
132  enable_fd_op_ = fmu.has_ad() && !all_regular(); // Use FD for second and higher order derivatives
133  step_ = 1e-6;
134  abstol_ = 1e-3;
135  reltol_ = 1e-3;
136  print_progress_ = false;
137  new_jacobian_ = true;
138  new_forward_ = false;
139  new_hessian_ = true;
140  hessian_coloring_ = true;
141  parallelization_ = Parallelization::SERIAL;
142  // Number of parallel tasks, by default
143  max_n_tasks_ = 1;
144  max_jac_tasks_ = max_hess_tasks_ = 0;
145 }

219 void FmuFunction::init(const Dict& opts) {
220  // Read options
221  for (auto&& op : opts) {
222  if (op.first=="enable_ad") {
223  enable_ad_ = op.second;
224  } else if (op.first=="validate_ad") {
225  validate_ad_ = op.second;
226  } else if (op.first=="validate_ad_file") {
227  validate_ad_file_ = op.second.to_string();
228  } else if (op.first=="check_hessian") {
229  check_hessian_ = op.second;
230  } else if (op.first=="make_symmetric") {
231  make_symmetric_ = op.second;
232  } else if (op.first=="step") {
233  step_ = op.second;
234  } else if (op.first=="abstol") {
235  abstol_ = op.second;
236  } else if (op.first=="reltol") {
237  reltol_ = op.second;
238  } else if (op.first=="parallelization") {
239  parallelization_ = to_enum<Parallelization>(op.second, "serial");
240  } else if (op.first=="print_progress") {
241  print_progress_ = op.second;
242  } else if (op.first=="new_jacobian") {
243  new_jacobian_ = op.second;
244  } else if (op.first=="new_hessian") {
245  new_hessian_ = op.second;
246  } else if (op.first=="hessian_coloring") {
247  hessian_coloring_ = op.second;
248  }
249  }
250 
251  // Call the initialization method of the base class
252  FunctionInternal::init(opts);
253 
254  // Read FD mode
255  fd_ = to_enum<FdMode>(fd_method_, "forward");
256 
257  // Consistency checks
258  if (enable_ad_) casadi_assert(fmu_.has_ad(),
259  "FMU does not provide support for analytic derivatives");
260  if (validate_ad_ && !enable_ad_) casadi_error("Inconsistent options");
261 
262  // New AD validation file, if any
263  if (!validate_ad_file_.empty()) {
264  std::ofstream valfile;
265  valfile.open(validate_ad_file_);
266  valfile << "Output Input Value Nominal Min Max AD FD Step Offset Stencil" << std::endl;
267  }
268  // Which inputs and outputs exist
269  has_fwd_ = has_adj_ = has_jac_ = has_hess_ = false;
270  for (auto&& i : out_) {
271  switch (i.type) {
272  case OutputType::JAC:
273  case OutputType::JAC_TRANS:
274  has_jac_ = true;
275  break;
276  case OutputType::FWD:
277  has_fwd_ = true;
278  break;
279  case OutputType::ADJ:
280  has_adj_ = true;
281  break;
282  case OutputType::HESS:
283  has_adj_ = true;
284  has_hess_ = true;
285  default:
286  break;
287  }
288  }
289 
290  // Forward derivatives only supported with analytic derivatives
291  if (has_fwd_ && !enable_ad_)
292  casadi_error("Analytic derivatives needed for forward directional derivatives");
293 
294  // Quick return if no Jacobian calculation
295  if (!has_jac_ && !has_adj_ && !has_hess_) return;
296 
297  // Parallelization
298  switch (parallelization_) {
299  case Parallelization::SERIAL:
300  if (verbose_) casadi_message("Serial evaluation");
301  break;
302 #ifdef WITH_OPENMP
303  case Parallelization::OPENMP:
304  max_n_tasks_ = omp_get_max_threads();
305  if (verbose_) casadi_message("OpenMP using at most " + str(max_n_tasks_) + " threads");
306  break;
307 #endif // WITH_OPENMP
308 #ifdef CASADI_WITH_THREAD
309  case Parallelization::THREAD:
310  max_n_tasks_ = std::thread::hardware_concurrency();
311  if (verbose_) casadi_message("std::thread using at most " + str(max_n_tasks_) + " threads");
312  break;
313 #endif // CASADI_WITH_THREAD
314  default:
315  casadi_warning("Parallelization " + to_string(parallelization_)
316  + " not enabled during compilation. Falling back to serial evaluation");
317  parallelization_ = Parallelization::SERIAL;
318  break;
319  }
320 
321  // Collect all inputs in any Jacobian, Hessian or adjoint block
322  std::vector<size_t> in_jac(fmu_.n_in(), 0);
323  jac_in_.clear();
324  jac_nom_in_.clear();
325  for (auto&& i : out_) {
326  if (i.type == OutputType::JAC || i.type == OutputType::JAC_TRANS
327  || i.type == OutputType::ADJ || i.type == OutputType::HESS) {
328  // Get input indices
329  const std::vector<size_t>& iind = fmu_.ired(i.wrt);
330  // Skip if no entries
331  if (iind.empty()) continue;
332  // Consistency check
333  bool exists = in_jac[iind.front()] > 0;
334  for (size_t j : iind) casadi_assert((in_jac[j] > 0) == exists, "Jacobian not a block");
335  // Add selection
336  if (!exists) {
337  for (size_t j : iind) {
338  jac_in_.push_back(j);
339  jac_nom_in_.push_back(fmu_.nominal_in(j));
340  in_jac[j] = jac_in_.size();
341  }
342  }
343  // Add column interval
344  i.cbegin = in_jac[iind.front()] - 1;
345  i.cend = i.cbegin + iind.size();
346  // Also rows for Hessian blocks
347  if (i.type == OutputType::HESS) {
348  // Get input indices
349  const std::vector<size_t>& iind = fmu_.ired(i.ind);
350  // Skip if no entries
351  if (iind.empty()) continue;
352  // Consistency check
353  bool exists = in_jac[iind.front()] > 0;
354  for (size_t j : iind) casadi_assert((in_jac[j] > 0) == exists, "Hessian not a block");
355  // Add selection
356  if (!exists) {
357  for (size_t j : iind) {
358  jac_in_.push_back(j);
359  jac_nom_in_.push_back(fmu_.nominal_in(j));
360  in_jac[j] = jac_in_.size();
361  }
362  }
363  // Add column interval
364  i.rbegin = in_jac[iind.front()] - 1;
365  i.rend = i.rbegin + iind.size();
366  }
367  }
368  }
369 
370  // Transpose of Jacobian sparsity
371  sp_trans_map_.resize(out_.size(), -1);
372  sp_trans_.clear();
373 
374  // Collect all outputs in any Jacobian or adjoint block
375  in_jac.resize(fmu_.n_out());
376  std::fill(in_jac.begin(), in_jac.end(), 0);
377  jac_out_.clear();
378  for (size_t k = 0; k < out_.size(); ++k) {
379  OutputStruct& i = out_[k];
380  if (i.type == OutputType::JAC || i.type == OutputType::JAC_TRANS) {
381  // Get output indices
382  const std::vector<size_t>& oind = fmu_.ored(i.ind);
383  // Skip if no entries
384  if (oind.empty()) continue;
385  // Consistency check
386  bool exists = in_jac[oind.front()] > 0;
387  for (size_t j : oind) casadi_assert((in_jac[j] > 0) == exists, "Jacobian not a block");
388  // Add selection
389  if (!exists) {
390  for (size_t j : oind) {
391  jac_out_.push_back(j);
392  in_jac[j] = jac_out_.size();
393  }
394  }
395  // Add row interval
396  i.rbegin = in_jac[oind.front()] - 1;
397  i.rend = i.rbegin + oind.size();
398  // Additional memory for transpose
399  if (i.type == OutputType::JAC_TRANS) {
400  // Retrieve the sparsity pattern
401  const Sparsity& sp = sparsity_out(k);
402  // Save transpose of sparsity pattern
403  sp_trans_map_.at(k) = sp_trans_.size();
404  sp_trans_.push_back(sp.T());
405  // Work vectors for casadi_trans
406  alloc_w(sp.nnz());
407  alloc_iw(sp.size2());
408  }
409  }
410  }
411  for (auto&& i : in_) {
412  if (i.type == InputType::ADJ) {
413  // Get output indices
414  const std::vector<size_t>& oind = fmu_.ored(i.ind);
415  // Skip if no entries
416  if (oind.empty()) continue;
417  // Consistency check
418  bool exists = in_jac[oind.front()] > 0;
419  for (size_t j : oind) casadi_assert((in_jac[j] > 0) == exists, "Jacobian not a block");
420  // Add selection
421  if (!exists) {
422  for (size_t j : oind) {
423  jac_out_.push_back(j);
424  in_jac[j] = jac_out_.size();
425  }
426  }
427  }
428  }
429 
430  // Get sparsity pattern for extended Jacobian
431  jac_sp_ = fmu_.jac_sparsity(jac_out_, jac_in_);
432 
433  // Calculate graph coloring
434  jac_colors_ = jac_sp_.uni_coloring();
435  if (verbose_) casadi_message("Jacobian graph coloring: " + str(jac_sp_.size2())
436  + " -> " + str(jac_colors_.size2()) + " directions");
437 
438  // Setup Jacobian memory
439  casadi_jac_setup(&p_, jac_sp_, jac_colors_);
440  p_.nom_in = get_ptr(jac_nom_in_);
441  p_.map_out = get_ptr(jac_out_);
442  p_.map_in = get_ptr(jac_in_);
443 
444  // Do not use more threads than there are colors in the Jacobian
445  max_jac_tasks_ = std::min(max_n_tasks_, jac_colors_.size2());
446 
447  // Work vector for storing extended Jacobian, shared between threads
448  if (has_jac_) {
449  alloc_w(jac_sp_.nnz(), true); // jac_nz
450  }
451 
452  // Work vectors for adjoint derivative calculation, shared between threads
453  if (has_adj_) {
454  alloc_w(fmu_.n_out(), true); // aseed
455  alloc_w(fmu_.n_in(), true); // asens
456  }
457 
458  // If Hessian calculation is needed
459  if (has_hess_) {
460  // Get sparsity pattern for extended Hessian
461  hess_sp_ = fmu_.hess_sparsity(jac_in_, jac_in_);
462  casadi_assert(hess_sp_.size1() == jac_in_.size(), "Inconsistent Hessian dimensions");
463  casadi_assert(hess_sp_.size2() == jac_in_.size(), "Inconsistent Hessian dimensions");
464  const casadi_int *hess_row = hess_sp_.row();
465  casadi_int hess_nnz = hess_sp_.nnz();
466 
467  // Get nonlinearly entering variables
468  std::vector<bool> is_nonlin(jac_in_.size(), false);
469  for (casadi_int k = 0; k < hess_nnz; ++k) is_nonlin[hess_row[k]] = true;
470  nonlin_.clear();
471  for (casadi_int c = 0; c < jac_in_.size(); ++c) {
472  if (is_nonlin[c]) nonlin_.push_back(c);
473  }
474  // Calculate graph coloring
475  if (hessian_coloring_) {
476  // Star coloring
477  hess_colors_ = hess_sp_.star_coloring();
478  if (verbose_) casadi_message("Hessian graph coloring: " + str(nonlin_.size())
479  + " -> " + str(hess_colors_.size2()) + " directions");
480  // Indices of non-nonlinearly entering variables
481  std::vector<casadi_int> zind;
482  for (casadi_int c = 0; c < jac_in_.size(); ++c) {
483  if (!is_nonlin[c]) zind.push_back(c);
484  }
485  // Zero out corresponding rows
486  hess_colors_.erase(zind, range(hess_colors_.size2()));
487  } else {
488  // One color for each nonlinear variable
489  hess_colors_ = Sparsity(jac_in_.size(), nonlin_.size(), range(nonlin_.size() + 1), nonlin_);
490  if (verbose_) casadi_message("Hessian calculation for " + str(nonlin_.size()) + " variables");
491  }
492 
493  // Number of threads to be used for Hessian calculation
494  max_hess_tasks_ = std::min(max_n_tasks_, hess_colors_.size2());
495 
496  // Work vector for storing extended Hessian, shared between threads
497  alloc_w(hess_sp_.nnz(), true); // hess_nz
498 
499  // Work vector for perturbed adjoint sensitivities
500  alloc_w(max_hess_tasks_ * fmu_.n_in(), true); // pert_asens
501 
502  // Work vector for avoiding conflicting assignments for star coloring
503  alloc_iw(max_hess_tasks_ * fmu_.n_in(), true); // star_iw
504 
505  // Work vector for making symmetric or checking symmetry
506  alloc_iw(hess_sp_.size2());
507  }
508 
509  // Total number of threads used for Jacobian/adjoint calculation
510  // Note: Adjoint calculation also used for Hessian
511  max_n_tasks_ = std::min(max_n_tasks_, std::max(max_jac_tasks_, max_hess_tasks_));
512  if (verbose_) casadi_message("Jacobian calculation with " + str(max_n_tasks_) + " threads");
513 
514  // Work vectors for Jacobian/adjoint/Hessian calculation, for each thread
515  casadi_int jac_iw, jac_w;
516  casadi_jac_work(&p_, &jac_iw, &jac_w);
517  alloc_iw(max_n_tasks_ * jac_iw, true);
518  alloc_w(max_n_tasks_ * jac_w, true);
519 }

557 InputStruct InputStruct::parse(const std::string& n, const Fmu* fmu,
558  std::vector<std::string>* name_in, std::vector<std::string>* name_out) {
559  // Return value
560  InputStruct s;
561  // Look for a prefix
562  if (has_prefix(n)) {
563  // Get the prefix
564  std::string pref, rem;
565  pref = pop_prefix(n, &rem);
566  if (pref == "out") {
567  if (has_prefix(rem)) {
568  // Second order function output (unused): Get the prefix
569  pref = pop_prefix(rem, &rem);
570  if (pref == "adj") {
571  s.type = InputType::ADJ_OUT;
572  s.ind = fmu ? fmu->index_in(rem) : -1;
573  if (name_in) name_in->push_back(rem);
574  } else {
575  casadi_error("Cannot process: " + n);
576  }
577  } else {
578  // Nondifferentiated function output (unused)
579  s.type = InputType::OUT;
580  s.ind = fmu ? fmu->index_out(rem) : -1;
581  if (name_out) name_out->push_back(rem);
582  }
583  } else if (pref == "fwd") {
584  // Forward seed
585  s.type = InputType::FWD;
586  s.ind = fmu ? fmu->index_in(rem) : 0;
587  if (name_in) name_in->push_back(rem);
588  } else if (pref == "adj") {
589  // Adjoint seed
590  s.type = InputType::ADJ;
591  s.ind = fmu ? fmu->index_out(rem) : 0;
592  if (name_out) name_out->push_back(rem);
593  } else {
594  // No such prefix
595  casadi_error("No such prefix: " + pref);
596  }
597  } else {
598  // No prefix - regular input
599  s.type = InputType::REG;
600  s.ind = fmu ? fmu->index_in(n) : 0;
601  if (name_in) name_in->push_back(n);
602  }
603  // Return input struct
604  return s;
605 }

607 OutputStruct OutputStruct::parse(const std::string& n, const Fmu* fmu,
608  std::vector<std::string>* name_in, std::vector<std::string>* name_out) {
609  // Return value
610  OutputStruct s;
611  // Look for prefix
612  if (has_prefix(n)) {
613  // Get the prefix
614  std::string pref, rem;
615  pref = pop_prefix(n, &rem);
616  if (pref == "jac") {
617  // Jacobian block
618  casadi_assert(has_prefix(rem), "Two arguments expected for Jacobian block");
619  pref = pop_prefix(rem, &rem);
620  if (pref == "adj") {
621  // Jacobian of adjoint sensitivity
622  casadi_assert(has_prefix(rem), "Two arguments expected for Jacobian block");
623  pref = pop_prefix(rem, &rem);
624  if (has_prefix(rem)) {
625  // Jacobian with respect to a sensitivity seed
626  std::string sens = pref;
627  pref = pop_prefix(rem, &rem);
628  if (pref == "adj") {
629  // Jacobian of adjoint sensitivity w.r.t. adjoint seed -> Transpose of Jacobian
630  s.type = OutputType::JAC_TRANS;
631  s.ind = fmu ? fmu->index_out(rem) : -1;
632  if (name_out) name_out->push_back(rem);
633  s.wrt = fmu ? fmu->index_in(sens) : -1;
634  if (name_in) name_in->push_back(sens);
635  } else if (pref == "out") {
636  // Jacobian w.r.t. to dummy output
637  s.type = OutputType::JAC_ADJ_OUT;
638  s.ind = fmu ? fmu->index_in(sens) : -1;
639  if (name_in) name_in->push_back(sens);
640  s.wrt = fmu ? fmu->index_out(rem) : -1;
641  if (name_in) name_out->push_back(rem);
642  } else {
643  casadi_error("No such prefix: " + pref);
644  }
645  } else {
646  // Hessian output
647  s.type = OutputType::HESS;
648  s.ind = fmu ? fmu->index_in(pref) : -1;
649  if (name_in) name_in->push_back(pref);
650  s.wrt = fmu ? fmu->index_in(rem) : -1;
651  if (name_in) name_in->push_back(rem);
652  }
653  } else {
654  if (has_prefix(rem)) {
655  std::string out = pref;
656  pref = pop_prefix(rem, &rem);
657  if (pref == "adj") {
658  // Jacobian of regular output w.r.t. adjoint sensitivity seed
659  s.type = OutputType::JAC_REG_ADJ;
660  s.ind = fmu ? fmu->index_out(out) : -1;
661  if (name_out) name_out->push_back(out);
662  s.wrt = fmu ? fmu->index_out(rem) : -1;
663  if (name_out) name_out->push_back(rem);
664  } else {
665  casadi_error("No such prefix: " + pref);
666  }
667  } else {
668  // Regular Jacobian
669  s.type = OutputType::JAC;
670  s.ind = fmu ? fmu->index_out(pref) : -1;
671  if (name_out) name_out->push_back(pref);
672  s.wrt = fmu ? fmu->index_in(rem) : -1;
673  if (name_in) name_in->push_back(rem);
674  }
675  }
676  } else if (pref == "fwd") {
677  // Forward sensitivity
678  s.type = OutputType::FWD;
679  s.ind = fmu ? fmu->index_out(rem) : -1;
680  if (name_out) name_out->push_back(rem);
681  } else if (pref == "adj") {
682  // Adjoint sensitivity
683  s.type = OutputType::ADJ;
684  s.wrt = fmu ? fmu->index_in(rem) : -1;
685  if (name_in) name_in->push_back(rem);
686  } else {
687  // No such prefix
688  casadi_error("No such prefix: " + pref);
689  }
690  } else {
691  // No prefix - regular output
692  s.type = OutputType::REG;
693  s.ind = fmu ? fmu->index_out(n) : -1;
694  if (name_out) name_out->push_back(n);
695  }
696  // Return output struct
697  return s;
698 }

784 int FmuFunction::eval(const double** arg, double** res, casadi_int* iw, double* w,
785  void* mem) const {
786  // Get memory struct
787  FmuMemory* m = static_cast<FmuMemory*>(mem);
788  casadi_assert(m != 0, "Memory is null");
789  // What blocks are there?
790  bool need_jac = false, need_fwd = false, need_adj = false, need_hess = false;
791  for (size_t k = 0; k < out_.size(); ++k) {
792  if (res[k]) {
793  switch (out_[k].type) {
794  case OutputType::JAC:
795  case OutputType::JAC_TRANS:
796  need_jac = true;
797  break;
798  case OutputType::FWD:
799  need_fwd = true;
800  break;
801  case OutputType::ADJ:
802  need_adj = true;
803  break;
804  case OutputType::HESS:
805  need_adj = true;
806  need_hess = true;
807  break;
808  default:
809  break;
810  }
811  }
812  }
813  // Work vectors, shared between threads
814  double *aseed = 0, *asens = 0, *jac_nz = 0, *hess_nz = 0;
815  if (need_jac) {
816  // Jacobian nonzeros, initialize to NaN
817  jac_nz = w; w += jac_sp_.nnz();
818  std::fill(jac_nz, jac_nz + jac_sp_.nnz(), casadi::nan);
819  }
820  if (need_adj) {
821  // Set up vectors
822  aseed = w; w += fmu_.n_out();
823  asens = w; w += fmu_.n_in();
824  // Clear seed/sensitivity vectors
825  std::fill(aseed, aseed + fmu_.n_out(), 0);
826  std::fill(asens, asens + fmu_.n_in(), 0);
827  // Copy adjoint seeds to aseed
828  for (size_t i = 0; i < in_.size(); ++i) {
829  if (arg[i] && in_[i].type == InputType::ADJ) {
830  const std::vector<size_t>& oind = fmu_.ored(in_[i].ind);
831  for (size_t k = 0; k < oind.size(); ++k) aseed[oind[k]] = arg[i][k];
832  }
833  }
834  }
835  if (need_hess) {
836  // Hessian nonzeros, initialize to NaN
837  hess_nz = w; w += hess_sp_.nnz();
838  std::fill(hess_nz, hess_nz + hess_sp_.nnz(), casadi::nan);
839  }
840  // Setup memory for threads
841  for (casadi_int task = 0; task < max_n_tasks_; ++task) {
842  FmuMemory* s = task == 0 ? m : m->slaves.at(task - 1);
843  // Shared memory
844  s->arg = arg;
845  s->res = res;
846  s->aseed = aseed;
847  s->asens = asens;
848  s->jac_nz = jac_nz;
849  s->hess_nz = hess_nz;
850  // Thread specific memory
851  casadi_jac_init(&p_, &s->d, &iw, &w);
852  if (task < max_hess_tasks_) {
853  // Perturbed adjoint sensitivities
854  s->pert_asens = w;
855  w += fmu_.n_in();
856  // Work vector for avoiding assignment of illegal nonzeros
857  s->star_iw = iw;
858  iw += fmu_.n_in();
859  }
860  }
861  // Evaluate everything except Hessian, possibly in parallel
862  if (verbose_) casadi_message("Evaluating regular outputs, forward sens, extended Jacobian");
863  if (eval_all(m, max_jac_tasks_, true, need_jac, need_fwd, need_adj, false)) return 1;
864  // Evaluate Hessian
865  if (need_hess) {
866  if (verbose_) casadi_message("Evaluating extended Hessian");
867  if (eval_all(m, max_hess_tasks_, false, false, false, false, true)) return 1;
868  // Post-process Hessian
869  remove_nans(hess_nz, iw);
870  if (check_hessian_) check_hessian(m, hess_nz, iw);
871  if (make_symmetric_) make_symmetric(hess_nz, iw);
872  }
873  // Fetch calculated blocks
874  for (size_t k = 0; k < out_.size(); ++k) {
875  // Get nonzeros, skip if not needed
876  double* r = res[k];
877  if (!r) continue;
878  // Get by type
879  switch (out_[k].type) {
880  case OutputType::JAC:
881  casadi_get_sub(r, jac_sp_, jac_nz,
882  out_[k].rbegin, out_[k].rend, out_[k].cbegin, out_[k].cend);
883  break;
884  case OutputType::JAC_TRANS:
885  casadi_get_sub(w, jac_sp_, jac_nz,
886  out_[k].rbegin, out_[k].rend, out_[k].cbegin, out_[k].cend);
887  casadi_trans(w, sp_trans_[sp_trans_map_[k]], r, sparsity_out(k), iw);
888  break;
889  case OutputType::ADJ:
890  for (size_t id : fmu_.ired(out_[k].wrt)) *r++ = asens[id];
891  break;
892  case OutputType::HESS:
893  casadi_get_sub(r, hess_sp_, hess_nz,
894  out_[k].rbegin, out_[k].rend, out_[k].cbegin, out_[k].cend);
895  break;
896  default:
897  break;
898  }
899  }
900  // Successful return
901  return 0;
902 }

904 int FmuFunction::eval_all(FmuMemory* m, casadi_int n_task,
905  bool need_nondiff, bool need_jac, bool need_fwd, bool need_adj, bool need_hess) const {
906  // Return flag
907  int flag = 0;
908  // Evaluate, serially or in parallel
909  if (parallelization_ == Parallelization::SERIAL || n_task == 1
910  || (!need_jac && !need_adj && !need_hess)) {
911  // Evaluate serially
912  flag = eval_task(m, 0, 1, need_nondiff, need_jac, need_fwd, need_adj, need_hess);
913  } else if (parallelization_ == Parallelization::OPENMP) {
914  #ifdef WITH_OPENMP
915  // Parallel region
916  #pragma omp parallel reduction(||:flag)
917  {
918  // Get thread number
919  casadi_int task = omp_get_thread_num();
920  // Get number of threads in region
921  casadi_int num_threads = omp_get_num_threads();
922  // Number of threads that are actually used
923  casadi_int num_used_threads = std::min(num_threads, n_task);
924  // Evaluate in parallel
925  if (task < num_used_threads) {
926  FmuMemory* s = task == 0 ? m : m->slaves.at(task - 1);
927  flag = eval_task(s, task, num_used_threads, need_nondiff && task == 0,
928  need_jac, need_fwd && task == 0, need_adj, need_hess);
929  } else {
930  // Nothing to do for thread
931  flag = 0;
932  }
933  }
934  #else // WITH_OPENMP
935  flag = 1;
936  #endif // WITH_OPENMP
937  } else if (parallelization_ == Parallelization::THREAD) {
938  #ifdef CASADI_WITH_THREAD
939  // Return value for each thread
940  std::vector<int> flag_task(n_task);
941  // Spawn threads
942  std::vector<std::thread> threads;
943  for (casadi_int task = 0; task < n_task; ++task) {
944  threads.emplace_back(
945  [&, task](int* fl) {
946  FmuMemory* s = task == 0 ? m : m->slaves.at(task - 1);
947  *fl = eval_task(s, task, n_task, need_nondiff && task == 0,
948  need_jac, need_fwd && task == 0, need_adj, need_hess);
949  }, &flag_task[task]);
950  }
951  // Join threads
952  for (auto&& th : threads) th.join();
953  // Join return flags
954  for (int fl : flag_task) flag = flag || fl;
955  #else // CASADI_WITH_THREAD
956  flag = 1;
957  #endif // CASADI_WITH_THREAD
958  } else {
959  casadi_error("Unknown parallelization: " + to_string(parallelization_));
960  }
961  // Return combined error flag
962  return flag;
963 }

965 int FmuFunction::eval_task(FmuMemory* m, casadi_int task, casadi_int n_task,
966  bool need_nondiff, bool need_jac, bool need_fwd, bool need_adj, bool need_hess) const {
967  // Pass all regular inputs
968  for (size_t k = 0; k < in_.size(); ++k) {
969  if (in_[k].type == InputType::REG) {
970  fmu_.set(m, in_[k].ind, m->arg[k]);
971  }
972  }
973  // Request all regular outputs to be evaluated
974  for (size_t k = 0; k < out_.size(); ++k) {
975  if (m->res[k] && out_[k].type == OutputType::REG) {
976  fmu_.request(m, out_[k].ind);
977  }
978  }
979  // Evaluate
980  if (fmu_.eval(m)) return 1;
981  // Get regular outputs (master thread only)
982  if (need_nondiff) {
983  for (size_t k = 0; k < out_.size(); ++k) {
984  if (m->res[k] && out_[k].type == OutputType::REG) {
985  fmu_.get(m, out_[k].ind, m->res[k]);
986  }
987  }
988  }
989  // Forward derivatives
990  if (need_fwd) {
991  // Pass all forward seeds
992  for (size_t k = 0; k < in_.size(); ++k) {
993  if (in_[k].type == InputType::FWD) {
994  fmu_.set_fwd(m, in_[k].ind, m->arg[k]);
995  }
996  }
997  // Request forward sensitivities
998  for (size_t k = 0; k < out_.size(); ++k) {
999  if (m->res[k] && out_[k].type == OutputType::FWD) {
1000  fmu_.request_fwd(m, out_[k].ind);
1001  }
1002  }
1003  // Calculate derivatives
1004  if (fmu_.eval_derivative(m, false)) return 1;
1005  // Collect forward sensitivities
1006  for (size_t k = 0; k < out_.size(); ++k) {
1007  if (m->res[k] && out_[k].type == OutputType::FWD) {
1008  fmu_.get_fwd(m, out_[k].ind, m->res[k]);
1009  }
1010  }
1011  }
1012  // Evalute extended Jacobian
1013  if (need_jac || need_adj) {
1014  // Selection of colors to be evaluated for the thread
1015  casadi_int c_begin = (task * jac_colors_.size2()) / n_task;
1016  casadi_int c_end = ((task + 1) * jac_colors_.size2()) / n_task;
1017  // Loop over colors
1018  for (casadi_int c = c_begin; c < c_end; ++c) {
1019  // Print progress
1020  if (print_progress_) print("Jacobian calculation, thread %d/%d: Seeding variable %d/%d\n",
1021  task + 1, n_task, c - c_begin + 1, c_end - c_begin);
1022  // Get derivative directions
1023  casadi_jac_pre(&p_, &m->d, c);
1024  // Calculate derivatives
1025  fmu_.set_seed(m, m->d.nseed, m->d.iseed, m->d.seed);
1026  fmu_.request_sens(m, m->d.nsens, m->d.isens, m->d.wrt);
1027  if (fmu_.eval_derivative(m, true)) return 1;
1028  fmu_.get_sens(m, m->d.nsens, m->d.isens, m->d.sens);
1029  // Scale derivatives
1030  casadi_jac_scale(&p_, &m->d);
1031  // Collect Jacobian nonzeros
1032  if (need_jac) {
1033  for (casadi_int i = 0; i < m->d.nsens; ++i) {
1034  m->jac_nz[m->d.nzind[i]] = m->d.sens[i];
1035  }
1036  }
1037  // Propagate adjoint sensitivities
1038  if (need_adj) {
1039  for (casadi_int i = 0; i < m->d.nsens; ++i)
1040  m->asens[m->d.wrt[i]] += m->aseed[m->d.isens[i]] * m->d.sens[i];
1041  }
1042  }
1043  }
1044  // Evaluate extended Hessian
1045  if (need_hess) {
1046  // Hessian coloring
1047  casadi_int n_hc = hess_colors_.size2();
1048  const casadi_int *hc_colind = hess_colors_.colind(), *hc_row = hess_colors_.row();
1049  // Hessian sparsity
1050  const casadi_int *hess_colind = hess_sp_.colind(), *hess_row = hess_sp_.row();
1051  // Selection of colors to be evaluated for the thread
1052  casadi_int c_begin = (task * n_hc) / n_task;
1053  casadi_int c_end = ((task + 1) * n_hc) / n_task;
1054  // Unperturbed values, step size
1055  std::vector<double> x, h;
1056  // Loop over colors
1057  for (casadi_int c = c_begin; c < c_end; ++c) {
1058  // Print progress
1059  if (print_progress_) print("Hessian calculation, thread %d/%d: Seeding variable %d/%d\n",
1060  task + 1, n_task, c - c_begin + 1, c_end - c_begin);
1061  // Variables being seeded
1062  casadi_int v_begin = hc_colind[c];
1063  casadi_int v_end = hc_colind[c + 1];
1064  casadi_int nv = v_end - v_begin;
1065  // Loop over variables being seeded for color
1066  x.resize(nv);
1067  h.resize(nv);
1068  for (casadi_int v = 0; v < nv; ++v) {
1069  // Corresponding input in Fmu
1070  casadi_int ind1 = hc_row[v_begin + v];
1071  casadi_int id = jac_in_.at(ind1);
1072  // Get unperturbed value
1073  x[v] = m->ibuf_.at(id);
1074  // Step size
1075  h[v] = m->self.step_ * fmu_.nominal_in(id);
1076  // Make sure a a forward step remains in bounds
1077  if (x[v] + h[v] > fmu_.max_in(id)) {
1078  // Ensure a negative step is possible
1079  if (m->ibuf_.at(id) - h[v] < fmu_.min_in(id)) {
1080  std::stringstream ss;
1081  ss << "Cannot perturb " << fmu_.desc_in(m, id) << " at " << x[v]
1082  << " with step size " << m->self.step_;
1083  casadi_warning(ss.str());
1084  return 1;
1085  }
1086  // Take reverse step instead
1087  h[v] = -h[v];
1088  }
1089  // Perturb the input
1090  m->ibuf_.at(id) += h[v];
1091  m->changed_.at(id) = true;
1092  // Inverse of step size
1093  h[v] = 1. / h[v];
1094  }
1095  // Request all outputs
1096  for (size_t i : jac_out_) {
1097  m->requested_.at(i) = true;
1098  m->wrt_.at(i) = -1;
1099  }
1100  // Calculate perturbed inputs
1101  if (fmu_.eval(m)) return 1;
1102  // Clear perturbed adjoint sensitivities
1103  std::fill(m->pert_asens, m->pert_asens + fmu_.n_in(), 0);
1104  // Loop over colors of the Jacobian
1105  for (casadi_int c1 = 0; c1 < jac_colors_.size2(); ++c1) {
1106  // Get derivative directions
1107  casadi_jac_pre(&p_, &m->d, c1);
1108  // Calculate derivatives
1109  fmu_.set_seed(m, m->d.nseed, m->d.iseed, m->d.seed);
1110  fmu_.request_sens(m, m->d.nsens, m->d.isens, m->d.wrt);
1111  if (fmu_.eval_derivative(m, true)) return 1;
1112  fmu_.get_sens(m, m->d.nsens, m->d.isens, m->d.sens);
1113  // Scale derivatives
1114  casadi_jac_scale(&p_, &m->d);
1115  // Propagate adjoint sensitivities
1116  for (casadi_int i = 0; i < m->d.nsens; ++i)
1117  m->pert_asens[m->d.wrt[i]] += m->aseed[m->d.isens[i]] * m->d.sens[i];
1118  }
1119  // Count how many times each input is calculated
1120  std::fill(m->star_iw, m->star_iw + fmu_.n_in(), 0);
1121  for (casadi_int v = 0; v < nv; ++v) {
1122  casadi_int ind1 = hc_row[v_begin + v];
1123  for (casadi_int k = hess_colind[ind1]; k < hess_colind[ind1 + 1]; ++k) {
1124  casadi_int id2 = jac_in_.at(hess_row[k]);
1125  m->star_iw[id2]++;
1126  }
1127  }
1128  // Loop over variables being seeded for color
