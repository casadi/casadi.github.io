95  int Einstein::sp_reverse(bvec_t** arg, bvec_t** res, casadi_int* iw, bvec_t* w) const {
96  //casadi_int* ind = iw;
97  //casadi_int cumprod;
98 
99  // Main loop
100  for (casadi_int i=0;i<n_iter_;++i) {
101 
102  // Data pointers
103  bvec_t* a = arg[1]+strides_a_[0];
104  bvec_t* b = arg[2]+strides_b_[0];
105  bvec_t* c = res[0]+strides_c_[0];
106 
107  // Construct indices
108  casadi_int sub = i;
109  for (casadi_int j=0;j<iter_dims_.size();++j) {
110  casadi_int ind = sub % iter_dims_[j];
111  sub/= iter_dims_[j];
112  a+= strides_a_[1+j]*ind;
113  b+= strides_b_[1+j]*ind;
114  c+= strides_c_[1+j]*ind;
115  }
116 
117  // Perform the actual multiplication
118  Contraction<bvec_t>(*c, 0, *a);
119  Contraction<bvec_t>(0, *c, *b);
120  }
121  copy_rev(arg[0], res[0], nnz());
122  return 0;
123  }

129  void Einstein::generate(CodeGenerator& g,
130  const std::vector<casadi_int>& arg,
131  const std::vector<casadi_int>& res) const {
132 
133  // Copy first argument if not inplace
134  if (arg[0]!=res[0]) {
135  g << g.copy(g.work(arg[0], nnz()), nnz(), g.work(res[0], nnz()));
136  }
137 
138  // main loop
139  g.local("i", "casadi_int");
140  g << "for (i=0; i<" << n_iter_ << "; ++i) {\n";
141 
142  // Data pointers
143  g.local("cr", "const casadi_real", "*");
144  g.local("cs", "const casadi_real", "*");
145  g.local("rr", "casadi_real", "*");
146  g << "cr = " << g.work(arg[1], dep(1).nnz()) << "+" << strides_a_[0] << ";\n";
147  g << "cs = " << g.work(arg[2], dep(2).nnz()) << "+" << strides_b_[0] << ";\n";
148  g << "rr = " << g.work(res[0], dep(0).nnz()) << "+" << strides_c_[0] << ";\n";
149 
150  // Construct indices
151  for (casadi_int j=0; j<iter_dims_.size(); ++j) {
152  if (j==0) {
153  g.local("k", "casadi_int");
154  g << "k = i;\n";
155  g.local("j", "casadi_int");
156  }
157  g << "j = k % " << iter_dims_[j] << ";\n";
158  if (j+1<iter_dims_.size()) g << "k /= " << iter_dims_[j] << ";\n";
159  if (strides_a_[1+j]) g << "cr += j*" << strides_a_[1+j] << ";\n";
160  if (strides_b_[1+j]) g << "cs += j*" << strides_b_[1+j] << ";\n";
161  if (strides_c_[1+j]) g << "rr += j*" << strides_c_[1+j] << ";\n";
162  }
163 
164  // Perform the actual multiplication
165  g << "*rr += *cr**cs;\n";
166 
167  g << "}\n";
168  }
