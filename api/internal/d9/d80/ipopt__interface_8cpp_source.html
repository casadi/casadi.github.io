137  void IpoptInterface::init(const Dict& opts) {
138  // Call the init method of the base class
139  Nlpsol::init(opts);
140 
141  // Default options
142  pass_nonlinear_variables_ = false;
143 
144  std::string convexify_strategy = "none";
145  double convexify_margin = 1e-7;
146  casadi_int max_iter_eig = 200;
147 
148  clip_inactive_lam_ = false;
149  inactive_lam_strategy_ = "reltol";
150  inactive_lam_value_ = 10;
151 
152  // Read user options
153  for (auto&& op : opts) {
154  if (op.first=="ipopt") {
155  opts_ = op.second;
156  } else if (op.first=="pass_nonlinear_variables") {
157  pass_nonlinear_variables_ = op.second;
158  } else if (op.first=="var_string_md") {
159  var_string_md_ = op.second;
160  } else if (op.first=="var_integer_md") {
161  var_integer_md_ = op.second;
162  } else if (op.first=="var_numeric_md") {
163  var_numeric_md_ = op.second;
164  } else if (op.first=="con_string_md") {
165  con_string_md_ = op.second;
166  } else if (op.first=="con_integer_md") {
167  con_integer_md_ = op.second;
168  } else if (op.first=="con_numeric_md") {
169  con_numeric_md_ = op.second;
170  } else if (op.first=="hess_lag") {
171  Function f = op.second;
172  casadi_assert_dev(f.n_in()==4);
173  casadi_assert_dev(f.n_out()==1);
174  set_function(f, "nlp_hess_l");
175  } else if (op.first=="jac_g") {
176  Function f = op.second;
177  casadi_assert_dev(f.n_in()==2);
178  casadi_assert_dev(f.n_out()==2);
179  set_function(f, "nlp_jac_g");
180  } else if (op.first=="grad_f") {
181  Function f = op.second;
182  casadi_assert_dev(f.n_in()==2);
183  casadi_assert_dev(f.n_out()==2);
184  set_function(f, "nlp_grad_f");
185  } else if (op.first=="convexify_strategy") {
186  convexify_strategy = op.second.to_string();
187  } else if (op.first=="convexify_margin") {
188  convexify_margin = op.second;
189  } else if (op.first=="max_iter_eig") {
190  max_iter_eig = op.second;
191  } else if (op.first=="clip_inactive_lam") {
192  clip_inactive_lam_ = op.second;
193  } else if (op.first=="inactive_lam_strategy") {
194  inactive_lam_strategy_ = op.second.to_string();
195  } else if (op.first=="inactive_lam_value") {
196  inactive_lam_value_ = op.second;
197  }
198  }
199 
200  // Do we need second order derivatives?
201  exact_hessian_ = true;
202  auto hessian_approximation = opts_.find("hessian_approximation");
203  if (hessian_approximation!=opts_.end()) {
204  exact_hessian_ = hessian_approximation->second == "exact";
205  }
206 
207  // Setup NLP functions
208  create_function("nlp_f", {"x", "p"}, {"f"});
209  create_function("nlp_g", {"x", "p"}, {"g"});
210  if (!has_function("nlp_grad_f")) {
211  create_function("nlp_grad_f", {"x", "p"}, {"f", "grad:f:x"});
212  }
213  if (!has_function("nlp_jac_g")) {
214  create_function("nlp_jac_g", {"x", "p"}, {"g", "jac:g:x"});
215  }
216  jacg_sp_ = get_function("nlp_jac_g").sparsity_out(1);
217 
218  convexify_ = false;
219 
220  // Allocate temporary work vectors
221  if (exact_hessian_) {
222  if (!has_function("nlp_hess_l")) {
223  create_function("nlp_hess_l", {"x", "p", "lam:f", "lam:g"},
224  {"triu:hess:gamma:x:x"}, {{"gamma", {"f", "g"}}});
225  }
226  hesslag_sp_ = get_function("nlp_hess_l").sparsity_out(0);
227  casadi_assert(hesslag_sp_.is_triu(), "Hessian must be upper triangular");
228  if (convexify_strategy!="none") {
229  convexify_ = true;
230  Dict opts;
231  opts["strategy"] = convexify_strategy;
232  opts["margin"] = convexify_margin;
233  opts["max_iter_eig"] = max_iter_eig;
234  opts["verbose"] = verbose_;
235  hesslag_sp_ = Convexify::setup(convexify_data_, hesslag_sp_, opts);
236  }
237  } else if (pass_nonlinear_variables_) {
238  nl_ex_ = oracle_.which_depends("x", {"f", "g"}, 2, false);
239  }
240 
241  // Allocate work vectors
242  alloc_w(ng_, true); // gk_
243  alloc_w(nx_, true); // grad_fk_
244  alloc_w(jacg_sp_.nnz(), true); // jac_gk_
245  if (exact_hessian_) {
246  alloc_w(hesslag_sp_.nnz(), true); // hess_lk_
247  }
248  if (convexify_) {
249  alloc_iw(convexify_data_.sz_iw);
250  alloc_w(convexify_data_.sz_w);
251  }
252  }

254  int IpoptInterface::init_mem(void* mem) const {
255  if (Nlpsol::init_mem(mem)) return 1;
256  auto m = static_cast<IpoptMemory*>(mem);
257 
258  // Start an IPOPT application
259  Ipopt::SmartPtr<Ipopt::IpoptApplication> *app = new Ipopt::SmartPtr<Ipopt::IpoptApplication>();
260  m->app = static_cast<void*>(app);
261  *app = new Ipopt::IpoptApplication(false);
262 
263  // Direct output through casadi::uout()
264  StreamJournal* jrnl_raw = new StreamJournal("console", J_ITERSUMMARY);
265  jrnl_raw->SetOutputStream(&casadi::uout());
266  jrnl_raw->SetPrintLevel(J_DBG, J_NONE);
267  SmartPtr<Journal> jrnl = jrnl_raw;
268  (*app)->Jnlst()->AddJournal(jrnl);
269 
270  // Create an Ipopt user class -- need to use Ipopts spart pointer class
271  Ipopt::SmartPtr<Ipopt::TNLP> *userclass = new Ipopt::SmartPtr<Ipopt::TNLP>();
272  m->userclass = static_cast<void*>(userclass);
273  *userclass = new IpoptUserClass(*this, m);
274 
275  if (verbose_) {
276  uout() << "There are " << nx_ << " variables and " << ng_ << " constraints." << std::endl;
277  if (exact_hessian_) uout() << "Using exact Hessian" << std::endl;
278  else uout() << "Using limited memory Hessian approximation" << std::endl;
279  }
280 
281  // Get all options available in (s)IPOPT
282  auto regops = (*app)->RegOptions()->RegisteredOptionsList();
283 
284  Dict options = Options::sanitize(opts_);
285  // Replace resto group with prefixes
286  auto it = options.find("resto");
287  if (it!=options.end()) {
288  Dict resto_options = it->second;
289  options.erase(it);
290  for (auto&& op : resto_options) {
291  options["resto." + op.first] = op.second;
292  }
293  }
294 
295  // Pass all the options to ipopt
296  for (auto&& op : options) {
297 
298  // There might be options with a resto prefix.
299  std::string option_name = op.first;
300  if (startswith(option_name, "resto.")) {
301  option_name = option_name.substr(6);
302  }
303 
304  // Find the option
305  auto regops_it = regops.find(option_name);
306  if (regops_it==regops.end()) {
307  casadi_error("No such IPOPT option: " + op.first);
308  }
309 
310  // Get the type
311  Ipopt::RegisteredOptionType ipopt_type = regops_it->second->Type();
312 
313  // Pass to IPOPT
314  bool ret;
315  switch (ipopt_type) {
316  case Ipopt::OT_Number:
317  ret = (*app)->Options()->SetNumericValue(op.first, op.second.to_double(), false);
318  break;
319  case Ipopt::OT_Integer:
320  ret = (*app)->Options()->SetIntegerValue(op.first, op.second.to_int(), false);
321  break;
322  case Ipopt::OT_String:
323  ret = (*app)->Options()->SetStringValue(op.first, op.second.to_string(), false);
324  break;
325  case Ipopt::OT_Unknown:
326  default:
327  casadi_warning("Cannot handle option \"" + op.first + "\", ignored");
328  continue;
329  }
330  if (!ret) casadi_error("Invalid options were detected by Ipopt.");
331  }
332 
333  // Override IPOPT's default linear solver
334  if (opts_.find("linear_solver") == opts_.end()) {
335  char * default_solver = getenv("IPOPT_DEFAULT_LINEAR_SOLVER");
336  if (default_solver) {
337  bool ret = (*app)->Options()->SetStringValue("linear_solver", default_solver, false);
338  casadi_assert(ret, "Corrupted IPOPT_DEFAULT_LINEAR_SOLVER environmental variable");
339  } else {
340  // Fall back to MUMPS (avoid user issues after SPRAL was added to binaries and
341  // chosen default by Ipopt)
342  bool ret = (*app)->Options()->SetStringValue("linear_solver", "mumps", false);
343  casadi_assert_dev(ret);
344  }
345 
346  }
347 
348  // Intialize the IpoptApplication and process the options
349  Ipopt::ApplicationReturnStatus status = (*app)->Initialize();
350  casadi_assert(status == Solve_Succeeded, "Error during IPOPT initialization");
351 
352  if (convexify_) m->add_stat("convexify");
353  return 0;
354  }

356  void IpoptInterface::set_work(void* mem, const double**& arg, double**& res,
357  casadi_int*& iw, double*& w) const {
358  auto m = static_cast<IpoptMemory*>(mem);
359

362

364  m->gk = w; w += ng_;
365  m->grad_fk = w; w += nx_;

368  m->hess_lk = w; w += hesslag_sp_.nnz();

381  return "Search_Direction_Becomes_Too_Small";
382  case Diverging_Iterates:
383  return "Diverging_Iterates";

385  return "User_Requested_Stop";

387  return "Maximum_Iterations_Exceeded";

389  return "Restoration_Failed";
390  case Error_In_Step_Computation:
391  return "Error_In_Step_Computation";

393  return "Not_Enough_Degrees_Of_Freedom";

395  return "Invalid_Problem_Definition";
396  case Invalid_Option:

399  return "Invalid_Number_Detected";
400  case Unrecoverable_Exception:
401  return "Unrecoverable_Exception";
402  case NonIpopt_Exception_Thrown:

405  return "Insufficient_Memory";

420  int IpoptInterface::solve(void* mem) const {
421  auto m = static_cast<IpoptMemory*>(mem);
422  auto d_nlp = &m->d_nlp;
423 
424  // Reset statistics
425  m->inf_pr.clear();
426  m->inf_du.clear();
427  m->mu.clear();
428  m->d_norm.clear();
429  m->regularization_size.clear();
430  m->alpha_pr.clear();
431  m->alpha_du.clear();
432  m->obj.clear();
433  m->ls_trials.clear();
434 
435  // Reset number of iterations
436  m->n_iter = 0;
437 
438  // Get back the smart pointers
439  Ipopt::SmartPtr<Ipopt::TNLP> *userclass =
440  static_cast<Ipopt::SmartPtr<Ipopt::TNLP>*>(m->userclass);
441  Ipopt::SmartPtr<Ipopt::IpoptApplication> *app =
442  static_cast<Ipopt::SmartPtr<Ipopt::IpoptApplication>*>(m->app);
443 
444  // Ask Ipopt to solve the problem
445  Ipopt::ApplicationReturnStatus status = (*app)->OptimizeTNLP(*userclass);
446  m->return_status = return_status_string(status);
447  m->success = status==Solve_Succeeded || status==Solved_To_Acceptable_Level
448  || status==Feasible_Point_Found;
449  if (status==Maximum_Iterations_Exceeded ||
450  status==Maximum_CpuTime_Exceeded) m->unified_return_status = SOLVER_RET_LIMITED;
451 
452 #if (IPOPT_VERSION_MAJOR > 3) || (IPOPT_VERSION_MAJOR == 3 && IPOPT_VERSION_MINOR >= 14)
453  if (status==Maximum_WallTime_Exceeded) m->unified_return_status = SOLVER_RET_LIMITED;
454 #endif
455 
456  // Save results to outputs
457  casadi_copy(m->gk, ng_, d_nlp->z + nx_);
458 
459  if (clip_inactive_lam_) {
460  // Compute a margin
461  double margin;
462  if (inactive_lam_strategy_=="abstol") {
463  margin = inactive_lam_value_;
464  } else if (inactive_lam_strategy_=="reltol") {
465  double constr_viol_tol;
466  (*app)->Options()->GetNumericValue("constr_viol_tol", constr_viol_tol, "");
467  if (status==Solved_To_Acceptable_Level) {
468  (*app)->Options()->GetNumericValue("acceptable_constr_viol_tol", constr_viol_tol, "");
469  }
470  margin = inactive_lam_value_*constr_viol_tol;
471  } else {
472  casadi_error("inactive_lam_strategy '" + inactive_lam_strategy_ +
473  "' unknown. Use 'abstol' or reltol'.");
474  }
475 
476  for (casadi_int i=0; i<nx_ + ng_; ++i) {
477  // Sufficiently inactive -> make multiplier exactly zero
478  if (d_nlp->lam[i]>0 && d_nlp->ubz[i] - d_nlp->z[i] > margin) d_nlp->lam[i]=0;
479  if (d_nlp->lam[i]<0 && d_nlp->z[i] - d_nlp->lbz[i] > margin) d_nlp->lam[i]=0;
480  }
481  }
482 
483  return 0;
484  }

486  bool IpoptInterface::
487  intermediate_callback(IpoptMemory* m, const double* x, const double* z_L, const double* z_U,
488  const double* g, const double* lambda, double obj_value, int iter,
489  double inf_pr, double inf_du, double mu, double d_norm,
490  double regularization_size, double alpha_du, double alpha_pr,
491  int ls_trials, bool full_callback) const {
492  auto d_nlp = &m->d_nlp;
493  m->n_iter += 1;
494  try {
495  m->inf_pr.push_back(inf_pr);
496  m->inf_du.push_back(inf_du);
497  m->mu.push_back(mu);
498  m->d_norm.push_back(d_norm);
499  m->regularization_size.push_back(regularization_size);
500  m->alpha_pr.push_back(alpha_pr);
501  m->alpha_du.push_back(alpha_du);
502  m->ls_trials.push_back(ls_trials);
503  m->obj.push_back(obj_value);
504  if (!fcallback_.is_null()) {
505  ScopedTiming tic(m->fstats.at("callback_fun"));
506  if (full_callback) {
507  casadi_copy(x, nx_, d_nlp->z);
508  for (casadi_int i=0; i<nx_; ++i) {
509  d_nlp->lam[i] = z_U[i]-z_L[i];
510  }
511  casadi_copy(lambda, ng_, d_nlp->lam + nx_);
512  casadi_copy(g, ng_, m->gk);
513  } else {
514  if (iter==0) {
515  uerr()
516  << "Warning: intermediate_callback is disfunctional in your installation. "
517  "You will only be able to use stats(). "
518  "See https://github.com/casadi/casadi/wiki/enableIpoptCallback to enable it."
519  << std::endl;
520  }
521  }
522 
523  // Inputs
524  std::fill_n(m->arg, fcallback_.n_in(), nullptr);
525  if (full_callback) {
526  // The values used below are meaningless
527  // when not doing a full_callback
528  m->arg[NLPSOL_X] = x;
529  m->arg[NLPSOL_F] = &obj_value;
530  m->arg[NLPSOL_G] = g;
531  m->arg[NLPSOL_LAM_P] = nullptr;
532  m->arg[NLPSOL_LAM_X] = d_nlp->lam;
533  m->arg[NLPSOL_LAM_G] = d_nlp->lam + nx_;
534  }
535 
536  // Outputs
537  std::fill_n(m->res, fcallback_.n_out(), nullptr);
538  double ret_double;
539  m->res[0] = &ret_double;
540 
541  fcallback_(m->arg, m->res, m->iw, m->w, 0);
542  casadi_int ret = static_cast<casadi_int>(ret_double);
543 
544  return !ret;
545  } else {
546  return 1;
547  }
548 
549  } catch(KeyboardInterruptException& ex) {
550  return 0;
551  } catch(std::exception& ex) {
552  casadi_warning("intermediate_callback: " + std::string(ex.what()));
553  if (iteration_callback_ignore_errors_) return 1;
554  return 0;
555  }
556  }

558  void IpoptInterface::
559  finalize_solution(IpoptMemory* m, const double* x, const double* z_L, const double* z_U,
560  const double* g, const double* lambda, double obj_value,
561  int iter_count) const {
562  auto d_nlp = &m->d_nlp;
563  try {
564  // Get primal solution
565  casadi_copy(x, nx_, d_nlp->z);
566 
567  // Get optimal cost
568  d_nlp->objective = obj_value;
569 
570  // Get dual solution (simple bounds)
571  for (casadi_int i=0; i<nx_; ++i) {
572  d_nlp->lam[i] = z_U[i]-z_L[i];
573  }
574 
575  // Get dual solution (nonlinear bounds)
576  casadi_copy(lambda, ng_, d_nlp->lam + nx_);
577 
578  // Get the constraints
579  casadi_copy(g, ng_, m->gk);
580 
581  // Get statistics
582  m->iter_count = iter_count;
583 
584  } catch(std::exception& ex) {
585  uerr() << "finalize_solution failed: " << ex.what() << std::endl;
586  }

606  get_starting_point(IpoptMemory* m, bool init_x, double* x,
607  bool init_z, double* z_L, double* z_U,
608  bool init_lambda, double* lambda) const {
609  auto d_nlp = &m->d_nlp;
610  try {
611  // Initialize primal variables
612  if (init_x) {
613  casadi_copy(d_nlp->z, nx_, x);
614  }
615 
616  // Initialize dual variables (simple bounds)
617  if (init_z) {
618  for (casadi_int i=0; i<nx_; ++i) {
619  z_L[i] = std::max(0., -d_nlp->lam[i]);
620  z_U[i] = std::max(0., d_nlp->lam[i]);
621  }
622  }
623 
624  // Initialize dual variables (nonlinear bounds)
625  if (init_lambda) {
626  casadi_copy(d_nlp->lam + nx_, ng_, lambda);
627  }
628 
629  return true;
630  } catch(std::exception& ex) {
631  uerr() << "get_starting_point failed: " << ex.what() << std::endl;
632  return false;
