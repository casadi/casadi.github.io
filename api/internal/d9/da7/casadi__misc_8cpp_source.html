90  bool is_range(const std::vector<casadi_int>& v,
91  casadi_int start, casadi_int stop, casadi_int step) {
92  casadi_int nret = (stop-start)/step + ((stop-start)%step!=0);
93  if (v.size()!=nret) return false;
94  casadi_int ind = start;
95  for (casadi_int e : v) {
96  if (e!=ind) return false;
97  ind += step;
98  }
99  return true;
100  }

104  casadi_int step, casadi_int len) {

106  stop = std::min(stop, len);

109  casadi_int ind = start;
110  for (std::vector<casadi_int>::iterator it=ret.begin(); it!=ret.end(); ++it) {
111  *it = ind;
112  ind += step;
113  }
114  return ret;

117  bool is_equally_spaced(const std::vector<double>& v) {
118  // Quick return if 2 or less entries
119  if (v.size()<=2) return true;
120  // Permitted error margin
121  // NOTE(@jaeandersson) 1e-14 good idea?
122  double margin = (v.back()-v.front())*1e-14;
123  // Make sure spacing is consistent throughout
124  double spacing = v[1]-v[0];
125  for (size_t i=2; i<v.size(); ++i) {
126  if (fabs(v[i]-v[i-1]-spacing)>margin) return false;
127  }
128  // Equal if reached this point
129  return true;
130  }

186  // Better have a bool return flag saying if we need reorer at all
187  std::vector<casadi_int> tensor_permute_mapping(const std::vector<casadi_int>& dims,
188  const std::vector<casadi_int>& order) {
189 
190  // Get problem dimensions
191  casadi_int N = casadi::product(dims);
192  casadi_int n = dims.size();
193  // Quick return if no elements
194  if (N==0) return std::vector<casadi_int>();
195 
196  // One dimension => null-permutation
197  if (n==1) return range(N);
198 
199  // Allocate space for resulting mapping
200  std::vector<casadi_int> mapping(N);
201  // Quick return if scalar
202  if (n==0) return mapping;
203 
204 
205  // Compute cumulative product
206  std::vector<casadi_int> cumprod(n+1, 1);
207  for (casadi_int k=1;k<dims.size();++k) cumprod[k]=cumprod[k-1]*dims[k-1];
208 
209  // Elementary stride
210  casadi_int stride = cumprod[order[0]];
211 
212  // Split problem in inner and outer part
213  casadi_int N_inner = dims[order[0]];
214  casadi_int N_outer = N/N_inner;
215 
216  // Reorder dims, cumprod
217  std::vector<casadi_int> new_dims(n-1), new_cumprod(n-1, 1);
218  for (casadi_int k=0;k<n-1;++k) {
219  new_dims[k] = dims[order[k+1]];
220  new_cumprod[k] = cumprod[order[k+1]];
221  }
222 
223  // Bank of counters
224  std::vector<casadi_int> index_counters(n-1);
225 
226  // Inex into mapping
227  casadi_int m_ind = 0;
228 
229  for (casadi_int i=0;i<N_outer;++i) {
230  // Compute index
231  casadi_int ind = 0;
232  for (casadi_int k=0;k<n-1;++k) ind+=index_counters[k]*new_cumprod[k];
233 
234  // Fill in mapping
235  for (casadi_int j=0;j<N_inner;++j) {
236  mapping.at(m_ind++) = ind;
237  ind+=stride;
238  }
239 
240  // Bump first counter
241  index_counters[0]++;
242 
243  // Overflow counters when needed
244  for (casadi_int k=0;k<n-2;++k) {
245  if (index_counters[k]==new_dims[k]) {
246  index_counters[k] = 0;
247  index_counters[k+1]++;
248  }
249  }
250  }
251  return mapping;
252  }

299 #ifdef HAVE_SIMPLE_MKSTEMPS
300 int simple_mkstemps_fd(const std::string& prefix, const std::string& suffix, std::string &result) {
301  // Characters available for inventing filenames
302  std::string chars = "abcdefghijklmnopqrstuvwxyz0123456789";
303  int char_size = static_cast<int>(chars.size());
304 
305  // How many tries do we allow?
306  casadi_int max_tries = std::numeric_limits<int>::max();
307 
308  // How long should the ID be to cover all tries?
309  double max_tries_d = static_cast<double>(max_tries);
310  double char_size_d = static_cast<double>(char_size);
311  int id_size = lround(ceil(log(max_tries_d)/log(char_size_d)));
312 
313  // Random number generator
314  std::default_random_engine rng(std::chrono::system_clock::now().time_since_epoch().count());
315  std::uniform_int_distribution<> r(0, char_size-1);
316 
317  for (casadi_int i=0;i<max_tries;++i) {
318  result = prefix;
319  for (casadi_int j=0;j<id_size;++j) {
320  result += chars.at(r(rng));
321  }
322  result += suffix;
323 
324 #ifdef _WIN32
325  int fd = _sopen(result.c_str(),
326  _O_BINARY | _O_CREAT | _O_EXCL | _O_RDWR, _SH_DENYNO, _S_IREAD | _S_IWRITE);
327  // Could add _O_TEMPORARY, but then no possiblity of !cleanup_
328 #else
329  int fd = open(result.c_str(), O_RDWR | O_CREAT | O_EXCL, S_IRUSR | S_IWUSR);
330 #endif
331  if (fd != -1) return fd;
332  if (fd == -1 && errno != EEXIST) return -1;
333  }
334  return 0;
335  }
