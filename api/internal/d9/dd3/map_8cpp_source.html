39  Function Map::create(const std::string& parallelization, const Function& f, casadi_int n) {
40  // Create instance of the right class
41  std::string suffix = str(n) + "_" + f.name();
42  if (parallelization == "serial") {
43  return Function::create(new Map("map" + suffix, f, n), Dict());
44  } else if (parallelization== "openmp") {
45  return Function::create(new OmpMap("ompmap" + suffix, f, n), Dict());
46  } else if (parallelization== "thread") {
47  return Function::create(new ThreadMap("threadmap" + suffix, f, n), Dict());
48  } else {
49  casadi_error("Unknown parallelization: " + parallelization);
50  }
51  }

134  template<typename T>
135  int Map::eval_gen(const T** arg, T** res, casadi_int* iw, T* w, int mem) const {
136  const T** arg1 = arg+n_in_;
137  std::copy_n(arg, n_in_, arg1);
138  T** res1 = res+n_out_;
139  std::copy_n(res, n_out_, res1);
140  for (casadi_int i=0; i<n_; ++i) {
141  if (f_(arg1, res1, iw, w, mem)) return 1;
142  for (casadi_int j=0; j<n_in_; ++j) {
143  if (arg1[j]) arg1[j] += f_.nnz_in(j);
144  }
145  for (casadi_int j=0; j<n_out_; ++j) {
146  if (res1[j]) res1[j] += f_.nnz_out(j);
147  }
148  }
149  return 0;
150  }

182  void Map::codegen_body(CodeGenerator& g) const {
183  g.local("i", "casadi_int");
184  g.local("arg1", "const casadi_real*", "*");
185  g.local("res1", "casadi_real*", "*");
186 
187  // Input buffer
188  g << "arg1 = arg+" << n_in_ << ";\n"
189  << "for (i=0; i<" << n_in_ << "; ++i) arg1[i]=arg[i];\n";
190  // Output buffer
191  g << "res1 = res+" << n_out_ << ";\n"
192  << "for (i=0; i<" << n_out_ << "; ++i) res1[i]=res[i];\n"
193  << "for (i=0; i<" << n_ << "; ++i) {\n";
194  // Evaluate
195  g << "if (" << g(f_, "arg1", "res1", "iw", "w") << ") return 1;\n";
196  // Update input buffers
197  for (casadi_int j=0; j<n_in_; ++j) {
198  if (f_.nnz_in(j))
199  g << "if (arg1[" << j << "]) arg1[" << j << "]+=" << f_.nnz_in(j) << ";\n";
200  }
201  // Update output buffers
202  for (casadi_int j=0; j<n_out_; ++j) {
203  if (f_.nnz_out(j))
204  g << "if (res1[" << j << "]) res1[" << j << "]+=" << f_.nnz_out(j) << ";\n";
205  }
206  g << "}\n";
207  }

209  Function Map
210  ::get_forward(casadi_int nfwd, const std::string& name,
211  const std::vector<std::string>& inames,
212  const std::vector<std::string>& onames,
213  const Dict& opts) const {
214  // Generate map of derivative
215  Function df = f_.forward(nfwd);
216  Function dm = df.map(n_, parallelization());
217 
218  // Input expressions
219  std::vector<MX> arg = dm.mx_in();
220 
221  // Need to reorder sensitivity inputs
222  std::vector<MX> res = arg;
223  std::vector<MX>::iterator it=res.begin()+n_in_+n_out_;
224  std::vector<casadi_int> ind;
225  for (casadi_int i=0; i<n_in_; ++i, ++it) {
226  casadi_int sz = f_.size2_in(i);
227  ind.clear();
228  for (casadi_int k=0; k<n_; ++k) {
229  for (casadi_int d=0; d<nfwd; ++d) {
230  for (casadi_int j=0; j<sz; ++j) {
231  ind.push_back((d*n_ + k)*sz + j);
232  }
233  }
234  }
235  *it = (*it)(Slice(), ind); // NOLINT
236  }
237 
238  // Get output expressions
239  res = dm(res);
240 
241  // Reorder sensitivity outputs
242  it = res.begin();
243  for (casadi_int i=0; i<n_out_; ++i, ++it) {
244  casadi_int sz = f_.size2_out(i);
245  ind.clear();
246  for (casadi_int d=0; d<nfwd; ++d) {
247  for (casadi_int k=0; k<n_; ++k) {
248  for (casadi_int j=0; j<sz; ++j) {
249  ind.push_back((k*nfwd + d)*sz + j);
250  }
251  }
252  }
253  *it = (*it)(Slice(), ind); // NOLINT
254  }
255 
256  Dict options = opts;
257  options["allow_duplicate_io_names"] = true;
258 
259  // Construct return function
260  return Function(name, arg, res, inames, onames, options);
261  }

329  int OmpMap::eval(const double** arg, double** res, casadi_int* iw, double* w, void* mem) const {
330 #ifndef WITH_OPENMP
331  return Map::eval(arg, res, iw, w, mem);
332 #else // WITH_OPENMP
333  size_t sz_arg, sz_res, sz_iw, sz_w;
334  f_.sz_work(sz_arg, sz_res, sz_iw, sz_w);
335 
336  // Error flag
337  casadi_int flag = 0;
338 
339  // Checkout memory objects
340  std::vector< scoped_checkout<Function> > ind; ind.reserve(n_);
341  for (casadi_int i=0; i<n_; ++i) ind.emplace_back(f_);
342 
343  // Evaluate in parallel
344 #pragma omp parallel for reduction(||:flag)
345  for (casadi_int i=0; i<n_; ++i) {
346  // Input buffers
347  const double** arg1 = arg + n_in_ + i*sz_arg;
348  for (casadi_int j=0; j<n_in_; ++j) {
349  arg1[j] = arg[j] ? arg[j] + i*f_.nnz_in(j) : 0;
350  }
351 
352  // Output buffers
353  double** res1 = res + n_out_ + i*sz_res;
354  for (casadi_int j=0; j<n_out_; ++j) {
355  res1[j] = res[j] ? res[j] + i*f_.nnz_out(j) : 0;
356  }
357 
358  // Evaluation
359  try {
360  flag = f_(arg1, res1, iw + i*sz_iw, w + i*sz_w, ind[i]) || flag;
361  } catch (std::exception& e) {
362  flag = 1;
363  casadi_warning("Exception raised: " + std::string(e.what()));
364  } catch (...) {
365  flag = 1;
366  casadi_warning("Uncaught exception.");
367  }
368 
369  }
370 
371  // Return error flag
372  return flag;
373 #endif // WITH_OPENMP
374  }

376  void OmpMap::codegen_body(CodeGenerator& g) const {
377  size_t sz_arg, sz_res, sz_iw, sz_w;
378  f_.sz_work(sz_arg, sz_res, sz_iw, sz_w);
379  g << "casadi_int i;\n"
380  << "const double** arg1;\n"
381  << "double** res1;\n"
382  << "casadi_int flag = 0;\n"
383  << "#pragma omp parallel for private(i,arg1,res1) reduction(||:flag)\n"
384  << "for (i=0; i<" << n_ << "; ++i) {\n"
385  << "arg1 = arg + " << n_in_ << "+i*" << sz_arg << ";\n";
386  for (casadi_int j=0; j<n_in_; ++j) {
387  g << "arg1[" << j << "] = arg[" << j << "] ? "
388  << g.arg(j) << "+i*" << f_.nnz_in(j) << ": 0;\n";
389  }
390  g << "res1 = res + " << n_out_ << "+i*" << sz_res << ";\n";
391  for (casadi_int j=0; j<n_out_; ++j) {
392  g << "res1[" << j << "] = res[" << j << "] ?"
393  << g.res(j) << "+i*" << f_.nnz_out(j) << ": 0;\n";
394  }
395  g << "flag = "
396  << g(f_, "arg1", "res1", "iw+i*" + str(sz_iw), "w+i*" + str(sz_w)) << " || flag;\n"
397  << "}\n"
398  << "if (flag) return 1;\n";
399  }

424  void ThreadsWork(const Function& f, casadi_int i,
425  const double** arg, double** res,
426  casadi_int* iw, double* w,
427  casadi_int ind, int& ret) {
428 
429  // Function dimensions
430  casadi_int n_in = f.n_in();
431  casadi_int n_out = f.n_out();
432 
433  // Function work sizes
434  size_t sz_arg, sz_res, sz_iw, sz_w;
435  f.sz_work(sz_arg, sz_res, sz_iw, sz_w);
436 
437  // Input buffers
438  const double** arg1 = arg + n_in + i*sz_arg;
439  for (casadi_int j=0; j<n_in; ++j) {
440  arg1[j] = arg[j] ? arg[j] + i*f.nnz_in(j) : nullptr;
441  }
442 
443  // Output buffers
444  double** res1 = res + n_out + i*sz_res;
445  for (casadi_int j=0; j<n_out; ++j) {
446  res1[j] = res[j] ? res[j] + i*f.nnz_out(j) : nullptr;
447  }
448 
449  try {
450  ret = f(arg1, res1, iw + i*sz_iw, w + i*sz_w, ind);
451  } catch (std::exception& e) {
452  ret = 1;
453  casadi_warning("Exception raised: " + std::string(e.what()));
454  } catch (...) {
455  ret = 1;
456  casadi_warning("Uncaught exception.");
457  }
458  }

460  int ThreadMap::eval(const double** arg, double** res, casadi_int* iw, double* w,
461  void* mem) const {
462 
463 #ifndef CASADI_WITH_THREAD
464  return Map::eval(arg, res, iw, w, mem);
465 #else // CASADI_WITH_THREAD
466  // Checkout memory objects
467  std::vector< scoped_checkout<Function> > ind; ind.reserve(n_);
468  for (casadi_int i=0; i<n_; ++i) ind.emplace_back(f_);
469 
470  // Allocate space for return values
471  std::vector<int> ret_values(n_);
472 
473  // Spawn threads
474  std::vector<std::thread> threads;
475  for (casadi_int i=0; i<n_; ++i) {
476  // Why the lambda function?
477  // Because it was the first iteration to pass tests on MingGW
478  // using mingw-std-threads.
479  threads.emplace_back(
480  [i](const Function& f, const double** arg, double** res,
481  casadi_int* iw, double* w, casadi_int ind, int& ret) {
482  ThreadsWork(f, i, arg, res, iw, w, ind, ret);
483  },
484  std::ref(f_), arg, res, iw, w, casadi_int(ind[i]), std::ref(ret_values[i]));
485  }
486 
487  // Join threads
488  for (auto && th : threads) th.join();
489 
490  // Anticipate success
491  int ret = 0;
492 
493  // Compute aggregate return value
494  for (int e : ret_values) ret = ret || e;
495 
496  return ret;
497 #endif // CASADI_WITH_THREAD
498  }
