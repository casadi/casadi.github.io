50  template<>
51  SX CASADI_EXPORT SX::_sym(const std::string& name, const Sparsity& sp) {
52  // Create a dense n-by-m matrix
53  std::vector<SXElem> retv;
54 
55  // Check if individial names have been provided
56  if (name[0]=='[') {
57 
58  // Make a copy of the string and modify it as to remove the special characters
59  std::string modname = name;
60  for (std::string::iterator it=modname.begin(); it!=modname.end(); ++it) {
61  switch (*it) {
62  case '(': case ')': case '[': case ']': case '{': case '}': case ',': case ';': *it = ' ';
63  }
64  }
65 
66  std::istringstream iss(modname);
67  std::string varname;
68 
69  // Loop over elements
70  while (!iss.fail()) {
71  // Read the name
72  iss >> varname;
73 
74  // Append to the return vector
75  if (!iss.fail())
76  retv.push_back(SXElem::sym(varname));
77  }
78  } else if (sp.is_scalar(true)) {
79  retv.push_back(SXElem::sym(name));
80  } else {
81  // Scalar
82  std::stringstream ss;
83  for (casadi_int k=0; k<sp.nnz(); ++k) {
84  ss.str("");
85  ss << name << "_" << k;
86  retv.push_back(SXElem::sym(ss.str()));
87  }
88  }
89 
90  // Determine dimensions automatically if empty
91  if (sp.is_scalar(true)) {
92  return SX(retv);
93  } else {
94  return SX(sp, retv, false);
95  }
96  }

100  // First pass: ignore symbolics
101  for (casadi_int i=0; i<nnz(); ++i) {

103  if (x.is_constant()) {

108  for (casadi_int i=0; i<nnz(); ++i) {
109  if (!nonzeros().at(i).is_regular()) return false;
110  }
111  return true;

201  template<>
202  void CASADI_EXPORT SX::expand(const SX& ex2, SX& ww, SX& tt) {
203  casadi_assert_dev(ex2.is_scalar());
204  SXElem ex = ex2.scalar();
205 
206  // Terms, weights and indices of the nodes that are already expanded
207  std::vector<std::vector<SXNode*> > terms;
208  std::vector<std::vector<double> > weights;
209  std::map<SXNode*, casadi_int> indices;
210 
211  // Stack of nodes that are not yet expanded
212  std::stack<SXNode*> to_be_expanded;
213  to_be_expanded.push(ex.get());
214 
215  while (!to_be_expanded.empty()) { // as long as there are nodes to be expanded
216 
217  // Check if the last element on the stack is already expanded
218  if (indices.find(to_be_expanded.top()) != indices.end()) {
219  // Remove from stack
220  to_be_expanded.pop();
221  continue;
222  }
223 
224  // Weights and terms
225  std::vector<double> w; // weights
226  std::vector<SXNode*> f; // terms
227 
228  if (to_be_expanded.top()->is_constant()) { // constant nodes are seen as multiples of one
229  w.push_back(to_be_expanded.top()->to_double());
230  f.push_back(casadi_limits<SXElem>::one.get());
231  } else if (to_be_expanded.top()->is_symbolic()) {
232  // symbolic nodes have weight one and itself as factor
233  w.push_back(1);
234  f.push_back(to_be_expanded.top());
235  } else { // unary or binary node
236 
237  casadi_assert_dev(to_be_expanded.top()->n_dep()); // make sure that the node is binary
238 
239  // Check if addition, subtracton or multiplication
240  SXNode* node = to_be_expanded.top();
241  // If we have a binary node that we can factorize
242  if (node->op() == OP_ADD || node->op() == OP_SUB ||
243  (node->op() == OP_MUL && (node->dep(0)->is_constant() ||
244  node->dep(1)->is_constant()))) {
245  // Make sure that both children are factorized, if not - add to stack
246  if (indices.find(node->dep(0).get()) == indices.end()) {
247  to_be_expanded.push(node->dep(0).get());
248  continue;
249  }
250  if (indices.find(node->dep(1).get()) == indices.end()) {
251  to_be_expanded.push(node->dep(1).get());
252  continue;
253  }
254 
255  // Get indices of children
256  casadi_int ind1 = indices[node->dep(0).get()];
257  casadi_int ind2 = indices[node->dep(1).get()];
258 
259  // If multiplication
260  if (node->op() == OP_MUL) {
261  double fac;
262  // Multiplication where the first factor is a constant
263  if (node->dep(0)->is_constant()) {
264  fac = node->dep(0)->to_double();
265  f = terms[ind2];
266  w = weights[ind2];
267  } else { // Multiplication where the second factor is a constant
268  fac = node->dep(1)->to_double();
269  f = terms[ind1];
270  w = weights[ind1];
271  }
272  for (casadi_int i=0; i<w.size(); ++i) w[i] *= fac;
273 
274  } else { // if addition or subtraction
275  if (node->op() == OP_ADD) { // Addition: join both sums
276  f = terms[ind1]; f.insert(f.end(), terms[ind2].begin(), terms[ind2].end());
277  w = weights[ind1]; w.insert(w.end(), weights[ind2].begin(), weights[ind2].end());
278  } else { // Subtraction: join both sums with negative weights for second term
279  f = terms[ind1]; f.insert(f.end(), terms[ind2].begin(), terms[ind2].end());
280  w = weights[ind1];
281  w.reserve(f.size());
282  for (casadi_int i=0; i<weights[ind2].size(); ++i) w.push_back(-weights[ind2][i]);
283  }
284  // Eliminate multiple elements
285  std::vector<double> w_new; w_new.reserve(w.size()); // weights
286  std::vector<SXNode*> f_new; f_new.reserve(f.size()); // terms
287  std::map<SXNode*, casadi_int> f_ind; // index in f_new
288 
289  for (casadi_int i=0; i<w.size(); i++) {
290  // Try to locate the node
291  auto it = f_ind.find(f[i]);
292  if (it == f_ind.end()) { // if the term wasn't found
293  w_new.push_back(w[i]);
294  f_new.push_back(f[i]);
295  f_ind[f[i]] = f_new.size()-1;
296  } else { // if the term already exists
297  w_new[it->second] += w[i]; // just add the weight
298  }
299  }
300  w = w_new;
301  f = f_new;
302  }
303  } else { // if we have a binary node that we cannot factorize
304  // By default,
305  w.push_back(1);
306  f.push_back(node);
307 
308  }
309  }
310 
311  // Save factorization of the node
312  weights.push_back(w);
313  terms.push_back(f);
314  indices[to_be_expanded.top()] = terms.size()-1;
315 
316  // Remove node from stack
317  to_be_expanded.pop();
318  }
319 
320  // Save expansion to output
321  casadi_int thisind = indices[ex.get()];
322  ww = SX(weights[thisind]);
323 
324  std::vector<SXElem> termsv(terms[thisind].size());
325  for (casadi_int i=0; i<termsv.size(); ++i)
326  termsv[i] = SXElem::create(terms[thisind][i]);
327  tt = SX(termsv);
328  }

331  SX CASADI_EXPORT SX::pw_const(const SX& t, const SX& tval, const SX& val) {
332  // number of intervals
333  casadi_int n = val.numel();
334 
335  casadi_assert(t.is_scalar(), "t must be a scalar");
336  casadi_assert(tval.numel() == n-1, "dimensions do not match");
337 
338  SX ret = val->at(0);
339  for (casadi_int i=0; i<n-1; ++i) {
340  ret += (val(i+1)-val(i)) * (t>=tval(i));
341  }
342 
343  return ret;

346  template<>
347  SX CASADI_EXPORT SX::pw_lin(const SX& t, const SX& tval, const SX& val) {
348  // Number of points
349  casadi_int N = tval.numel();
350  casadi_assert(N>=2, "pw_lin: N>=2");
351  casadi_assert(val.numel() == N, "dimensions do not match");
352 
353  // Gradient for each line segment
354  SX g = SX(1, N-1);
355  for (casadi_int i=0; i<N-1; ++i)
356  g(i) = (val(i+1)- val(i))/(tval(i+1)-tval(i));
357 
358  // Line segments
359  SX lseg = SX(1, N-1);
360  for (casadi_int i=0; i<N-1; ++i)
361  lseg(i) = val(i) + g(i)*(t-tval(i));
362 
363  // Return piecewise linear function
364  return pw_const(t, tval(range(1, N-1)), lseg);
365  }

367  template<>
368  SX CASADI_EXPORT SX::gauss_quadrature(const SX& f, const SX& x, const SX& a,
369  const SX& b, casadi_int order, const SX& w) {
370  casadi_assert(order == 5, "gauss_quadrature: order must be 5");
371  casadi_assert(w.is_empty(), "gauss_quadrature: empty weights");
372 
373  // Change variables to [-1, 1]
374  if (!is_equal(a.scalar(), -1) || !is_equal(b.scalar(), 1)) {
375  SX q1 = (b-a)/2;
376  SX q2 = (b+a)/2;
377 
378  Function fcn("gauss_quadrature", {x}, {f});
379 
380  return q1*gauss_quadrature(fcn(q1*x+q2).at(0), x, -1, 1);
381  }
382 
383  // Gauss points
384  std::vector<double> xi;
385  xi.push_back(-std::sqrt(5 + 2*std::sqrt(10.0/7))/3);
386  xi.push_back(-std::sqrt(5 - 2*std::sqrt(10.0/7))/3);
387  xi.push_back(0);
388  xi.push_back(std::sqrt(5 - 2*std::sqrt(10.0/7))/3);
389  xi.push_back(std::sqrt(5 + 2*std::sqrt(10.0/7))/3);
390 
391  // Gauss weights
392  std::vector<double> wi;
393  wi.push_back((322-13*std::sqrt(70.0))/900.0);
394  wi.push_back((322+13*std::sqrt(70.0))/900.0);
395  wi.push_back(128/225.0);
396  wi.push_back((322+13*std::sqrt(70.0))/900.0);
397  wi.push_back((322-13*std::sqrt(70.0))/900.0);
398 
399  // Evaluate at the Gauss points
400  Function fcn("gauss_quadrature", {x}, {f});
401  std::vector<SXElem> f_val(5);
402  for (casadi_int i=0; i<5; ++i)
403  f_val[i] = fcn(SX(xi[i])).at(0).scalar();
404 
405  // Weighted sum
406  SXElem sum;
407  for (casadi_int i=0; i<5; ++i)
408  sum += wi[i]*f_val[i];
409 
410  return sum;
411  }

413  template<>
414  SX CASADI_EXPORT SX::simplify(const SX& x) {
415  SX r = x;
416  for (casadi_int el=0; el<r.nnz(); ++el) {
417  // Start by expanding the node to a weighted sum
418  SX terms, weights;
419  expand(r.nz(el), weights, terms);
420 
421  // Make a scalar product to get the simplified expression
422  r.nz(el) = mtimes(terms.T(), weights);
423  }
424  return r;

427  template<>
428  std::vector<SX> CASADI_EXPORT
429  SX::substitute(const std::vector<SX>& ex, const std::vector<SX>& v, const std::vector<SX>& vdef) {
430 
431  // Assert consistent dimensions
432  if (v.size()!=vdef.size()) {
433  casadi_warning("subtitute: number of symbols to replace ( " + str(v.size()) + ") "
434  "must match number of expressions (" + str(vdef.size()) + ") "
435  "to replace them with.");
436  }
437 
438  // Quick return if all equal
439  bool all_equal = true;
440  for (casadi_int k=0; k<v.size(); ++k) {
441  if (v[k].size()!=vdef[k].size() || !is_equal(v[k], vdef[k])) {
442  all_equal = false;
443  break;
444  }
445  }
446  if (all_equal) return ex;
447 
448  // Check sparsities
449  for (casadi_int k=0; k<v.size(); ++k) {
450  if (v[k].sparsity()!=vdef[k].sparsity()) {
451  // Expand vdef to sparsity of v if vdef is scalar
452  if (vdef[k].is_scalar() && vdef[k].nnz()==1) {
453  std::vector<SX> vdef_mod = vdef;
454  vdef_mod[k] = SX(v[k].sparsity(), vdef[k]->at(0), false);
455  return substitute(ex, v, vdef_mod);
456  } else {
457  casadi_error("Sparsities of v and vdef must match. Got v: "
458  + v[k].dim() + " and vdef: " + vdef[k].dim() + ".");
459  }
460  }
461  }
462 
463 
464  // Otherwise, evaluate symbolically
465  Function F("tmp_substitute", v, ex, Dict{{"max_io", 0}, {"allow_free", true}});
466  return F(vdef);
467  }

474  template<>
475  void CASADI_EXPORT SX::substitute_inplace(const std::vector<SX >& v, std::vector<SX >& vdef,
476  std::vector<SX >& ex, bool reverse) {
477  // Assert correctness
478  casadi_assert_dev(v.size()==vdef.size());
479  for (casadi_int i=0; i<v.size(); ++i) {
480  casadi_assert(v[i].is_symbolic(), "the variable is not symbolic");
481  casadi_assert(v[i].sparsity() == vdef[i].sparsity(), "the sparsity patterns of the "
482  "expression and its defining bexpression do not match");
483  }
484 
485  // Quick return if empty or single expression
486  if (v.empty()) return;
487 
488  // Function inputs
489  std::vector<SX> f_in;
490  if (!reverse) f_in.insert(f_in.end(), v.begin(), v.end());
491 
492  // Function outputs
493  std::vector<SX> f_out = vdef;
494  f_out.insert(f_out.end(), ex.begin(), ex.end());
495 
496  // Write the mapping function
497  Function f("tmp_substitute_inplace", f_in, f_out, Dict{{"max_io", 0}, {"allow_free", true}});
498 
499  // Get references to the internal data structures
500  SXFunction *ff = f.get<SXFunction>();
501  const std::vector<ScalarAtomic>& algorithm = ff->algorithm_;
502  std::vector<SXElem> work(f.sz_w());
503 
504  // Iterator to the binary operations
505  std::vector<SXElem>::const_iterator b_it=ff->operations_.begin();
506 
507  // Iterator to stack of constants
508  std::vector<SXElem>::const_iterator c_it = ff->constants_.begin();
509 
510  // Iterator to free variables
511  std::vector<SXElem>::const_iterator p_it = ff->free_vars_.begin();
512 
513  // Evaluate the algorithm
514  for (std::vector<ScalarAtomic>::const_iterator it=algorithm.begin(); it<algorithm.end(); ++it) {
515  switch (it->op) {
516  case OP_INPUT:
517  // reverse is false, substitute out
518  work[it->i0] = vdef.at(it->i1)->at(it->i2);
519  break;
520  case OP_OUTPUT:
521  if (it->i0 < v.size()) {
522  vdef.at(it->i0)->at(it->i2) = work[it->i1];
523  if (reverse) {
524  // Use the new variable henceforth, substitute in
525  work[it->i1] = v.at(it->i0)->at(it->i2);
526  }
527  } else {
528  // Auxiliary output
529  ex.at(it->i0 - v.size())->at(it->i2) = work[it->i1];
530  }
531  break;
532  case OP_CONST: work[it->i0] = *c_it++; break;
533  case OP_PARAMETER: work[it->i0] = *p_it++; break;
534  default:
535  {
536  switch (it->op) {
537  CASADI_MATH_FUN_BUILTIN(work[it->i1], work[it->i2], work[it->i0])
538  }
539 
540  // Avoid creating duplicates
541  const casadi_int depth = 2; // NOTE: a higher depth could possibly give more savings
542  work[it->i0].assignIfDuplicate(*b_it++, depth);
543  }
544  }
545  }
546  }

614  std::vector<SXElem*> res(f.sz_res());
615  for (casadi_int i=0;i<e.size();++i) {
616  res[i] = get_ptr(ret.at(i).nonzeros());
617  }
618 
619  std::unordered_map<std::string, SXElem > cache;
620  IncrementalSerializer s;
621 
622  // Iterator to stack of constants
623  std::vector<SXElem>::const_iterator c_it = ff->constants_.begin();
624 
625  // Iterator to free variables
626  std::vector<SXElem>::const_iterator p_it = ff->free_vars_.begin();
627 
628  // Evaluate algorithm
629  for (auto&& a : ff->algorithm_) {
630  switch (a.op) {
631  case OP_INPUT:
632  w[a.i0] = arg[a.i1]==nullptr ? 0 : arg[a.i1][a.i2];
633  if (arg[a.i1]!=nullptr) cache[s.pack(w[a.i0])] = w[a.i0];
634  break;
635  case OP_OUTPUT:
636  if (res[a.i0]!=nullptr) res[a.i0][a.i2] = w[a.i1];
637  break;
638  case OP_CONST:
639  w[a.i0] = *c_it++;
640  cache[s.pack(w[a.i0])] = w[a.i0];
641  break;
642  case OP_PARAMETER:
643  w[a.i0] = *p_it++;
644  cache[s.pack(w[a.i0])] = w[a.i0];
645  break;
646  default:
647  {
648 
649  // Evaluate the function to a temporary value
650  // (as it might overwrite the children in the work vector)
651  SXElem f;
652  // Missing simplifications like [x+y]->[twice]
653  switch (a.op) {
654  CASADI_MATH_FUN_BUILTIN(w[a.i1], w[a.i2], f)
655  }
656 
657  std::string key = s.pack(f);
658 
659  auto itk = cache.find(key);
660  if (itk==cache.end()) {
661  cache[key] = f;
662  } else {
663  f = itk->second;
664  }
665 
666  // Finally save the function value
667  w[a.i0] = f;
668  }
669  }
670  }
671  return ret;
672  }

764  SX CASADI_EXPORT SX::taylor(const SX& f, const SX& x,
765  const SX& a, casadi_int order) {
766  casadi_assert_dev(x.is_scalar() && a.is_scalar());
767  if (f.nnz()!=f.numel())
768  throw CasadiException("taylor: not implemented for sparse matrices");
769  SX ff = vec(f.T());
770 
771  SX result = substitute(ff, x, a);
772  double nf=1;
773  SX dx = (x-a);
774  SX dxa = (x-a);
775  for (casadi_int i=1; i<=order; i++) {
776  ff = jacobian(ff, x);
777  nf*=static_cast<double>(i);
778  result+=1/nf * substitute(ff, x, a) * dxa;
779  dxa*=dx;
780  }
781  return reshape(result, f.size2(), f.size1()).T();
782  }

784  SX mtaylor_recursive(const SX& ex, const SX& x, const SX& a, casadi_int order,
785  const std::vector<casadi_int>&order_contributions,
786  const SXElem & current_dx=casadi_limits<SXElem>::one,
787  double current_denom=1, casadi_int current_order=1) {
788  SX result = substitute(ex, x, a)*current_dx/current_denom;
789  for (casadi_int i=0;i<x.nnz();i++) {
790  if (order_contributions[i]<=order) {
791  result += mtaylor_recursive(SX::jacobian(ex, x->at(i)),
792  x, a,
793  order-order_contributions[i],
794  order_contributions,
795  current_dx*(x->at(i)-a->at(i)),
796  current_denom*static_cast<double>(current_order),
797  current_order+1);
798  }
799  }
800  return result;

852  template<>
853  void CASADI_EXPORT SX::extract(std::vector<SX>& ex, std::vector<SX>& v_sx,
854  std::vector<SX>& vdef_sx, const Dict& opts) {
855  // Read options
856  std::string v_prefix = "v_", v_suffix = "";
857  bool lift_shared = true, lift_calls = false;
858  casadi_int v_ind = 0;
859  for (auto&& op : opts) {
860  if (op.first == "prefix") {
861  v_prefix = std::string(op.second);
862  } else if (op.first == "suffix") {
863  v_suffix = std::string(op.second);
864  } else if (op.first == "lift_shared") {
865  lift_shared = op.second;
866  } else if (op.first == "lift_calls") {
867  lift_calls = op.second;
868  } else if (op.first == "offset") {
869  v_ind = op.second;
870  } else {
871  casadi_error("No such option: " + std::string(op.first));
872  }
873  }
874  // Partially implemented
875  casadi_assert(lift_shared, "Not implemented");
876  casadi_assert(!lift_calls, "Not implemented");
877  // Sort the expression
878  Function f("tmp_extract", std::vector<SX>(), ex, Dict{{"max_io", 0}, {"allow_free", true}});
879  SXFunction *ff = f.get<SXFunction>();
880  // Get references to the internal data structures
881  const std::vector<ScalarAtomic>& algorithm = ff->algorithm_;
882  std::vector<SXElem> work(f.sz_w());
883  std::vector<SXElem> work2 = work;
884  // Iterator to the binary operations
885  std::vector<SXElem>::const_iterator b_it=ff->operations_.begin();
886  // Iterator to stack of constants
887  std::vector<SXElem>::const_iterator c_it = ff->constants_.begin();
888  // Iterator to free variables
889  std::vector<SXElem>::const_iterator p_it = ff->free_vars_.begin();
890  // Count how many times an expression has been used
891  std::vector<casadi_int> usecount(work.size(), 0);
892  // Evaluate the algorithm
893  std::vector<SXElem> v, vdef;
894  for (std::vector<ScalarAtomic>::const_iterator it=algorithm.begin(); it<algorithm.end(); ++it) {
895  // Increase usage counters
896  switch (it->op) {
897  case OP_CONST:
898  case OP_PARAMETER:
899  break;
900  CASADI_MATH_BINARY_BUILTIN // Binary operation
901  case OP_IF_ELSE_ZERO:
902  if (usecount[it->i2]==0) {
903  usecount[it->i2]=1;
904  } else if (usecount[it->i2]==1) {
905  // Get a suitable name
906  vdef.push_back(work[it->i2]);
907  usecount[it->i2]=-1; // Extracted, do not extract again
908  }
909  // fall-through
910  case OP_OUTPUT:
911  default: // Unary operation, binary operation or output
912  if (usecount[it->i1]==0) {
913  usecount[it->i1]=1;
914  } else if (usecount[it->i1]==1) {
915  vdef.push_back(work[it->i1]);
916  usecount[it->i1]=-1; // Extracted, do not extract again
917  }
918  }
919  // Perform the operation
920  switch (it->op) {
921  case OP_OUTPUT:
922  break;
923  case OP_CONST:
924  case OP_PARAMETER:
925  usecount[it->i0] = -1; // Never extract since it is a primitive type
926  break;
927  default:
928  work[it->i0] = *b_it++;
929  usecount[it->i0] = 0; // Not (yet) extracted
930  break;
931  }
932  }
933  // Create intermediate variables
934  std::stringstream v_name;
935  for (casadi_int i=0; i<vdef.size(); ++i) {
936  v_name.str(std::string());
937  v_name << v_prefix << (v_ind++) << v_suffix;
938  v.push_back(SXElem::sym(v_name.str()));
939  }
940  // Consistency check
941  casadi_assert(vdef.size() < std::numeric_limits<int>::max(), "Integer overflow");
942  // Mark the above expressions
943  for (casadi_int i=0; i<vdef.size(); ++i) {
944  vdef[i].set_temp(static_cast<int>(i)+1);
945  }
946  // Save the marked nodes for later cleanup
947  std::vector<SXElem> marked = vdef;
948  // Reset iterator
949  b_it=ff->operations_.begin();
950  // Evaluate the algorithm
951  for (std::vector<ScalarAtomic>::const_iterator it=algorithm.begin(); it<algorithm.end(); ++it) {
952  switch (it->op) {
953  case OP_OUTPUT: ex.at(it->i0)->at(it->i2) = work[it->i1]; break;
954  case OP_CONST: work2[it->i0] = work[it->i0] = *c_it++; break;
955  case OP_PARAMETER: work2[it->i0] = work[it->i0] = *p_it++; break;
956  default:
957  {
958  switch (it->op) {
959  CASADI_MATH_FUN_BUILTIN(work[it->i1], work[it->i2], work[it->i0])
960  }
961  work2[it->i0] = *b_it++;
962  // Replace with intermediate variables
963  casadi_int ind = work2[it->i0].get_temp()-1;
964  if (ind>=0) {
965  vdef.at(ind) = work[it->i0];
966  work[it->i0] = v.at(ind);
967  }
968  }
969  }
970  }
971  // Unmark the expressions
972  for (std::vector<SXElem>::iterator it=marked.begin(); it!=marked.end(); ++it) {
973  it->set_temp(0);
974  }
975  // Save v, vdef
976  v_sx.resize(v.size());
977  std::copy(v.begin(), v.end(), v_sx.begin());
978  vdef_sx.resize(vdef.size());
979  std::copy(vdef.begin(), vdef.end(), vdef_sx.begin());
980  }

1021  template<>
1022  SX CASADI_EXPORT SX::poly_roots(const SX& p) {
1023  casadi_assert(p.size2()==1,
1024  "poly_root(): supplied parameter must be column vector but got "
1025  + p.dim() + ".");
1026  casadi_assert_dev(p.is_dense());
1027  if (p.size1()==2) { // a*x + b
1028  SX a = p(0);
1029  SX b = p(1);
1030  return -b/a;
1031  } else if (p.size1()==3) { // a*x^2 + b*x + c
1032  SX a = p(0);
1033  SX b = p(1);
1034  SX c = p(2);
1035  SX ds = sqrt(b*b-4*a*c);
1036  SX bm = -b;
1037  SX a2 = 2*a;
1038  SX ret = SX::vertcat({(bm-ds)/a2, (bm+ds)/a2});
1039  return ret;
1040  } else if (p.size1()==4) {
1041  // www.cs.iastate.edu/~cs577/handouts/polyroots.pdf
1042  SX ai = 1/p(0);
1043 
1044  SX p_ = p(1)*ai;
1045  SX q = p(2)*ai;
1046  SX r = p(3)*ai;
1047 
1048  SX pp = p_*p_;
1049 
1050  SX a = q - pp/3;
1051  SX b = r + 2.0/27*pp*p_-p_*q/3;
1052 
1053  SX a3 = a/3;
1054 
1055  SX phi = acos(-b/2/sqrt(-a3*a3*a3));
1056 
1057  SX ret = SX::vertcat({cos(phi/3), cos((phi+2*pi)/3), cos((phi+4*pi)/3)});
1058  ret*= 2*sqrt(-a3);
1059 
1060  ret-= p_/3;
1061  return ret;
1062  } else if (p.size1()==5) {
1063  SX ai = 1/p(0);
1064  SX b = p(1)*ai;
1065  SX c = p(2)*ai;
1066  SX d = p(3)*ai;
1067  SX e = p(4)*ai;
1068 
1069  SX bb= b*b;
1070  SX f = c - (3*bb/8);
1071  SX g = d + (bb*b / 8) - b*c/2;
1072  SX h = e - (3*bb*bb/256) + (bb * c/16) - (b*d/4);
1073  SX poly = SX::vertcat({1, f/2, ((f*f -4*h)/16), -g*g/64});
1074  SX y = poly_roots(poly);
1075 
1076  SX r0 = y(0); // NOLINT(cppcoreguidelines-slicing)
1077  SX r1 = y(2); // NOLINT(cppcoreguidelines-slicing)
1078 
1079  SX p = sqrt(r0); // two non-zero-roots
1080  SX q = sqrt(r1);
1081 
1082  SX r = -g/(8*p*q);
1083 
1084  SX s = b/4;
1085 
1086  SX ret = SX::vertcat({
1087  p + q + r -s,
1088  p - q - r -s,
1089  -p + q - r -s,
1090  -p - q + r -s});
1091  return ret;
1092  } else if (is_equal(p(p.nnz()-1)->at(0), 0)) {
1093  SX ret = SX::vertcat({poly_roots(p(range(p.nnz()-1))), 0});
1094  return ret;
1095  } else {
1096  casadi_error("poly_root(): can only solve cases for first or second order polynomial. "
1097  "Got order " + str(p.size1()-1) + ".");
1098  }
