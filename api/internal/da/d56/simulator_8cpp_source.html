129 Sparsity Simulator::get_sparsity_in(casadi_int i) {
130  // Regular inputs
131  switch (i) {
132  case SIMULATOR_X0: return x();
133  case SIMULATOR_U: return repmat(u(), 1, ng_ - 1);
134  case SIMULATOR_Z0: return z();
135  case SIMULATOR_P: return p();
136  }
137  i -= SIMULATOR_NUM_IN;
138  // Nondifferentiated outputs
139  if (!nondiff_ && nfwd_ > 0) {
140  switch (i) {
141  case SIMULATOR_X: return repmat(Sparsity(x().size()), 1, ng_);
142  case SIMULATOR_Z: return repmat(Sparsity(z().size()), 1, ng_);
143  }
144  i -= SIMULATOR_NUM_OUT;
145  }
146  // Forward seeds
147  if (nfwd_ > 0) {
148  switch (i) {
149  case SIMULATOR_X0: return repmat(x(), 1, nfwd_);
150  case SIMULATOR_U: return repmat(u(), 1, (ng_ - 1) * nfwd_);
151  case SIMULATOR_Z0: return repmat(z(), 1, nfwd_);
152  case SIMULATOR_P: return repmat(p(), 1, nfwd_);
153  }
154  i -= SIMULATOR_NUM_IN;
155  }
156  // Default return
157  return Sparsity();
158 }

218 void Simulator::set_work(void* mem, const double**& arg, double**& res,
219  casadi_int*& iw, double*& w) const {
220  auto m = static_cast<SimulatorMemory*>(mem);
221  // Call the base class method
222  OracleFunction::set_work(m, arg, res, iw, w);
223  // Inputs
224  m->x0 = arg[SIMULATOR_X0];
225  m->u = arg[SIMULATOR_U];
226  m->z0 = arg[SIMULATOR_Z0];
227  m->p = arg[SIMULATOR_P];
228  arg += SIMULATOR_NUM_IN;
229  // Nondifferentiated outputs, if given
230  if (!nondiff_ && nfwd_ > 0) {
231  m->out_x = arg[SIMULATOR_X];
232  m->out_z = arg[SIMULATOR_Z];
233  arg += SIMULATOR_NUM_OUT;
234  } else {
235  m->out_x = m->out_z = 0;
236  }
237  // Forward seeds
238  if (nfwd_ > 0) {
239  m->fwd_x0 = arg[SIMULATOR_X0];
240  m->fwd_u = arg[SIMULATOR_U];
241  m->fwd_z0 = arg[SIMULATOR_Z0];
242  m->fwd_p = arg[SIMULATOR_P];
243  arg += SIMULATOR_NUM_IN;
244  } else {
245  m->fwd_x0 = m->fwd_u = m->fwd_z0 = m->fwd_p = 0;
246  }
247  // Outputs, if requested
248  if (nondiff_) {
249  m->x = res[SIMULATOR_X];
250  m->z = res[SIMULATOR_Z];
251  res += SIMULATOR_NUM_OUT;
252  } else {
253  m->x = m->z = 0;
254  }
255  // Forward sensitivities
256  if (nfwd_ > 0) {
257  m->fwd_x = res[SIMULATOR_X];
258  m->fwd_z = res[SIMULATOR_Z];
259  res += SIMULATOR_NUM_OUT;
260  } else {
261  m->fwd_x = m->fwd_z = 0;
262  }
263  // Current state
264  m->xk = w; w += nx_;
265  m->zk = w; w += nz_;
266  m->fwd_xk = w; w += nfwd_ * nx_;
267  m->fwd_zk = w; w += nfwd_ * nz_;
268 }

270 int Simulator::eval(const double** arg, double** res, casadi_int* iw, double* w, void* mem) const {
271  auto m = static_cast<SimulatorMemory*>(mem);
272  // Setup memory object
273  setup(m, arg, res, iw, w);
274  // Get initial state, algebraic variable guess
275  casadi_copy(m->x0, nx_, m->xk);
276  casadi_copy(m->z0, nz_, m->zk);
277  // Forward seeds w.r.t. initial state
278  casadi_copy(m->fwd_x0, nfwd_ * nx_, m->fwd_xk);
279  casadi_copy(m->fwd_z0, nfwd_ * nz_, m->fwd_zk);
280  // Reset solver, take time to t0, calculate outputs at t0
281  m->t = grid_.front();
282  reset(m);
283  // Get state
284  casadi_copy(m->xk, nx_, m->x);
285  casadi_copy(m->zk, nz_, m->z);
286  // Get state sensitivities
287  for (casadi_int i = 0; i < nfwd_; ++i) {
288  casadi_copy(m->fwd_xk + nx_ * i, nx_, m->fwd_x + nx_ * ng_ * i);
289  casadi_copy(m->fwd_zk + nz_ * i, nz_, m->fwd_z + nz_ * ng_ * i);
290  }
291  // Advance output time
292  if (m->x) m->x += nx_;
293  if (m->z) m->z += nz_;
294  // Next stop time due to step change in input
295  casadi_int k_stop = next_stop(1, m->u);
296  // Integrate forward
297  for (casadi_int k = 1; k < ng_; ++k) {
298  // Update stopping time, if needed
299  if (k > k_stop) k_stop = next_stop(k, m->u);
300  // Integrate forward
301  if (verbose_) casadi_message("Integrating to " + str(grid_[k])
302  + ", stopping time " + str(grid_[k_stop]));
303  advance(m, grid_[k], grid_[k_stop]);
304  casadi_copy(m->xk, nx_, m->x);
305  casadi_copy(m->zk, nz_, m->z);
306  // Collect state sensitivities
307  for (casadi_int i = 0; i < nfwd_; ++i) {
308  casadi_copy(m->fwd_xk + nx_ * i, nx_, m->fwd_x + nx_ * (ng_ * i + k));
309  casadi_copy(m->fwd_zk + nz_ * i, nz_, m->fwd_z + nz_ * (ng_ * i + k));
310  }
311  // Advance output time
312  if (m->x) m->x += nx_;
313  if (m->u) m->u += nu_;
314  if (m->fwd_u) m->fwd_u += nu_;
315  if (m->z) m->z += nz_;
316  }
317  // Print stats
318  if (print_stats_) print_stats(m);
319  return 0;
320 }

336 void Simulator::init(const Dict& opts) {
337  // Read options
338  for (auto&& op : opts) {
339  if (op.first=="print_stats") {
340  print_stats_ = op.second;
341  } else if (op.first=="nondiff") {
342  nondiff_ = op.second;
343  } else if (op.first=="nfwd") {
344  nfwd_ = op.second;
345  }
346  }
347 
348  // Number of grid points
349  ng_ = grid_.size();
350 
351  // Consistency checks
352  casadi_assert(nondiff_ || nfwd_ > 0, "Inconsistent options");
353  casadi_assert(ng_ >= 2, "Need at least two grid points for simulation");
354 
355  // Call the base class method
356  OracleFunction::init(opts);
357 
358  // Oracle can be evaluated directly
359  set_function(oracle_, "dae");
360 
361  // Error if sparse input
362  casadi_assert(x().is_dense(), "Sparse DAE not supported");
363  casadi_assert(u().is_dense(), "Sparse DAE not supported");
364  casadi_assert(z().is_dense(), "Sparse DAE not supported");
365  casadi_assert(p().is_dense(), "Sparse DAE not supported");
366 
367  // Get dimensions (including sensitivity equations)
368  nx_ = x().nnz();
369  nu_ = u().nnz();
370  nz_ = z().nnz();
371  np_ = p().nnz();
372 
373  // Sensitivities not implemented
374  if (nfwd_ > 0) casadi_warning("Forward sensitivities experimental");
375 
376  // Nominal values for states
377  nom_x_ = oracle_.nominal_in(DYN_X);
378  nom_z_ = oracle_.nominal_in(DYN_Z);
379 
380  // Get the sparsities of the forward and reverse DAE
381  sp_jac_dae_ = sp_jac_dae();
382  casadi_assert(!sp_jac_dae_.is_singular(),
383  "Jacobian of the forward problem is structurally rank-deficient. "
384  "sprank(J)=" + str(sprank(sp_jac_dae_)) + "<" + str(nx_+nz_));
385 
386  // Work vectors
387  alloc_w(nx_, true); // xk
388  alloc_w(nz_, true); // zk
389  alloc_w(nx_ * nfwd_, true); // fwd_xk
390  alloc_w(nz_ * nfwd_, true); // fwd_zk
391 }

400 Sparsity Simulator::sp_jac_dae() {
401  // Start with the sparsity pattern of the ODE part
402  Sparsity jac_ode_x = oracle_.jac_sparsity(DYN_ODE, DYN_X);
403 
404  // Add diagonal to get interdependencies
405  jac_ode_x = jac_ode_x + Sparsity::diag(nx_);
406 
407  // Quick return if no algebraic variables
408  if (nz_==0) return jac_ode_x;
409 
410  // Add contribution from algebraic variables and equations
411  Sparsity jac_ode_z = oracle_.jac_sparsity(DYN_ODE, DYN_Z);
412  Sparsity jac_alg_x = oracle_.jac_sparsity(DYN_ALG, DYN_X);
413  Sparsity jac_alg_z = oracle_.jac_sparsity(DYN_ALG, DYN_Z);
414  return blockcat(jac_ode_x, jac_ode_z,
415  jac_alg_x, jac_alg_z);
416 }

422 template<typename XType>
423 Function Simulator::map2oracle(const std::string& name,
424  const std::map<std::string, XType>& d, const Dict& opts) {
425  // Gather symbolic inputs and outputs
426  std::vector<XType> de_in(enum_traits<DynIn>::n_enum), de_out(enum_traits<DynOut>::n_enum);
427  for (auto&& i : d) {
428  if (has_enum<DynIn>(i.first)) {
429  de_in[to_enum<DynIn>(i.first)] = i.second;
430  } else if (has_enum<DynOut>(i.first)) {
431  de_out[to_enum<DynOut>(i.first)] = i.second;
432  } else {
433  casadi_error("No such field: " + i.first);
434  }
435  }
436  // Make sure x and ode exist
437  casadi_assert(!de_in[DYN_X].is_empty(), "Ill-posed ODE - no state");
438  // Number of right-hand-sides
439  casadi_int nrhs = de_in[DYN_X].size2();
440  // Make sure consistent number of right-hand-sides
441  for (bool b : {true, false}) {
442  for (auto&& e : b ? de_in : de_out) {
443  // Skip time
444  if (&e == &de_in[DYN_T]) continue;
445  // Number of rows
446  casadi_int nr = e.size1();
447  // Make sure no change in number of elements
448  casadi_assert(e.numel()==nr*nrhs, "Inconsistent number of rhs");
449  e = reshape(e, nr, nrhs);
450  }
451  }
452  // Consistent sparsity for x
453  casadi_assert(de_in[DYN_X].size()==de_out[DYN_ODE].size(),
454  "Dimension mismatch for 'ode'");
455  de_out[DYN_ODE] = project(de_out[DYN_ODE], de_in[DYN_X].sparsity());
456  // Consistent sparsity for z
457  casadi_assert(de_in[DYN_Z].size()==de_out[DYN_ALG].size(),
458  "Dimension mismatch for 'alg'");
459  de_out[DYN_ALG] = project(de_out[DYN_ALG], de_in[DYN_Z].sparsity());
460  // Construct
461  return Function(name, de_in, de_out, dyn_in(), dyn_out(), opts);
462 }

464 casadi_int Simulator::next_stop(casadi_int k, const double* u) const {
465  // Integrate till the end if no input signals
466  if (nu_ == 0 || u == 0) return ng_ - 1;
467  // Find the next discontinuity, if any
468  for (; k + 1 < ng_; ++k) {
469  // Next control value
470  const double *u_next = u + nu_;
471  // Check if there is any change in input from k to k + 1
472  for (casadi_int i = 0; i < nu_; ++i) {
473  // Step change detected: stop integration at k
474  if (u[i] != u_next[i]) return k;
475  }
476  // Shift u
477  u = u_next;
478  }
479  // No step changes detected
480  return k;
481 }
