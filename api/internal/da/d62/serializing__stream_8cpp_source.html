64  str(serialization_protocol_version) + " was expected.");

198  unsigned char ref = 'a';
199  // Note: outputstreams work neatly with std::hex,
200  // but inputstreams don't
201  out.put(ref + (reinterpret_cast<unsigned char&>(e) % 16));
202  out.put(ref + (reinterpret_cast<unsigned char&>(e) >> 4));
203  }

303  void SerializingStream::pack(std::istream& s) {
304  decorate('B');
305  s.seekg(0, std::ios::end);
306  size_t len = s.tellg();
307  s.seekg(0, std::ios::beg);
308  pack(len);
309  char buffer[1024];
310  for (size_t i=0;i<len;++i) {
311  s.read(buffer, 1024);
312  size_t c = s.gcount();
313  for (size_t j=0;j<c;++j) {
314  pack(buffer[j]);
315  }
316  if (s.rdstate() & std::ifstream::eofbit) break;
317  }
318  }
