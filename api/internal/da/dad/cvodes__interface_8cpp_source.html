79 void CvodesInterface::init(const Dict& opts) {
80  if (verbose_) casadi_message(name_ + "::init");
81 
82  // Initialize the base classes
83  SundialsInterface::init(opts);
84 
85  // Default options
86  std::string linear_multistep_method = "bdf";
87  std::string nonlinear_solver_iteration = "newton";
88  min_step_size_ = 0;
89  always_recalculate_jacobian_ = true;
90 
91  // Read options
92  for (auto&& op : opts) {
93  if (op.first=="linear_multistep_method") {
94  linear_multistep_method = op.second.to_string();
95  } else if (op.first=="min_step_size") {
96  min_step_size_ = op.second;
97  } else if (op.first=="nonlinear_solver_iteration") {
98  nonlinear_solver_iteration = op.second.to_string();
99  } else if (op.first=="always_recalculate_jacobian") {
100  always_recalculate_jacobian_ = op.second;
101  }
102  }
103 
104  // Algebraic variables not supported
105  casadi_assert(nz_==0 && nrz_==0,
106  "CVODES does not support algebraic variables");
107 
108  if (linear_multistep_method=="adams") {
109  lmm_ = CV_ADAMS;
110  } else if (linear_multistep_method=="bdf") {
111  lmm_ = CV_BDF;
112  } else {
113  casadi_error("Unknown linear multistep method: " + linear_multistep_method);
114  }
115 
116  if (nonlinear_solver_iteration=="newton") {
117  iter_ = CV_NEWTON;
118  } else if (nonlinear_solver_iteration=="functional") {
119  iter_ = CV_FUNCTIONAL;
120  } else {
121  casadi_error("Unknown nonlinear solver iteration: " + nonlinear_solver_iteration);
122  }
123 
124  // Misc
125  alloc_w(nx_); // casadi_project
126  alloc_w(nrx_); // casadi_project
127 }

129 int CvodesInterface::init_mem(void* mem) const {
130  if (SundialsInterface::init_mem(mem)) return 1;
131  auto m = to_mem(mem);
132 
133  // Create CVodes memory block
134  m->mem = CVodeCreate(lmm_, iter_);
135  casadi_assert(m->mem!=nullptr, "CVodeCreate: Creation failed");
136 
137  // Set error handler function
138  THROWING(CVodeSetErrHandlerFn, m->mem, ehfun, m);
139 
140  // Set user data
141  THROWING(CVodeSetUserData, m->mem, m);
142 
143  // Initialize CVodes
144  double t0 = 0;
145  THROWING(CVodeInit, m->mem, rhsF, t0, m->xz);
146 
147  // Set tolerances
148  if (scale_abstol_) {
149  THROWING(CVodeSVtolerances, m->mem, reltol_, m->abstolv);
150  } else {
151  THROWING(CVodeSStolerances, m->mem, reltol_, abstol_);
152  }
153 
154  // Maximum number of steps
155  THROWING(CVodeSetMaxNumSteps, m->mem, max_num_steps_);
156 
157  // Initial step size
158  if (step0_!=0) THROWING(CVodeSetInitStep, m->mem, step0_);
159 
160  // Min step size
161  if (min_step_size_!=0) THROWING(CVodeSetMinStep, m->mem, min_step_size_);
162 
163  // Max step size
164  if (max_step_size_!=0) THROWING(CVodeSetMaxStep, m->mem, max_step_size_);
165 
166  // Maximum order of method
167  if (max_order_) THROWING(CVodeSetMaxOrd, m->mem, max_order_);
168 
169  // Coeff. in the nonlinear convergence test
170  if (nonlin_conv_coeff_!=0) THROWING(CVodeSetNonlinConvCoef, m->mem, nonlin_conv_coeff_);
171 
172  // attach a linear solver
173  if (newton_scheme_==SD_DIRECT) {
174  // Direct scheme
175  CVodeMem cv_mem = static_cast<CVodeMem>(m->mem);
176  cv_mem->cv_lmem = m;
177  cv_mem->cv_lsetup = lsetupF;
178  cv_mem->cv_lsolve = lsolveF;
179  cv_mem->cv_setupNonNull = TRUE;
180  } else {
181  // Iterative scheme
182  casadi_int pretype = use_precon_ ? PREC_LEFT : PREC_NONE;
183  switch (newton_scheme_) {
184  case SD_DIRECT: casadi_assert_dev(0);
185  case SD_GMRES: THROWING(CVSpgmr, m->mem, pretype, max_krylov_); break;
186  case SD_BCGSTAB: THROWING(CVSpbcg, m->mem, pretype, max_krylov_); break;
187  case SD_TFQMR: THROWING(CVSptfqmr, m->mem, pretype, max_krylov_); break;
188  }
189  THROWING(CVSpilsSetJacTimesVecFn, m->mem, jtimesF);
190  if (use_precon_) THROWING(CVSpilsSetPreconditioner, m->mem, psetupF, psolveF);
191  }
192 
193  // Quadrature equations
194  if (nq_>0) {
195  // Initialize quadratures in CVodes
196  THROWING(CVodeQuadInit, m->mem, rhsQF, m->q);
197 
198  // Should the quadrature errors be used for step size control?
199  if (quad_err_con_) {
200  THROWING(CVodeSetQuadErrCon, m->mem, true);
201 
202  // Quadrature error tolerances
203  // TODO(Joel): vector absolute tolerances
204  THROWING(CVodeQuadSStolerances, m->mem, reltol_, abstol_);
205  }
206  }
207 
208  // Initialize adjoint sensitivities
209  if (nrx_>0) {
210  casadi_int interpType = interp_ == SD_HERMITE ? CV_HERMITE : CV_POLYNOMIAL;
211  THROWING(CVodeAdjInit, m->mem, steps_per_checkpoint_, interpType);
212  }
213 
214  m->first_callB = true;
215  return 0;
216 }

254 void CvodesInterface::advance(IntegratorMemory* mem,
255  const double* u, double* x, double* z, double* q) const {
256  auto m = to_mem(mem);
257 
258  // Set controls
259  casadi_copy(u, nu_, m->u);
260 
261  // Do not integrate past change in input signals or past the end
262  THROWING(CVodeSetStopTime, m->mem, m->t_stop);
263 
264  // Integrate, unless already at desired time
265  const double ttol = 1e-9;
266  if (fabs(m->t - m->t_next) >= ttol) {
267  // Integrate forward ...
268  double tret = m->t;
269  if (nrx_>0) {
270  // ... with taping
271  THROWING(CVodeF, m->mem, m->t_next, m->xz, &tret, CV_NORMAL, &m->ncheck);
272  } else {
273  // ... without taping
274  THROWING(CVode, m->mem, m->t_next, m->xz, &tret, CV_NORMAL);
275  }
276 
277  // Get quadratures
278  if (nq_ > 0) {
279  THROWING(CVodeGetQuad, m->mem, &tret, m->q);
280  }
281  }
282 
283  // Set function outputs
284  casadi_copy(NV_DATA_S(m->xz), nx_, x);
285  casadi_copy(NV_DATA_S(m->q), nq_, q);
286 
287  // Get stats
288  THROWING(CVodeGetIntegratorStats, m->mem, &m->nsteps, &m->nfevals, &m->nlinsetups,
289  &m->netfails, &m->qlast, &m->qcur, &m->hinused,
290  &m->hlast, &m->hcur, &m->tcur);
291  THROWING(CVodeGetNonlinSolvStats, m->mem, &m->nniters, &m->nncfails);
292 }

294 void CvodesInterface::impulseB(IntegratorMemory* mem,
295  const double* rx, const double* rz, const double* rp) const {
296  auto m = to_mem(mem);
297 
298  // Call method in base class
299  SundialsInterface::impulseB(mem, rx, rz, rp);
300 
301  if (m->first_callB) {
302  // Create backward problem
303  THROWING(CVodeCreateB, m->mem, lmm_, iter_, &m->whichB);
304  THROWING(CVodeInitB, m->mem, m->whichB, rhsB, m->t, m->rxz);
305  THROWING(CVodeSStolerancesB, m->mem, m->whichB, reltol_, abstol_);
306  THROWING(CVodeSetUserDataB, m->mem, m->whichB, m);
307  if (newton_scheme_==SD_DIRECT) {
308  // Direct scheme
309  CVodeMem cv_mem = static_cast<CVodeMem>(m->mem);
310  CVadjMem ca_mem = cv_mem->cv_adj_mem;
311  CVodeBMem cvB_mem = ca_mem->cvB_mem;
312  cvB_mem->cv_lmem = m;
313  cvB_mem->cv_mem->cv_lmem = m;
314  cvB_mem->cv_mem->cv_lsetup = lsetupB;
315  cvB_mem->cv_mem->cv_lsolve = lsolveB;
316  cvB_mem->cv_mem->cv_setupNonNull = TRUE;
317  } else {
318  // Iterative scheme
319  casadi_int pretype = use_precon_ ? PREC_LEFT : PREC_NONE;
320  switch (newton_scheme_) {
321  case SD_DIRECT: casadi_assert_dev(0);
322  case SD_GMRES: THROWING(CVSpgmrB, m->mem, m->whichB, pretype, max_krylov_); break;
323  case SD_BCGSTAB: THROWING(CVSpbcgB, m->mem, m->whichB, pretype, max_krylov_); break;
324  case SD_TFQMR: THROWING(CVSptfqmrB, m->mem, m->whichB, pretype, max_krylov_); break;
325  }
326  THROWING(CVSpilsSetJacTimesVecFnB, m->mem, m->whichB, jtimesB);
327  if (use_precon_) THROWING(CVSpilsSetPreconditionerB, m->mem, m->whichB, psetupB, psolveB);
328  }
329 
330  // Quadratures for the backward problem
331  THROWING(CVodeQuadInitB, m->mem, m->whichB, rhsQB, m->ruq);
332  if (quad_err_con_) {
333  THROWING(CVodeSetQuadErrConB, m->mem, m->whichB, true);
334  THROWING(CVodeQuadSStolerancesB, m->mem, m->whichB, reltol_, abstol_);
335  }
336 
337  // Mark initialized
338  m->first_callB = false;
339  } else {
340  // Save solver stats offsets before reset
341  save_offsets(m);
342 
343  // Reinitialize solver
344  THROWING(CVodeReInitB, m->mem, m->whichB, m->t, m->rxz);
345  THROWING(CVodeQuadReInitB, m->mem, m->whichB, m->ruq);
346  }
347 }

349 void CvodesInterface::retreat(IntegratorMemory* mem, const double* u,
350  double* rx, double* rq, double* uq) const {
351  auto m = to_mem(mem);
352 
353  // Set controls
354  casadi_copy(u, nu_, m->u);
355 
356  // Integrate, unless already at desired time
357  if (m->t_next < m->t) {
358  THROWING(CVodeB, m->mem, m->t_next, CV_NORMAL);
359  double tret;
360  THROWING(CVodeGetB, m->mem, m->whichB, &tret, m->rxz);
361  if (nrq_ > 0 || nuq_ > 0) {
362  THROWING(CVodeGetQuadB, m->mem, m->whichB, &tret, m->ruq);
363  }
364  }
365 
366  // Save outputs
367  casadi_copy(NV_DATA_S(m->rxz), nrx_, rx);
368  casadi_copy(NV_DATA_S(m->ruq), nrq_, rq);
369  casadi_copy(NV_DATA_S(m->ruq) + nrq_, nuq_, uq);
370 
371  // Get stats
372  CVodeMem cv_mem = static_cast<CVodeMem>(m->mem);
373  CVadjMem ca_mem = cv_mem->cv_adj_mem;
374  CVodeBMem cvB_mem = ca_mem->cvB_mem;
375  THROWING(CVodeGetIntegratorStats, cvB_mem->cv_mem, &m->nstepsB,
376  &m->nfevalsB, &m->nlinsetupsB, &m->netfailsB, &m->qlastB,
377  &m->qcurB, &m->hinusedB, &m->hlastB, &m->hcurB, &m->tcurB);
378  THROWING(CVodeGetNonlinSolvStats, cvB_mem->cv_mem, &m->nnitersB, &m->nncfailsB);
379 
380  // Add offset corresponding to counters that were set to zero at reinitializations
381  add_offsets(m);
382 }

485 int CvodesInterface::psolveF(double t, N_Vector x, N_Vector xdot, N_Vector r,
486  N_Vector z, double gamma, double delta, int lr, void *user_data, N_Vector tmp) {
487  try {
488  auto m = to_mem(user_data);
489  auto& s = m->self;
490 
491  // Get right-hand sides in m->v1
492  double* v = NV_DATA_S(r);
493  casadi_copy(v, s.nx_, m->v1);
494 
495  // Solve for undifferentiated right-hand-side, save to output
496  if (s.linsolF_.solve(m->jacF, m->v1, 1, false, m->mem_linsolF)) return 1;
497  v = NV_DATA_S(z); // possibly different from r
498  casadi_copy(m->v1, s.nx1_, v);
499 
500  // Sensitivity equations
501  if (s.nfwd_ > 0) {
502  // Second order correction
503  if (s.second_order_correction_) {
504  // The outputs will double as seeds for jtimesF
505  casadi_clear(v + s.nx1_, s.nx_ - s.nx1_);
506  if (s.calc_jtimesF(m, t, NV_DATA_S(x), nullptr, v, nullptr, m->v2, nullptr)) return 1;
507 
508  // Subtract m->v2 from m->v1, scaled with -gamma
509  casadi_axpy(s.nx_ - s.nx1_, m->gamma, m->v2 + s.nx1_, m->v1 + s.nx1_);
510  }
511 
512  // Solve for sensitivity right-hand-sides
513  if (s.linsolF_.solve(m->jacF, m->v1 + s.nx1_, s.nfwd_, false, m->mem_linsolF)) return 1;
514 
515  // Save to output, reordered
516  casadi_copy(m->v1 + s.nx1_, s.nx_ - s.nx1_, v + s.nx1_);
517  }
518 
519  return 0;
520  } catch(std::exception& e) { // non-recoverable error
521  uerr() << "psolve failed: " << e.what() << std::endl;
522  return -1;
523  }
524 }

526 int CvodesInterface::psolveB(double t, N_Vector x, N_Vector xB, N_Vector xdotB, N_Vector rvecB,
527  N_Vector zvecB, double gammaB, double deltaB, int lr, void *user_data, N_Vector tmpB) {
528  try {
529  auto m = to_mem(user_data);
530  auto& s = m->self;
531 
532  // Get right-hand sides in m->v1
533  double* v = NV_DATA_S(rvecB);
534  casadi_copy(v, s.nrx_, m->v1);
535 
536  // Solve for undifferentiated right-hand-side, save to output
537  if (s.linsolF_.solve(m->jacF, m->v1, s.nadj_, true, m->mem_linsolF)) return 1;
538  v = NV_DATA_S(zvecB); // possibly different from rvecB
539  casadi_copy(m->v1, s.nrx1_ * s.nadj_, v);
540 
541  // Sensitivity equations
542  if (s.nfwd_ > 0) {
543  // Second order correction
544  if (s.second_order_correction_) {
545  // The outputs will double as seeds for calc_daeB
546  casadi_clear(v + s.nrx1_ * s.nadj_, s.nrx_ - s.nrx1_ * s.nadj_);
547  if (s.calc_daeB(m, t, NV_DATA_S(x), nullptr, v, nullptr, nullptr, m->v2, nullptr)) return 1;
548  // Subtract m->v2 from m->v1, scaled with gammaB
549  casadi_axpy(s.nrx_ - s.nrx1_ * s.nadj_, -m->gammaB, m->v2 + s.nrx1_ * s.nadj_,
550  m->v1 + s.nrx1_ * s.nadj_);
551  }
552 
553  // Solve for sensitivity right-hand-sides
554  if (s.linsolF_.solve(m->jacF, m->v1 + s.nx1_, s.nadj_ * s.nfwd_,
555  true, m->mem_linsolF)) return 1;
556 
557  // Save to output, reordered
558  casadi_copy(m->v1 + s.nx1_, s.nx_ - s.nx1_, v + s.nx1_);
559  }
560 
561  return 0;
562  } catch(std::exception& e) { // non-recoverable error
563  uerr() << "psolveB failed: " << e.what() << std::endl;
564  return -1;
565  }

568 int CvodesInterface::psetupF(double t, N_Vector x, N_Vector xdot, booleantype jok,
569  booleantype *jcurPtr, double gamma, void *user_data,
570  N_Vector tmp1, N_Vector tmp2, N_Vector tmp3) {
571  try {
572  auto m = to_mem(user_data);
573  auto& s = m->self;
574  // Store gamma for later
575  m->gamma = gamma;
576 
577  // Sparsity patterns
578  const Sparsity& sp_jac_ode_x = s.get_function("jacF").sparsity_out(0);
579  const Sparsity& sp_jacF = s.linsolF_.sparsity();
580 
581  // Calculate Jacobian, if necessary
582  if (s.always_recalculate_jacobian_ || !jcurPtr || *jcurPtr == 0) {
583  // Re(calculate) Jacobian
584  if (s.calc_jacF(m, t, NV_DATA_S(x), nullptr,
585  m->jac_ode_x, nullptr, nullptr, nullptr)) return 1;
586 
587  // Jacobian is now current
588  if (jcurPtr) *jcurPtr = 1;
589  }
590 
591  // Project to expected sparsity pattern (with diagonal)
592  casadi_project(m->jac_ode_x, sp_jac_ode_x, m->jacF, sp_jacF, m->w);
593 
594  // Scale and shift diagonal
595  const casadi_int *colind = sp_jacF.colind(), *row = sp_jacF.row();
596  for (casadi_int c = 0; c < sp_jacF.size2(); ++c) {
597  for (casadi_int k = colind[c]; k < colind[c + 1]; ++k) {
598  casadi_int r = row[k];
599  // Scale Jacobian
600  m->jacF[k] *= -gamma;
601  // Add contribution to diagonal
602  if (r == c) m->jacF[k] += 1;
603  }
604  }
605 
606  // Prepare the solution of the linear system (e.g. factorize)
607  if (s.linsolF_.nfact(m->jacF, m->mem_linsolF)) return 1;
608 
609  return 0;
610  } catch(std::exception& e) { // non-recoverable error
611  uerr() << "psetup failed: " << e.what() << std::endl;
612  return -1;
613  }
