31  Function MapSum::create(const std::string& name, const std::string& parallelization,
32  const Function& f, casadi_int n,
33  const std::vector<bool>& reduce_in,
34  const std::vector<bool>& reduce_out,
35  const Dict& opts) {
36  if (reduce_out.empty()) return create(name, parallelization, f, n,
37  reduce_in, std::vector<bool>(f.n_out(), false));
38  casadi_assert(reduce_in.size()==f.n_in(), "Dimension mismatch");
39  casadi_assert(reduce_out.size()==f.n_out(), "Dimension mismatch");
40 
41  if (parallelization == "serial") {
42  std::string suffix = str(reduce_in)+str(reduce_out);
43  Function ret;
44  if (!f->incache(name, ret, suffix)) {
45  // Create new serial map
46  ret = Function::create(new MapSum(name, f, n, reduce_in, reduce_out), opts);
47  casadi_assert_dev(ret.name()==name);
48  // Save in cache
49  f->tocache(ret, suffix);
50  }
51  return ret.wrap_as_needed(opts);
52  } else {
53  casadi_error("Unknown parallelization: " + parallelization);
54  }
55  }

147  template<typename T>
148  int MapSum::eval_gen(const T** arg, T** res, casadi_int* iw, T* w, int mem) const {
149  const T** arg1 = arg+n_in_;
150  std::copy_n(arg, n_in_, arg1);
151  T** res1 = res+n_out_;
152 
153  T* w_scratch = w + f_.sz_w();
154  for (casadi_int j=0;j<n_out_;++j) {
155  if (res[j] && reduce_out_[j]) {
156  casadi_clear(res[j], f_.nnz_out(j)); // clear sums
157  res1[j] = w_scratch; // Make the function dump result in scratch space
158  w_scratch += f_.nnz_out(j);
159  } else {
160  res1[j] = res[j];
161  }
162  }
163  for (casadi_int i=0; i<n_; ++i) {
164  if (f_(arg1, res1, iw, w, mem)) return 1;
165  for (casadi_int j=0; j<n_in_; ++j) {
166  if (arg1[j] && !reduce_in_[j]) arg1[j] += f_.nnz_in(j);
167  }
168  for (casadi_int j=0; j<n_out_; ++j) {
169  if (res1[j]) {
170  if (reduce_out_[j]) {
171  casadi_add(f_.nnz_out(j), res1[j], res[j]); // Perform sum
172  } else {
173  res1[j] += f_.nnz_out(j);
174  }
175  }
176  }
177  }
178  return 0;
179  }

191  int MapSum::sp_reverse(bvec_t** arg, bvec_t** res, casadi_int* iw, bvec_t* w, void* mem) const {
192  // Note: f_.rev(arg,res,iw,w)
193  // has a side effect of clearing res
194  // Reduced outputs should not be cleared;
195  // they must influence each iteration
196 
197  // Store reduced res in scratch space
198  bvec_t* w_scratch = w + f_.sz_w();
199  for (casadi_int j=0;j<n_out_;++j) {
200  if (res[j] && reduce_out_[j]) {
201  casadi_copy(res[j], f_.nnz_out(j), w_scratch);
202  w_scratch += f_.nnz_out(j);
203  }
204  }
205  bvec_t** arg1 = arg+n_in_;
206  std::copy_n(arg, n_in_, arg1);
207  bvec_t** res1 = res+n_out_;
208  std::copy_n(res, n_out_, res1);
209  for (casadi_int i=0; i<n_; ++i) {
210  // Restore res1[j] from scratch space
211  w_scratch = w + f_.sz_w();
212  for (casadi_int j=0;j<n_out_;++j) {
213  if (res[j] && reduce_out_[j]) {
214  casadi_copy(w_scratch, f_.nnz_out(j), res1[j]);
215  w_scratch += f_.nnz_out(j);
216  }
217  }
218  if (f_.rev(arg1, res1, iw, w)) return 1;
219  for (casadi_int j=0; j<n_in_; ++j) {
220  if (arg1[j] && !reduce_in_[j]) arg1[j] += f_.nnz_in(j);
221  }
222  for (casadi_int j=0; j<n_out_; ++j) {
223  if (res1[j] && !reduce_out_[j]) res1[j] += f_.nnz_out(j);
224  }
225  }
226  return 0;
227  }

233  void MapSum::codegen_body(CodeGenerator& g) const {
234  g.add_auxiliary(CodeGenerator::AUX_CLEAR);
235  g.local("i", "casadi_int");
236  g.local("arg1", "const casadi_real*", "*");
237  g.local("res1", "casadi_real*", "*");
238  g.local("w_scratch", "casadi_real*", "*");
239  // Input buffer
240  g << "arg1 = arg+" << n_in_ << ";\n"
241  << "for (i=0; i<" << n_in_ << "; ++i) arg1[i]=arg[i];\n";
242  // Output buffer
243  g << "res1 = res+" << n_out_ << ";\n";
244  g << "w_scratch = w+" << f_.sz_w() << ";\n";
245  for (casadi_int j=0;j<n_out_;++j) {
246  if (reduce_out_[j]) {
247  g << "if (res[" << j << "]) {\n";
248  g << "casadi_clear(res[" << j << "], " << f_.nnz_out(j) << ");\n";
249  g << "res1[" << j << "] = w_scratch;\n";
250  g << "w_scratch+=" << f_.nnz_out(j) << ";\n";
251  g << "} else {\n";
252  g << "res1[" << j << "] = res[" << j << "];\n";
253  g << "}\n";
254  } else {
255  g << "res1[" << j << "] = res[" << j << "];\n";
256  }
257  }
258 
259  g << "for (i=0; i<" << n_ << "; ++i) {\n";
260  // Evaluate
261  g << "if (" << g(f_, "arg1", "res1", "iw", "w") << ") return 1;\n";
262  // Update input buffers
263  for (casadi_int j=0; j<n_in_; ++j) {
264  if (!reduce_in_[j] && f_.nnz_in(j)) {
265  g << "if (arg1[" << j << "]) arg1[" << j << "]+=" << f_.nnz_in(j) << ";\n";
266  }
267  }
268  // Update output buffers
269  for (casadi_int j=0; j<n_out_; ++j) {
270  if (reduce_out_[j]) {
271  g << "if (res1[" << j << "]) ";
272  g << g.axpy(f_.nnz_out(j), "1.0", "res1[" + str(j) + "]", "res[" + str(j) + "]") << "\n";
273  } else {
274  if (f_.nnz_out(j)) {
275  g << "if (res1[" << j << "]) ";
276  g << "res1[" << j << "]+=" << f_.nnz_out(j) << ";\n";
277  }
278  }
279  }
280  g << "}\n";
281  }

283  Function MapSum
284  ::get_forward(casadi_int nfwd, const std::string& name,
285  const std::vector<std::string>& inames,
286  const std::vector<std::string>& onames,
287  const Dict& opts) const {
288  // Generate map of derivative
289  Function df = f_.forward(nfwd);
290 
291  for (casadi_int i=0;i<n_out_;++i) {
292  if (reduce_out_[i]) casadi_assert(df.nnz_in(n_in_+i)==0, "Case not implemented");
293  }
294 
295  std::vector<bool> reduce_in = join(reduce_in_, reduce_out_, reduce_in_);
296  Function dm = MapSum::create("mapsum" + str(n_) + "_" + df.name(), parallelization(),
297  df, n_, reduce_in, reduce_out_);
298 
299  // Input expressions
300  std::vector<MX> arg = dm.mx_in();
301 
302  // Need to reorder sensitivity inputs
303  std::vector<MX> res = arg;
304  std::vector<MX>::iterator it=res.begin()+n_in_+n_out_;
305  std::vector<casadi_int> ind;
306  for (casadi_int i=0; i<n_in_; ++i, ++it) {
307  if (reduce_in_[i]) continue;
308  casadi_int sz = f_.size2_in(i);
309  ind.clear();
310  for (casadi_int k=0; k<n_; ++k) {
311  for (casadi_int d=0; d<nfwd; ++d) {
312  for (casadi_int j=0; j<sz; ++j) {
313  ind.push_back((d*n_ + k)*sz + j);
314  }
315  }
316  }
317  *it = (*it)(Slice(), ind); // NOLINT
318  }
319 
320  // Get output expressions
321  res = dm(res);
322 
323  // Reorder sensitivity outputs
324  it = res.begin();
325  for (casadi_int i=0; i<n_out_; ++i, ++it) {
326  if (reduce_out_[i]) continue;
327  casadi_int sz = f_.size2_out(i);
328  ind.clear();
329  for (casadi_int d=0; d<nfwd; ++d) {
330  for (casadi_int k=0; k<n_; ++k) {
331  for (casadi_int j=0; j<sz; ++j) {
332  ind.push_back((k*nfwd + d)*sz + j);
333  }
334  }
335  }
336  *it = (*it)(Slice(), ind); // NOLINT
337  }
338 
339  // Construct return function
340  Dict custom_opts = opts;
341  custom_opts["always_inline"] = true;
342  custom_opts["allow_duplicate_io_names"] = true;
343  return Function(name, arg, res, inames, onames, custom_opts);
344  }
