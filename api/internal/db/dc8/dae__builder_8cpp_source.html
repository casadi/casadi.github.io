385 void DaeBuilder::set_z(const std::vector<std::string>& name, const std::vector<std::string>& alg) {
386  try {
387  // Consistency check
388  if (!alg.empty()) {
389  casadi_assert(alg.size() == name.size(), "Inconsistent number of algebraic variables");
390  }
391 
392  // Update z
393  set_all("z", name);
394 
395  // Update algebraic equations
396  if (!alg.empty()) {
397  for (size_t i = 0; i < name.size(); ++i) variable(name[i]).alg = find(alg[i]);
398  }
399  } catch (std::exception& e) {
400  THROW_ERROR("set_z", e.what());
401  }
402 }

633 std::string DaeBuilder::type(const std::string& name, casadi_int fmi_version) const {
634  // Check version

636  // Handle FMI 2
637  if (fmi_version == 2) {

640  // Assume FMI 3

642 }
