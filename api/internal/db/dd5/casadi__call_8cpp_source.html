37  MX Call::projectArg(const MX& x, const Sparsity& sp, casadi_int i) {
38  if (x.size()==sp.size()) {
39  // Insert sparsity projection nodes if needed
40  return project(x, sp);
41  } else {
42  // Different dimensions
43  if (x.is_empty() || sp.is_empty()) { // NOTE: To permissive?
44  // Replace nulls with zeros of the right dimension
45  return MX::zeros(sp);
46  } else if (x.is_scalar()) {
47  // Scalar argument means set all
48  return MX(sp, x);
49  } else if (x.size1()==sp.size2() && x.size2()==sp.size1() && sp.is_vector()) {
50  // Transposed vector
51  return projectArg(x.T(), sp, i);
52  } else {
53  // Mismatching dimensions
54  casadi_error("Cannot create function call node: Dimension mismatch for argument "
55  + str(i) + ". Argument has shape " + str(x.size())
56  + " but function input has shape " + str(sp.size()));
57  }
58  }
59  }

61  Call::Call(const Function& fcn, const std::vector<MX>& arg) : fcn_(fcn) {
62 
63  // Number inputs and outputs
64  casadi_int num_in = fcn.n_in();
65  casadi_assert(arg.size()==num_in, "Argument list length (" + str(arg.size())
66  + ") does not match number of inputs (" + str(num_in)
67  + ") for function " + fcn.name());
68 
69  // Create arguments of the right dimensions and sparsity
70  std::vector<MX> arg1(num_in);
71  for (casadi_int i=0; i<num_in; ++i) {
72  arg1[i] = projectArg(arg[i], fcn_.sparsity_in(i), i);
73  }
74  set_dep(arg1);
75  set_sparsity(Sparsity::scalar());
76  }

109  void Call::ad_forward(const std::vector<std::vector<MX>>& fseed,
110  std::vector<std::vector<MX>>& fsens) const {
111  try {
112  // Nondifferentiated inputs and outputs
113  std::vector<MX> arg(n_dep());
114  for (casadi_int i=0; i<arg.size(); ++i) arg[i] = dep(i);
115  std::vector<MX> res(nout());
116  for (casadi_int i=0; i<res.size(); ++i) res[i] = get_output(i);
117 
118  // Call the cached functions
119  fcn_->call_forward(arg, res, fseed, fsens, false, false);
120  } catch (std::exception& e) {
121  CASADI_THROW_ERROR("ad_forward", e.what());
122  }
123  }

125  void Call::ad_reverse(const std::vector<std::vector<MX>>& aseed,
126  std::vector<std::vector<MX>>& asens) const {
127  try {
128  // Find a common conditional argument among the seeds, if any
129  MX cond = common_cond(aseed);
130  // Nondifferentiated inputs and outputs
131  std::vector<MX> arg(n_dep());
132  for (casadi_int i=0; i<arg.size(); ++i) arg[i] = dep(i);
133  std::vector<MX> res(nout());
134  for (casadi_int i=0; i<res.size(); ++i) res[i] = get_output(i);
135  // Call the cached functions
136  std::vector<std::vector<MX>> v;
137  fcn_->call_reverse(arg, res, aseed, v, false, false);
138  for (casadi_int i=0; i<v.size(); ++i) {
139  for (casadi_int j=0; j<v[i].size(); ++j) {
140  // Skip structurally zero contributions (necessary?)
141  if (v[i][j].is_empty()) continue;
142  // Prevent propagation of NaNs through if/else
143  if (!cond.is_empty()) v[i][j] = if_else(cond, v[i][j], 0);
144  // Add seeds
145  asens[i][j] += v[i][j];
146  }
147  }
148  } catch (std::exception& e) {
149  CASADI_THROW_ERROR("ad_reverse", e.what());
150  }
151  }

235  MX Call::common_cond(const std::vector<std::vector<MX> >& seed) {
236  // Check if all seeds are conditional with the same seed
237  MX c;
238  for (const std::vector<MX>& seed_dir : seed) {
239  for (const MX& s : seed_dir) {
240  // Skip zero seeds
241  if (s.is_zero()) continue;
242  // If not a conditional, no common condition
243  if (!s.is_op(OP_IF_ELSE_ZERO)) return MX();
244  // Get conditional
245  MX c1 = s.dep(0);
246  // Has c already been set
247  if (c.is_empty(true)) {
248  // First time encountered
249  c = c1;
250  } else if (!MX::is_equal(c, c1)) {
251  // Different conditionals
252  return MX();
253  }
254  }
255  }
256  return c;
257  }
