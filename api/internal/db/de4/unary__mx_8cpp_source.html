66  void UnaryMX::ad_forward(const std::vector<std::vector<MX> >& fseed,
67  std::vector<std::vector<MX> >& fsens) const {
68  // Get partial derivatives
69  MX pd[2];
70  MX dummy; // Function value, dummy second argument
71  casadi_math<MX>::der(op_, dep(), dummy, shared_from_this<MX>(), pd);
72 
73  // Propagate forward seeds
74  for (casadi_int d=0; d<fsens.size(); ++d) {
75  fsens[d][0] = pd[0]*fseed[d][0];
76  }
77  }

102  void UnaryMX::generate(CodeGenerator& g,
103  const std::vector<casadi_int>& arg,
104  const std::vector<casadi_int>& res) const {
105  std::string r, x;
106  if (nnz()==1) {
107  // Scalar assignment
108  r = g.workel(res[0]);
109  x = g.workel(arg[0]);
110  } else {
111  // Vector assignment
112  g.local("cs", "const casadi_real", "*");
113  g.local("rr", "casadi_real", "*");
114  g.local("i", "casadi_int");
115  g << "for (i=0, rr=" << g.work(res[0], nnz()) << ", cs=" << g.work(arg[0], nnz())
116  << "; i<" << sparsity().nnz() << "; ++i) ";
117  r = "*rr++";
118  x = "*cs++";
119  }
120 
121  // Output the operation
122  g << r << " = " << g.print_op(op_, " " + x + " ") << ";\n";
123  }

125  MX UnaryMX::get_unary(casadi_int op) const {
126  if (!GlobalOptions::simplification_on_the_fly) return MXNode::get_unary(op);
127 
128  switch (op_) {
129  case OP_NEG:
130  if (op==OP_NEG) return dep();
131  else if (op==OP_SQ) return dep()->get_unary(OP_SQ);
132  else if (op==OP_FABS) return dep()->get_unary(OP_FABS);
133  else if (op==OP_COS) return dep()->get_unary(OP_COS);
134  break;
135  case OP_SQRT:
136  if (op==OP_SQ) return dep();
137  else if (op==OP_FABS) return shared_from_this<MX>();
138  break;
139  case OP_SQ:
140  if (op==OP_SQRT) return dep()->get_unary(OP_FABS);
141  else if (op==OP_FABS) return shared_from_this<MX>();
142  break;
143  case OP_EXP:
144  if (op==OP_LOG) return dep();
145  else if (op==OP_FABS) return shared_from_this<MX>();
146  break;
147  case OP_LOG:
148  if (op==OP_EXP) return dep();
149  break;
150  case OP_FABS:
151  if (op==OP_FABS) return shared_from_this<MX>();
152  else if (op==OP_SQ) return dep()->get_unary(OP_SQ);
153  else if (op==OP_COS) return dep()->get_unary(OP_COS);
154  break;
155  case OP_INV:
156  if (op==OP_INV) return dep();
157  break;
158  default: break; // no rule
159  }
160 
161  // Fallback to default implementation
162  return MXNode::get_unary(op);
163  }

165  MX UnaryMX::_get_binary(casadi_int op, const MX& y, bool scX, bool scY) const {
166  switch (op_) {
167  case OP_NEG:
168  if (op==OP_ADD) return y->_get_binary(OP_SUB, dep(), scY, scX);
169  else if (op==OP_MUL) return -dep()->_get_binary(OP_MUL, y, scX, scY);
170  else if (op==OP_DIV) return -dep()->_get_binary(OP_DIV, y, scX, scY);
171  break;
172  case OP_INV:
173  if (op==OP_MUL) return y->_get_binary(OP_DIV, dep(), scY, scX);
174  break;
175  case OP_TWICE:
176  if (op==OP_SUB && MX::is_equal(y, dep(), maxDepth())) return dep();
177  break;
178  case OP_SQ:
179  if (op==OP_ADD && y.op()==OP_SQ) /*sum of squares:*/
180  if ((dep().op()==OP_SIN && y->dep().op()==OP_COS) ||
181  (dep().op()==OP_COS && y->dep()->op()==OP_SIN)) /* sin^2(x)+sin^2(y) */
182  if (MX::is_equal(dep()->dep(), y->dep()->dep(), maxDepth())) /*sin^2(x) + cos^2(x) */
183  return MX::ones(y.sparsity());
184  break;
185  default: break; // no rule
186  }
187 
188  // Fallback to default implementation
189  return MXNode::_get_binary(op, y, scX, scY);
190  }
