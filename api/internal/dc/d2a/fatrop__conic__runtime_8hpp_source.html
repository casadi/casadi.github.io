40 // SYMBOL "fatrop_dense_transfer"
41 template<typename T1>
42 void casadi_fatrop_conic_dense_transfer(double factor, const T1* x,
43  const casadi_int* sp_x, T1* y,
44  const casadi_int* sp_y, T1* w) {
45  casadi_sparsify(x, w, sp_x, 0);
46  casadi_int nrow_y = sp_y[0];
47  casadi_int ncol_y = sp_y[1];
48  const casadi_int *colind_y = sp_y+2, *row_y = sp_y + 2 + ncol_y+1;
49  /* Loop over columns of y */
50  casadi_int i, el;
51  for (i=0; i<ncol_y; ++i) {
52  for (el=colind_y[i]; el<colind_y[i+1]; ++el) y[nrow_y*i + row_y[el]] += factor*(*w++);
53  }
54 }

144 // SYMBOL "fatrop_solve"
145 template<typename T1>
146 int casadi_fatrop_conic_solve(casadi_fatrop_conic_data<T1>* d, const double** arg, double** res, casadi_int* iw, double* w) {
147  casadi_int k, i, j, n_row, offset, start, stop;
148  T1 f;
149  const casadi_fatrop_conic_prob<T1>* p = d->prob;
150  const casadi_qp_prob<T1>* p_qp = p->qp;
151  casadi_qp_data<T1>* d_qp = d->qp;
152 
153  casadi_fatrop_conic_mproject(-1.0, d_qp->a, p_qp->sp_a, d->AB, p->ABsp, d->pv);
154  casadi_project(d_qp->a, p_qp->sp_a, d->CD, p->CDsp, d->pv);
155 
156  casadi_project(d_qp->h, p_qp->sp_h, d->RSQ, p->RSQsp, d->pv);
157 
158  d->a_eq_idx[0] = 0;
159  d->a_ineq_idx[0] = 0;
160  d->x_eq_idx[0] = 0;
161  d->x_ineq_idx[0] = 0;
162 
163  // Loop over CD blocks
164  for (k=0;k<p->N+1;++k) {
165  d->a_eq_idx[k+1] = d->a_eq_idx[k];
166  d->a_ineq_idx[k+1] = d->a_ineq_idx[k];
167  start = p->CD[k].offset_r;
168  stop = p->CD[k].offset_r+p->CD[k].rows;
169  for (i=start;i<stop;++i) {
170  if (d_qp->lba[i]==d_qp->uba[i]) {
171  d->a_eq[d->a_eq_idx[k+1]++] = i;
172  } else {
173  if (d_qp->lba[i]==-inf && d_qp->uba[i]==inf) continue;
174  d->a_ineq[d->a_ineq_idx[k+1]++] = i;
175  }
176  }
177  d->x_eq_idx[k+1] = d->x_eq_idx[k];
178  d->x_ineq_idx[k+1] = d->x_ineq_idx[k];
179  start = p->CD[k].offset_c;
180  stop = p->CD[k].offset_c+p->CD[k].cols;
181  //uout() << "start" << start << "->" << stop << std::endl;
182  for (i=start;i<stop;++i) {
183  if (d_qp->lbx[i]==d_qp->ubx[i]) {
184  d->x_eq[d->x_eq_idx[k+1]++] = i;
185  } else {
186  if (d_qp->lbx[i]==-inf && d_qp->ubx[i]==inf) continue;
187  d->x_ineq[d->x_ineq_idx[k+1]++] = i;
188  }
189  }
190  //uout() << "k=" << k << std::endl;
191  //uout() << "a_eq" << std::vector<double>(d->a_eq+d->a_eq_idx[k], d->a_eq+d->a_eq_idx[k+1]) << std::endl;
192  //uout() << "a_ineq" << std::vector<double>(d->a_ineq+d->a_ineq_idx[k], d->a_ineq+d->a_ineq_idx[k+1]) << std::endl;
193  //uout() << "x_eq" << std::vector<double>(d->x_eq+d->x_eq_idx[k], d->x_eq+d->x_eq_idx[k+1]) << std::endl;
194  //uout() << "x_ineq" << std::vector<double>(d->x_ineq+d->x_ineq_idx[k], d->x_ineq+d->x_ineq_idx[k+1]) << std::endl;
195 
196  //uout() << "AB=" << std::vector<double>(d->AB,d->AB+100) << std::endl;
197 
198  //uout() << "CD=" << std::vector<double>(d->CD,d->CD+100) << std::endl;
199 
200  //uout() << "RSQ=" << std::vector<double>(d->RSQ,d->RSQ+100) << std::endl;
201  }
202 
203 
204  return 0;
205 }

210  casadi_qp_work(p->qp, sz_arg, sz_res, sz_iw, sz_w);
