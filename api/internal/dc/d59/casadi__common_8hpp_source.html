232  inline std::string fmtstr(const std::string& fmt, const std::vector<std::string>& args) {
233  std::string s = fmt;
234  for (auto&& e : args) {
235  std::string::size_type n = s.find("%s");
236  if (n==std::string::npos) return "** Ill-formatted string ** " + fmt;
237  s.replace(n, 2, e);
238  }
239  return s;
240  }
