128 Function simpleRK(Function f, casadi_int N, casadi_int order) {
129  // Consistency check
130  casadi_assert(N>=1,
131  "Parameter N (number of steps) must be at least 1, but got " + str(N) + ".");
132  casadi_assert(order==4, "Only RK order 4 is supported now.");
133  casadi_assert(f.n_in()==2, "Function must have two inputs: x and p");
134  casadi_assert(f.n_out()==1, "Function must have one outputs: dot(x)");
135 
136  MX x0 = MX::sym("x0", f.sparsity_in(0));
137  MX p = MX::sym("p", f.sparsity_in(1));
138  MX h = MX::sym("h");
139 
140  std::vector<double> b(order);
141  b[0]=1.0/6;
142  b[1]=1.0/3;
143  b[2]=1.0/3;
144  b[3]=1.0/6;
145 
146  std::vector<double> c(order);
147  c[0]=0;
148  c[1]=1.0/2;
149  c[2]=1.0/2;
150  c[3]=1;
151 
152  std::vector< std::vector<double> > A(order-1);
153  A[0].resize(1);
154  A[0][0]=1.0/2;
155  A[1].resize(2);
156  A[1][0]=0;A[1][1]=1.0/2;
157  A[2].resize(3);
158  A[2][0]=0;
159  A[2][1]=0;A[2][2]=1;
160 
161  // Time step
162  MX dt = h/N;
163 
164  std::vector<MX> k(order);
165  std::vector<MX> f_arg(2);
166 
167  // Integrate
168  MX xf = x0;
169  for (casadi_int i=0; i<N; ++i) {
170  for (casadi_int j=0; j<order; ++j) {
171  MX xL = 0;
172  for (casadi_int jj=0; jj<j; ++jj) {
173  xL += k.at(jj)*A.at(j-1).at(jj);
174  }
175  f_arg[0] = xf+xL;
176  f_arg[1] = p;
177  k[j] = dt*f(f_arg).at(0);
178  }
179 
180  for (casadi_int j=0; j<order; ++j) {
181  xf += b.at(j)*k.at(j);
182  }
183  }
184 
185  // Form discrete-time dynamics
186  return Function("F", {x0, p, h}, {xf}, {"x0", "p", "h"}, {"xf"});
187 }

189 void collocation_interpolators(const std::vector<double> & tau,
190  std::vector< std::vector<double> > &C, std::vector< double > &D) {
191  // Find the degree of the interpolation
192  casadi_int deg = tau.size();
193 
194  // Include zero
195  std::vector<double> etau_root = tau;
196  etau_root.insert(etau_root.begin(), 0);
197 
198  // Allocate storage space for resulting coefficients
199  C.resize(deg+1);
200  for (casadi_int i=0;i<deg+1;++i) {
201  C[i].resize(deg+1);
202  }
203  D.resize(deg+1);
204 
205  // Collocation point
206  SX tau_sym = SX::sym("tau");
207 
208  // For all collocation points
209  for (casadi_int j=0; j<deg+1; ++j) {
210  // Construct Lagrange polynomials to get the polynomial basis at the collocation point
211  SX L = 1;
212  for (casadi_int j2=0; j2<deg+1; ++j2) {
213  if (j2 != j) {
214  L *= (tau_sym-etau_root[j2])/(etau_root[j]-etau_root[j2]);
215  }
216  }
217 
218  Function lfcn("lfcn", {tau_sym}, {L});
219 
220  // Evaluate the polynomial at the final time to get the
221  // coefficients of the continuity equation
222  D[j] = lfcn(std::vector<DM>{1.}).at(0)->front();
223 
224  // Evaluate the time derivative of the polynomial at all collocation points to
225  // get the coefficients of the continuity equation
226  Function tfcn("tfcn", {tau_sym}, {tangent(L, tau_sym)});
227  for (casadi_int j2=0; j2<deg+1; ++j2) {
228  C[j2][j] = tfcn(std::vector<DM>{etau_root[j2]}).at(0)->front();
229  }
230  }

233 void collocation_coeff(const std::vector<double> & tau,
234  DM &C, DM &D, DM &B) {
235  // Find the degree of the interpolation
236  casadi_int deg = tau.size();
237 
238  // Include zero
239  std::vector<double> etau_root = tau;
240  etau_root.insert(etau_root.begin(), 0);
241 
242  // Check if tau ends with '1' (cfr. radau scheme)
243  bool has_end = tau.back()==1;
244 
245  // Coefficients of the collocation equation
246  std::vector<std::vector<double> > C_(deg+1, std::vector<double>(deg+1, 0));
247 
248  // Coefficients of the continuity equation
249  std::vector<double> D_(deg+1, 0);
250 
251  // Coefficients of the quadratures
252  std::vector<double> B_(deg+1, 0);
253 
254  // For all collocation points
255  for (casadi_int j=0; j<deg+1; ++j) {
256 
257  // Construct Lagrange polynomials to get the polynomial basis at the collocation point
258  Polynomial p = 1;
259  for (casadi_int r=0; r<deg+1; ++r) {
260  if (r!=j) {
261  p *= Polynomial(-etau_root[r], 1)/(etau_root[j]-etau_root[r]);
262  }
263  }
264 
265  // Evaluate the polynomial at the final time to get the
266  // coefficients of the continuity equation
267  if (has_end) {
268  D_[j] = j==deg ? 1 : 0;
269  } else {
270  D_[j] = p(1.0);
271  }
272  // Evaluate the time derivative of the polynomial at all collocation points to
273  // get the coefficients of the continuity equation
274  Polynomial dp = p.derivative();
275  for (casadi_int r=0; r<deg+1; ++r) {
276  C_[j][r] = dp(etau_root[r]);
277  }
278 
279  // Integrate polynomial to get the coefficients of the quadratures
280  Polynomial ip = p.anti_derivative();
281  B_[j] = ip(1.0);
282  }
283  C = DM(C_);
284  C = C(Slice(), Slice(1, deg+1)); // NOLINT(cppcoreguidelines-slicing)
285  D = DM(D_);
286  B = DM(std::vector<double>(B_.begin()+1, B_.end()));
287 }

289 Function simpleIRK(Function f, casadi_int N, casadi_int order, const std::string& scheme,
290  const std::string& solver,
291  const Dict& solver_options) {
292  // Consistency check
293  casadi_assert(N>=1,
294  "Parameter N (number of steps) must be at least 1, but got " + str(N) + ".");
295  casadi_assert(f.n_in()==2, "Function must have two inputs: x and p");
296  casadi_assert(f.n_out()==1, "Function must have one outputs: dot(x)");
297 
298  // Obtain collocation points
299  std::vector<double> tau_root = collocation_points(order, scheme);
300 
301  // Retrieve collocation interpolating matrices
302  std::vector < std::vector <double> > C;
303  std::vector < double > D;
304  collocation_interpolators(tau_root, C, D);
305 
306  // Inputs of constructed function
307  MX x0 = MX::sym("x0", f.sparsity_in(0));
308  MX p = MX::sym("p", f.sparsity_in(1));
309  MX h = MX::sym("h");
310 
311  // Time step
312  MX dt = h/N;
313 
314  // Implicitly defined variables
315  MX v = MX::sym("v", repmat(x0.sparsity(), order));
316  std::vector<MX> x = vertsplit(v, x0.size1());
317  x.insert(x.begin(), x0);
318 
319  // Collect the equations that implicitly define v
320  std::vector<MX> V_eq, f_in(2), f_out;
321  for (casadi_int j=1; j<order+1; ++j) {
322  // Expression for the state derivative at the collocation point
323  MX xp_j = 0;
324  for (casadi_int r=0; r<=order; ++r) xp_j+= C[j][r]*x[r];
325 
326  // Collocation equations
327  f_in[0] = x[j];
328  f_in[1] = p;
329  f_out = f(f_in);
330  V_eq.push_back(dt*f_out.at(0)-xp_j);
331  }
332 
333  // Root-finding function
334  Function rfp("rfp", {v, x0, p, h}, {vertcat(V_eq)});
335 
336  // Create a implicit function instance to solve the system of equations
337  Function ifcn = rootfinder("ifcn", solver, rfp, solver_options);
338 
339  // Get state at end time
340  MX xf = x0;
341  for (casadi_int k=0; k<N; ++k) {
342  std::vector<MX> ifcn_out = ifcn({repmat(xf, order), xf, p, h});
343  x = vertsplit(ifcn_out[0], xf.size1());
344 
345  // State at end of step
346  xf = D[0]*xf;
347  for (casadi_int i=1; i<=order; ++i) {
348  xf += D[i]*x[i-1];
349  }
350  }
351 
352  // Form discrete-time dynamics
353  return Function("F", {x0, p, h}, {xf}, {"x0", "p", "h"}, {"xf"});
354 }

356 Function simpleIntegrator(Function f, const std::string& plugin,
357  const Dict& plugin_options) {
358  // Consistency check
359  casadi_assert(f.n_in()==2, "Function must have two inputs: x and p");
360  casadi_assert(f.n_out()==1, "Function must have one outputs: dot(x)");
361 
362  // Sparsities
363  Sparsity x_sp = f.sparsity_in(0);
364  Sparsity p_sp = f.sparsity_in(1);
365 
366  // Wrapper function inputs
367  MX x = MX::sym("x", x_sp);
368  MX u = MX::sym("u", vertcat(Sparsity::scalar(), vec(p_sp))); // augment p with t
369 
370  // Normalized xdot
371  casadi_int u_offset[] = {0, 1, 1+p_sp.size1()};
372  std::vector<MX> pp = vertsplit(u, std::vector<casadi_int>(u_offset, u_offset+3));
373  MX h = pp[0];
374  MX p = reshape(pp[1], p_sp.size());
375  MX f_in[] = {x, p};
376  MX xdot = f(std::vector<MX>(f_in, f_in+2)).at(0);
377  xdot *= h;
378 
379  // Form DAE function
380  MXDict dae = {{"x", x}, {"p", u}, {"ode", xdot}};
381 
382  // Create integrator function with normalized time from 0 to 1
383  Function ifcn = integrator("integrator", plugin, dae, plugin_options);
384 
385  // Inputs of constructed function
386  MX x0 = MX::sym("x0", x_sp);
387  p = MX::sym("p", p_sp);
388  h = MX::sym("h");
389 
390  // State at end
391  MX xf = ifcn(MXDict{{"x0", x0}, {"p", vertcat(h, vec(p))}}).at("xf");
392 
393  // Form discrete-time dynamics
394  return Function("F", {x0, p, h}, {xf}, {"x0", "p", "h"}, {"xf"});
395 }

515  bool dfs_match_pantelides(Equation* i) {
516 
517  // Pantelides alg 3.2: Colour i (1)
518  i->visited = true;
519 
520  // Look for unassigned candidate variables
521  // Pantelides alg 3.2: (2) If a V-node j exists such that edge (i-j) exists...
522  for (auto* j : i->vars) {
523  // Pantelides alg 3.2: ... and ASSIGN (j)= 0 then:
524  if (j->assign==nullptr) {
525  // Found such variable
526  // Assign
527  // Pantelides alg 3.2: (2b) Set ASSIGN(j)=i
528  j->assign = i; i->assign = j;
529 
530  // Pantelides alg 3.2: (2a) set PATHFOUND = TRUE, (2c) return
531  return true; // End successfully
532  }
533  }
534 
535  // Look for assigned candidate variables
536  // Pantelides alg 3.2: (3) For every j such that edge (i-j) exists
537  for (auto j : i->vars) {
538  // Pantelides alg 3.2: ... and j is uncoloured do:
539  if (!j->visited) {
540  // Pantelides alg 3.2: (3a) Colour j
541  j->visited = true;
542 
543  // Pantelides alg 3.2: (3b) Set k = ASSIGN (j)
544  Equation* k = j->assign;
545 
546  // Pantelides alg 3.2: (3c) AUGMENTPATH (k, PATHFOUND)
547  if (dfs_match_pantelides(k)) {
548  // Re-assignment
549  // Pantelides alg 3.2: (3d-1) Set ASSIGN (j) = i
550  j->assign = i; i->assign = j;
551  // Pantelides alg 3.2: (3d-2) Return
552  return true;
553  }
554 
555  }
556  }
557 
558  // Exhausted all options; return unsuccessfully
559  return false;
560  }

581  // Pantelides alg 4.1: For k=1 to N' do (3)
582  for (casadi_int k=0;k<Np;++k) {
583  // Pantelides alg 4.1: Set i=k (3a)
584  Equation* i = G.E[k];
585 
586  casadi_int iter = 0;
587  // Pantelides alg 4.1: Repeat (3b)
588  while (true) {
589  if (iter++ > max_iter) return false;
590  // Pantelides alg 4.1: delete all V-nodes with A!=0 and their incident edges
591  // * from the graph (3b-1)
592  for (auto* v : G.V) {
593  if (v->der) {
594  v->deleted = true;
595  v->eqs.clear();
596  for (auto* e : G.E) {
597  auto it = std::find(e->vars.begin(), e->vars.end(), v);
598  if (it!=e->vars.end()) e->vars.erase(it);
599  }
600  }
601  }
602 
603  // Pantelides alg 4.1: Designate all nodes as "uncoloured" (3b-2)
604  for (auto* v : G.V) v->visited = false;
605  for (auto* e : G.E) e->visited = false;
606 
607  // Pantelides alg 4.1: Set PATHFOUND= FALSE (3b-3)
608  // Pantelides alg 4.1: AUGMENTPATH (i, PATHFOUND) (3b-4)
609  // Pantelides alg 4.1: Until PATHFOUND (3c)
610  if (dfs_match_pantelides(i)) break; // out of while loop
611 
612  // Pantelides alg 4.1: If PATHFOUND FALSE then (3b-5)
613  casadi_int n = G.V.size();
614  // Pantelides alg 4.1: For every coloured V-node j do (3b-5-i)
615  for (casadi_int jj=0;jj<n;++jj) {
616  Variable* j = G.V[jj];
617  if (j->visited && !j->deleted) {
618  graph_add_der(G, j);
619  }
620  }
621 
622  // Pantelides alg 4.1: For every coloured E-node l do (3b-5-ii)
623  n = G.E.size();
624  for (casadi_int ll=0;ll<n;++ll) {
625  Equation* l = G.E[ll];
626  if (l->visited) graph_add_der(G, l, true);
627  }
628 
629  for (casadi_int ll=n;ll<G.E.size();++ll) {
630  Equation* l = G.E[ll];
631  bool valid = false;
632  for (auto* v : l->vars0) {
633  if (!v->eqs.empty()) valid = true;
634  }
635  if (!valid) return false;
636  }
637 
638  // Pantelides alg 4.1: For every coloured V-node j set
639  // ASSIGN (A(j))= B(ASSIGN (j)) (3b-5-iii)
640  for (auto* j : G.V) {
641  if (j->visited && !j->deleted) {
642  j->der->assign = j->assign->der;
643  j->assign->der->assign = j->der;
644  }
645  }
646 
647  // Pantelides alg 4.1: Set i=B(i) (3b-5-iv)
648  i = i->der;
649 
650  }
651  }
652  return true;
653  }

655  void dae_struct_detect(const std::string& algorithm,
656  const Sparsity& graph, std::vector<casadi_int>& var_map,
657  std::vector<casadi_int>& eq_map,
658  casadi_int max_iter) {
659  // Input sanitization
660  casadi_assert(var_map.size()==graph.size2(),
661  "var_map size must match graph columns.");
662 
663  // Graph structure
664  Graph G;
665 
666  // Allocate space for node representation
667  G.E.resize(graph.size1(), nullptr);
668  G.V.resize(graph.size2(), nullptr);
669  G.nrow_orig = graph.size1();
670  G.ncol_orig = graph.size2();
671 
672  for (auto*& e : G.E) e = new Equation();
673  for (auto*& v : G.V) v = new Variable();
674 
675  // Set positions
676  casadi_int i;
677  i=0; for (auto* e : G.E) e->i = i++;
678  i=0; for (auto* v : G.V) v->i = i++;
679 
680  // Create edges using incidence sparsity
681  const casadi_int* colind = graph.colind();
682  const casadi_int* row = graph.row();
683 
684  // Loop over incidence columns
685  for (casadi_int c=0;c<graph.size2();++c) {
686  // Loop over nonzeros
687  for (casadi_int j=colind[c]; j<colind[c+1]; ++j) {
688  casadi_int r = row[j];
689  G.V[c]->eqs.push_back(G.E[r]); // Non-monotone
690  G.E[r]->vars.push_back(G.V[c]);
691  }
692  }
693 
694  // Process var_map
695  for (casadi_int i=0;i<var_map.size();++i) {
696  Variable* v = G.V[i];
697  casadi_int i_der = var_map[i];
698  if (i_der>=0) {
699  Variable* e = G.V[i_der];
700  v->der = e;
701  e->der_inv = v;
702  }
703  }
704 
705  for (auto* v : G.V) v->eqs0 = v->eqs;
706  for (auto* e : G.E) e->vars0 = e->vars;
707 
708  bool detect = false;
709  if (algorithm=="pantelides") {
710  detect = dae_struct_detect_pantelides(G, var_map, eq_map, max_iter);
711  } else {
712  casadi_error("Algorithm '" + algorithm + "' not recognized.");
713  }
714  casadi_assert(detect, "Structural detection failed.");
715 
716  // Prepare outputs
717  var_map.resize(G.V.size());
718  eq_map.resize(G.E.size());
719 
720  int k = 0;
721  // Populate var_map
722  for (auto* v : G.V) {
723  if (v->der) {
724  var_map[v->i] = v->der->i;
725  } else {
726  var_map[v->i] = -1;
727  }
728  casadi_assert_dev(v->i==k);
729  k++;
730  }
731 
732  k = 0;
733  // Populate eq_map
734  for (auto* e : G.E) {
735  if (e->der) {
736  eq_map[e->i] = e->der->i;
737  } else {
738  eq_map[e->i] = -1;
739  }
740  casadi_assert_dev(e->i==k);
741  k++;
742  }
743 
744  }

795 const std::map<std::string, X>
796 reduce_index_gen(const std::map<std::string, X>& dae, Dict& stats, const Dict& opts) {
797  double baumgarte_pole_ = 0;
798  std::string algorithm = "pantelides";
799  casadi_int max_iter = 500;
800  // Option parsing
801  for (auto&& op : opts) {
802  if (op.first=="baumgarte_pole") {
803  baumgarte_pole_ = op.second;
804  } else if (op.first=="algorithm") {
805  algorithm = op.second.to_string();
806  } else if (op.first=="max_iter") {
807  max_iter = op.second;
808  } else {
809  casadi_error("Unknown option '" + op.first + "'.");
810  }
811  }
812 
813  // Dae input sanitization
814 
815  for (const auto& e : dae) {
816  casadi_assert(e.second.is_column() && e.second.is_dense(),
817  "Dense column vector expected for key '" + e.first + "'. "
818  "Got " + e.second.dim(true) + " instead.");
819  }
820 
821  X x = get_from_dict(dae, "x", X(0, 1));
822  X ode = get_from_dict(dae, "ode", X(0, 1));
823  casadi_assert(x.numel()==ode.numel(),
824  "Size of explicit differential state vector (x: " + str(x.numel()) + ") must "
825  "match size of ode right-hand-side (ode: " + str(ode.numel()) + ").");
826  casadi_assert(x.size1()==0, "Explicit differential states not supported yet.");
827 
828  X x_impl = get_from_dict(dae, "x_impl", X(0, 1));
829  X dx_impl = get_from_dict(dae, "dx_impl", X(0, 1));
830  casadi_assert(x_impl.numel()==dx_impl.numel(),
831  "Size of implicit differential state vector (x_impl: " + str(x_impl.numel()) + ") must "
832  "match size of implicit differential state derivative vector (dx_impl: "
833  + str(dx_impl.numel()) + ").");
834 
835  X z = get_from_dict(dae, "z", X(0, 1));
836  X alg = get_from_dict(dae, "alg", X(0, 1));
837  casadi_assert(z.numel()+x_impl.numel()==alg.numel(),
838  "Size of algebraic state vector (z: " + str(z.numel()) + ") + "
839  "size of implicit states (x_impl: " + str(x_impl.numel()) + ") must "
840  "match size of algebraic equations (alg: " + str(alg.numel()) + ").");
841 
842  X t = get_from_dict(dae, "t", X::sym("t"));
843  casadi_assert(t.is_scalar(), "Time must be scalar. Got " + t.dim() + " instead.");
844 
845  X p = get_from_dict(dae, "p", X(0, 1));
846 
847  // Check that dx_impl are classified correctly
848  std::vector<bool> dep = X::which_depends(alg, dx_impl);
849 
850  if (!all(dep)) {
851  std::vector<X> dx_impl_split = vertsplit(dx_impl);
852  std::vector<std::string> prob;
853  for (casadi_int i=0;i<dep.size();++i) {
854  if (!dep[i]) prob.push_back(dx_impl_split[i].name());
855  }
856  casadi_error("Found dx_impl variables that do not appear in alg: " + join(prob, ",") +
857  ". They should be classified as z instead.");
858  }
859 
860 
861  bool normal_order = true;
862 
863  // Determine graph structure of problem for structural index reduction
864  X V;
865  if (normal_order) {
866  V = vertcat(x_impl, dx_impl, z);
867  } else {
868  V = vertcat(x_impl, z, dx_impl);
869  }
870 
871  Function temp("temp", {V, p, t}, {alg});
872  Sparsity G = temp.sparsity_jac(0, 0);
873 
874  // Populate var_map: a list that associates variables with their derivatives
875  int nx_impl = x_impl.numel();
876  int nz = z.numel();
877  std::vector<casadi_int> var_map(V.numel(), -1);
878  for (casadi_int i=0;i<nx_impl;++i) {
879  var_map[i] = i+nx_impl+(normal_order? 0: nz);
880  }
881 
882  // Allocate space for eq_map: a list that associates equations with their derivatives
883  std::vector<casadi_int> eq_map;
884 
885  // Modifies var_map and eq_map
886  dae_struct_detect(algorithm, G, var_map, eq_map, max_iter);
887 
888  // Variables should not be removed
889  casadi_assert_dev(var_map.size()>=2*nx_impl+nz);
890 
891  // List of scalarized variables
892  std::vector<X> var_ext = vertsplit(V);
893  // Allocate extra space for extension due to index reduction
894  var_ext.resize(var_map.size());
895 
896  // Populate extra space
897  for (casadi_int i=0;i<var_map.size();++i) {
898  casadi_int i_der = var_map[i];
899  // Derivate index is in the extra space?
900  if (i_der>=V.numel())
901  // Consstruct a new symbol with derived name
902  var_ext[i_der] = X::sym("d"+var_ext[i].name());
903  }
904 
905  // Prepare arguments for jtimes
906  // In essence, this is like to the novel x_impl and dx_impl,
907  // but a subset may be promoted to x (explicit diff. state)
908  std::vector<X> xs; xs.reserve(var_map.size());
909  std::vector<X> dxs; dxs.reserve(var_map.size());
910 
911  for (casadi_int i=0;i<var_map.size();++i) {
912  casadi_int i_der = var_map[i];
913  if (i_der>=0) {
914  xs.push_back(var_ext[i]);
915  dxs.push_back(var_ext[i_der]);
916  }
917  }
918 
919  // Stack to be used as jtimes arguments
920  X x_cat = vertcat(xs);
921  X dx_cat = vertcat(dxs);
922 
923  // Don't forget to differentiate time itself
924  X j1 = vertcat(x_cat, t);
925  X j2 = vertcat(dx_cat, 1);
926 
927  // List of scalarized equations
928  std::vector<X> eq_ext = vertsplit(alg);
929  // Allocate extra space for extension due to index reduction
930  eq_ext.resize(eq_map.size());
931 
932  // For each equation, figure out how many times it was differentiated (=order)
933  std::vector<casadi_int> order = get_orders(eq_map);
934  casadi_int max_order = *std::max_element(order.begin(), order.end());
935 
936  // We lump together all relevant equations before doing differentiation in a target group,
937  // in order to make use of shared subexpressions.
938  // order -> all equations that give rise to an equation of this order
939  std::map<casadi_int, std::vector<casadi_int> > targets;
940  for (casadi_int i=0;i<eq_map.size();++i) {
941  if (eq_map[i]>=0) {
942  targets[order[eq_map[i]]].push_back(i);
943  }
944  }
945 
946  // Non-active equations will be dropped from the final set
947  std::vector<bool> active(eq_map.size(), true);
948 
949  // Equations that get dropped, but should still hold for initial guess
950  std::vector<X> dropped_constraints;
951 
952  // Loop over equation elements in increasing order
953  for (casadi_int ord=1;ord<=max_order;++ord) {
954  const std::vector<casadi_int> & target = targets[ord];
955 
956  // Remember to-be-dropped equations
957  for (casadi_int i : target) dropped_constraints.push_back(eq_ext[i]);
958 
959  // Gather target's equations
960  X block = vertcat(vector_slice(eq_ext, target));
961  // Perform differentiation
962  std::vector<X> dblocks = vertsplit(jtimes(block, j1, j2, false));
963  // Scatter computed derivatives into target resultant equation slots
964  for (casadi_int i=0;i<target.size();++i) {
965  casadi_int t = target[i];
966  eq_ext[eq_map[t]] = dblocks[i];
967  active[t] = false;
968  }
969  }
970 
971  std::vector<casadi_int> eq_map_inverse = get_inverse(eq_map);
972 
973  // Construct the new list of algebraic equations
974  // In theory, just collect all active members of eq_ext
975  // In practice, we add some Baumgarte stabilization to avoid constraint drifting
976  std::vector<X> alg_new;
977  for (casadi_int i=0;i<eq_map.size();++i) {
978  if (active[i]) {
979 
980  // Start from the pure equation
981  X new_eq = eq_ext[i];
982 
983  // Construct coefficients of the polynomial (x-gamma)^order
984  Polynomial p(1);
985  for (casadi_int k=0;k<order[i];++k) {
986  p *= Polynomial(1, -baumgarte_pole_);
987  }
988 
989  const std::vector<double>& coeff = p.coeff();
990 
991  // Find indices of originating equations of equation i
992  // e.g. if eq_i = der(eq_j), eq_j = der(eq_k), obtain [j,k]
993  std::vector<casadi_int> gen_i = path(eq_map_inverse, i);
994 
995  // Add inner product of coefficients with originating equations of new_eq
996  for (casadi_int k=0;k<gen_i.size();++k) {
997  new_eq += coeff[k+1]*eq_ext[gen_i[k]];
998  }
999 
1000  // Ready to add
1001  alg_new.push_back(new_eq);
1002  }
1003  }
1004 
1005  std::vector<casadi_int> var_order = get_orders(var_map);
1006  std::vector<casadi_int> var_map_inverse = get_inverse(var_map);
1007 
1008 
1009  // Which implicit states become explicit pure integrators?
1010  std::vector<casadi_int> impl_to_expl;
1011  for (casadi_int i=0;i<var_map.size();++i) {
1012  if (var_order[i]>=2) {
1013  // Find indices of originating variables of variable i
1014  // e.g. if var_i = der(var_j), var_j = der(var_k), obtain [j,k]
1015  std::vector<casadi_int> gen_i = path(var_map_inverse, i);
1016 
1017  // Anything too deep should become explicit integrator
1018  if (gen_i.size()>1)
1019  impl_to_expl.insert(impl_to_expl.end(), gen_i.begin()+1, gen_i.end());
1020 
1021  }
1022  }
1023 
1024  // Take complement
1025  std::vector<casadi_int> impl_to_expl_compl = complement(impl_to_expl, xs.size());
1026  std::map<std::string, X> dae_result;
1027 
1028  dae_result["x"] = vertcat(vector_slice(xs, impl_to_expl));
1029  dae_result["ode"] = vertcat(vector_slice(dxs, impl_to_expl));
1030 
1031  dae_result["x_impl"] = vertcat(vector_slice(xs, impl_to_expl_compl));
1032  dae_result["dx_impl"] = vertcat(vector_slice(dxs, impl_to_expl_compl));
1033 
1034  dae_result["t"] = t;
1035  dae_result["p"] = p;
1036 
1037  // Which algebraic variables were not promoted to differential states?
1038  std::vector<X> z_orig = vertsplit(z);
1039  std::vector<X> z_new;
1040  for (casadi_int i=0;i<var_map.size();++i) {
1041  // Algebraic
1042  if (var_order[i]==0 && var_map[i]==-1) {
1043  z_new.push_back(z_orig.at(i-(normal_order?2:1)*nx_impl));
1044  }
1045  }
1046  dae_result["z"] = vertcat(z_new);
1047  dae_result["alg"] = vertcat(alg_new);
1048 
1049  dae_result["I"] = vertcat(dropped_constraints);
1050 
1051  stats["index"] = max_order+1;
1052 
1053  return dae_result;
1054 }
