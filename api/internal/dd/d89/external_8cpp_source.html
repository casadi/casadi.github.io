68 void External::init_external() {
69  // Increasing/decreasing reference counter
70  incref_ = (signal_t)li_.get_function(name_ + "_incref");
71  decref_ = (signal_t)li_.get_function(name_ + "_decref");
72 
73  casadi_assert(static_cast<bool>(incref_) == static_cast<bool>(decref_),
74  "External must either define both incref and decref or neither.");
75 
76  // Getting default arguments
77  get_default_in_ = (default_t)li_.get_function(name_ + "_default_in");
78 
79  // Getting number of inputs and outputs
80  get_n_in_ = (getint_t)li_.get_function(name_ + "_n_in");
81  get_n_out_ = (getint_t)li_.get_function(name_ + "_n_out");
82 
83  // Getting names of inputs and outputs
84  get_name_in_ = (name_t)li_.get_function(name_ + "_name_in");
85  get_name_out_ = (name_t)li_.get_function(name_ + "_name_out");
86 
87  // Work vector sizes
88  work_ = (work_t)li_.get_function(name_ + "_work");
89 
90  // Increase reference counter - external function memory initialized at this point
91  if (incref_) incref_();
92 }

270 void External::init(const Dict& opts) {
271  // Call the initialization method of the base class
272  FunctionInternal::init(opts);
273 
274  casadi_assert(any_symbol_found(),
275  "Could not find any function/symbol starting with '" + name_ + "_'. "
276  "Make sure to read documentation of `external()` for proper usage.");
277 
278  // Reference counting?
279  has_refcount_ = li_.has_function(name_ + "_incref");
280  casadi_assert(has_refcount_==li_.has_function(name_ + "_decref"),
281  "External functions must provide functions for both increasing "
282  "and decreasing the reference count, or neither.");
283 
284  // Allocate work vectors
285  casadi_int sz_arg=0, sz_res=0, sz_iw=0, sz_w=0;
286  if (work_) {
287  casadi_int flag = work_(&sz_arg, &sz_res, &sz_iw, &sz_w);
288  casadi_assert(flag==0, "External: \"work\" failed");
289  } else if (li_.has_meta(name_ + "_WORK")) {
290  std::vector<casadi_int> v = li_.meta_vector<casadi_int>(name_ + "_WORK");
291  casadi_assert_dev(v.size()==4);
292  sz_arg = v[0];
293  sz_res = v[1];
294  sz_iw = v[2];
295  sz_w = v[3];
296  }
297 
298  // Work vectors
299  alloc_arg(sz_arg);
300  alloc_res(sz_res);
301  alloc_iw(sz_iw);
302  alloc_w(sz_w);
303 }

388 Function External
389 ::get_forward(casadi_int nfwd, const std::string& name,
390  const std::vector<std::string>& inames,
391  const std::vector<std::string>& onames,
392  const Dict& opts) const {
393  // Consistency check
394  casadi_int n=1;
395  while (n<nfwd) n*=2;
396  if (n!=nfwd || !has_forward(nfwd)) {
397  // Inefficient code to be replaced later
398  Function fwd1 = forward(1);
399  return fwd1.map(name, "serial", nfwd, range(n_in_+n_out_), std::vector<casadi_int>(), opts);
400  //casadi_error("Internal error: Refactoring needed, cf. #1055");
401  }
402  return external(name, li_, opts);
403 }

430 Function External::factory(const std::string& name,
431  const std::vector<std::string>& s_in,
432  const std::vector<std::string>& s_out,
433  const Function::AuxOut& aux,
434  const Dict& opts) const {
435  // If not available, call base class function
436  if (!li_.has_function(name)) {
437  return FunctionInternal::factory(name, s_in, s_out, aux, opts);
438  }
439 
440  // Retrieve function
441  Function ret = external(name, li_, opts);
442 
443  // Replace colons in input names
444  std::vector<std::string> s_io(s_in);
445  for (std::string& s : s_io) replace(s.begin(), s.end(), ':', '_');
446 
447  // Inputs consistency checks
448  casadi_assert(s_in.size() == ret.n_in(),
449  "Inconsistent number of inputs. Expected " + str(s_in.size())+ " "
450  "(" + str(s_io) + "), got " + str(ret.n_in()) + ".");
451  for (size_t i = 0; i < s_in.size(); ++i) {
452  casadi_assert(s_io[i] == ret.name_in(i),
453  "Inconsistent input name. Expected: " + str(s_io) + ", "
454  "got " + ret.name_in(i) + " for input " + str(i));
455  }
456 
457  // Replace colons in output names
458  s_io = s_out;
459  for (std::string& s : s_io) replace(s.begin(), s.end(), ':', '_');
460 
461  // Outputs consistency checks
462  casadi_assert(s_out.size() == ret.n_out(),
463  "Inconsistent number of outputs. Expected " + str(s_out.size()) + " "
464  "(" + str(s_io) + "), got " + str(ret.n_out()) + ".");
465  for (size_t i = 0; i < s_out.size(); ++i) {
466  casadi_assert(s_io[i] == ret.name_out(i),
467  "Inconsistent output name. Expected: " + str(s_io) + ", "
468  "got " + ret.name_out(i) + " for output " + str(i));
469  }
470 
471  return ret;
472 }
