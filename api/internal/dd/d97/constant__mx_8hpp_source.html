602  MX Constant<Value>::get_horzcat(const std::vector<MX>& x) const {
603  // Check if all arguments have the same constant value
604  for (auto&& i : x) {
605  if (!i->is_value(to_double())) {
606  // Not all the same value, fall back to base class
607  return ConstantMX::get_horzcat(x);
608  }
609  }
610 
611  // Assemble the sparsity pattern
612  std::vector<Sparsity> sp;
613  for (auto&& i : x) sp.push_back(i.sparsity());
614  return MX(horzcat(sp), v_.value, false);

643  template<typename Value>
644  MX Constant<Value>::get_unary(casadi_int op) const {
645  // Constant folding
646  double ret(0);
647  casadi_math<double>::fun(op, to_double(), 0.0, ret);
648  if (operation_checker<F0XChecker>(op) || sparsity().is_dense()) {
649  return MX(sparsity(), ret);
650  } else {
651  if (v_.value==0) {
652  if (is_zero() && operation_checker<F0XChecker>(op)) {
653  return MX(sparsity(), ret, false);
654  } else {
655  return repmat(MX(ret), size1(), size2());
656  }
657  }
658  double ret2;
659  casadi_math<double>::fun(op, 0, 0.0, ret2);
660  return DM(sparsity(), ret, false)
661  + DM(sparsity().pattern_inverse(), ret2, false);
662  }
663  }

665  template<typename Value>
666  MX Constant<Value>::_get_binary(casadi_int op, const MX& y, bool ScX, bool ScY) const {
667  casadi_assert_dev(sparsity()==y.sparsity() || ScX || ScY);
668 
669  if (ScX && !operation_checker<FX0Checker>(op)) {
670  double ret;
671  casadi_math<double>::fun(op, nnz()> 0 ? to_double(): 0.0, 0, ret);
672 
673  if (ret!=0) {
674  Sparsity f = Sparsity::dense(y.size1(), y.size2());
675  MX yy = project(y, f);
676  return MX(f, shared_from_this<MX>())->_get_binary(op, yy, false, false);
677  }
678  } else if (ScY && !operation_checker<F0XChecker>(op)) {
679  bool grow = true;
680  if (y->op()==OP_CONST && dynamic_cast<const ConstantDM*>(y.get())==nullptr) {
681  double ret;
682  casadi_math<double>::fun(op, 0, y.nnz()>0 ? y->to_double() : 0, ret);
683  grow = ret!=0;
684  }
685  if (grow) {
686  Sparsity f = Sparsity::dense(size1(), size2());
687  MX xx = project(shared_from_this<MX>(), f);
688  return xx->_get_binary(op, MX(f, y), false, false);
689  }
690  }
691 
692  switch (op) {
693  case OP_ADD:
694  if (v_.value==0) return ScY && !y->is_zero() ? repmat(y, size1(), size2()) : y;
695  break;
696  case OP_SUB:
697  if (v_.value==0) return ScY && !y->is_zero() ? repmat(-y, size1(), size2()) : -y;
698  break;
699  case OP_MUL:
700  if (v_.value==1) return y;
701  if (v_.value==-1) return -y;
702  if (v_.value==2) return y->get_unary(OP_TWICE);
703  break;
704  case OP_DIV:
705  if (v_.value==1) return y->get_unary(OP_INV);
706  if (v_.value==-1) return -y->get_unary(OP_INV);
707  break;
708  case OP_POW:
709  if (v_.value==0) return MX::zeros(y.sparsity());
710  if (v_.value==1) return MX::ones(y.sparsity());
711  if (v_.value==std::exp(1.0)) return y->get_unary(OP_EXP);
712  break;
713  default: break; //no rule
714  }
715 
716  // Constant folding
717  // NOTE: ugly, should use a function instead of a cast
718  if (y->op()==OP_CONST && dynamic_cast<const ConstantDM*>(y.get())==nullptr) {
719  double y_value = y.nnz()>0 ? y->to_double() : 0;
720  double ret;
721  casadi_math<double>::fun(op, nnz()> 0.0 ? to_double(): 0, y_value, ret);
722 
723  return MX(y.sparsity(), ret, false);
724  }
725 
726  // Fallback
727  return MXNode::_get_binary(op, y, ScX, ScY);
728  }
