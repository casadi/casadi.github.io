432 int FmuInternal::eval_derivative(FmuMemory* m, bool independent_seeds) const {
433  // Gather input and output indices
434  gather_sens(m);
435  // Calculate derivatives using FMU directional derivative support
436  if (m->self.enable_ad_) {
437  // Evaluate using AD
438  if (eval_ad(m)) return 1;
439  }
440  // Calculate derivatives using finite differences
441  if (!m->self.enable_ad_ || m->self.validate_ad_) {
442  // Evaluate using FD
443  if (eval_fd(m, independent_seeds)) return 1;
444  }
445  return 0;
446 }

448 void FmuInternal::set(FmuMemory* m, size_t ind, const double* value) const {
449  if (value) {
450  // Argument is given
451  for (size_t id : ired_[ind]) {
452  if (*value != m->ibuf_.at(id)) {
453  m->ibuf_.at(id) = *value;
454  m->changed_.at(id) = true;
455  }
456  value++;
457  }
458  } else {
459  // Argument is null - all zeros
460  for (size_t id : ired_[ind]) {
461  if (0 != m->ibuf_.at(id)) {
462  m->ibuf_.at(id) = 0;
463  m->changed_.at(id) = true;
464  }
465  }
466  }
467 }

538 void FmuInternal::gather_io(FmuMemory* m) const {
539  // Collect input indices and corresponding value references and values
540  m->id_in_.clear();
541  m->vr_in_.clear();
542  m->v_in_.clear();
543  for (size_t id = 0; id < m->changed_.size(); ++id) {
544  if (m->changed_[id]) {
545  m->id_in_.push_back(id);
546  m->vr_in_.push_back(vr_in_[id]);
547  m->v_in_.push_back(m->ibuf_[id]);
548  m->changed_[id] = false;
549  }
550  }
551  // Collect output indices, corresponding value references
552  m->id_out_.clear();
553  m->vr_out_.clear();
554  for (size_t id = 0; id < m->requested_.size(); ++id) {
555  if (m->requested_[id]) {
556  m->id_out_.push_back(id);
557  m->vr_out_.push_back(vr_out_[id]);
558  m->requested_[id] = false;
559  }
560  }
561 }

563 void FmuInternal::gather_sens(FmuMemory* m) const {
564  // Gather input and output indices
565  gather_io(m);
566  // Number of inputs and outputs
567  size_t n_known = m->id_in_.size();
568  size_t n_unknown = m->id_out_.size();
569  // Get/clear seeds
570  m->d_in_.clear();
571  for (size_t id : m->id_in_) {
572  m->d_in_.push_back(m->seed_[id]);
573  m->seed_[id] = 0;
574  }
575  // Ensure at least one seed
576  casadi_assert(n_known != 0, "No seeds");
577  // Allocate result vectors
578  m->v_out_.resize(n_unknown);
579  m->d_out_.resize(n_unknown);
580 }
