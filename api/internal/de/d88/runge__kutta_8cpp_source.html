64  void RungeKutta::setup_step() {
65  // Continuous-time dynamics, forward problem
66  Function f = get_function("dae");
67 
68  // Symbolic inputs
69  MX t0 = MX::sym("t0", f.sparsity_in(DYN_T));
70  MX h = MX::sym("h");
71  MX x0 = MX::sym("x0", f.sparsity_in(DYN_X));
72  MX p = MX::sym("p", f.sparsity_in(DYN_P));
73  MX u = MX::sym("u", f.sparsity_in(DYN_U));
74 
75  // Intermediate variables (does not enter in F_, only in G_)
76  MX v = MX::sym("v", x0.size1(), x0.size2() * 3);
77  std::vector<MX> x = horzsplit(v, x0.size2());
78  casadi_assert_dev(x.size() == 3);
79 
80  // Definitions of x
81  std::vector<MX> x_def(3);
82 
83  // Time points
84  std::vector<MX> tt(3);
85 
86  // Half a step, 6-th of a step
87  MX h_half = h / 2, h_sixth = h / 6;
88 
89  // Forward integration
90  {
91  // Arguments when calling f
92  std::vector<MX> f_arg(DYN_NUM_IN);
93  std::vector<MX> f_res;
94  f_arg[DYN_P] = p;
95  f_arg[DYN_U] = u;
96 
97  // k1
98  f_arg[DYN_T] = t0;
99  f_arg[DYN_X] = x0;
100  f_res = f(f_arg);
101  MX k1 = f_res[DYN_ODE];
102  MX k1q = f_res[DYN_QUAD];
103 
104  // k2
105  tt[0] = f_arg[DYN_T] = t0 + h_half;
106  x_def[0] = f_arg[DYN_X] = x0 + h_half * k1;
107  f_res = f(f_arg);
108  MX k2 = f_res[DYN_ODE];
109  MX k2q = f_res[DYN_QUAD];
110 
111  // k3
112  tt[1] = tt[0];
113  x_def[1] = f_arg[DYN_X] = x0 + h_half * k2;
114  f_res = f(f_arg);
115  MX k3 = f_res[DYN_ODE];
116  MX k3q = f_res[DYN_QUAD];
117 
118  // k4
119  tt[2] = f_arg[DYN_T] = t0 + h;
120  x_def[2] = f_arg[DYN_X] = x0 + h * k3;
121  f_res = f(f_arg);
122  MX k4 = f_res[DYN_ODE];
123  MX k4q = f_res[DYN_QUAD];
124 
125  // Take step
126  MX xf = x0 + h_sixth * (k1 + 2*k2 + 2*k3 + k4);
127  MX qf = h_sixth * (k1q + 2*k2q + 2*k3q + k4q);
128 
129  // Define discrete time dynamics
130  f_arg.resize(FSTEP_NUM_IN);
131  f_arg[FSTEP_T] = t0;
132  f_arg[FSTEP_H] = h;
133  f_arg[FSTEP_X0] = x0;
134  f_arg[FSTEP_V0] = v;
135  f_arg[FSTEP_P] = p;
136  f_arg[FSTEP_U] = u;
137  f_res.resize(FSTEP_NUM_OUT);
138  f_res[FSTEP_XF] = xf;
139  f_res[FSTEP_QF] = qf;
140  f_res[FSTEP_VF] = horzcat(x_def);
141  Function F("stepF", f_arg, f_res,
142  {"t", "h", "x0", "v0", "p", "u"}, {"xf", "vf", "qf"});
143  set_function(F, F.name(), true);
144  if (nfwd_ > 0) create_forward("stepF", nfwd_);
145  }
146 
147  // Backward integration
148  if (nadj_ > 0) {
149  // Continuous-time dynamics, backward problem
150  Function g = get_function("rdae");
151 
152  // Symbolic inputs
153  MX rx0 = MX::sym("rx0", g.sparsity_in(BDYN_ADJ_ODE));
154  MX rp = MX::sym("rp", g.sparsity_in(BDYN_ADJ_QUAD));
155 
156  // Intermediate variables (do not enter in G_)
157  MX rv = MX::sym("rv", rx0.size1(), 3 * rx0.size2());
158  std::vector<MX> rx_def(3);
159 
160  // Arguments when calling g
161  std::vector<MX> g_arg(BDYN_NUM_IN);
162  std::vector<MX> g_res;
163  g_arg[BDYN_P] = p;
164  g_arg[BDYN_U] = u;
165  g_arg[BDYN_ADJ_QUAD] = rp;
166 
167  // k1
168  g_arg[BDYN_T] = tt[2];
169  g_arg[BDYN_X] = x[2];
170  g_arg[BDYN_ADJ_ODE] = rx0;
171  g_res = g(g_arg);
172  MX k1 = g_res[BDYN_ADJ_X];
173  MX k1rq = g_res[BDYN_ADJ_P];
174  MX k1uq = g_res[BDYN_ADJ_U];
175 
176  // k2
177  g_arg[BDYN_T] = tt[1];
178  g_arg[BDYN_X] = x[1];
179  g_arg[BDYN_ADJ_ODE] = rx_def[2] = rx0 + h_half * k1;
180  g_res = g(g_arg);
181  MX k2 = g_res[BDYN_ADJ_X];
182  MX k2rq = g_res[BDYN_ADJ_P];
183  MX k2uq = g_res[BDYN_ADJ_U];
184 
185  // k3
186  g_arg[BDYN_T] = tt[0];
187  g_arg[BDYN_X] = x[0];
188  g_arg[BDYN_ADJ_ODE] = rx_def[1] = rx0 + h_half * k2;
189  g_res = g(g_arg);
190  MX k3 = g_res[BDYN_ADJ_X];
191  MX k3rq = g_res[BDYN_ADJ_P];
192  MX k3uq = g_res[BDYN_ADJ_U];
193 
194  // k4
195  g_arg[BDYN_T] = t0;
196  g_arg[BDYN_X] = x0;
197  g_arg[BDYN_ADJ_ODE] = rx_def[0] = rx0 + h * k3;
198  g_res = g(g_arg);
199  MX k4 = g_res[BDYN_ADJ_X];
200  MX k4rq = g_res[BDYN_ADJ_P];
201  MX k4uq = g_res[BDYN_ADJ_U];
202 
203  // Take step
204  MX rxf = rx0 + h_sixth * (k1 + 2*k2 + 2*k3 + k4);
205  MX rqf = h_sixth * (k1rq + 2*k2rq + 2*k3rq + k4rq);
206  MX uqf = h_sixth * (k1uq + 2*k2uq + 2*k3uq + k4uq);
207 
208  // Define discrete time dynamics
209  g_arg.resize(BSTEP_NUM_IN);
210  g_arg[BSTEP_T] = t0;
211  g_arg[BSTEP_H] = h;
212  g_arg[BSTEP_X] = x0;
213  g_arg[BSTEP_P] = p;
214  g_arg[BSTEP_U] = u;
215  g_arg[BSTEP_V] = v;
216  g_arg[BSTEP_RX0] = rx0;
217  g_arg[BSTEP_RP] = rp;
218  g_arg[BSTEP_RV0] = rv;
219  g_res.resize(BSTEP_NUM_OUT);
220  g_res[BSTEP_RXF] = rxf;
221  g_res[BSTEP_RVF] = horzcat(rx_def);
222  g_res[BSTEP_RQF] = rqf;
223  g_res[BSTEP_UQF] = uqf;
224  Function G("stepB", g_arg, g_res,
225  {"t", "h", "rx0", "rv0", "rp", "x", "v", "p", "u"},
226  {"rxf", "rvf", "rqf", "uqf"});
227  set_function(G, G.name(), true);
228  if (nfwd_ > 0) create_forward("stepB", nfwd_);
229  }
230  }
