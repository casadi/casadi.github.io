54  bool GenericType::can_cast_to(TypeID other) const {
55  switch (other) {
56  case OT_BOOL:
57  return is_bool() || is_int() || is_double();
58  case OT_BOOLVECTOR:
59  return is_int_vector() || is_double_vector();
60  case OT_INT:
61  case OT_DOUBLE:
62  return is_int() || is_double();
63  case OT_INTVECTOR:
64  case OT_DOUBLEVECTOR:
65  return is_double_vector() || is_int_vector();
66  case OT_INTVECTORVECTOR:
67  case OT_DOUBLEVECTORVECTOR:
68  return is_double_vector_vector() || is_int_vector_vector();
69  case OT_STRINGVECTOR:
70  return is_string_vector() || is_string() || is_double_vector() || is_int_vector();
71  case OT_VOIDPTR:
72  return is_void_pointer() || is_int();
73  default:
74  return getType() == other;
75  }
76  }

470  bool GenericType::operator!=(const GenericType& op2) const {
471  if (is_string() && op2.is_string()) {
472  return to_string() != op2.to_string();
473  }
474 
475  if (is_int() && op2.is_int()) {
476  return to_int() != op2.to_int();
477  }
478 
479  if (is_double() && op2.is_double()) {
480  return to_double() != op2.to_double();
481  }
482 
483  if (is_double_vector() && op2.is_double_vector()) {
484  const std::vector<double> &v1 = to_double_vector();
485  const std::vector<double> &v2 = op2.to_double_vector();
486  if (v1.size() != v2.size()) return true;
487  for (casadi_int i=0; i<v1.size(); ++i)
488  if (v1[i] != v2[i]) return true;
489  return false;
490  }
491 
492  if (is_int_vector() && op2.is_int_vector()) {
493  const std::vector<casadi_int> &v1 = to_int_vector();
494  const std::vector<casadi_int> &v2 = op2.to_int_vector();
495  if (v1.size() != v2.size()) return true;
496  for (casadi_int i=0; i<v1.size(); ++i)
497  if (v1[i] != v2[i]) return true;
498  return false;
499  }
500 
501  if (is_int_vector_vector() && op2.is_int_vector_vector()) {
502  const std::vector< std::vector<casadi_int> > &v1 = to_int_vector_vector();
503  const std::vector< std::vector<casadi_int> > &v2 = op2.to_int_vector_vector();
504  if (v1.size() != v2.size()) return true;
505  for (casadi_int i=0; i<v1.size(); ++i) {
506  if (v1[i].size() != v2[i].size()) return true;
507  for (casadi_int j=0; j<v1[i].size(); ++j) {
508  if (v1[i][j] != v2[i][j]) return true;
509  }
510  }
511  return false;
512  }
513 
514  if (is_double_vector_vector() && op2.is_double_vector_vector()) {
515  const std::vector< std::vector<double> > &v1 = to_double_vector_vector();
516  const std::vector< std::vector<double> > &v2 = op2.to_double_vector_vector();
517  if (v1.size() != v2.size()) return true;
518  for (casadi_int i=0; i<v1.size(); ++i) {
519  if (v1[i].size() != v2[i].size()) return true;
520  for (casadi_int j=0; j<v1[i].size(); ++j) {
521  if (v1[i][j] != v2[i][j]) return true;
522  }
523  }
524  return false;
525  }
526 
527  // Different types
528  return true;
529  }

593  Dict combine(const Dict& first, const Dict& second, bool recurse) {
594  if (first.empty()) return second;
595  if (second.empty()) return first;
596  Dict ret = second;
597  update_dict(ret, first, recurse);
598  return ret;
599  }

601  void update_dict(Dict& target, const Dict& source, bool recurse) {
602  for (auto&& e : source) {
603  if (recurse) {
604  auto it = target.find(e.first);
605  if (it!=target.end() && it->second.is_dict()) {
606  Dict local = it->second;
607  update_dict(local, e.second, recurse);
608  it->second = local;
609  continue;
610  }
611  }
612  target[e.first] = e.second;
613  }
614  }
