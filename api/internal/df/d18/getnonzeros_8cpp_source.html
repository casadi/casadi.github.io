32  MX GetNonzeros::create(const Sparsity& sp, const MX& x, const std::vector<casadi_int>& nz) {
33  // No elements at all
34  if (nz.empty()) return MX::zeros(sp);
35  // Simplify to slice
36  if (is_slice(nz)) return create(sp, x, to_slice(nz));
37  // Simplify to slice2
38  if (is_slice2(nz)) {
39  std::pair<Slice, Slice> sl = to_slice2(nz);
40  return create(sp, x, sl.first, sl.second);
41  }
42  return MX::create(new GetNonzerosVector(sp, x, nz));
43  }

124  int GetNonzerosSlice2::
125  eval_gen(const T* const* arg, T* const* res, casadi_int* iw, T* w) const {
126  const T* outer = arg[0] + outer_.start;
127  const T* outer_stop = arg[0] + outer_.stop;
128  T* odata = res[0];
129  for (; outer != outer_stop; outer += outer_.step) {
130  for (const T* inner = outer+inner_.start;
131  inner != outer+inner_.stop;
132  inner += inner_.step) {
133  *odata++ = *inner;
134  }
135  }
136  return 0;

222  void GetNonzeros::eval_mx(const std::vector<MX>& arg, std::vector<MX>& res) const {
223  // Get all the nonzeros
224  std::vector<casadi_int> nz = all();
225 
226  // Output sparsity
227  const Sparsity& osp = sparsity();
228  const casadi_int* orow = osp.row();
229  std::vector<casadi_int> ocol = osp.get_col();
230 
231  // Input sparsity
232  const Sparsity& isp = dep().sparsity();
233  //const std::vector<casadi_int>& irow = isp.row();
234  std::vector<casadi_int> icol = isp.get_col();
235 
236  // Get all input elements
237  std::vector<casadi_int> el_input;
238  isp.find(el_input);
239 
240  // Sparsity pattern being formed and corresponding nonzero mapping
241  std::vector<casadi_int> r_colind, r_row, r_nz, r_ind;
242 
243  // Get the matching nonzeros
244  r_ind.resize(el_input.size());
245  std::copy(el_input.begin(), el_input.end(), r_ind.begin());
246  arg[0].sparsity().get_nz(r_ind);
247 
248  // Sparsity pattern for the result
249  r_colind.resize(osp.size2()+1); // Col count
250  std::fill(r_colind.begin(), r_colind.end(), 0);
251  r_row.clear();
252 
253  // Perform the assignments
254  r_nz.clear();
255  for (casadi_int k=0; k<nz.size(); ++k) {
256 
257  // Get the corresponding nonzero for the input
258  casadi_int el = nz[k];
259 
260  // Skip if zero assignment
261  if (el==-1) continue;
262 
263  // Get the corresponding nonzero in the argument
264  casadi_int el_arg = r_ind[el];
265 
266  // Skip if no argument
267  if (el_arg==-1) continue;
268 
269  // Save the assignment
270  r_nz.push_back(el_arg);
271 
272  // Get the corresponding element
273  casadi_int i=ocol[k], j=orow[k];
274 
275  // Add to sparsity pattern
276  r_row.push_back(j);
277  r_colind[1+i]++;
278  }
279 
280  // col count -> col offset
281  for (casadi_int i=1; i<r_colind.size(); ++i) r_colind[i] += r_colind[i-1];
282 
283  // Create a sparsity pattern from vectors
284  if (r_nz.empty()) {
285  res[0] = MX(osp.size());
286  } else {
287  Sparsity f_sp(osp.size1(), osp.size2(), r_colind, r_row);
288  res[0] = arg[0]->get_nzref(f_sp, r_nz);
289  }
290  }

292  void GetNonzeros::ad_forward(const std::vector<std::vector<MX> >& fseed,
293  std::vector<std::vector<MX> >& fsens) const {
294 
295  // Get all the nonzeros
296  std::vector<casadi_int> nz = all();
297 
298  // Number of derivative directions
299  casadi_int nfwd = fsens.size();
300 
301  // Output sparsity
302  const Sparsity& osp = sparsity();
303  const casadi_int* orow = osp.row();
304  std::vector<casadi_int> ocol = osp.get_col();
305 
306  // Input sparsity
307  const Sparsity& isp = dep().sparsity();
308  //const std::vector<casadi_int>& irow = isp.row();
309  std::vector<casadi_int> icol;
310 
311  // Get all input elements
312  std::vector<casadi_int> el_input;
313 
314  // Sparsity pattern being formed and corresponding nonzero mapping
315  std::vector<casadi_int> r_colind, r_row, r_nz, r_ind;
316 
317  // Nondifferentiated function and forward sensitivities
318  for (casadi_int d=0; d<nfwd; ++d) {
319 
320  // Get references to arguments and results
321  const MX& arg = fseed[d][0];
322  MX& res = fsens[d][0];
323 
324  if (arg.sparsity()==isp) { // Matching sparsity
325  if (nz.empty()) {
326  res = MX(osp.size());
327  } else {
328  res = arg->get_nzref(osp, nz);
329  }
330  } else {
331  // Expensive operations
332  if (el_input.empty()) isp.find(el_input);
333  if (icol.empty()) icol = isp.get_col();
334 
335  // Get the matching nonzeros
336  r_ind.resize(el_input.size());
337  std::copy(el_input.begin(), el_input.end(), r_ind.begin());
338  arg.sparsity().get_nz(r_ind);
339 
340  // Sparsity pattern for the result
341  r_colind.resize(osp.size2()+1); // Col count
342  std::fill(r_colind.begin(), r_colind.end(), 0);
343  r_row.clear();
344 
345  // Perform the assignments
346  r_nz.clear();
347  for (casadi_int k=0; k<nz.size(); ++k) {
348 
349  // Get the corresponding nonzero for the input
350  casadi_int el = nz[k];
351 
352  // Skip if zero assignment
353  if (el==-1) continue;
354 
355  // Get the corresponding nonzero in the argument
356  casadi_int el_arg = r_ind[el];
357 
358  // Skip if no argument
359  if (el_arg==-1) continue;
360 
361  // Save the assignment
362  r_nz.push_back(el_arg);
363 
364  // Get the corresponding element
365  casadi_int i=ocol[k], j=orow[k];
366 
367  // Add to sparsity pattern
368  r_row.push_back(j);
369  r_colind[1+i]++;
370  }
371 
372  // col count -> col offset
373  for (casadi_int i=1; i<r_colind.size(); ++i) r_colind[i] += r_colind[i-1];
374 
375  // Create a sparsity pattern from vectors
376  if (r_nz.empty()) {
377  res = MX(osp.size());
378  } else {
379  Sparsity f_sp(osp.size1(), osp.size2(), r_colind, r_row);
380  res = arg->get_nzref(f_sp, r_nz);
381  }
382  }
383  }

386  void GetNonzeros::ad_reverse(const std::vector<std::vector<MX> >& aseed,
387  std::vector<std::vector<MX> >& asens) const {
388  // Get all the nonzeros
389  std::vector<casadi_int> nz = all();
390 
391  // Number of derivative directions
392  casadi_int nadj = aseed.size();
393 
394  // Output sparsity
395  const Sparsity& osp = sparsity();
396  std::vector<casadi_int> ocol;
397 
398  // Input sparsity
399  const Sparsity& isp = dep().sparsity();
400  //const std::vector<casadi_int>& irow = isp.row();
401  std::vector<casadi_int> icol;
402 
403  // Get all input elements
404  std::vector<casadi_int> el_input;
405 
406  // Sparsity pattern being formed and corresponding nonzero mapping
407  std::vector<casadi_int> r_colind, r_row, r_nz, r_ind;
408 
409  // Adjoint sensitivities
410  for (casadi_int d=0; d<nadj; ++d) {
411 
412  // Get an owning references to the seeds and sensitivities
413  // and clear the seeds for the next run
414  MX aseed0 = aseed[d][0];
415  MX asens0 = asens[d][0]; // Sensitivity before addition
416 
417  if (aseed0.sparsity()==osp && asens0.sparsity().nnz()==0) { // Matching sparsity
418  asens[d][0] = aseed0->get_nzadd(DM::zeros(isp), nz);
419  } else {
420  // Expensive operations
421  if (el_input.empty()) isp.find(el_input);
422  if (icol.empty()) icol = isp.get_col();
423  if (ocol.empty()) ocol = osp.get_col();
424 
425  // Get the corresponding nz locations in the output sparsity pattern
426  aseed0.sparsity().find(r_nz);
427  osp.get_nz(r_nz);
428 
429  // Filter out ignored entries and check if there is anything to add at all
430  bool elements_to_add = false;
431  for (std::vector<casadi_int>::iterator k=r_nz.begin(); k!=r_nz.end(); ++k) {
432  if (*k>=0) {
433  if (nz[*k]>=0) {
434  elements_to_add = true;
435  } else {
436  *k = -1;
437  }
438  }
439  }
440 
441  // Quick continue of no elements to add
442  if (!elements_to_add) continue;
443 
444  // Get the nz locations in the adjoint sensitivity corresponding to the inputs
445  r_ind.resize(el_input.size());
446  std::copy(el_input.begin(), el_input.end(), r_ind.begin());
447  asens0.sparsity().get_nz(r_ind);
448 
449  // Enlarge the sparsity pattern of the sensitivity if not all additions fit
450  for (std::vector<casadi_int>::iterator k=r_nz.begin(); k!=r_nz.end(); ++k) {
451  if (*k>=0 && r_ind[nz[*k]]<0) {
452 
453  // Create a new pattern which includes both the the previous seed and the addition
454  Sparsity sp = asens0.sparsity().unite(dep().sparsity());
455  asens0 = asens0->get_project(sp);
456 
457  // Recalculate the nz locations in the adjoint sensitivity corresponding to the inputs
458  std::copy(el_input.begin(), el_input.end(), r_ind.begin());
459  asens0.sparsity().get_nz(r_ind);
460 
461  break;
462  }
463  }
464 
465  // Have r_nz point to locations in the sensitivity instead of the output
466  for (std::vector<casadi_int>::iterator k=r_nz.begin(); k!=r_nz.end(); ++k) {
467  if (*k>=0) {
468  *k = r_ind[nz[*k]];
469  }
470  }
471 
472  asens[d][0] = aseed0->get_nzadd(asens0, r_nz);
473  }
474  }
475  }

483  const std::vector<casadi_int>& arg,
484  const std::vector<casadi_int>& res) const {
485  // Codegen the indices
486  std::string ind = g.constant(nz_);
487 
488  // Codegen the assignments
489  g.local("cii", "const casadi_int", "*");
490  g.local("rr", "casadi_real", "*");
491  g.local("ss", "casadi_real", "*");
492  g << "for (cii=" << ind << ", rr=" << g.work(res[0], nnz())
493  << ", ss=" << g.work(arg[0], dep(0).nnz())
494  << "; cii!=" << ind << "+" << nz_.size()
495  << "; ++cii) *rr++ = ";
496  if (has_negative(nz_)) {
497  g << "*cii>=0 ? ss[*cii] : 0;\n";
498  } else {
499  g << "ss[*cii];\n";
500  }

503  MX GetNonzeros::get_nzref(const Sparsity& sp, const std::vector<casadi_int>& nz) const {
504  // Get all the nonzeros
505  std::vector<casadi_int> nz_all = all();
506 
507  // Eliminate recursive calls
508  std::vector<casadi_int> nz_new(nz);
509  for (std::vector<casadi_int>::iterator i=nz_new.begin(); i!=nz_new.end(); ++i) {
510  if (*i>=0) *i = nz_all[*i];
511  }
512  return dep()->get_nzref(sp, nz_new);
513  }

538  bool GetNonzerosVector::is_equal(const MXNode* node, casadi_int depth) const {
