64  void Project::ad_forward(const std::vector<std::vector<MX> >& fseed,
65  std::vector<std::vector<MX> >& fsens) const {
66  casadi_int nfwd = fsens.size();
67  for (casadi_int d=0; d<nfwd; ++d) {
68  fsens[d][0] = project(fseed[d][0], sparsity() * dep().sparsity(), true);
69  }
70  }

113  MXNode* Project::deserialize(DeserializingStream& s) {
114  char t;
115  s.unpack("Project::type", t);
116  switch (t) {
117  case 'n':
118  return new Project(s);
119  case 'd':
120  return new Densify(s);
121  case 's':
122  return new Sparsify(s);
123  default:
124  casadi_assert_dev(false);
125  }
126  }
