106  template<typename XType>
107  Function Rootfinder::create_oracle(const std::map<std::string, XType>& d,
108  const Dict& opts) {
109  std::vector<XType> rfp_in(RFP_NUM_IN), rfp_out(RFP_NUM_OUT);
110  for (auto&& i : d) {
111  if (i.first=="x") {
112  rfp_in[RFP_X]=i.second;
113  } else if (i.first=="p") {
114  rfp_in[RFP_P]=i.second;
115  } else if (i.first=="g") {
116  rfp_out[RFP_G]=i.second;
117  } else {
118  casadi_error("No such field: " + i.first);
119  }
120  }
121 
122  // Options for the oracle
123  Dict oracle_options;
124  Dict::const_iterator it = opts.find("oracle_options");
125  if (it!=opts.end()) {
126  // "oracle_options" has been set
127  oracle_options = it->second;
128  } else if ((it=opts.find("verbose")) != opts.end()) {
129  // "oracle_options" has not been set, but "verbose" has
130  oracle_options["verbose"] = it->second;
131  }
132 
133  // Create oracle
134  return Function("rfp", rfp_in, rfp_out, {"x0", "p"}, {"x"}, oracle_options);
135  }

182  void Rootfinder::init(const Dict& opts) {
183 
184  // Default (temporary) options
185  Dict linear_solver_options;
186  std::string linear_solver = "qr";
187  Function jac; // Jacobian of f with respect to z
188 
189  // Read options
190  for (auto&& op : opts) {
191  if (op.first=="implicit_input") {
192  iin_ = op.second;
193  } else if (op.first=="implicit_output") {
194  iout_ = op.second;
195  } else if (op.first=="jacobian_function") {
196  jac = op.second;
197  } else if (op.first=="linear_solver_options") {
198  linear_solver_options = op.second;
199  } else if (op.first=="linear_solver") {
200  linear_solver = op.second.to_string();
201  } else if (op.first=="constraints") {
202  u_c_ = op.second;
203  }
204  }
205 
206  // Get the number of equations and check consistency
207  casadi_assert(iin_>=0 && iin_<oracle_.n_in() && oracle_.n_in()>0,
208  "Implicit input not in range");
209  casadi_assert(iout_>=0 && iout_<oracle_.n_out() && oracle_.n_out()>0,
210  "Implicit output not in range");
211  casadi_assert(oracle_.sparsity_out(iout_).is_dense()
212  && oracle_.sparsity_out(iout_).is_column(),
213  "Residual must be a dense vector");
214  casadi_assert(oracle_.sparsity_in(iin_).is_dense()
215  && oracle_.sparsity_in(iin_).is_column(),
216  "Unknown must be a dense vector");
217  n_ = oracle_.nnz_out(iout_);
218  casadi_assert(n_ == oracle_.nnz_in(iin_),
219  "Dimension mismatch. Input size is " + str(oracle_.nnz_in(iin_)) + ", "
220  "while output size is " + str(oracle_.nnz_out(iout_)));
221 
222  // Call the base class initializer
223  OracleFunction::init(opts);
224 
225  // Generate Jacobian if not provided
226  if (jac.is_null()) {
227  std::vector<std::string> s_in = oracle_.name_in();
228  std::vector<std::string> s_out = oracle_.name_out();
229  s_out.insert(s_out.begin(), "jac:" + oracle_.name_out(iout_) + ":" + oracle_.name_in(iin_));
230  jac = oracle_.factory(oracle_.name() + "_jac", s_in, s_out);
231  }
232  set_function(jac, "jac_f_z");
233  sp_jac_ = jac.sparsity_out(0);
234  // Check for structural singularity in the Jacobian
235  casadi_assert(!sp_jac_.is_singular(),
236  "Rootfinder::init: singularity - the jacobian is structurally rank-deficient. "
237  "sprank(J)=" + str(sprank(sp_jac_)) + " (instead of " + str(sp_jac_.size1()) + ")");
238 
239  // Get the linear solver creator function
240  linsol_ = Linsol("linsol", linear_solver, sp_jac_, linear_solver_options);
241 
242  // Constraints
243  casadi_assert(u_c_.size()==n_ || u_c_.empty(),
244  "Constraint vector if supplied, must be of length n, but got "
245  + str(u_c_.size()) + " and n = " + str(n_));
246 
247  // Allocate sufficiently large work vectors
248  alloc(oracle_);
249  size_t sz_w = oracle_.sz_w();
250  if (!jac.is_null()) {
251  sz_w = std::max(sz_w, jac.sz_w());
252  }
253  alloc_w(sz_w + 2*static_cast<size_t>(n_));
254  }

268  int Rootfinder::eval(const double** arg, double** res,
269  casadi_int* iw, double* w, void* mem) const {
270  // Reset the solver, prepare for solution
271  setup(mem, arg, res, iw, w);
272 
273  // Solve the NLP
274  int ret = solve(mem);
275  auto m = static_cast<RootfinderMemory*>(mem);
276  if (error_on_fail_ && !m->success)
277  casadi_error("rootfinder process failed. "
278  "Set 'error_on_fail' option to false to ignore this error.");
279 
280  return ret;
281  }

283  void Rootfinder::set_work(void* mem, const double**& arg, double**& res,
284  casadi_int*& iw, double*& w) const {
285  auto m = static_cast<RootfinderMemory*>(mem);
286 
287  // Problem has not been solved at this point
288  m->success = false;
289  m->unified_return_status = SOLVER_RET_UNKNOWN;

300  Function Rootfinder
301  ::get_forward(casadi_int nfwd, const std::string& name,
302  const std::vector<std::string>& inames,
303  const std::vector<std::string>& onames,
304  const Dict& opts) const {
305  // Symbolic expression for the input
306  std::vector<MX> arg = mx_in(), res = mx_out();
307  std::vector<std::vector<MX>> fseed = fwd_seed<MX>(nfwd), fsens;
308  arg[iin_] = MX::sym(arg[iin_].name(), Sparsity(arg[iin_].size()));
309  for (auto&& e : fseed) e[iin_] = MX::sym(e[iin_].name(), e[iin_].size());
310  ad_forward(arg, res, fseed, fsens, false, false);
311 
312  // Construct return function
313  arg.insert(arg.end(), res.begin(), res.end());
314  std::vector<MX> v(nfwd);
315  for (casadi_int i=0; i<n_in_; ++i) {
316  for (casadi_int d=0; d<nfwd; ++d) v[d] = fseed[d][i];
317  arg.push_back(horzcat(v));
318  }
319  res.clear();
320  for (casadi_int i=0; i<n_out_; ++i) {
321  for (casadi_int d=0; d<nfwd; ++d) v[d] = fsens[d][i];
322  res.push_back(horzcat(v));
323  }
324 
325  Dict options = opts;
326  options["allow_duplicate_io_names"] = true;
327 
328  return Function(name, arg, res, inames, onames, options);
329  }

331  Function Rootfinder
332  ::get_reverse(casadi_int nadj, const std::string& name,
333  const std::vector<std::string>& inames,
334  const std::vector<std::string>& onames,
335  const Dict& opts) const {
336  // Symbolic expression for the input
337  std::vector<MX> arg = mx_in();
338  arg[iin_] = MX::sym(arg[iin_].name() + "_guess",
339  Sparsity(arg[iin_].size()));
340  std::vector<MX> res = mx_out();
341  std::vector<std::vector<MX> > aseed = symbolicAdjSeed(nadj, res), asens;
342  ad_reverse(arg, res, aseed, asens, false, false);
343 
344  // Construct return function
345  arg.insert(arg.end(), res.begin(), res.end());
346  std::vector<MX> v(nadj);
347  for (casadi_int i=0; i<n_out_; ++i) {
348  for (casadi_int d=0; d<nadj; ++d) v[d] = aseed[d][i];
349  arg.push_back(horzcat(v));
350  }
351  res.clear();
352  for (casadi_int i=0; i<n_in_; ++i) {
353  for (casadi_int d=0; d<nadj; ++d) v[d] = asens[d][i];
354  res.push_back(horzcat(v));
355  }
356 
357  Dict options = opts;
358  options["allow_duplicate_io_names"] = true;
359  return Function(name, arg, res, inames, onames, options);
360  }

362  int Rootfinder::
363  sp_forward(const bvec_t** arg, bvec_t** res, casadi_int* iw, bvec_t* w, void* mem) const {
364  bvec_t* tmp1 = w; w += n_;
365  bvec_t* tmp2 = w; w += n_;
366 
367  // Propagate dependencies through the function
368  const bvec_t** arg1 = arg+n_in_;
369  std::copy(arg, arg+n_in_, arg1);
370  arg1[iin_] = nullptr;
371  bvec_t** res1 = res+n_out_;
372  std::fill_n(res1, n_out_, static_cast<bvec_t*>(nullptr));
373  res1[iout_] = tmp1;
374  oracle_(arg1, res1, iw, w, 0);
375 
376  // "Solve" in order to propagate to z
377  std::fill_n(tmp2, n_, 0);
378  sp_jac_.spsolve(tmp2, tmp1, false);
379  if (res[iout_]) std::copy(tmp2, tmp2+n_, res[iout_]);
380 
381  // Propagate to auxiliary outputs
382  if (n_out_>1) {
383  arg1[iin_] = tmp2;
384  std::copy(res, res+n_out_, res1);
385  res1[iout_] = nullptr;
386  oracle_(arg1, res1, iw, w, 0);
387  }
388  return 0;
389  }

391  int Rootfinder::sp_reverse(bvec_t** arg, bvec_t** res,
392  casadi_int* iw, bvec_t* w, void* mem) const {
393  bvec_t* tmp1 = w; w += n_;
394  bvec_t* tmp2 = w; w += n_;
395 
396  // Get & clear seed corresponding to implicitly defined variable
397  if (res[iout_]) {
398  std::copy(res[iout_], res[iout_]+n_, tmp1);
399  std::fill_n(res[iout_], n_, 0);
400  } else {
401  std::fill_n(tmp1, n_, 0);
402  }
403 
404  // Propagate dependencies from auxiliary outputs to z
405  bvec_t** res1 = res+n_out_;
406  std::copy(res, res+n_out_, res1);
407  res1[iout_] = nullptr;
408  bvec_t** arg1 = arg+n_in_;
409  std::copy(arg, arg+n_in_, arg1);
410  arg1[iin_] = tmp1;
411  if (n_out_>1) {
412  if (oracle_.rev(arg1, res1, iw, w, 0)) return 1;
413  }
414 
415  // "Solve" in order to get seed
416  std::fill_n(tmp2, n_, 0);
417  sp_jac_.spsolve(tmp2, tmp1, true);
418 
419  // Propagate dependencies through the function
420  for (casadi_int i=0; i<n_out_; ++i) res1[i] = nullptr;
421  res1[iout_] = tmp2;
422  arg1[iin_] = nullptr; // just a guess
423  if (oracle_.rev(arg1, res1, iw, w, 0)) return 1;
424  return 0;
425  }

431  void Rootfinder::
432  ad_forward(const std::vector<MX>& arg, const std::vector<MX>& res,
433  const std::vector<std::vector<MX> >& fseed,
434  std::vector<std::vector<MX> >& fsens,
435  bool always_inline, bool never_inline) const {
436  // Number of directional derivatives
437  casadi_int nfwd = fseed.size();
438  fsens.resize(nfwd);
439 
440  // Quick return if no seeds
441  if (nfwd==0) return;
442 
443  // Propagate through f_
444  std::vector<MX> f_arg(arg);
445  f_arg.at(iin_) = res.at(iout_);
446  std::vector<MX> f_res(res);
447  f_res.at(iout_) = MX(size_in(iin_)); // zero residual
448  std::vector<std::vector<MX> > f_fseed(fseed);
449  for (casadi_int d=0; d<nfwd; ++d) {
450  f_fseed[d].at(iin_) = MX(size_in(iin_)); // ignore seeds for guess
451  }
452  oracle_->call_forward(f_arg, f_res, f_fseed, fsens,
453  always_inline, never_inline);
454 
455  // Get expression of Jacobian
456  Function jac = get_function("jac_f_z");
457  MX J = jac(f_arg).front();
458 
459  // Solve for all the forward derivatives at once
460  std::vector<MX> rhs(nfwd);
461  for (casadi_int d=0; d<nfwd; ++d) rhs[d] = vec(fsens[d][iout_]);
462  rhs = horzsplit(J->get_solve(-horzcat(rhs), false, linsol_));
463  for (casadi_int d=0; d<nfwd; ++d) fsens[d][iout_] = reshape(rhs[d], size_in(iin_));
464 
465  // Propagate to auxiliary outputs
466  if (n_out_>1) {
467  for (casadi_int d=0; d<nfwd; ++d) f_fseed[d][iin_] = fsens[d][iout_];
468  oracle_->call_forward(f_arg, f_res, f_fseed, fsens,
469  always_inline, never_inline);
470  for (casadi_int d=0; d<nfwd; ++d) fsens[d][iout_] = f_fseed[d][iin_]; // Otherwise overwritten
471  }
472  }

474  void Rootfinder::
475  ad_reverse(const std::vector<MX>& arg, const std::vector<MX>& res,
476  const std::vector<std::vector<MX> >& aseed,
477  std::vector<std::vector<MX> >& asens,
478  bool always_inline, bool never_inline) const {
479 
480  // Number of directional derivatives
481  casadi_int nadj = aseed.size();
482  asens.resize(nadj);
483 
484  // Quick return if no seeds
485  if (nadj==0) return;
486 
487  // Get expression of Jacobian
488  std::vector<MX> f_arg(arg);
489  f_arg[iin_] = res.at(iout_);
490  Function jac = get_function("jac_f_z");
491  MX J = jac(f_arg).front();
492 
493  // Get adjoint seeds for calling f
494  std::vector<MX> f_res(res);
495  f_res[iout_] = MX(size_in(iin_)); // zero residual
496  std::vector<std::vector<MX> > f_aseed(nadj);
497  for (casadi_int d=0; d<nadj; ++d) {
498  f_aseed[d].resize(n_out_);
499  for (casadi_int i=0; i<n_out_; ++i) f_aseed[d][i] = i==iout_ ? f_res[iout_] : aseed[d][i];
500  }
501 
502  // Propagate dependencies from auxiliary outputs
503  std::vector<MX> rhs(nadj);
504  std::vector<std::vector<MX> > asens_aux;
505  if (n_out_>1) {
506  oracle_->call_reverse(f_arg, f_res, f_aseed, asens_aux, always_inline, never_inline);
507  for (casadi_int d=0; d<nadj; ++d) rhs[d] = vec(asens_aux[d][iin_] + aseed[d][iout_]);
508  } else {
509  for (casadi_int d=0; d<nadj; ++d) rhs[d] = vec(aseed[d][iout_]);
510  }
511 
512  // Solve for all the adjoint seeds at once
513  rhs = horzsplit(J->get_solve(-horzcat(rhs), true, linsol_));
514  for (casadi_int d=0; d<nadj; ++d) {
515  for (casadi_int i=0; i<n_out_; ++i) {
516  if (i==iout_) {
517  f_aseed[d][i] = reshape(rhs[d], size_out(i));
518  } else {
519  // Avoid counting the auxiliary seeds twice
520  f_aseed[d][i] = MX(size_out(i));
521  }
522  }
523  }
524 
525  // No dependency on guess (1)
526  std::vector<MX> tmp(nadj);
527  for (casadi_int d=0; d<nadj; ++d) {
528  asens[d].resize(n_in_);
529  tmp[d] = asens[d][iin_].is_empty(true) ? MX(size_in(iin_)) : asens[d][iin_];
530  }
531 
532  // Propagate through f_
533  oracle_->call_reverse(f_arg, f_res, f_aseed, asens, always_inline, never_inline);
534 
535  // No dependency on guess (2)
536  for (casadi_int d=0; d<nadj; ++d) {
537  asens[d][iin_] = tmp[d];
538  }
539 
540  // Add contribution from auxiliary outputs
541  if (n_out_>1) {
542  for (casadi_int d=0; d<nadj; ++d) {
543  for (casadi_int i=0; i<n_in_; ++i) if (i!=iin_) asens[d][i] += asens_aux[d][i];
544  }
545  }
546  }
