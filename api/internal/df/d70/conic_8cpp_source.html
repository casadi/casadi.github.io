110  template<typename M>
111  Function qpsol_nlp(const std::string& name, const std::string& solver,
112  const std::map<std::string, M>& qp, const Dict& opts) {
113  // We have: minimize f(x) = 1/2 * x' H x + c'x
114  // subject to lbx <= x <= ubx
115  // lbg <= g(x) = A x + b <= ubg
116  // h(x) >=0 (psd)
117 
118  // Extract 'expand' option
119  Dict opt = opts;
120  auto it = opt.find("expand");
121  bool expand = false;
122  bool error_on_fail = get_from_dict(opts, "error_on_fail", true);
123  if (it!=opt.end()) {
124  expand = it->second;
125  opt.erase(it);
126  }
127  if (expand && M::type_name()=="MX") {
128  Function f = Function("f", qp, {"x", "p"}, {"f", "g", "h"});
129  std::vector<SX> arg = f.sx_in();
130  std::vector<SX> res = f(arg);
131  SXDict qp_mod;
132  for (casadi_int i=0;i<f.n_in();++i) qp_mod[f.name_in(i)] = arg[i];
133  for (casadi_int i=0;i<f.n_out();++i) qp_mod[f.name_out(i)] = res[i];
134  return qpsol_nlp(name, solver, qp_mod, opt);
135  }
136 
137  M x, p, f, g, h;
138  for (auto&& i : qp) {
139  if (i.first=="x") {
140  x = i.second;
141  } else if (i.first=="p") {
142  p = i.second;
143  } else if (i.first=="f") {
144  f = i.second;
145  } else if (i.first=="g") {
146  g = i.second;
147  } else if (i.first=="h") {
148  h = i.second;
149  } else {
150  casadi_error("No such field: " + i.first);
151  }
152  }
153 
154  if (f.is_empty()) f = 0;
155  if (g.is_empty()) g = M(0, 1);
156 
157  // Dimension checks
158  casadi_assert(g.is_dense() && g.is_vector(),
159  "Expected a dense vector 'g', but got " + g.dim() + ".");
160 
161  casadi_assert(f.is_dense() && f.is_scalar(),
162  "Expected a dense scalar 'f', but got " + f.dim() + ".");
163 
164  casadi_assert(x.is_dense() && x.is_vector(),
165  "Expected a dense vector 'x', but got " + x.dim() + ".");
166 
167  casadi_assert(h.is_square(),
168  "Expected a symmetric matrix 'h', but got " + h.dim() + ".");
169 
170  if (g.is_empty(true)) g = M(0, 1); // workaround
171 
172  // Gradient of the objective: gf == Hx + g
173  M gf = M::gradient(f, x);
174 
175  // Identify the linear term in the objective
176  M c = substitute(gf, x, M::zeros(x.sparsity()));
177 
178  // Identify the quadratic term in the objective
179  M H = M::jacobian(gf, x, {{"symmetric", true}});
180 
181  // Identify constant term in the objective
182  Function r("constant_qp", {x, p}, {substitute(f, x, M::zeros(x.sparsity()))});
183 
184  // Identify the constant term in the constraints
185  M b = substitute(g, x, M::zeros(x.sparsity()));
186 
187  // Identify the linear term in the constraints
188  M A = M::jacobian(g, x);
189 
190  // Identify the constant term in the psd constraints
191  M P = substitute(h, x, M::zeros(x.sparsity()));
192 
193  // Identify the linear term in the psd constraints
194  M Q = M::jacobian(h, x);
195 
196  // Create a function for calculating the required matrices vectors
197  Function prob(name + "_qp", {x, p}, {H, c, A, b, Q, P});
198 
199  // Make sure that the problem is sound
200  casadi_assert(!prob.has_free(), "Cannot create '" + prob.name() + "' "
201  "since " + str(prob.get_free()) + " are free.");
202 
203  // Create the QP solver
204  Function conic_f = conic(name + "_qpsol", solver,
205  {{"h", H.sparsity()}, {"a", A.sparsity()},
206  {"p", P.sparsity()}, {"q", Q.sparsity()}}, opt);
207 
208  // Create an MXFunction with the right signature
209  std::vector<MX> ret_in(NLPSOL_NUM_IN);
210  ret_in[NLPSOL_X0] = MX::sym("x0", x.sparsity());
211  ret_in[NLPSOL_P] = MX::sym("p", p.sparsity());
212  ret_in[NLPSOL_LBX] = MX::sym("lbx", x.sparsity());
213  ret_in[NLPSOL_UBX] = MX::sym("ubx", x.sparsity());
214  ret_in[NLPSOL_LBG] = MX::sym("lbg", g.sparsity());
215  ret_in[NLPSOL_UBG] = MX::sym("ubg", g.sparsity());
216  ret_in[NLPSOL_LAM_X0] = MX::sym("lam_x0", x.sparsity());
217  ret_in[NLPSOL_LAM_G0] = MX::sym("lam_g0", g.sparsity());
218  std::vector<MX> ret_out(NLPSOL_NUM_OUT);
219 
220 
221  std::vector<MX> v(NL_NUM_IN);
222  v[NL_X] = ret_in[NLPSOL_X0];
223  v[NL_P] = ret_in[NLPSOL_P];
224  // Evaluate constant part of objective
225  MX rv = r(v)[0];
226  // Get expressions for the QP matrices and vectors
227  v = prob(v);
228 
229  // Call the QP solver
230  std::vector<MX> w(CONIC_NUM_IN);
231  w[CONIC_H] = v.at(0);
232  w[CONIC_G] = v.at(1);
233  w[CONIC_A] = v.at(2);
234  w[CONIC_Q] = v.at(4);
235  w[CONIC_P] = v.at(5);
236  w[CONIC_LBX] = ret_in[NLPSOL_LBX];
237  w[CONIC_UBX] = ret_in[NLPSOL_UBX];
238  w[CONIC_LBA] = ret_in[NLPSOL_LBG] - v.at(3);
239  w[CONIC_UBA] = ret_in[NLPSOL_UBG] - v.at(3);
240  w[CONIC_X0] = ret_in[NLPSOL_X0];
241  w[CONIC_LAM_X0] = ret_in[NLPSOL_LAM_X0];
242  w[CONIC_LAM_A0] = ret_in[NLPSOL_LAM_G0];
243  w = conic_f(w);
244 
245  // Get expressions for the solution
246  ret_out[NLPSOL_X] = reshape(w[CONIC_X], x.size());
247  ret_out[NLPSOL_F] = rv + w[CONIC_COST];
248  ret_out[NLPSOL_G] = reshape(mtimes(v.at(2), w[CONIC_X]), g.size()) + v.at(3);
249  ret_out[NLPSOL_LAM_X] = reshape(w[CONIC_LAM_X], x.size());
250  ret_out[NLPSOL_LAM_G] = reshape(w[CONIC_LAM_A], g.size());
251  ret_out[NLPSOL_LAM_P] = MX::nan(p.sparsity());
252 
253  Dict fun_opts;
254  fun_opts["default_in"] = nlpsol_default_in();
255  fun_opts["error_on_fail"] = error_on_fail;
256 
257  return Function(name, ret_in, ret_out, nlpsol_in(), nlpsol_out(), fun_opts);
258  }

270  // Constructor
271  Conic::Conic(const std::string& name, const std::map<std::string, Sparsity> &st)
272  : FunctionInternal(name) {
273 
274  // Set default options
275  error_on_fail_ = true;
276 
277  P_ = Sparsity(0, 0);
278  for (auto i=st.begin(); i!=st.end(); ++i) {
279  if (i->first=="a") {
280  A_ = i->second;
281  } else if (i->first=="h") {
282  H_ = i->second;
283  } else if (i->first=="q") {
284  Q_ = i->second;
285  } else if (i->first=="p") {
286  P_ = i->second;
287  } else {
288  casadi_error("Unrecognized field in QP structure: " + str(i->first));
289  }
290  }
291 
292  // We need either A or H
293  casadi_assert(!A_.is_null() || !H_.is_null(),
294  "Cannot determine dimension");
295 
296  // Generate A or H
297  if (A_.is_null()) {
298  A_ = Sparsity(0, H_.size2());
299  } else if (H_.is_null()) {
300  H_ = Sparsity(A_.size2(), A_.size2());
301  } else {
302  // Consistency check
303  casadi_assert(A_.size2()==H_.size2(),
304  "Got incompatible dimensions.\n"
305  "min x'Hx + G'x s.t. LBA <= Ax <= UBA :\n"
306  "H: " + H_.dim() + " - A: " + A_.dim() + "\n"
307  "We need: H.size2()==A.size2()");
308  }
309 
310  casadi_assert(H_.is_symmetric(),
311  "Got incompatible dimensions. min x'Hx + G'x\n"
312  "H: " + H_.dim() +
313  "We need H square & symmetric");
314 
315  nx_ = A_.size2();
316  na_ = A_.size1();
317 
318  // Check psd constraints (linear part)
319  if (Q_.is_null()) {
320  Q_ = Sparsity(0, 0);
321  np_ = 0;
322  } else {
323  casadi_assert(Q_.size2()==nx_,
324  "Got incompatible dimensions.\n"
325  "Q: " + Q_.dim() +
326  "We need the product Qx to exist.");
327  np_ = static_cast<casadi_int>(sqrt(static_cast<double>(Q_.size1())));
328  casadi_assert(np_*np_==Q_.size1(),
329  "Got incompatible dimensions.\n"
330  "Q: " + Q_.dim() +
331  "We need Q.size1() to have an integer square root.");
332 
333  Sparsity qsum = reshape(sum2(Q_), np_, np_);
334 
335  casadi_assert(qsum.is_symmetric(),
336  "Got incompatible dimensions.");
337  }
338 
339  // Check psd constraints (constant part)
340  if (P_.is_null()) P_ = Sparsity(np_, np_);
341 
342  casadi_assert(P_.is_symmetric(),
343  "Got incompatible dimensions.\n"
344  "P: " + P_.dim() +
345  "We need P square & symmetric.");
346 
347  casadi_assert(P_.size1()==np_,
348  "Got incompatible dimensions.\n"
349  "P: " + P_.dim() +
350  "We need P " + str(np_) + "-by-" + str(np_) + ".");
351 
352  }

405  // Call the init method of the base class

408  print_problem_ = false;

410  // Read options

415  print_problem_ = op.second;

472  void Conic::check_inputs(const double* lbx, const double* ubx,
473  const double* lba, const double* uba) const {
474  for (casadi_int i=0; i<nx_; ++i) {
475  double lb = lbx ? lbx[i] : 0., ub = ubx ? ubx[i] : 0.;
476  casadi_assert(lb <= ub && lb!=inf && ub!=-inf,
477  "Ill-posed problem detected: "
478  "LBX[" + str(i) + "] <= UBX[" + str(i) + "] was violated. "
479  "Got LBX[" + str(i) + "]=" + str(lb) + " and UBX[" + str(i) + "] = " + str(ub) + ".");
480  }
481  for (casadi_int i=0; i<na_; ++i) {
482  double lb = lba ? lba[i] : 0., ub = uba ? uba[i] : 0.;
483  casadi_assert(lb <= ub && lb!=inf && ub!=-inf,
484  "Ill-posed problem detected: "
485  "LBA[" + str(i) + "] <= UBA[" + str(i) + "] was violated. "
486  "Got LBA[" + str(i) + "] = " + str(lb) + " and UBA[" + str(i) + "] = " + str(ub) + ".");
487  }

511  int Conic::
512  eval(const double** arg, double** res, casadi_int* iw, double* w, void* mem) const {
513  if (print_problem_) {
514  uout() << "H:";
515  DM::print_dense(uout(), H_, arg[CONIC_H], false);
516  uout() << std::endl;
517  uout() << "G:" << std::vector<double>(arg[CONIC_G], arg[CONIC_G]+nx_) << std::endl;
518  uout() << "A:";
519  DM::print_dense(uout(), A_, arg[CONIC_A], false);
520  uout() << std::endl;
521  uout() << "lba:" << std::vector<double>(arg[CONIC_LBA], arg[CONIC_LBA]+na_) << std::endl;
522  uout() << "uba:" << std::vector<double>(arg[CONIC_UBA], arg[CONIC_UBA]+na_) << std::endl;
523  uout() << "lbx:" << std::vector<double>(arg[CONIC_LBX], arg[CONIC_LBX]+nx_) << std::endl;
524  uout() << "ubx:" << std::vector<double>(arg[CONIC_UBX], arg[CONIC_UBX]+nx_) << std::endl;
525  }
526  auto m = static_cast<ConicMemory*>(mem);
527 
528  if (inputs_check_) {
529  check_inputs(arg[CONIC_LBX], arg[CONIC_UBX], arg[CONIC_LBA], arg[CONIC_UBA]);
530  }
531 
532  setup(mem, arg, res, iw, w);
533 
534  int ret = solve(arg, res, iw, w, mem);
535 
536  if (error_on_fail_ && !m->d_qp.success)
537  casadi_error("conic process failed. "
538  "Set 'error_on_fail' option to false to ignore this error.");
539  return ret;
540  }

558  void Conic::sdp_to_socp_init(SDPToSOCPMem& mem) const {
559 
560  Sparsity qsum = reshape(sum2(Q_), np_, np_);
561 
562  // Block detection
563  Sparsity aggregate = qsum+P_;
564 
565  std::vector<casadi_int> p;
566  casadi_int nb = aggregate.scc(p, mem.r);
567 
568  std::string pattern_message = "Pattern not recognised";
569 
570  casadi_assert(p==range(p.size()), pattern_message);
571 
572  const casadi_int* row = aggregate.row();
573  const casadi_int* colind = aggregate.colind();
574 
575  // Check fishbone-structure
576  for (casadi_int i=0;i<nb;++i) {
577  casadi_int block_size = mem.r[i+1]-mem.r[i];
578  // number of nonzeros in column mem.r[i+1]-1
579  casadi_int nz = colind[mem.r[i+1]]-colind[mem.r[i+1]-1];
580  // Last column of block should be dense
581  casadi_assert(nz==block_size, pattern_message);
582  for (casadi_int k=0;k<block_size-1;++k) {
583  casadi_assert(colind[mem.r[i]+k+1]-colind[mem.r[i]+k], pattern_message);
584  casadi_assert(*(row++)==k+mem.r[i], pattern_message);
585  casadi_assert(*(row++)==mem.r[i]+block_size-1, pattern_message);
586  }
587 
588  for (casadi_int k=0;k<block_size;++k)
589  casadi_assert(*(row++)==k+mem.r[i], pattern_message);
590  }
591 
611  /*
612 
613  Aggregate pattern:
614 
615  x (x)
616  x(x)
617  xx(x)
618  x (x)
619  x(x)
620  xx(x)
621 
622  We are interested in the parts in parenthesis (target).
623 
624  Find out which rows in Q correspond to those targets
625  */
626 
627  // Lookup vector for target start and end
628  std::vector<casadi_int> target_start(nb), target_stop(nb);
629  for (casadi_int i=0;i<nb;++i) {
630  target_start[i] = (mem.r[i+1]-1)*aggregate.size1()+mem.r[i];
631  target_stop[i] = target_start[i]+mem.r[i+1]-mem.r[i];
632  }
633 
634  // Collect the nonzero indices in Q that that correspond to the target area
635  std::vector<casadi_int> q_nz;
636  // Triplet form for map_Q sparsity
637  std::vector<casadi_int> q_row, q_col;
638 
639  // Loop over Q's columns (decision variables)
640  for (casadi_int j=0; j<Q_.size2(); ++j) {
641  casadi_int block_index = 0;
642  // Loop over Q's rows
643  for (casadi_int k=Q_.colind(j); k<Q_.colind(j+1); ++k) {
644  casadi_int i = Q_.row(k);
645 
646  // Increment block_index if i runs ahead
647  while (i>target_stop[block_index] && block_index<nb-1) block_index++;
648 
649  if (i>=target_start[block_index] && i<target_stop[block_index]) {
650  // Got a nonzero in the target region
651  q_nz.push_back(k);
652  q_row.push_back(mem.r[block_index]+i-target_start[block_index]);
653  q_col.push_back(j);
654  }
655  }
656  }
657 
658  mem.map_Q = IM::triplet(q_row, q_col, q_nz, mem.r[nb], nx_);
659 
660  // Add the [X1;Z1;X2;Z2;...] part
661  mem.map_Q = horzcat(mem.map_Q, -IM::eye(mem.r[nb])).T();
662 
663  // Get maximum nonzero count of any column
664  casadi_int max_nnz = 0;
665  for (casadi_int i=0;i<mem.map_Q.size2();++i) {
666  max_nnz = std::max(max_nnz, mem.map_Q.colind(i+1)-mem.map_Q.colind(i));
667  }
668 
669  // ind/val size needs to cover max nonzero count
670  mem.indval_size = std::max(nx_, max_nnz);
671 
672  // Collect the indices for the P target area
673  mem.map_P.resize(mem.r[nb], -1);
674  for (casadi_int i=0;i<nb;++i) {
675  for (casadi_int k=P_.colind(mem.r[i+1]-1); k<P_.colind(mem.r[i+1]); ++k) {
676  casadi_int r = P_.row(k);
677  mem.map_P[r] = k;
678  }
679  }
680 
681  // ind/val size needs to cover blocksize
682  for (casadi_int i=0;i<nb;++i)
683  mem.indval_size = std::max(mem.indval_size, mem.r[i+1]-mem.r[i]);
684 
685  // Get the transpose and mapping
686  mem.AT = A_.transpose(mem.A_mapping);
687  }
