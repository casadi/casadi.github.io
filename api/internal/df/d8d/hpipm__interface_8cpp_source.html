85

95  } else if (op.first=="nx") {
96  nxs_ = op.second;
97  struct_cnt++;
98  } else if (op.first=="nu") {
99  nus_ = op.second;
100  struct_cnt++;
101  } else if (op.first=="ng") {
102  ngs_ = op.second;
103  struct_cnt++;
104  } else if (op.first=="inf") {
105  inf_ = op.second;
106  } else if (op.first=="hpipm") {

108  auto it = hopts.find("mode");

123  d_ocp_qp_ipm_arg_set_default(mode, &hpipm_options_);
124 
125  for (auto&& op : hopts) {
126  if (op.first=="mu0") {
127  hpipm_options_.mu0 = op.second;
128  } else if (op.first=="alpha_min") {
129  hpipm_options_.alpha_min = op.second;
130  } else if (op.first=="res_g_max") {
131  hpipm_options_.res_g_max = op.second;
132  } else if (op.first=="res_b_max") {
133  hpipm_options_.res_b_max = op.second;
134  } else if (op.first=="res_d_max") {
135  hpipm_options_.res_d_max = op.second;
136  } else if (op.first=="res_m_max") {
137  hpipm_options_.res_m_max = op.second;
138  } else if (op.first=="iter_max") {
139  hpipm_options_.iter_max = op.second;
140  } else if (op.first=="stat_max") {
141  hpipm_options_.stat_max = op.second;
142  } else if (op.first=="pred_corr") {
143  hpipm_options_.pred_corr = op.second;
144  } else if (op.first=="cond_pred_corr") {
145  hpipm_options_.cond_pred_corr = op.second;
146  } else if (op.first=="itref_pred_max") {
147  hpipm_options_.itref_pred_max = op.second;
148  } else if (op.first=="itref_corr_max") {
149  hpipm_options_.itref_corr_max = op.second;
150  } else if (op.first=="reg_prim") {
151  hpipm_options_.reg_prim = op.second;
152  } else if (op.first=="lq_fact") {
153  hpipm_options_.lq_fact = op.second;
154  } else if (op.first=="lam_min") {
155  hpipm_options_.lam_min = op.second;
156  } else if (op.first=="t_min") {
157  hpipm_options_.t_min = op.second;
158  } else if (op.first=="warm_start") {
159  hpipm_options_.warm_start = op.second;
160  } else if (op.first=="abs_form") {
161  hpipm_options_.abs_form = op.second;
162  } else if (op.first=="comp_dual_sol_eq") {
163  hpipm_options_.comp_dual_sol_eq = op.second;
164  } else if (op.first=="comp_res_exit") {
165  hpipm_options_.comp_res_exit = op.second;
166  } else if (op.first=="mode") {
167  // already covered
168  } else {
169  casadi_error("Unknown option.");
170  }
171  }
172  }
173  }
174 
175  bool detect_structure = struct_cnt==0;
176  casadi_assert(struct_cnt==0 || struct_cnt==4,
177  "You must either set all of N, nx, nu, ng; "
178  "or set none at all (automatic detection).");
179 
180  const std::vector<int>& nx = nxs_;
181  const std::vector<int>& ng = ngs_;
182  const std::vector<int>& nu = nus_;
183 
184  Sparsity lamg_csp_, lam_ulsp_, lam_uusp_, lam_xlsp_, lam_xusp_, lam_clsp_;
185 
186  if (detect_structure) {
187  /* General strategy: look for the xk+1 diagonal part in A
188  */
189 
190  // Find the right-most column for each row in A -> A_skyline
191  // Find the second-to-right-most column -> A_skyline2
192  // Find the left-most column -> A_bottomline
193  Sparsity AT = A_.T();
194  std::vector<casadi_int> A_skyline;
195  std::vector<casadi_int> A_skyline2;
196  std::vector<casadi_int> A_bottomline;
197  for (casadi_int i=0;i<AT.size2();++i) {
198  casadi_int pivot = AT.colind()[i+1];
199  A_bottomline.push_back(AT.row()[AT.colind()[i]]);
200  if (pivot>AT.colind()[i]) {
201  A_skyline.push_back(AT.row()[pivot-1]);
202  if (pivot>AT.colind()[i]+1) {
203  A_skyline2.push_back(AT.row()[pivot-2]);
204  } else {
205  A_skyline2.push_back(-1);
206  }
207  } else {
208  A_skyline.push_back(-1);
209  A_skyline2.push_back(-1);
210  }
211  }
212 
213  /*
214  Loop over the right-most columns of A:
215  they form the diagonal part due to xk+1 in gap constraints.
216  detect when the diagonal pattern is broken -> new stage
217  */
218  casadi_int pivot = 0; // Current right-most element
219  casadi_int start_pivot = pivot; // First right-most element that started the stage
220  casadi_int cg = 0; // Counter for non-gap-closing constraints
221  for (casadi_int i=0;i<na_;++i) { // Loop over all rows
222  bool commit = false; // Set true to jump to the stage
223  if (A_skyline[i]>pivot+1) { // Jump to a diagonal in the future
224  nus_.push_back(A_skyline[i]-pivot-1); // Size of jump equals number of states
225  commit = true;
226  } else if (A_skyline[i]==pivot+1) { // Walking the diagonal
227  if (A_skyline2[i]<start_pivot) { // Free of below-diagonal entries?
228  pivot++;
229  } else {
230  nus_.push_back(0); // We cannot but conclude that we arrived at a new stage
231  commit = true;
232  }
233  } else { // non-gap-closing constraint detected
234  cg++;
235  }
236 
237  if (commit) {
238  nxs_.push_back(pivot-start_pivot+1);
239  ngs_.push_back(cg); cg=0;
240  start_pivot = A_skyline[i];
241  pivot = A_skyline[i];
242  }
243  }
244  nxs_.push_back(pivot-start_pivot+1);
245 
246  // Correction for k==0
247  nxs_[0] = A_skyline[0];
248  nus_[0] = 0;
249  ngs_.erase(ngs_.begin());
250  casadi_int cN=0;
251  for (casadi_int i=na_-1;i>=0;--i) {
252  if (A_bottomline[i]<start_pivot) break;
253  cN++;
254  }
255  ngs_.push_back(cg-cN);
256  ngs_.push_back(cN);
257 
258  N_ = nus_.size();
259  if (verbose_) {
260  casadi_message("Detected structure: N " + str(N_) + ", nx " + str(nx) + ", "
261  "nu " + str(nu) + ", ng " + str(ng) + ".");
262  }
263  nus_.push_back(0);
264  }
265 
266  uout() << "nx,nu,ng" << nx << nu << ng << std::endl;
267 
268  casadi_assert_dev(nx.size()==N_+1);
269  casadi_assert_dev(nu.size()==N_+1);
270  casadi_assert_dev(ng.size()==N_+1);
271 
272  casadi_assert(nx_ == std::accumulate(nx.begin(), nx.end(), 0) + // NOLINT
273  std::accumulate(nu.begin(), nu.end(), 0),
274  "sum(nx)+sum(nu) = must equal total size of variables (" + str(nx_) + "). "
275  "Structure is: N " + str(N_) + ", nx " + str(nx) + ", "
276  "nu " + str(nu) + ", ng " + str(ng) + ".");
277  casadi_assert(na_ == std::accumulate(nx.begin()+1, nx.end(), 0) + // NOLINT
278  std::accumulate(ng.begin(), ng.end(), 0),
279  "sum(nx+1)+sum(ng) = must equal total size of constraints (" + str(na_) + "). "
280  "Structure is: N " + str(N_) + ", nx " + str(nx) + ", "
281  "nu " + str(nu) + ", ng " + str(ng) + ".");
282  // Load library HPIPM when applicable
283  std::string searchpath;
284 
285  /* Disassemble A input into:
286  A B I
287  C D
288  A B I
289  C D
290  */
291  casadi_int offset_r = 0, offset_c = 0;
292  for (casadi_int k=0;k<N_;++k) { // Loop over blocks
293  A_blocks.push_back({offset_r, offset_c, nx[k+1], nx[k]});
294  B_blocks.push_back({offset_r, offset_c+nx[k], nx[k+1], nu[k]});
295  C_blocks.push_back({offset_r+nx[k+1], offset_c, ng[k], nx[k]});
296  D_blocks.push_back({offset_r+nx[k+1], offset_c+nx[k], ng[k], nu[k]});
297 
298  offset_c+= nx[k]+nu[k];
299  if (k+1<N_)
300  I_blocks.push_back({offset_r, offset_c, nx[k+1], nx[k+1]});
301  else
302  I_blocks.push_back({offset_r, offset_c, nx[k+1], nx[k+1]});
303  offset_r+= nx[k+1]+ng[k];
304  }
305 
306  C_blocks.push_back({offset_r, offset_c, ng[N_], nx[N_]});
307  D_blocks.push_back({offset_r, offset_c+nx[N_], ng[N_], nu[N_]});
308 
309  Asp_ = blocksparsity(na_, nx_, A_blocks);
310  Bsp_ = blocksparsity(na_, nx_, B_blocks);
311  Csp_ = blocksparsity(na_, nx_, C_blocks);
312  Dsp_ = blocksparsity(na_, nx_, D_blocks);
313  Isp_ = blocksparsity(na_, nx_, I_blocks, true);
314 
315  Sparsity total = Asp_ + Bsp_ + Csp_ + Dsp_ + Isp_;
316 
317  casadi_assert((A_ + total).nnz() == total.nnz(),
318  "HPIPM: specified structure of A does not correspond to what the interface can handle. "
319  "Structure is: N " + str(N_) + ", nx " + str(nx) + ", nu " + str(nu) + ", "
320  "ng " + str(ng) + ".");
321  casadi_assert_dev(total.nnz() == Asp_.nnz() + Bsp_.nnz() + Csp_.nnz() + Dsp_.nnz()
322  + Isp_.nnz());
323 
324  /* Disassemble H input into:
325  Q S'
326  S R
327  Q S'
328  S R
329 
330  Multiply by 2
331  */
332  casadi_int offset = 0;
333  for (casadi_int k=0;k<N_+1;++k) { // Loop over blocks
334  R_blocks.push_back({offset+nx[k], offset+nx[k], nu[k], nu[k]});
335  S_blocks.push_back({offset+nx[k], offset, nu[k], nx[k]});
336  Q_blocks.push_back({offset, offset, nx[k], nx[k]});
337  offset+= nx[k]+nu[k];
338  }
339 
340  Rsp_ = blocksparsity(nx_, nx_, R_blocks);
341  Ssp_ = blocksparsity(nx_, nx_, S_blocks);
342  Qsp_ = blocksparsity(nx_, nx_, Q_blocks);
343 
344  total = Rsp_ + Ssp_ + Qsp_ + Ssp_.T();
345  casadi_assert((H_ + total).nnz() == total.nnz(),
346  "HPIPM: specified structure of H does not correspond to what the interface can handle. "
347  "Structure is: N " + str(N_) + ", nx " + str(nx) + ", nu " + str(nu) + ", "
348  "ng " + str(ng) + ".");
349  casadi_assert_dev(total.nnz() == Rsp_.nnz() + 2*Ssp_.nnz() + Qsp_.nnz());
350 
351  /* Disassemble LBA/UBA input into:
352  b
353  lg/ug
354 
355  b
356  lg/ug
357  */
358  offset = 0;
359 
360  for (casadi_int k=0;k<N_;++k) {
361  b_blocks.push_back({offset, 0, nx[k+1], 1}); offset+= nx[k+1];
362  lug_blocks.push_back({offset, 0, ng[k], 1}); offset+= ng[k];
363  }
364  lug_blocks.push_back({offset, 0, ng[N_], 1});
365 
366  bsp_ = blocksparsity(na_, 1, b_blocks);
367  lugsp_ = blocksparsity(na_, 1, lug_blocks);
368  total = bsp_ + lugsp_;
369  casadi_assert_dev(total.nnz() == bsp_.nnz() + lugsp_.nnz());
370  casadi_assert_dev(total.nnz() == na_);
371 
372  /* Disassemble G/X0 input into:
373  r/u
374  q/x
375 
376  r/u
377  q/x
378  */
379  offset = 0;
380 
381  for (casadi_int k=0;k<N_+1;++k) {
382  x_blocks.push_back({offset, 0, nx[k], 1}); offset+= nx[k];
383  u_blocks.push_back({offset, 0, nu[k], 1}); offset+= nu[k];
384  }
385 
386  usp_ = blocksparsity(nx_, 1, u_blocks);
387  xsp_ = blocksparsity(nx_, 1, x_blocks);
388  total = usp_ + xsp_;
389  casadi_assert_dev(total.nnz() == usp_.nnz() + xsp_.nnz());
390  casadi_assert_dev(total.nnz() == nx_);
391 
392  std::vector< casadi_hpipm_block > theirs_u_blocks, theirs_x_blocks;
393  offset = 0;
394 
395  for (casadi_int k=0;k<N_;++k) {
396  theirs_u_blocks.push_back({offset, 0, nu[k], 1}); offset+= nu[k];
397  theirs_x_blocks.push_back({offset, 0, nx[k], 1}); offset+= nx[k];
398  }
399  theirs_x_blocks.push_back({offset, 0, nx[N_], 1});
400 
401  theirs_usp_ = blocksparsity(nx_, 1, theirs_u_blocks);
402  theirs_xsp_ = blocksparsity(nx_, 1, theirs_x_blocks);
403  total = theirs_usp_ + theirs_xsp_;
404  casadi_assert_dev(total.nnz() == theirs_usp_.nnz() + theirs_xsp_.nnz());
405  casadi_assert_dev(total.nnz() == nx_);
406 
407  offset = 0;
408  std::vector< casadi_hpipm_block > lamg_gap_blocks;
409  for (casadi_int k=0;k<N_;++k) {
410  lamg_gap_blocks.push_back({offset, 0, nx[k+1], 1});offset+= nx[k+1] + ng[k];
411  }
412  lamg_gapsp_ = blocksparsity(na_, 1, lamg_gap_blocks);
413  lamg_csp_ = lamg_gapsp_.pattern_inverse();
414 
415  offset = 0;
416 
417  for (casadi_int k=0;k<N_;++k) {
418  lam_ul_blocks.push_back({offset, 0, nu[k], 1}); offset+= nu[k];
419  lam_xl_blocks.push_back({offset, 0, nx[k], 1}); offset+= nx[k];
420  lam_uu_blocks.push_back({offset, 0, nu[k], 1}); offset+= nu[k];
421  lam_xu_blocks.push_back({offset, 0, nx[k], 1}); offset+= nx[k];
422  lam_cl_blocks.push_back({offset, 0, ng[k], 1}); offset+= ng[k];
423  lam_cu_blocks.push_back({offset, 0, ng[k], 1}); offset+= ng[k];
424  }
425  lam_xl_blocks.push_back({offset, 0, nx[N_], 1}); offset+= nx[N_];
426  lam_xu_blocks.push_back({offset, 0, nx[N_], 1}); offset+= nx[N_];
427  lam_cl_blocks.push_back({offset, 0, ng[N_], 1}); offset+= ng[N_];
428  lam_cu_blocks.push_back({offset, 0, ng[N_], 1}); offset+= ng[N_];
429 
430  lam_ulsp_ = blocksparsity(offset, 1, lam_ul_blocks);
431  lam_uusp_ = blocksparsity(offset, 1, lam_uu_blocks);
432  lam_xlsp_ = blocksparsity(offset, 1, lam_xl_blocks);
433  lam_xusp_ = blocksparsity(offset, 1, lam_xu_blocks);
434  lam_clsp_ = blocksparsity(offset, 1, lam_cl_blocks);
435  lam_cusp_ = blocksparsity(offset, 1, lam_cu_blocks);
436 
437  pisp_ = Sparsity::dense(std::accumulate(nx.begin()+1, nx.end(), 0), 1); // NOLINT
438 
439  total = lam_ulsp_ + lam_uusp_ + lam_xlsp_ + lam_xusp_ + lam_clsp_ + lam_cusp_;
440  casadi_assert_dev(total.nnz() == lam_ulsp_.nnz() + lam_uusp_.nnz() + lam_xlsp_.nnz() +
441  lam_xusp_.nnz() + lam_clsp_.nnz() + lam_cusp_.nnz());
442  casadi_assert_dev(total.nnz() == offset);
443 
444  theirs_Xsp_ = Sparsity::dense(std::accumulate(nx.begin(), nx.end(), 0), 1); // NOLINT
445  theirs_Usp_ = Sparsity::dense(std::accumulate(nu.begin(), nu.end(), 0), 1); // NOLINT
446 
447  nus_.push_back(0);
448  zeros_.resize(N_+1, 0);
449 
450  uout() << "nus" << nus_ << std::endl;
451 
452  set_hpipm_prob();
453 
454  // Allocate memory
455  casadi_int sz_arg, sz_res, sz_w, sz_iw;
456  casadi_hpipm_work(&p_, &sz_arg, &sz_res, &sz_iw, &sz_w);
457 
458  uout() << sz_arg << std::endl;
459  uout() << sz_res << std::endl;
460  uout() << sz_iw << std::endl;
461  uout() << sz_w << std::endl;
462 
463  alloc_arg(sz_arg, true);
464  alloc_res(sz_res, true);
465  alloc_iw(sz_iw, true);
466  alloc_w(sz_w, true);

472  std::vector<casadi_int> hpipm_blocks_pack(const std::vector<casadi_hpipm_block>& blocks) {
473  size_t N = blocks.size();
474  std::vector<casadi_int> ret(4*N);
475  casadi_int* r = get_ptr(ret);
476  for (casadi_int i=0;i<N;++i) {
477  *r++ = blocks[i].offset_r;
478  *r++ = blocks[i].offset_c;
479  *r++ = blocks[i].rows;
480  *r++ = blocks[i].cols;
481  }
482  return ret;
483  }

746  Sparsity HpipmInterface::blocksparsity(casadi_int rows, casadi_int cols,
747  const std::vector<casadi_hpipm_block>& blocks, bool eye) {
748  DM r(rows, cols);
749  for (auto && b : blocks) {
750  if (eye) {
751  r(range(b.offset_r, b.offset_r+b.rows),
752  range(b.offset_c, b.offset_c+b.cols)) = DM::eye(b.rows);
753  casadi_assert_dev(b.rows==b.cols);
754  } else {
755  r(range(b.offset_r, b.offset_r+b.rows),
756  range(b.offset_c, b.offset_c+b.cols)) = DM::zeros(b.rows, b.cols);
757  }
758  }
759  return r.sparsity();
760  }

762  const std::vector<casadi_hpipm_block>& blocks, bool eye) {

769  casadi_assert_dev(blocks[k].rows==blocks[k].cols);
770  offset+=blocks[k].rows;
771  } else {
772  offset+=blocks[k].rows*blocks[k].cols;
773  }
774  }
